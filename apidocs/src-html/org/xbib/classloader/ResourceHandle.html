002    package org.xbib.classloader;
003    
004    import java.io.IOException;
005    import java.io.InputStream;
006    import java.net.URL;
007    import java.security.cert.Certificate;
008    import java.util.jar.Attributes;
009    import java.util.jar.Manifest;
010    
011    /**
012     * This is a handle (a connection) to some resource, which may
013     * be a class, native library, text file, image, etc. Handles are returned
014     * by a ResourceFinder. A resource handle allows easy access to the resource data
015     * (using methods {@link #getInputStream} or {@link #getBytes}) as well as
016     * access resource metadata, such as attributes, certificates, etc.
017     * <p/>
018     * As soon as the handle is no longer in use, it should be explicitly
019     * {@link #close}d, similarly to I/O streams.
020     */
021    public interface ResourceHandle {
022        /**
023         * Return the name of the resource. The name is a "/"-separated path
024         * name that identifies the resource.
025         */
026        String getName();
027    
028        /**
029         * Returns the URL of the resource.
030         */
031        URL getUrl();
032    
033        /**
034         * Does this resource refer to a directory.  Directory resources are commly used
035         * as the basis for a URL in client application.  A directory resource has 0 bytes for it's content.
036         */
037        boolean isDirectory();
038    
039        /**
040         * Returns the CodeSource URL for the class or resource.
041         */
042        URL getCodeSourceUrl();
043    
044        /**
045         * Returns and InputStream for reading this resource data.
046         */
047        InputStream getInputStream() throws IOException;
048    
049        /**
050         * Returns the length of this resource data, or -1 if unknown.
051         */
052        int getContentLength();
053    
054        /**
055         * Returns this resource data as an array of bytes.
056         */
057        byte[] getBytes() throws IOException;
058    
059        /**
060         * Returns the Manifest of the JAR file from which this resource
061         * was loaded, or null if none.
062         */
063        Manifest getManifest() throws IOException;
064    
065        /**
066         * Return the Certificates of the resource, or null if none.
067         */
068        Certificate[] getCertificates();
069    
070        /**
071         * Return the Attributes of the resource, or null if none.
072         */
073        Attributes getAttributes() throws IOException;
074    
075        /**
076         * Closes a connection to the resource indentified by this handle. Releases
077         * any I/O objects associated with the handle.
078         */
079        void close();
080    }
