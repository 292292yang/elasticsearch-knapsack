002    package org.xbib.classloader;
003    
004    import java.net.URL;
005    import java.util.Enumeration;
006    
007    /**
008     * Abstraction of resource searching policy. Given resource name, the resource
009     * finder performs implementation-specific lookup, and, if it is able to locate
010     * the resource, returns the {@link AbstractResourceHandle handle(s)} or URL(s)
011     * of it.
012     */
013    public interface ResourceFinder {
014    
015        /**
016         * Find the resource by name and return URL of it if found.
017         *
018         * @param name the resource name
019         * @return resource URL or null if resource was not found
020         */
021        URL findResource(String name);
022    
023        /**
024         * Find all resources with given name and return enumeration of their URLs.
025         *
026         * @param name the resource name
027         * @return enumeration of resource URLs (possibly empty).
028         */
029        Enumeration<URL> findResources(String name);
030    
031        /**
032         * Get the resource by name and, if found, open connection to it and return
033         * the {@link AbstractResourceHandle handle} of it.
034         *
035         * @param name the resource name
036         * @return resource handle or null if resource was not found
037         */
038        ResourceHandle getResource(String name);
039    }
