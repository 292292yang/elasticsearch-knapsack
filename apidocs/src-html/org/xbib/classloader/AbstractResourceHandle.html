002    package org.xbib.classloader;
003    
004    import java.io.ByteArrayOutputStream;
005    import java.io.IOException;
006    import java.io.InputStream;
007    import java.security.cert.Certificate;
008    import java.util.jar.Attributes;
009    import java.util.jar.Manifest;
010    
011    public abstract class AbstractResourceHandle implements ResourceHandle {
012    
013        public byte[] getBytes() throws IOException {
014            InputStream in = getInputStream();
015            try {
016                return load(in);
017            } finally {
018                if (in != null) {
019                    try {
020                        in.close();
021                    } catch (Exception ignored) {
022                    }
023                }
024            }
025        }
026    
027        public Manifest getManifest() throws IOException {
028            return null;
029        }
030    
031        public Certificate[] getCertificates() {
032            return null;
033        }
034    
035        public Attributes getAttributes() throws IOException {
036            Manifest m = getManifest();
037            if (m == null) {
038                return null;
039            }
040    
041            String entry = getUrl().getFile();
042            return m.getAttributes(entry);
043        }
044    
045        public void close() {
046        }
047    
048        public String toString() {
049            return "[" + getName() + ": " + getUrl() + "; code source: " + getCodeSourceUrl() + "]";
050        }
051    
052        static byte[] load(InputStream inputStream) throws IOException {
053            try {
054                byte[] buffer = new byte[4096];
055                ByteArrayOutputStream out = new ByteArrayOutputStream();
056                for (int count = inputStream.read(buffer); count >= 0; count = inputStream.read(buffer)) {
057                    out.write(buffer, 0, count);
058                }
059                return out.toByteArray();
060            } finally {
061                if (inputStream != null) {
062                    try {
063                        inputStream.close();
064                    } catch (Exception ignored) {
065                    }
066                }
067            }
068        }
069    }
