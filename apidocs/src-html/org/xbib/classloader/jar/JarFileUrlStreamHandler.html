002    package org.xbib.classloader.jar;
003    
004    import java.io.File;
005    import java.io.FileNotFoundException;
006    import java.io.IOException;
007    import java.net.MalformedURLException;
008    import java.net.URL;
009    import java.net.URLConnection;
010    import java.net.URLStreamHandler;
011    import java.util.jar.JarEntry;
012    import java.util.jar.JarFile;
013    
014    public class JarFileUrlStreamHandler extends URLStreamHandler {
015    
016        private URL expectedUrl;
017    
018        private final JarFile jarFile;
019    
020        private final JarEntry jarEntry;
021    
022        public static URL createURL(JarFile jarFile, JarEntry jarEntry) throws MalformedURLException {
023            return createURL(jarFile, jarEntry, new File(jarFile.getName()).toURI().toURL());
024        }
025    
026        public static URL createURL(JarFile jarFile, JarEntry jarEntry, URL codeSource) throws MalformedURLException {
027            JarFileUrlStreamHandler handler = new JarFileUrlStreamHandler(jarFile, jarEntry);
028            URL url = new URL("jar", "", -1, codeSource + "!/" + jarEntry.getName(), handler);
029            handler.setExpectedUrl(url);
030            return url;
031        }
032    
033        public JarFileUrlStreamHandler(JarFile jarFile, JarEntry jarEntry) {
034            if (jarFile == null) {
035                throw new NullPointerException("jarFile is null");
036            }
037            if (jarEntry == null) {
038                throw new NullPointerException("jarEntry is null");
039            }
040    
041            this.jarFile = jarFile;
042            this.jarEntry = jarEntry;
043        }
044    
045        public void setExpectedUrl(URL expectedUrl) {
046            if (expectedUrl == null) {
047                throw new NullPointerException("expectedUrl is null");
048            }
049            this.expectedUrl = expectedUrl;
050        }
051    
052        public URLConnection openConnection(URL url) throws IOException {
053            if (expectedUrl == null) {
054                throw new IllegalStateException("expectedUrl was not set");
055            }
056            // the caller copied the URL reusing a stream handler from a previous call
057            if (!expectedUrl.equals(url)) {
058                // the new url is supposed to be within our context, so it must have a jar protocol
059                if (!"jar".equals(url.getProtocol())) {
060                    throw new IllegalArgumentException("Unsupported protocol " + url.getProtocol());
061                }
062                // split the path at "!/" into the file part and entry part
063                String path = url.getPath();
064                String[] chunks = path.split("!/", 2);
065                // if we only got only one chunk, it didn't contain the required "!/" delimiter
066                if (chunks.length == 1) {
067                    throw new MalformedURLException("Url does not contain a '!' character: " + url);
068                }
069                String file = chunks[0];
070                String entryPath = chunks[1];
071                // this handler only supports jars on the local file system
072                if (!file.startsWith("file:")) {
073                    // let the system handler deal with this
074                    return new URL(url.toExternalForm()).openConnection();
075                }
076                file = file.substring("file:".length());
077                // again the new url is supposed to be within our context so it must reference the same jar file
078                if (!jarFile.getName().equals(file)) {
079                    // let the system handler deal with this
080                    return new URL(url.toExternalForm()).openConnection();
081                }
082                // get the entry
083                JarEntry newEntry = jarFile.getJarEntry(entryPath);
084                if (newEntry == null) {
085                    throw new FileNotFoundException("Entry not found: " + url);
086                }
087                return new JarFileUrlConnection(url, jarFile, newEntry);
088            }
089            return new JarFileUrlConnection(url, jarFile, jarEntry);
090        }
091    }
