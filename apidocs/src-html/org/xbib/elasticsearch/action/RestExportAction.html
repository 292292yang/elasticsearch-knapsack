072    /**
073     * The knapsack export action performs a scan/scroll action over a user defined query
074     * and stores the result in an archive or transfers it to an ES cluster
075     */
076    public class RestExportAction extends BaseRestHandler {

088        @Inject
089        public RestExportAction(Settings settings,
090                                Environment environment,
091                                Client client,
092                                RestController controller,
093                                SettingsFilter settingsFilter,
094                                ClusterService clusterService,
095                                ClusterName clusterName) {
096            super(settings, client);
097            this.environment = environment;
098            this.clusterName = clusterName;
099            this.settingsFilter = settingsFilter;
100            this.knapsackHelper = new KnapsackHelper(client, clusterService);
101            this.executor = EsExecutors.newFixed(10, 0, EsExecutors.daemonThreadFactory(settings, "knapsack-export"));
102    
103            controller.registerHandler(POST, "/_export", this);
104            controller.registerHandler(POST, "/{index}/_export", this);
105            controller.registerHandler(POST, "/{index}/{type}/_export", this);
106    
107            controller.registerHandler(POST, "/_export/{mode}", this);
108            controller.registerHandler(POST, "/{index}/_export/{mode}", this);
109            controller.registerHandler(POST, "/{index}/{type}/_export/{mode}", this);
110    
111            controller.registerHandler(GET, "/_export/state", new RestKnapsackExportState());
112            controller.registerHandler(POST, "/_export/abort", new RestKnapsackExportAbort());
113        }

115        @Override
116        public void handleRequest(final RestRequest request, RestChannel channel) {
117            try {
118                final XContentBuilder builder = restContentBuilder(request);
119                final boolean copy = "copy".equals(request.param("mode"));
120                if (copy) {
121                    final URI destination = KnapsackHelper.getAddress(client, clusterName, request.param("host"), request.paramAsInt("port", 0),
122                            request.param("cluster"));
123                    executor.execute(new ExportThread(request, destination));
124                    builder.startObject()
125                            .field("running", true)
126                            .field("mode", request.param("mode") != null ? request.param("mode") : "export")
127                            .field("path", destination)
128                            .endObject();
129                } else {
130                    final String index = request.param("index", "_all");
131                    final String type = request.param("type");
132                    final String defaultSpec = index + (type != null ? "_" + type : "") + ".tar.gz";
133                    final String path = request.param("path", defaultSpec);
134                    URI uri = URI.create("file:" + path);
135                    ConnectionService<Session<KnapsackPacket>> service = ConnectionService.getInstance();
136                    ConnectionFactory<Session<KnapsackPacket>> factory = service.getConnectionFactory(uri);
137                    final Connection<Session<KnapsackPacket>> connection = factory.getConnection(uri);
138                    final Session<KnapsackPacket> session = connection.createSession();
139                    session.open(Session.Mode.WRITE);
140                    if (session.isOpen()) {
141                        executor.execute(new ExportThread(request, connection, session));
142                        builder.startObject()
143                                .field("running", true)
144                                .field("mode", request.param("mode") != null ? request.param("mode") : "export")
145                                .field("archive", factory.getName())
146                                .field("path", connection.getURI())
147                                .endObject();
148                    } else {
149                        builder.startObject()
150                                .field("running", false)
151                                .endObject();
152                        channel.sendResponse(new XContentRestResponse(request, BAD_REQUEST, builder));
153                        return;
154                    }
155                }
156                channel.sendResponse(new XContentRestResponse(request, OK, builder));
157            } catch (Throwable ex) {
158                try {
159                    logger.error(ex.getMessage(), ex);
160                    channel.sendResponse(new XContentThrowableRestResponse(request, ex));
161                } catch (Exception ex2) {
162                    logger.error(ex2.getMessage(), ex2);
163                }
164            }
165        }

167        private class ExportThread extends Thread {
168    
169            private final RestRequest request;
170    
171            private BulkClient bulkClient;
172    
173            private Connection<Session<KnapsackPacket>> connection;
174    
175            private Session<KnapsackPacket> session;
176    
177            private TimeValue timeout;
178    
179            private URI destination;
180    
181            private Map<String,Object> map;
182    
183            ExportThread(RestRequest request, URI destination) {
184                this.request = request;
185                this.destination = destination;
186            }
187    
188            ExportThread(RestRequest request, Connection<Session<KnapsackPacket>> connection,
189                         Session<KnapsackPacket> session) {
190                this.request = request;
191                this.connection = connection;
192                this.session = session;
193            }
194    
195            @Override
196            public void run() {
197                logger.info("params = {}", request.params());
198                this.timeout = request.paramAsTime("timeout", TimeValue.timeValueSeconds(30L));
199                final int maxActionsPerBulkRequest = request.paramAsInt("maxActionsPerBulkRequest", 1000);
200                final int maxBulkConcurrency = request.paramAsInt("maxBulkConcurrency",
201                        Runtime.getRuntime().availableProcessors() * 2);
202                final boolean skipIndexCreation = request.paramAsBoolean("skipIndexCreation", false);
203                map = toMap(request.param("map"));
204                logger.info("index/type map = {}", map);
205                String path = connection != null ? connection.getURI().getSchemeSpecificPart() : null;
206                boolean copy = "copy".equals(request.param("mode"));
207                boolean s3 = "s3".equals(request.param("mode"));
208                final KnapsackStatus status = new KnapsackStatus()
209                        .setTimestamp(new Date())
210                        .setMap(map)
211                        .setPath(path)
212                        .setURI(destination)
213                        .setCopy(copy)
214                        .setS3(s3);
215                try {
216                    logger.info("start of export: {}", status);
217                    knapsackHelper.addExport(status);
218                    this.bulkClient = new BulkClient();
219                    Map<String,Set<String>> indices = newHashMap();
220                    for (String s : Strings.commaDelimitedListToSet(request.param("index", "_all"))) {
221                        indices.put(s, Strings.commaDelimitedListToSet(request.param("type")));
222                    }
223                    if (copy) {
224                        Settings settings = KnapsackHelper.clientSettings(environment, destination);
225                        logger.info("client settings classloader = " + settings.getClassLoader());
226                        bulkClient.flushInterval(TimeValue.timeValueSeconds(5))
227                                .maxActionsPerBulkRequest(maxActionsPerBulkRequest)
228                                .maxConcurrentBulkRequests(maxBulkConcurrency)
229                                .newClient(destination, settings);
230                        logger.info("waiting for healthy cluster...");
231                        bulkClient.waitForCluster(ClusterHealthStatus.YELLOW, timeout);
232                        logger.info("... cluster is ready");
233                    }
234                    if (!skipIndexCreation) {
235                        if (map != null) {
236                            for (String spec : map.keySet()) {
237                                if (spec == null) {
238                                    continue;
239                                }
240                                String[] s = spec.split("/");
241                                String index = s[0];
242                                String type = s.length > 1 ? s[1] : null;
243                                if (!"_all".equals(index)) {
244                                    Set<String> types = indices.get(index);
245                                    if (types == null) {
246                                        types = newHashSet();
247                                    }
248                                    if (type != null) {
249                                        types.add(type);
250                                    }
251                                    indices.put(index, types);
252                                }
253                            }
254                        }
255                        // get settings for all indices
256                        logger.info("getting settings for indices {}", indices.keySet());
257                        Set<String> settingsIndices = newHashSet(indices.keySet());
258                        settingsIndices.remove("_all");
259                        Map<String, String> settings = getSettings(settingsIndices.toArray(new String[settingsIndices.size()]));
260                        logger.info("found indices: {}", settings.keySet());
261                        // we resolved the specs in indices to the real indices in the settings
262                        // get mapping per index, create index
263                        for (String index : settings.keySet()) {
264                            CreateIndexRequest createIndexRequest = createIndexRequest(mapIndex(index));
265                            if (!copy) {
266                                session.write(new KnapsackPacket(mapIndex(index),
267                                        "_settings",
268                                        null,
269                                        null,
270                                        settings.get(index)));
271                            }
272                            Set<String> types = indices.get(index);
273                            createIndexRequest.settings(settings.get(index));
274                            logger.info("getting mappings for index {} and types {}", index, types);
275                            Map<String, String> mappings = getMapping(index, types != null ? ImmutableSet.copyOf(types) : null);
276                            logger.info("found mappings: {}", mappings.keySet());
277                            for (String type : mappings.keySet()) {
278                                if (!copy) {
279                                     session.write(new KnapsackPacket(mapIndex(index),
280                                             mapType(index, type),
281                                             "_mapping",
282                                             null,
283                                             mappings.get(type)));
284                                }
285                                logger.info("adding mapping: {}", mapType(index, type));
286                                createIndexRequest.mapping(mapType(index, type), mappings.get(type));
287                            }
288                            if (copy) {
289                                // create index
290                                logger.info("creating index: {}", mapIndex(index));
291                                bulkClient.client().admin().indices().create(createIndexRequest).actionGet();
292                                logger.info("index created: {}", mapIndex(index));
293                            }
294                        }
295                    }
296                    SearchRequest searchRequest;
297                    // override size = 10 default
298                    request.params().put("size", request.param("maxActionsPerBulkRequest", "1000"));
299                    searchRequest = RestSearchAction.parseSearchRequest(request);
300                    searchRequest.listenerThreaded(false);
301                    SearchOperationThreading operationThreading =
302                            SearchOperationThreading.fromString(request.param("operation_threading"), null);
303                    if (operationThreading != null) {
304                        if (operationThreading == SearchOperationThreading.NO_THREADS) {
305                            operationThreading = SearchOperationThreading.SINGLE_THREAD;
306                        }
307                        searchRequest.operationThreading(operationThreading);
308                    }
309                    for (String index : indices.keySet()) {
310                        searchRequest.searchType(SearchType.SCAN).scroll(timeout);
311                        if (!"_all".equals(index)) {
312                            searchRequest.indices(index);
313                        }
314                        Set<String> types = indices.get(index);
315                        if (types != null) {
316                            searchRequest.types(types.toArray(new String[types.size()]));
317                        }
318                        // use local node client
319                        SearchResponse searchResponse = client.search(searchRequest).actionGet();
320                        long total = 0L;
321                        while (searchResponse.getScrollId() != null) {
322                            searchResponse = client.prepareSearchScroll(searchResponse.getScrollId())
323                                    .setScroll(timeout)
324                                    .execute()
325                                    .actionGet();
326                            long hits = searchResponse.getHits().getHits().length;
327                            if (hits == 0) {
328                                break;
329                            }
330                            total += hits;
331                            if (logger.isDebugEnabled()) {
332                                logger.debug("total={} hits={} took={}", total, hits, searchResponse.getTookInMillis());
333                            }
334                            for (SearchHit hit : searchResponse.getHits()) {
335                                if (copy) {
336                                    IndexRequest indexRequest = new IndexRequest(mapIndex(hit.getIndex()),
337                                            mapType(hit.getIndex(), hit.getType()), hit.getId());
338                                    for (String f : hit.getFields().keySet()) {
339                                        if (f.equals("_parent")) {
340                                            indexRequest.parent(hit.getFields().get(f).getValue().toString());
341                                        } else if (f.equals("_routing")) {
342                                            indexRequest.routing(hit.getFields().get(f).getValue().toString());
343                                        } else if (f.equals("_timestamp")) {
344                                            indexRequest.timestamp(hit.getFields().get(f).getValue().toString());
345                                        } else if (f.equals("_version")) {
346                                            indexRequest.versionType(VersionType.EXTERNAL)
347                                                    .version(Long.parseLong(hit.getFields().get(f).getValue().toString()));
348                                        } else if (f.equals("_source")) {
349                                            indexRequest.source(hit.getSourceAsString());
350                                        } else {
351                                            indexRequest.source(f, hit.getFields().get(f).getValue().toString());
352                                        }
353                                    }
354                                    if (!hit.getFields().keySet().contains("_source")) {
355                                        indexRequest.source(hit.getSourceAsString());
356                                    }
357                                    bulkClient.index(indexRequest);
358                                } else {
359                                    for (String f : hit.getFields().keySet()) {
360                                        session.write(new KnapsackPacket(mapIndex(hit.getIndex()),
361                                            mapType(hit.getIndex(), hit.getType()),
362                                            hit.getId(), f, hit.getFields().get(f).getValue().toString()));
363                                    }
364                                    if (!hit.getFields().keySet().contains("_source")) {
365                                        session.write(new KnapsackPacket(mapIndex(hit.getIndex()),
366                                            mapType(hit.getIndex(), hit.getType()),
367                                            hit.getId(), "_source", hit.getSourceAsString()));
368                                    }
369                                }
370                            }
371                        }
372                    }
373                    logger.info("end of export: {}", status);
374                    if (!copy) {
375                        session.close();
376                    }
377                    if (copy) {
378                        bulkClient.refresh().shutdown();
379                        bulkClient = null;
380                    }
381                    if (s3 && request.param("uri") != null) {
382                        // s3://auth:host, s3://host
383                        URI s3URI = URI.create(request.param("uri")
384                                + "?bucketName=" + request.param("bucketName")
385                                + "&key=" + request.param("key"));
386                        logger.info("trying to transfer file {} to s3 {}", path, s3URI);
387                        ConnectionService<Session<ObjectPacket>> service = ConnectionService.getInstance();
388                        ConnectionFactory<Session<ObjectPacket>> factory = service.getConnectionFactory(s3URI);
389                        final Connection<Session<ObjectPacket>> connection = factory.getConnection(s3URI);
390                        final Session<ObjectPacket> session = connection.createSession();
391                        session.open(Session.Mode.WRITE);
392                        session.write(new ObjectPacket().name(path));
393                        session.close();
394                    }
395                } catch (Throwable e) {
396                    logger.error(e.getMessage(), e);
397                } finally {
398                    try {
399                        knapsackHelper.removeExport(status);
400                    } catch (IOException e) {
401                        logger.error(e.getMessage(), e);
402                    }
403                }
404            }

406            private Map<String, Object> toMap(String s) {
407                if (s == null) {
408                    logger.warn("no map defined");
409                    return newHashMap();
410                }
411                BytesArray ref = new BytesArray(s);
412                try {
413                    byte[] b = ref.hasArray() ? ref.array() : ref.toBytes();
414                    XContentParser parser = XContentFactory.xContent(XContentType.JSON).createParser(b, 0, b.length);
415                    return parser.mapAndClose();
416                } catch (IOException e) {
417                    logger.warn("parse error while reading map", e);
418                    return newHashMap();
419                }
420            }
421    
422            private String mapIndex(String index) {
423                return map.containsKey(index) ? map.get(index).toString() : index;
424            }
425    
426            private String mapType(String index, String type) {
427                String s = index + "/" + type;
428                return map.containsKey(s) ? map.get(s).toString() : type;
429            }

432        private Map<String, String> getSettings(String... index) throws IOException {
433            Map<String, String> settings = newHashMap();
434            ClusterStateRequestBuilder request = client.admin().cluster().prepareState()
435                    .setRoutingTable(true)
436                    .setNodes(true)
437                    .setIndices(index);
438            ClusterStateResponse response = request.execute().actionGet();
439            MetaData metaData = response.getState().metaData();
440            if (!metaData.getIndices().isEmpty()) {
441                // filter out the settings from the metadata
442                for (IndexMetaData indexMetaData : metaData) {
443                    final XContentBuilder builder = jsonBuilder();
444                    builder.startObject();
445                    for (Map.Entry<String, String> entry :
446                            settingsFilter.filterSettings(indexMetaData.getSettings()).getAsMap().entrySet()) {
447                        builder.field(entry.getKey(), entry.getValue());
448                    }
449                    builder.endObject();
450                    settings.put(indexMetaData.getIndex(), builder.string());
451                }
452            }
453            return settings;
454        }

456        private Map<String, String> getMapping(String index, Set<String> types) throws IOException {
457            Map<String, String> mappings = newHashMap();
458            ClusterStateRequestBuilder request = client.admin().cluster().prepareState()
459                    .setRoutingTable(true)
460                    .setNodes(true);
461            if (!"_all".equals(index)) {
462                request.setIndices(index);
463            }
464            ClusterStateResponse response = request.execute().actionGet();
465            MetaData metaData = response.getState().getMetaData();
466            if (!metaData.getIndices().isEmpty()) {
467                for (IndexMetaData indexMetaData : metaData) {
468                    for (ObjectCursor<MappingMetaData> c : indexMetaData.getMappings().values()) {
469                        MappingMetaData mappingMetaData = c.value;
470                        if (types == null || types.isEmpty() || types.contains(mappingMetaData.type())) {
471                            final XContentBuilder builder = jsonBuilder();
472                            builder.startObject();
473                            builder.field(mappingMetaData.type());
474                            builder.map(mappingMetaData.sourceAsMap());
475                            builder.endObject();
476                            mappings.put(mappingMetaData.type(), builder.string());
477                        }
478                    }
479                }
480            }
481            return mappings;
482        }

484        private class RestKnapsackExportState implements RestHandler {
485    
486            @Override
487            public void handleRequest(RestRequest request, RestChannel channel) {
488                try {
489                    XContentBuilder builder = restContentBuilder(request);
490                    builder.startObject()
491                            .field("exports")
492                            .startArray();
493                    for (KnapsackStatus export : knapsackHelper.getExports()) {
494                        export.toXContent(builder, EMPTY_PARAMS);
495                    }
496                    builder.endArray()
497                            .endObject();
498                    channel.sendResponse(new XContentRestResponse(request, OK, builder));
499                } catch (IOException ioe) {
500                    try {
501                        channel.sendResponse(new XContentThrowableRestResponse(request, ioe));
502                    } catch (IOException e) {
503                        logger.error("unable to send response to client");
504                    }
505                }
506            }
507        }
508    
509        private class RestKnapsackExportAbort implements RestHandler {
510    
511            @Override
512            public void handleRequest(RestRequest request, RestChannel channel) {
513                try {
514                    XContentBuilder builder = restContentBuilder(request);
515                    builder.startObject()
516                            .field("exports")
517                            .startArray();
518                    for (KnapsackStatus export : knapsackHelper.getExports()) {
519                        export.toXContent(builder, EMPTY_PARAMS);
520                    }
521                    builder.endArray()
522                            .endObject();
523                    executor.shutdownNow();
524                    executor = EsExecutors.newFixed(10, 0, EsExecutors.daemonThreadFactory(settings, "knapsack-export"));
525                    channel.sendResponse(new XContentRestResponse(request, OK, builder));
526                } catch (IOException ioe) {
527                    try {
528                        channel.sendResponse(new XContentThrowableRestResponse(request, ioe));
529                    } catch (IOException e) {
530                        logger.error("unable to send response to client");
531                    }
532                }
533            }
