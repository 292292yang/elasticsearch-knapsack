053
054public class RestExportAction extends BaseRestHandler {
055
056    private final SettingsFilter settingsFilter;
057
058    private final KnapsackService knapsackService;
059
060    @Inject
061    public RestExportAction(Settings settings, Client client,
062            RestController controller, SettingsFilter settingsFilter, KnapsackService knapsackService) {
063        super(settings, client);
064        this.settingsFilter = settingsFilter;
065        this.knapsackService = knapsackService;
066
067        controller.registerHandler(POST, "/_export", this);
068        controller.registerHandler(POST, "/{index}/_export", this);
069        controller.registerHandler(POST, "/{index}/{type}/_export", this);
070        controller.registerHandler(GET, "/_export/state",  new RestKnapsackExportStatus());
071    }
072
073    @Override
074    public void handleRequest(final RestRequest request, RestChannel channel) {
075        final String index = request.param("index");
076        final String type = request.param("type");
077        final String[] indices = Strings.splitStringByCommaToArray(request.param("index", "_all"));
078        final String[] types = Strings.splitStringByCommaToArray(request.param("type"));
079        final String desc = (index != null ? index : "_all")
080                + (type != null ? "_" + type : "");
081        try {
082            XContentBuilder builder = restContentBuilder(request)
083                    .startObject()
084                    .field("ok", true)
085                    .endObject();
086
087            ClusterHealthResponse healthResponse =
088                    client.admin().cluster().prepareHealth().setWaitForYellowStatus()
089                            .setTimeout("30s").execute().actionGet(30000);
090
091            if (healthResponse.isTimedOut()) {
092                String msg = "cluster not healthy, cowardly refusing to continue with export";
093                logger.error(msg);
094                throw new IOException(msg);
095            }
096
097            final String target = request.param("target", desc);
098            String scheme = request.param("scheme", "targz");
099            for (String codec : StreamCodecService.getCodecs()) {
100                if (target.endsWith(codec)) {
101                    scheme = "tar" + codec;
102                }
103            }
104
105            ConnectionService service = ConnectionService.getInstance();
106            ConnectionFactory factory = service.getConnectionFactory(scheme);
107            final Connection<Session> connection = factory.getConnection(URI.create(scheme + ":" + target));
108            final Session session = connection.createSession();
109            session.open(Session.Mode.WRITE);
110
111            channel.sendResponse(new XContentRestResponse(request, OK, builder));
112
113            EsExecutors.daemonThreadFactory(settings, "Knapsack export [" + desc + "]")
114                    .newThread(new Thread() {
115                @Override
116                public void run() {
117                    setName("[Exporter Thread " + desc + "]");
118                    long millis = request.paramAsLong("millis", 30000L);
119                    int size = request.paramAsInt("size", 1000);
120
121                    final KnapsackStatus status = new KnapsackStatus(indices, types, target);
122                    try {
123                        logger.info("starting export to {}", target);
124                        knapsackService.addExport(status);
125                        // export settings & mappings
126                        for (String s : indices) {
127                            Map<String, String> settings = getSettings(s);
128                            for (String index : settings.keySet()) {
129                                session.write(new ElasticPacket(index, "_settings", null, settings.get(index)));
130                                Map<String, String> mappings = getMapping(index, ImmutableSet.copyOf(types));
131                                for (String type : mappings.keySet()) {
132                                    session.write(new ElasticPacket(index, type, "_mapping", mappings.get(type)));
133                                }
134                            }
135                        }
136
137                        // export document _source fields
138                        SearchRequestBuilder searchRequest = client.prepareSearch()
139                                .setSize(size)
140                                .setSearchType(SearchType.SCAN)
141                                .setScroll(TimeValue.timeValueMillis(millis));
142                        if (indices != null && !"_all".equals(index)) {
143                            searchRequest.setIndices(indices);
144                        }
145                        if (types != null) {
146                            searchRequest.setTypes(types);
147                        }
148                        SearchResponse searchResponse = searchRequest.execute().actionGet();
149                        while (searchResponse.getScrollId() != null) {
150                            searchResponse = client.prepareSearchScroll(searchResponse.getScrollId())
151                                    .setScroll(TimeValue.timeValueMillis(millis))
152                                    .execute().actionGet();
153                            if (searchResponse.getHits().getHits().length == 0) {
154                                break;
155                            }
156                            for (SearchHit hit : searchResponse.getHits()) {
157                                ElasticPacket packet = new ElasticPacket(hit.getIndex(), hit.getType(), hit.getId(), hit.getSourceAsString());
158                                session.write(packet);
159                            }
160                        }
161
162                        session.close();
163                        connection.close();
164
165                        logger.info("export to {} completed", target);
166
167                    } catch (Exception e) {
168                        logger.error(e.getMessage(), e);
169                    } finally {
170                        try {
171                            knapsackService.removeExport(status);
172                        } catch (IOException e) {
173                            logger.error(e.getMessage(), e);
174                        }
175                    }
176                }
177            }).start();
178
179        } catch (IOException ex) {
180            try {
181                channel.sendResponse(new XContentThrowableRestResponse(request, ex));
182            } catch (Exception ex2) {
183                logger.error(ex2.getMessage(), ex2);
184            }
185        }
186    }
187
188    private Map<String, String> getSettings(String index) throws IOException {
189        logger.info("getting settings for index {}", index);
190        Map<String, String> settings = newHashMap();
191        ClusterStateRequestBuilder request = client.admin().cluster().prepareState()
192                .setFilterRoutingTable(true)
193                .setFilterNodes(true);
194        if (!"_all".equals(index)) {
195            request.setFilterIndices(index);
196        }
197        ClusterStateResponse response = request.execute().actionGet();
198        MetaData metaData = response.getState().metaData();
199        if (!metaData.getIndices().isEmpty()) {
200            for (IndexMetaData indexMetaData : metaData) {
201                final XContentBuilder builder = jsonBuilder();
202                builder.startObject();
203                for (Map.Entry<String, String> entry :
204                        settingsFilter.filterSettings(indexMetaData.getSettings()).getAsMap().entrySet()) {
205                    builder.field(entry.getKey(), entry.getValue());
206                }
207                builder.endObject();
208                settings.put(indexMetaData.getIndex(), builder.string());
209            }
210        }
211        return settings;
212    }
213
214    private Map<String, String> getMapping(String index, Set<String> types) throws IOException {
215        logger.info("getting mapping for index {} and types {}", index, types);
216        Map<String, String> mappings = newHashMap();
217        ClusterStateResponse response = client.admin().cluster().prepareState()
218                .setFilterRoutingTable(true)
219                .setFilterNodes(true)
220                .setFilterIndices(index)
221                .execute().actionGet();
222        MetaData metaData = response.getState().getMetaData();
223        if (!metaData.getIndices().isEmpty()) {
224            IndexMetaData indexMetaData = metaData.iterator().next();
225            for (MappingMetaData mappingMd : indexMetaData.getMappings().values()) {
226                if (types == null || types.isEmpty() || types.contains(mappingMd.type())) {
227                    final XContentBuilder builder = jsonBuilder();
228                    builder.startObject();
229                    builder.field(mappingMd.type());
230                    builder.map(mappingMd.sourceAsMap());
231                    builder.endObject();
232                    mappings.put(mappingMd.type(), builder.string());
233                }
234            }
235        }
236        return mappings;
237    }
238
239    private class ElasticPacket implements Packet<String> {
240
241        String name;
242        String packet;
243
244        ElasticPacket(String index, String type, String id, String packet) {
245            this.name = index + "/" + type + (id != null ? "/" + id : "");
246            this.packet = packet;
247        }
248
249        @Override
250        public String getName() {
251            return name;
252        }
253
254        @Override
255        public String getPacket() {
256            return packet;
257        }
258
259        @Override
260        public String toString() {
261            return packet;
262        }
263    }
264
265    private class RestKnapsackExportStatus implements RestHandler {
266
267        @Override
268        public void handleRequest(RestRequest request, RestChannel channel) {
269            try {
270                XContentBuilder builder = restContentBuilder(request);
271                builder.startObject()
272                        .field("exports")
273                        .startArray();
274                for (KnapsackStatus export : knapsackService.getExports()) {
275                    export.toXContent(builder, EMPTY_PARAMS);
276                }
277                builder.endArray()
278                        .endObject();
279                channel.sendResponse(new XContentRestResponse(request, OK, builder));
280            } catch (IOException ioe) {
281                try {
282                    channel.sendResponse(new XContentThrowableRestResponse(request, ioe));
283                } catch (IOException e) {
284                    // ignore
285                }
286            }
287        }
288    }
