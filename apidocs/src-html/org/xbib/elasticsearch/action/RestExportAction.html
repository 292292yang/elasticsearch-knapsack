072/**
073 * The knapsack export action performs a scan/scroll action over a user defined query
074 * and stores the result in an archive or transfers it to an ES cluster
075 */
076public class RestExportAction extends BaseRestHandler {
077
078    private final ClusterName clusterName;
079
080    private final Environment environment;
081

083

088    @Inject
089    public RestExportAction(Settings settings,
090                            Environment environment,
091                            Client client,
092                            RestController controller,
093                            SettingsFilter settingsFilter,
094                            ClusterService clusterService,
095                            ClusterName clusterName) {
096        super(settings, client);
097        this.environment = environment;
098        this.clusterName = clusterName;
099        this.settingsFilter = settingsFilter;
100        this.knapsackHelper = new KnapsackHelper(client, clusterService);
101        this.executor = EsExecutors.newScaling(0, 10, 7L, TimeUnit.DAYS,
102                EsExecutors.daemonThreadFactory(settings, "knapsack-export"));
103
104        controller.registerHandler(POST, "/_export", this);
105        controller.registerHandler(POST, "/{index}/_export", this);
106        controller.registerHandler(POST, "/{index}/{type}/_export", this);
107
108        controller.registerHandler(POST, "/_export/{mode}", this);
109        controller.registerHandler(POST, "/{index}/_export/{mode}", this);
110        controller.registerHandler(POST, "/{index}/{type}/_export/{mode}", this);
111
112        controller.registerHandler(GET, "/_export/state", new RestKnapsackExportState());
113        controller.registerHandler(POST, "/_export/abort", new RestKnapsackExportAbort());
114    }

116    @Override
117    public void handleRequest(final RestRequest request, RestChannel channel) {
118        try {
119            final XContentBuilder builder = restContentBuilder(request);
120            final boolean copy = "copy".equals(request.param("mode"));
121            if (copy) {
122                final URI destination = KnapsackHelper.getAddress(client, clusterName, request.param("host"), request.paramAsInt("port", 0),
123                        request.param("cluster"));
124                executor.execute(new ExportThread(request, destination));
125                builder.startObject()
126                        .field("running", true)
127                        .field("mode", request.param("mode") != null ? request.param("mode") : "export")
128                        .field("path", destination)
129                        .endObject();
130            } else {
131                final String index = request.param("index", "_all");
132                final String type = request.param("type");
133                final String defaultSpec = index + (type != null ? "_" + type : "") + ".tar.gz";
134                final String path = request.param("path", defaultSpec);
135                URI uri = URI.create("file:" + path);
136                ConnectionService<Session<KnapsackPacket>> service = ConnectionService.getInstance();
137                ConnectionFactory<Session<KnapsackPacket>> factory = service.getConnectionFactory(uri);
138                final Connection<Session<KnapsackPacket>> connection = factory.getConnection(uri);
139                final Session<KnapsackPacket> session = connection.createSession();
140                session.open(Session.Mode.WRITE);
141                if (session.isOpen()) {
142                    executor.execute(new ExportThread(request, connection, session));
143                    builder.startObject()
144                            .field("running", true)
145                            .field("mode", request.param("mode") != null ? request.param("mode") : "export")
146                            .field("archive", factory.getName())
147                            .field("path", connection.getURI())
148                            .endObject();
149                } else {
150                    builder.startObject()
151                            .field("running", false)
152                            .endObject();
153                    channel.sendResponse(new XContentRestResponse(request, BAD_REQUEST, builder));
154                    return;
155                }
156            }
157            channel.sendResponse(new XContentRestResponse(request, OK, builder));
158        } catch (Throwable ex) {
159            try {
160                logger.error(ex.getMessage(), ex);
161                channel.sendResponse(new XContentThrowableRestResponse(request, ex));
162            } catch (Exception ex2) {
163                logger.error(ex2.getMessage(), ex2);
164            }
165        }
166    }

168    private class ExportThread extends Thread {
169
170        private final RestRequest request;
171
172        private BulkTransportClient bulkClient;
173
174        private Connection<Session<KnapsackPacket>> connection;
175
176        private Session<KnapsackPacket> session;
177
178        private TimeValue timeout;
179
180        private URI destination;
181
182        private Map<String,Object> map;
183
184        ExportThread(RestRequest request, URI destination) {
185            this.request = request;
186            this.destination = destination;
187        }
188
189        ExportThread(RestRequest request, Connection<Session<KnapsackPacket>> connection,
190                     Session<KnapsackPacket> session) {
191            this.request = request;
192            this.connection = connection;
193            this.session = session;
194        }
195
196        @Override
197        public void run() {
198            logger.info("params = {}", request.params());
199            this.timeout = request.paramAsTime("timeout", TimeValue.timeValueSeconds(30L));
200            final int maxActionsPerBulkRequest = request.paramAsInt("maxActionsPerBulkRequest", 1000);
201            final int maxBulkConcurrency = request.paramAsInt("maxBulkConcurrency",
202                    Runtime.getRuntime().availableProcessors() * 2);
203            final boolean skipIndexCreation = request.paramAsBoolean("skipIndexCreation", false);
204            map = toMap(request.param("map"));
205            logger.info("index/type map = {}", map);
206            String path = connection != null ? connection.getURI().getSchemeSpecificPart() : null;
207            boolean copy = "copy".equals(request.param("mode"));
208            boolean s3 = "s3".equals(request.param("mode"));
209            final KnapsackStatus status = new KnapsackStatus()
210                    .setTimestamp(new Date())
211                    .setMap(map)
212                    .setPath(path)
213                    .setURI(destination)
214                    .setCopy(copy)
215                    .setS3(s3);
216            try {
217                logger.info("start of export: {}", status);
218                knapsackHelper.addExport(status);
219                this.bulkClient = new BulkTransportClient();
220                Map<String,Set<String>> indices = newHashMap();
221                for (String s : Strings.commaDelimitedListToSet(request.param("index", "_all"))) {
222                    indices.put(s, Strings.commaDelimitedListToSet(request.param("type")));
223                }
224                if (copy) {
225                    Settings settings = KnapsackHelper.clientSettings(environment, destination);
226                    logger.info("client settings classloader = " + settings.getClassLoader());
227                    bulkClient.flushInterval(TimeValue.timeValueSeconds(5))
228                            .maxActionsPerBulkRequest(maxActionsPerBulkRequest)
229                            .maxConcurrentBulkRequests(maxBulkConcurrency)
230                            .newClient(destination, settings);
231                    logger.info("waiting for healthy cluster...");
232                    bulkClient.waitForCluster(ClusterHealthStatus.YELLOW, timeout);
233                    logger.info("... cluster is ready");
234                }
235                if (!skipIndexCreation) {
236                    if (map != null) {
237                        for (String spec : map.keySet()) {
238                            if (spec == null) {
239                                continue;
240                            }
241                            String[] s = spec.split("/");
242                            String index = s[0];
243                            String type = s.length > 1 ? s[1] : null;
244                            if (!"_all".equals(index)) {
245                                Set<String> types = indices.get(index);
246                                if (types == null) {
247                                    types = newHashSet();
248                                }
249                                if (type != null) {
250                                    types.add(type);
251                                }
252                                indices.put(index, types);
253                            }
254                        }
255                    }
256                    // get settings for all indices
257                    logger.info("getting settings for indices {}", indices.keySet());
258                    Set<String> settingsIndices = newHashSet(indices.keySet());
259                    settingsIndices.remove("_all");
260                    Map<String, String> settings = getSettings(settingsIndices.toArray(new String[settingsIndices.size()]));
261                    logger.info("found indices: {}", settings.keySet());
262                    // we resolved the specs in indices to the real indices in the settings
263                    // get mapping per index, create index
264                    for (String index : settings.keySet()) {
265                        CreateIndexRequest createIndexRequest = createIndexRequest(mapIndex(index));
266                        if (!copy) {
267                            session.write(new KnapsackPacket(mapIndex(index),
268                                    "_settings",
269                                    null,
270                                    null,
271                                    settings.get(index)));
272                        }
273                        Set<String> types = indices.get(index);
274                        createIndexRequest.settings(settings.get(index));
275                        logger.info("getting mappings for index {} and types {}", index, types);
276                        Map<String, String> mappings = getMapping(index, types != null ? ImmutableSet.copyOf(types) : null);
277                        logger.info("found mappings: {}", mappings.keySet());
278                        for (String type : mappings.keySet()) {
279                            if (!copy) {
280                                 session.write(new KnapsackPacket(mapIndex(index),
281                                         mapType(index, type),
282                                         "_mapping",
283                                         null,
284                                         mappings.get(type)));
285                            }
286                            logger.info("adding mapping: {}", mapType(index, type));
287                            createIndexRequest.mapping(mapType(index, type), mappings.get(type));
288                        }
289                        if (copy) {
290                            // create index
291                            logger.info("creating index: {}", mapIndex(index));
292                            bulkClient.client().admin().indices().create(createIndexRequest).actionGet();
293                            logger.info("index created: {}", mapIndex(index));
294                        }
295                    }
296                }
297                SearchRequest searchRequest;
298                // override size = 10 default
299                request.params().put("size", request.param("maxActionsPerBulkRequest", "1000"));
300                searchRequest = RestSearchAction.parseSearchRequest(request);
301                searchRequest.listenerThreaded(false);
302                for (String index : indices.keySet()) {
303                    searchRequest.searchType(SearchType.SCAN).scroll(timeout);
304                    if (!"_all".equals(index)) {
305                        searchRequest.indices(index);
306                    }
307                    Set<String> types = indices.get(index);
308                    if (types != null) {
309                        searchRequest.types(types.toArray(new String[types.size()]));
310                    }
311                    // use local node client
312                    SearchResponse searchResponse = client.search(searchRequest).actionGet();
313                    long total = 0L;
314                    while (searchResponse.getScrollId() != null) {
315                        searchResponse = client.prepareSearchScroll(searchResponse.getScrollId())
316                                .setScroll(timeout)
317                                .execute()
318                                .actionGet();
319                        long hits = searchResponse.getHits().getHits().length;
320                        if (hits == 0) {
321                            break;
322                        }
323                        total += hits;
324                        if (logger.isDebugEnabled()) {
325                            logger.debug("total={} hits={} took={}", total, hits, searchResponse.getTookInMillis());
326                        }
327                        for (SearchHit hit : searchResponse.getHits()) {
328                            if (copy) {
329                                IndexRequest indexRequest = new IndexRequest(mapIndex(hit.getIndex()),
330                                        mapType(hit.getIndex(), hit.getType()), hit.getId());
331                                for (String f : hit.getFields().keySet()) {
332                                    if (f.equals("_parent")) {
333                                        indexRequest.parent(hit.getFields().get(f).getValue().toString());
334                                    } else if (f.equals("_routing")) {
335                                        indexRequest.routing(hit.getFields().get(f).getValue().toString());
336                                    } else if (f.equals("_timestamp")) {
337                                        indexRequest.timestamp(hit.getFields().get(f).getValue().toString());
338                                    } else if (f.equals("_version")) {
339                                        indexRequest.versionType(VersionType.EXTERNAL)
340                                                .version(Long.parseLong(hit.getFields().get(f).getValue().toString()));
341                                    } else if (f.equals("_source")) {
342                                        indexRequest.source(hit.getSourceAsString());
343                                    } else {
344                                        indexRequest.source(f, hit.getFields().get(f).getValue().toString());
345                                    }
346                                }
347                                if (!hit.getFields().keySet().contains("_source")) {
348                                    indexRequest.source(hit.getSourceAsString());
349                                }
350                                bulkClient.index(indexRequest);
351                            } else {
352                                for (String f : hit.getFields().keySet()) {
353                                    session.write(new KnapsackPacket(mapIndex(hit.getIndex()),
354                                        mapType(hit.getIndex(), hit.getType()),
355                                        hit.getId(), f, hit.getFields().get(f).getValue().toString()));
356                                }
357                                if (!hit.getFields().keySet().contains("_source")) {
358                                    session.write(new KnapsackPacket(mapIndex(hit.getIndex()),
359                                        mapType(hit.getIndex(), hit.getType()),
360                                        hit.getId(), "_source", hit.getSourceAsString()));
361                                }
362                            }
363                        }
364                    }
365                }
366                logger.info("end of export: {}", status);
367                if (!copy) {
368                    session.close();
369                }
370                if (copy) {
371                    for (String index : indices.keySet()) {
372                        bulkClient.refresh(index);
373                    }
374                    bulkClient.shutdown();
375                }
376                if (s3 && request.param("uri") != null) {
377                    // s3://auth:host, s3://host
378                    URI s3URI = URI.create(request.param("uri")
379                            + "?bucketName=" + request.param("bucketName")
380                            + "&key=" + request.param("key"));
381                    logger.info("trying to transfer file {} to s3 {}", path, s3URI);
382                    ConnectionService<Session<ObjectPacket>> service = ConnectionService.getInstance();
383                    ConnectionFactory<Session<ObjectPacket>> factory = service.getConnectionFactory(s3URI);
384                    final Connection<Session<ObjectPacket>> connection = factory.getConnection(s3URI);
385                    final Session<ObjectPacket> session = connection.createSession();
386                    session.open(Session.Mode.WRITE);
387                    session.write(new ObjectPacket().name(path));
388                    session.close();
389                }
390            } catch (Throwable e) {
391                logger.error(e.getMessage(), e);
392            } finally {
393                try {
394                    knapsackHelper.removeExport(status);
395                } catch (IOException e) {
396                    logger.error(e.getMessage(), e);
397                }
398            }
399        }

401        private Map<String, Object> toMap(String s) {
402            if (s == null) {
403                logger.warn("no map defined");
404                return newHashMap();
405            }
406            BytesArray ref = new BytesArray(s);
407            try {
408                byte[] b = ref.hasArray() ? ref.array() : ref.toBytes();
409                XContentParser parser = XContentFactory.xContent(XContentType.JSON).createParser(b, 0, b.length);
410                return parser.mapAndClose();
411            } catch (IOException e) {
412                logger.warn("parse error while reading map", e);
413                return newHashMap();
414            }
415        }
416
417        private String mapIndex(String index) {
418            return map.containsKey(index) ? map.get(index).toString() : index;
419        }
420
421        private String mapType(String index, String type) {
422            String s = index + "/" + type;
423            return map.containsKey(s) ? map.get(s).toString() : type;
424        }
425    }

427    private Map<String, String> getSettings(String... index) throws IOException {
428        Map<String, String> settings = newHashMap();
429        ClusterStateRequestBuilder request = client.admin().cluster().prepareState()
430                .setRoutingTable(true)
431                .setNodes(true)
432                .setIndices(index);
433        ClusterStateResponse response = request.execute().actionGet();
434        MetaData metaData = response.getState().metaData();
435        if (!metaData.getIndices().isEmpty()) {
436            // filter out the settings from the metadata
437            for (IndexMetaData indexMetaData : metaData) {
438                final XContentBuilder builder = jsonBuilder();
439                builder.startObject();
440                for (Map.Entry<String, String> entry :
441                        settingsFilter.filterSettings(indexMetaData.getSettings()).getAsMap().entrySet()) {
442                    builder.field(entry.getKey(), entry.getValue());
443                }
444                builder.endObject();
445                settings.put(indexMetaData.getIndex(), builder.string());
446            }
447        }
448        return settings;
449    }

451    private Map<String, String> getMapping(String index, Set<String> types) throws IOException {
452        Map<String, String> mappings = newHashMap();
453        ClusterStateRequestBuilder request = client.admin().cluster().prepareState()
454                .setRoutingTable(true)
455                .setNodes(true);
456        if (!"_all".equals(index)) {
457            request.setIndices(index);
458        }
459        ClusterStateResponse response = request.execute().actionGet();
460        MetaData metaData = response.getState().getMetaData();
461        if (!metaData.getIndices().isEmpty()) {
462            for (IndexMetaData indexMetaData : metaData) {
463                for (ObjectCursor<MappingMetaData> c : indexMetaData.getMappings().values()) {
464                    MappingMetaData mappingMetaData = c.value;
465                    if (types == null || types.isEmpty() || types.contains(mappingMetaData.type())) {
466                        final XContentBuilder builder = jsonBuilder();
467                        builder.startObject();
468                        builder.field(mappingMetaData.type());
469                        builder.map(mappingMetaData.sourceAsMap());
470                        builder.endObject();
471                        mappings.put(mappingMetaData.type(), builder.string());
472                    }
473                }
474            }
475        }
476        return mappings;
477    }

479    private class RestKnapsackExportState implements RestHandler {
480
481        @Override
482        public void handleRequest(RestRequest request, RestChannel channel) {
483            try {
484                XContentBuilder builder = restContentBuilder(request);
485                builder.startObject()
486                        .field("exports")
487                        .startArray();
488                for (KnapsackStatus export : knapsackHelper.getExports()) {
489                    export.toXContent(builder, EMPTY_PARAMS);
490                }
491                builder.endArray()
492                        .endObject();
493                channel.sendResponse(new XContentRestResponse(request, OK, builder));
494            } catch (IOException ioe) {
495                try {
496                    channel.sendResponse(new XContentThrowableRestResponse(request, ioe));
497                } catch (IOException e) {
498                    logger.error("unable to send response to client");
499                }
500            }
501        }
502    }

504    private class RestKnapsackExportAbort implements RestHandler {
505
506        @Override
507        public void handleRequest(RestRequest request, RestChannel channel) {
508            try {
509                XContentBuilder builder = restContentBuilder(request);
510                builder.startObject()
511                        .field("exports")
512                        .startArray();
513                for (KnapsackStatus export : knapsackHelper.getExports()) {
514                    export.toXContent(builder, EMPTY_PARAMS);
515                }
516                builder.endArray()
517                        .endObject();
518                executor.shutdownNow();
519                executor = EsExecutors.newScaling(0, 10, 7L, TimeUnit.DAYS,
520                        EsExecutors.daemonThreadFactory(settings, "knapsack-export"));
521                channel.sendResponse(new XContentRestResponse(request, OK, builder));
522            } catch (IOException ioe) {
523                try {
524                    channel.sendResponse(new XContentThrowableRestResponse(request, ioe));
525                } catch (IOException e) {
526                    logger.error("unable to send response to client");
527                }
528            }
529        }
