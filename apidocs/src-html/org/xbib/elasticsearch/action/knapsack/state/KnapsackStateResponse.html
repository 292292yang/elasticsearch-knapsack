016package org.xbib.elasticsearch.action.knapsack.state;
017
018import org.elasticsearch.action.ActionResponse;
019import org.elasticsearch.common.io.stream.StreamInput;
020import org.elasticsearch.common.io.stream.StreamOutput;
021import org.elasticsearch.common.logging.ESLogger;
022import org.elasticsearch.common.logging.ESLoggerFactory;
023import org.elasticsearch.common.xcontent.ToXContent;
024import org.elasticsearch.common.xcontent.XContentBuilder;
025import org.xbib.elasticsearch.knapsack.KnapsackState;
026
027import java.io.IOException;
028import java.nio.file.Path;
029import java.util.LinkedList;
030import java.util.List;
031
032import static org.elasticsearch.common.xcontent.XContentFactory.jsonBuilder;
033
034public class KnapsackStateResponse extends ActionResponse implements ToXContent {
035
036    private final static ESLogger logger= ESLoggerFactory.getLogger("state");
037
038    private List<KnapsackState> states = new LinkedList<>();
039
040    public KnapsackStateResponse addState(KnapsackState state) {
041        this.states.add(state);
042        return this;
043    }
044
045    public List<KnapsackState> getStates() {
046        return states;
047    }
048
049    public boolean isExportActive(Path path) {
050        if (states != null && path != null) {
051            for (KnapsackState ks : states) {
052                if (ks != null && "export".equals(ks.getMode()) && ks.getPath() != null && path.toString().equals(ks.getPath().toString())) {
053                    return true;
054                }
055            }
056        }
057        return false;
058    }
059
060    public boolean isImportActive(Path path) {
061        if (states != null && path != null) {
062            for (KnapsackState ks : states) {
063                if (ks != null && "import".equals(ks.getMode()) && ks.getPath() != null && path.toString().equals(ks.getPath().toString())) {
064                    return true;
065                }
066            }
067        }
068        return false;
069    }
070
071    public boolean isPushActive() {
072        if (states != null) {
073            for (KnapsackState ks : states) {
074                if (ks != null && "push".equals(ks.getMode())) {
075                    return true;
076                }
077            }
078        }
079        return false;
080    }
081
082    public boolean isPullActive() {
083        if (states != null) {
084            for (KnapsackState ks : states) {
085                if (ks != null && "pull".equals(ks.getMode())) {
086                    return true;
087                }
088            }
089        }
090        return false;
091    }
092
093    @Override
094    public XContentBuilder toXContent(XContentBuilder builder, Params params) throws IOException {
095        builder.field("count", states.size())
096                .startArray("states");
097        for (KnapsackState ks : states) {
098            ks.toXContent(builder, params);
099        }
100        builder.endArray();
101        return builder;
102    }
103
104    @Override
105    public void readFrom(StreamInput in) throws IOException {
106        super.readFrom(in);
107        states = new LinkedList<>();
108        int size = in.readInt();
109        for (int i = 0; i < size; i++) {
110            KnapsackState ks = new KnapsackState();
111            ks.readFrom(in);
112            states.add(ks);
113        }
114    }
115
116    @Override
117    public void writeTo(StreamOutput out) throws IOException {
118        super.writeTo(out);
119        if (states != null) {
120            out.writeInt(states.size());
121            for (KnapsackState state : states) {
122                state.writeTo(out);
123            }
124        }
125    }
126
127    @Override
128    public String toString() {
129        try {
130            XContentBuilder builder = jsonBuilder();
131            builder.startObject();
132            builder = toXContent(builder, EMPTY_PARAMS);
133            builder.endObject();
134            return builder.string();
135        } catch (IOException e) {
136            return "";
137        }
138    }
139}
