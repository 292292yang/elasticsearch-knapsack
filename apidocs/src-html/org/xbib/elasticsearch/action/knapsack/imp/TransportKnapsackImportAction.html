068    private final static ESLogger logger = ESLoggerFactory.getLogger(KnapsackImportAction.class.getSimpleName());

076    @Inject
077    public TransportKnapsackImportAction(Settings settings,
078                                         ThreadPool threadPool,
079                                         Client client,
080                                         NodeService nodeService, ActionFilters actionFilters,
081                                         IndexNameExpressionResolver indexNameExpressionResolver,
082                                         KnapsackService knapsack) {
083        super(settings, KnapsackImportAction.NAME, threadPool, actionFilters, indexNameExpressionResolver);
084        this.client = client;
085        this.nodeService = nodeService;
086        this.knapsack = knapsack;
087    }

089    @Override
090    protected void doExecute(final KnapsackImportRequest request, ActionListener<KnapsackImportResponse> listener) {
091        final KnapsackState state = new KnapsackState()
092                .setMode("import")
093                .setNodeName(nodeService.nodeName());
094        final KnapsackImportResponse response = new KnapsackImportResponse()
095                .setState(state);
096        try {
097            Path path = request.getPath();
098            if (path == null) {
099                path = new File("_all.tar.gz").toPath();
100            }
101            ByteSizeValue bytesToTransfer = request.getBytesToTransfer();
102            BytesProgressWatcher watcher = new BytesProgressWatcher(bytesToTransfer.bytes());
103            final Session<StringPacket> session = ArchiveService.newSession(path, watcher);
104            EnumSet<Session.Mode> mode = EnumSet.of(Session.Mode.READ);
105            session.open(mode, path);
106            if (session.isOpen()) {
107                final BulkNodeClient bulkClient = new BulkNodeClient();
108                bulkClient.flushIngestInterval(TimeValue.timeValueSeconds(5))
109                        .maxActionsPerRequest(request.getMaxActionsPerBulkRequest())
110                        .maxConcurrentRequests(request.getMaxBulkConcurrency())
111                        .init(client);
112                state.setTimestamp(new DateTime())
113                        .setPath(path);
114                response.setRunning(true);
115                knapsack.submit(new Thread() {
116                    public void run() {
117                        try {
118                            performImport(request, state, session, bulkClient);
119                        } catch (Throwable t) {
120
121                        }
122                    }
123                });
124            } else {
125                response.setRunning(false).setReason("session can not be opened: mode=" + mode + " path=" + path);
126            }
127            listener.onResponse(response);
128        } catch (Throwable e) {
129            logger.error(e.getMessage(), e);
130            listener.onFailure(e);
131        }
132    }

134    /**
135     * Import thread
136     *
137     * @param request request
138     * @param state   state
139     * @param session session
140     */
141    final void performImport(final KnapsackImportRequest request,
142                             final KnapsackState state,
143                             final Session<StringPacket> session,
144                             final Ingest bulkClient) {
145        try {
146            logger.info("start of import: {}", state);
147            knapsack.addImport(state);
148            final Map<String, CreateIndexRequest> indexRequestMap = new HashMap<>();
149            final Set<String> indexCreated = new HashSet<>();
150            final Map<String, String> indexReplicaMap = new HashMap<>();
151            final Map<String, Map<String, String>> aliasRequestMap = new HashMap<>();
152            // per field
153            Map<String, StringPacket> packets = new LinkedHashMap<>();
154            StringPacket packet;
155            String lastCoord = null;
156            long count = 0L;
157            while ((packet = session.read()) != null && !Thread.interrupted()) {
158                count++;
159                String index = (String) packet.meta().get("index");
160                String type = (String) packet.meta().get("type");
161                String id = (String) packet.meta().get("id");
162                String field = (String) packet.meta().get("field");
163                if (field == null) {
164                    field = "_source";
165                }
166                if ("_settings".equals(type)) {
167                    index = mapIndex(request, index);
168                    String settingsStr;
169                    // override settings by user settings
170                    if (request.hasIndexSettings(index)) {
171                        InputStreamReader reader =
172                                new InputStreamReader(new FileInputStream(request.getIndexSettings(index)), "UTF-8");
173                        settingsStr = Streams.copyToString(reader);
174                        reader.close();
175                    } else {
176                        settingsStr = packet.payload().toString();
177                    }
178                    if (!"_all".equals(index)) {
179                        logger.info("index {}: found settings {}", index, settingsStr);
180                        CreateIndexRequest createIndexRequest = indexRequestMap.get(index);
181                        if (createIndexRequest == null) {
182                            createIndexRequest = createIndexRequest(index);
183                            indexRequestMap.put(index, createIndexRequest);
184                        }
185                        Settings.Builder indexSettingsBuilder = Settings.settingsBuilder()
186                                .loadFromSource(settingsStr);
187                        indexReplicaMap.put(index, indexSettingsBuilder.get("index.number_of_replicas"));
188                        // get settings, but overwrite replica, and disable refresh for faster bulk
189                        Settings indexSettings = indexSettingsBuilder
190                                .put("index.refresh_interval", "-1s")
191                                .put("index.number_of_replicas", 0)
192                                .build();
193                        logger.info("switching index {} for bulk indexing: {}", index, indexSettings.getAsMap());
194                        createIndexRequest.settings(indexSettings);
195                    }
196                } else if ("_mapping".equals(id)) {
197                    // first map type, then index
198                    type = mapType(request, index, type);
199                    index = mapIndex(request, index);
200                    String mapping;
201                    // override mappings by user request
202                    if (request.hasIndexTypeMapping(index, type)) {
203                        InputStreamReader reader =
204                                new InputStreamReader(new FileInputStream(request.getIndexTypeMapping(index, type)), "UTF-8");
205                        mapping = Streams.copyToString(reader);
206                        reader.close();
207                    } else {
208                        mapping = packet.payload();
209                    }
210                    if (!"_all".equals(index)) {
211                        logger.info("index {}: found mapping {}", index, mapping);
212                        CreateIndexRequest createIndexRequest = indexRequestMap.get(index);
213                        if (createIndexRequest == null) {
214                            createIndexRequest = createIndexRequest(index);
215                            indexRequestMap.put(index, createIndexRequest);
216                        }
217                        createIndexRequest.mapping(type, mapping);
218                    }
219                } else if ("_alias".equals(id)) {
220                    Map<String, String> aliases = new HashMap<>();
221                    if (aliasRequestMap.containsKey(index)) {
222                        aliases = aliasRequestMap.get(index);
223                    }
224                    aliases.put(type, packet.payload());
225                    aliasRequestMap.put(index, aliases);
226                } else {
227                    // index normal document fields. Check for sane entries here.
228                    if (index != null && type != null && id != null && packet.payload() != null) {
229                        // additional check for Mac tar "." artifacts and skip them (should we check for lowercase here?)
230                        if (!type.startsWith(".") && !id.startsWith(".")) {
231                            String coord = index + File.separator + type + File.separator + id;
232                            if (!coord.equals(lastCoord) && !packets.isEmpty()) {
233                                indexPackets(bulkClient, indexRequestMap, indexCreated, aliasRequestMap, request, packets);
234                                packets.clear();
235                            }
236                            packets.put(field, packet);
237                            lastCoord = coord;
238                        }
239                    }
240                }
241            }
242            if (!packets.isEmpty()) {
243                indexPackets(bulkClient, indexRequestMap, indexCreated, aliasRequestMap, request, packets);
244            }
245            bulkClient.flushIngest();
246            bulkClient.waitForResponses(TimeValue.timeValueSeconds(60));
247            for (String index : indexReplicaMap.keySet()) {
248                try {
249                    logger.info("resetting refresh rate for index {}", index);
250                    bulkClient.stopBulk(index);
251                    Integer replica = Integer.parseInt(indexReplicaMap.get(index));
252                    logger.info("resetting replica level {} for index {}", replica, index);
253                    bulkClient.updateReplicaLevel(index, replica);
254                } catch (Exception e) {
255                    logger.error(e.getMessage(), e);
256                }
257            }
258            for (String index : indexCreated) {
259                bulkClient.refreshIndex(index);
260            }
261            bulkClient.shutdown();
262            logger.info("end of import: {}, count = {}", state, count);
263        } catch (Throwable e) {
264            logger.error(e.getMessage(), e);
265        } finally {
266            try {
267                knapsack.removeImport(state);
268                session.close();
269            } catch (IOException e) {
270                logger.error(e.getMessage(), e);
271            }
272        }
273    }

275    private void indexPackets(Ingest bulkClient, Map<String, CreateIndexRequest> indexRequestMap, Set<String> indexCreated,
276                              Map<String, Map<String, String>> aliasRequestMap,
277                              KnapsackImportRequest request, Map<String, StringPacket> packets) {
278        StringPacket packet = packets.values().iterator().next(); // first packet
279        String index = (String) packet.meta().get("index");
280        String type = (String) packet.meta().get("type");
281        String id = (String) packet.meta().get("id");
282        // check if index must be created
283        if (indexRequestMap.containsKey(index)) {
284            CreateIndexRequest createIndexRequest = indexRequestMap.remove(index);
285            if (request.withMetadata()) {
286                logger.info("creating index {}", index);
287                createIndexRequest.timeout(request.getTimeout());
288                try {
289                    CreateIndexResponse response =
290                            bulkClient.client().execute(CreateIndexAction.INSTANCE, createIndexRequest).actionGet();
291                    if (!response.isAcknowledged()) {
292                        logger.warn("index creation was not acknowledged");
293                    }
294                    indexCreated.add(index);
295                } catch (IndexAlreadyExistsException e) {
296                    logger.warn("index already exists: {}", index);
297                }
298            }
299        } else {
300            // we assume this one is automatically created by the bulk API
301            indexCreated.add(index);
302        }
303        if (aliasRequestMap.containsKey(index)) {
304            Map<String, String> aliases = aliasRequestMap.remove(index);
305            if (request.withMetadata()) {
306                IndicesAliasesRequestBuilder requestBuilder =
307                        new IndicesAliasesRequestBuilder(bulkClient.client(), IndicesAliasesAction.INSTANCE);
308                for (String alias : aliases.keySet()) {
309                    requestBuilder.addAlias(index, alias, aliases.get(alias));
310                }
311                logger.info("creating {} aliases for index {}", aliases.size(), index);
312                requestBuilder.execute().actionGet();
313            }
314        }
315        // index document begins here
316        IndexRequest indexRequest = new IndexRequest(mapIndex(request, index), mapType(request, index, type), id);
317        for (String f : packets.keySet()) {
318            if (f == null) {
319                continue;
320            }
321            Object o = packets.get(f).payload();
322            if (o == null) {
323                logger.error("empty payload detected");
324                continue;
325            }
326            String payload = o.toString();
327            switch (f) {
328                case "_parent":
329                    indexRequest.parent(payload);
330                    break;
331                case "_routing":
332                    indexRequest.routing(payload);
333                    break;
334                case "_timestamp":
335                    indexRequest.timestamp(payload);
336                    break;
337                case "_version":
338                    indexRequest.versionType(VersionType.EXTERNAL).version(Long.parseLong(payload));
339                    break;
340                case "_source":
341                    indexRequest.source(payload);
342                    break;
343                default:
344                    if (!f.startsWith(".")) {
345                        indexRequest.source(f, payload);
346                    }
347                    break;
348            }
349        }
350        bulkClient.bulkIndex(indexRequest);
351    }
