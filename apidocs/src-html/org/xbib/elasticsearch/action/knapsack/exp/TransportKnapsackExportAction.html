063public class TransportKnapsackExportAction extends TransportAction<KnapsackExportRequest, KnapsackExportResponse> {
064
065    private final Client client;
066
067    private final NodeService nodeService;
068
069    private final KnapsackService knapsack;
070
071    @Inject
072    public TransportKnapsackExportAction(Settings settings,
073                                         ThreadPool threadPool,
074                                         Client client, NodeService nodeService, ActionFilters actionFilters,
075                                         IndexNameExpressionResolver indexNameExpressionResolver,
076                                         KnapsackService knapsack) {
077        super(settings, KnapsackExportAction.NAME, threadPool, actionFilters, indexNameExpressionResolver);
078        this.client = client;
079        this.nodeService = nodeService;
080        this.knapsack = knapsack;
081    }
082
083    @Override
084    protected void doExecute(final KnapsackExportRequest request, ActionListener<KnapsackExportResponse> listener) {
085        final KnapsackState state = new KnapsackState()
086                .setMode("export")
087                .setNodeName(nodeService.nodeName());
088        final KnapsackExportResponse response = new KnapsackExportResponse()
089                .setState(state);
090        try {
091            Path path = request.getPath();
092            if (path == null) {
093                path = new File("_all.tar.gz").toPath();
094            }
095            ByteSizeValue bytesToTransfer = request.getBytesToTransfer();
096            BytesProgressWatcher watcher = new BytesProgressWatcher(bytesToTransfer.bytes());
097            final ArchiveSession session = ArchiveService.newSession(path, watcher);
098            EnumSet<Session.Mode> mode = EnumSet.of(request.isOverwriteAllowed() ?
099                            Session.Mode.OVERWRITE : Session.Mode.WRITE);
100            session.open(mode, path);
101            if (session.isOpen()) {
102                state.setPath(path).setTimestamp(new DateTime());
103                response.setRunning(true);
104                knapsack.submit(new Thread() {
105                    public void run() {
106                        performExport(request, state, session);
107                    }
108                });
109            } else {
110                response.setRunning(false).setReason("session can not be opened: mode=" + mode + " path=" + path);
111            }
112            listener.onResponse(response);
113        } catch (Throwable e) {
114            logger.error(e.getMessage(), e);
115            listener.onFailure(e);
116        }
117    }
118
119    /**
120     * Export thread
121     *
122     * @param request request
123     * @param state   state
124     * @param session session
125     */
126    final void performExport(final KnapsackExportRequest request,
127                             final KnapsackState state,
128                             final ArchiveSession session) {
129        try {
130            logger.info("start of export: {}", state);
131            knapsack.addExport(state);
132            Map<String, Set<String>> indices = new HashMap<>();
133            for (String s : Strings.commaDelimitedListToSet(request.getIndex())) {
134                indices.put(s, Strings.commaDelimitedListToSet(request.getType()));
135            }
136            // never write _settings / _mapping to bulk format
137            if (request.isWithMetadata() && !(session instanceof EsBulkSession)) {
138                if (request.getIndexTypeNames() != null) {
139                    for (Object spec : request.getIndexTypeNames().keySet()) {
140                        if (spec == null) {
141                            continue;
142                        }
143                        String[] s = spec.toString().split("/");
144                        String index = s[0];
145                        String type = s.length > 1 ? s[1] : null;
146                        if (!"_all".equals(index)) {
147                            Set<String> types = indices.get(index);
148                            if (types == null) {
149                                types = new HashSet<>();
150                            }
151                            if (type != null) {
152                                types.add(type);
153                            }
154                            indices.put(index, types);
155                        }
156                    }
157                }
158                // get settings for all indices
159                logger.info("getting settings for indices {}", indices.keySet());
160                Set<String> settingsIndices = new HashSet<>(indices.keySet());
161                settingsIndices.remove("_all");
162                Map<String, String> settings = getSettings(client, settingsIndices.toArray(new String[settingsIndices.size()]));
163                logger.info("found indices: {}", settings.keySet());
164                // we resolved the specs in indices to the real indices in the settings
165                // get mapping and alias per index and create index if copy mode is enabled
166                for (String index : settings.keySet()) {
167                    CreateIndexRequest createIndexRequest = createIndexRequest(mapIndex(request, index));
168                    StringPacket packet = new StringPacket();
169                    packet.meta("index", mapIndex(request, index));
170                    packet.meta("type", "_settings");
171                    packet.payload(settings.get(index));
172                    session.write(packet);
173                    Set<String> types = indices.get(index);
174                    createIndexRequest.settings(settings.get(index));
175                    logger.info("getting mappings for index {} and types {}", index, types);
176                    Map<String, String> mappings = getMapping(client, index, types != null ? new HashSet<>(types) : null);
177                    logger.info("found mappings: {}", mappings.keySet());
178                    for (String type : mappings.keySet()) {
179                        packet = new StringPacket();
180                        packet.meta("index", mapIndex(request, index));
181                        packet.meta("type", mapType(request, index, type));
182                        packet.meta("id", "_mapping");
183                        packet.payload(mappings.get(type));
184                        session.write(packet);
185                        logger.info("adding mapping: {}", mapType(request, index, type));
186                        createIndexRequest.mapping(mapType(request, index, type), mappings.get(type));
187                    }
188                    if (request.isWithAliases()) {
189                        logger.info("getting aliases for index {}", index);
190                        Map<String, String> aliases = getAliases(client, index);
191                        logger.info("found {} aliases", aliases.size());
192                        for (String alias : aliases.keySet()) {
193                            packet = new StringPacket();
194                            packet.meta("index", mapIndex(request, index));
195                            packet.meta("type", alias);
196                            packet.meta("id", "_alias");
197                            packet.payload(aliases.get(alias));
198                            session.write(packet);
199                        }
200                    }
201                }
202            }
203            SearchRequest searchRequest = request.getSearchRequest();
204            if (searchRequest == null) {
205                searchRequest = new SearchRequestBuilder(client, SearchAction.INSTANCE)
206                        .setQuery(QueryBuilders.matchAllQuery()).request();
207            }
208            for (String index : indices.keySet()) {
209                searchRequest.searchType(SearchType.SCAN).scroll(request.getTimeout());
210                if (!"_all".equals(index)) {
211                    searchRequest.indices(index);
212                }
213                Set<String> types = indices.get(index);
214                if (types != null) {
215                    searchRequest.types(types.toArray(new String[types.size()]));
216                }
217                // use local node client here
218                SearchResponse searchResponse = client.search(searchRequest).actionGet();
219                long total = 0L;
220                while (searchResponse.getScrollId() != null && !Thread.interrupted()) {
221                    searchResponse = client.prepareSearchScroll(searchResponse.getScrollId())
222                            .setScroll(request.getTimeout())
223                            .execute()
224                            .actionGet();
225                    long hits = searchResponse.getHits().getHits().length;
226                    if (hits == 0) {
227                        break;
228                    }
229                    total += hits;
230                    logger.debug("total={} hits={} took={}", total, hits, searchResponse.getTookInMillis());
231                    for (SearchHit hit : searchResponse.getHits()) {
232                        for (String f : hit.getFields().keySet()) {
233                            StringPacket packet = new StringPacket();
234                            packet.meta("index", mapIndex(request, hit.getIndex()));
235                            packet.meta("type", mapType(request, hit.getIndex(), hit.getType()));
236                            packet.meta("id", hit.getId());
237                            packet.meta("field", f);
238                            packet.payload(hit.getFields().get(f).getValue().toString());
239                            session.write(packet);
240                        }
241                        if (!hit.getFields().keySet().contains("_source")) {
242                            StringPacket packet = new StringPacket();
243                            packet.meta("index", mapIndex(request, hit.getIndex()));
244                            packet.meta("type", mapType(request, hit.getIndex(), hit.getType()));
245                            packet.meta("id", hit.getId());
246                            packet.meta("field", "_source");
247                            packet.payload(hit.getSourceAsString());
248                            session.write(packet);
249                        }
250                    }
251                }
252            }
253            session.close();
254            logger.info("end of export: {}, packets = {}, total bytes transferred = {}, rate = {}",
255                    state, session.getPacketCounter(),
256                    session.getWatcher().getTotalBytesInAllTransfers(),
257                    String.format("%f", session.getWatcher().getRecentByteRatePerSecond()));
258        } catch (Throwable e) {
259            logger.error(e.getMessage(), e);
260        } finally {
261            try {
262                knapsack.removeExport(state);
263            } catch (IOException e) {
264                logger.error(e.getMessage(), e);
265            }
266        }
267    }
