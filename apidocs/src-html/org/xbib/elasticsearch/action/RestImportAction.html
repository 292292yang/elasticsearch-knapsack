050
051public class RestImportAction extends BaseRestHandler {
052
053    private final KnapsackService knapsackService;
054
055    @Inject
056    public RestImportAction(Settings settings, Client client,
057            RestController controller, KnapsackService knapsackService) {
058        super(settings, client);
059        this.knapsackService = knapsackService;
060
061        controller.registerHandler(POST, "/_import", this);
062        controller.registerHandler(POST, "/{index}/_import", this);
063        controller.registerHandler(POST, "/{index}/{type}/_import", this);
064        controller.registerHandler(GET, "/_import/state", new RestKnapsackImportStatus());
065    }
066
067    @Override
068    public void handleRequest(final RestRequest request, RestChannel channel) {
069        final String newIndex = request.param("index", "_all");
070        final String newType = request.param("type");
071        final String desc = newIndex + (newType != null ? "_" + newType : "");
072        try {
073            XContentBuilder builder = restContentBuilder(request)
074                    .startObject()
075                    .field("ok", true)
076                    .endObject();
077
078            ClusterHealthResponse healthResponse =
079                    client.admin().cluster().prepareHealth().setWaitForYellowStatus()
080                            .setTimeout("30s").execute().actionGet(30000);
081
082            if (healthResponse.isTimedOut()) {
083                throw new IOException("cluster not healthy, cowardly refusing to continue with export");
084            }
085
086            final String target = request.param("target", desc);
087            String scheme = request.param("scheme", "targz");
088            for (String codec : StreamCodecService.getCodecs()) {
089                if (target.endsWith(codec)) {
090                    scheme = "tar" + codec;
091                }
092            }
093
094            ConnectionService service = ConnectionService.getInstance();
095            ConnectionFactory factory = service.getConnectionFactory(scheme);
096            final Connection<Session> connection = factory.getConnection(URI.create(scheme + ":" + target));
097            final Session session = connection.createSession();
098            session.open(Session.Mode.READ);
099
100            EsExecutors.daemonThreadFactory(settings, "Knapsack import [" + desc + "]")
101                    .newThread(new ImportThread(request, connection, session, target)).start();
102
103            channel.sendResponse(new XContentRestResponse(request, OK, builder));
104        } catch (IOException ex) {
105            try {
106                logger.error(ex.getMessage(), ex);
107                channel.sendResponse(new XContentThrowableRestResponse(request, ex));
108            } catch (Exception ex2) {
109                logger.error(ex2.getMessage(), ex2);
110            }
111        }
112    }
113
114    private class ImportThread extends Thread {
115
116        private final Map<String, String> indices = newHashMap();
117
118        private final Map<String, String> mappings = newHashMap();
119
120        private final Set<String> created = newHashSet();
121
122        private final RestRequest request;
123
124        private final Connection<Session> connection;
125
126        private final Session session;
127
128        private final String target;
129
130        private String timeout;
131
132        ImportThread(RestRequest request, Connection<Session> connection, Session session, String target) {
133            this.request = request;
134            this.connection = connection;
135            this.session = session;
136            this.target = target;
137        }
138
139        @Override
140        public void run() {
141
142            final Map<String, String> params = request.params();
143            final int size = request.paramAsInt("bulk_size", 100);
144            final int maxActiveBulkRequest = request.paramAsInt("max_active_bulks", 10);
145            final String newIndex = request.param("index", "_all");
146            final String newType = request.param("type");
147            timeout = request.param("timeout", "30s");
148
149            BulkOperation op = new BulkOperation(client, logger)
150                    .setBulkSize(size)
151                    .setMaxActiveRequests(maxActiveBulkRequest);
152
153            final KnapsackStatus status = new KnapsackStatus(new String[] { newIndex },
154                    newType != null ? new String[] { newType } : null, target);
155            try {
156                logger.info("starting import of {}", target);
157                knapsackService.addImport(status);
158                Packet<String> packet;
159                while ((packet = session.read()) != null) {
160                    String[] entry = packet.getName().split("/");
161                    String index = entry.length > 0 ? entry[0] : null;
162                    String type = entry.length > 1 ? entry[1] : null;
163                    String id = entry.length > 2 ? entry[2] : null;
164                    // re-map to new index/type
165                    if (newIndex != null && !newIndex.equals("_all")) {
166                        index = newIndex;
167                    }
168                    if (newType != null) {
169                        type = newType;
170                    }
171                    if (entry.length == 2) {
172                        if ("_settings".equals(entry[1])) {
173                            String settings;
174                            if (params.containsKey(index + "_settings")) {
175                                InputStreamReader reader = new InputStreamReader(new FileInputStream(params.get(index + "_settings")), "UTF-8");
176                                settings = Streams.copyToString(reader);
177                                reader.close();
178                            } else {
179                                settings = packet.getPacket();
180                            }
181                            logger.info("index {}: got settings {}", index, settings);
182                            indices.put(index, settings);
183                            if (!createIndex(index)) {
184                                throw new IOException("unable to create index '" + index + "' with settings " + settings);
185                            }
186                        }
187                    } else if (entry.length == 3) {
188                        String mapping;
189                        if ("_mapping".equals(entry[2])) {
190                            if (params.containsKey(index + "_" + type + "_mapping")) {
191                                InputStreamReader reader = new InputStreamReader(new FileInputStream(params.get(index + "_" + type + "_mapping")), "UTF-8");
192                                mapping = Streams.copyToString(reader);
193                                reader.close();
194                            } else {
195                                mapping = packet.getPacket();
196                            }
197                            mappings.put(index + "/" + type, mapping);
198                            if (!createMapping(index, type)) {
199                                throw new IOException("unable to create index type '" + index + "/" + type + "' with mapping " + mapping);
200                            }
201                        } else {
202                            // index document
203                            op.index(index, type, id, packet.getPacket());
204                        }
205                    }
206                }
207                logger.info("import of {} completed", target);
208                session.close();
209                connection.close();
210
211            } catch (Exception e) {
212                logger.error(e.getMessage(), e);
213            } finally {
214                try {
215                    op.flush();
216                    knapsackService.removeImport(status);
217                } catch (IOException ex) {
218                    logger.error(ex.getMessage(), ex);
219                }
220            }
221        }
222
223        private boolean createIndex(String index) {
224            if ("_all".equals(index)) {
225                logger.warn("not creating index, index name is _all");
226                return true;
227            }
228            if (created.contains(index)) {
229                logger.warn("index {} already created", index);
230                return true;
231            }
232            String settings = indices.get(index);
233            created.add(index);
234            logger.info("creating index {} from import with settings {}", index, settings);
235            CreateIndexRequest createIndexRequest = createIndexRequest(index);
236            createIndexRequest.timeout(timeout);
237            if (settings != null) {
238                createIndexRequest.source(settings);
239            }
240            CreateIndexResponse response = client.admin().indices().create(createIndexRequest).actionGet();
241            return response.isAcknowledged();
242        }
243
244        private boolean createMapping(String index, String type) throws IOException {
245            if ("_all".equals(index)) {
246                logger.warn("not creating mapping, index name is _all");
247                return true;
248            }
249            if (!createIndex(index)) {
250                throw new IOException("unable to create index " + index);
251            }
252            String desc = index + "/" + type;
253            if (created.contains(desc)) {
254                logger.warn("mapping {} already created", desc);
255                return true;
256            }
257            String mapping = mappings.get(desc);
258            created.add(desc);
259            logger.info("creating mapping {} from import", desc);
260            PutMappingRequest putMappingRequest = putMappingRequest(index);
261            putMappingRequest.timeout(timeout);
262            putMappingRequest.type(type);
263            if (mapping != null) {
264                putMappingRequest.source(mapping);
265            }
266            PutMappingResponse response = client.admin().indices().putMapping(putMappingRequest).actionGet();
267            return response.isAcknowledged();
268        }
269    }
270
271    private class RestKnapsackImportStatus implements RestHandler {
272
273        @Override
274        public void handleRequest(RestRequest request, RestChannel channel) {
275            try {
276                XContentBuilder builder = restContentBuilder(request);
277                builder.startObject()
278                        .field("imports")
279                        .startArray();
280                for (KnapsackStatus export : knapsackService.getImports()) {
281                    export.toXContent(builder, EMPTY_PARAMS);
282                }
283                builder.endArray()
284                        .endObject();
285                channel.sendResponse(new XContentRestResponse(request, OK, builder));
286            } catch (IOException ioe) {
287                try {
288                    channel.sendResponse(new XContentThrowableRestResponse(request, ioe));
289                } catch (IOException e) {
290                    // ignore
291                }
292            }
293        }
294    }
