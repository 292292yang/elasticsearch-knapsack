059    /**
060     * The knapsack import action opens an archive and transfers the content into Elasticsearch
061     */
062    public class RestImportAction extends BaseRestHandler {
063    
064        private final ClusterName clusterName;
065    
066        private final KnapsackHelper knapsackHelper;
067    
068        @Inject
069        public RestImportAction(Settings settings,
070                                Client client,
071                                RestController controller,
072                                ClusterService clusterService,
073                                ClusterName clusterName) {
074            super(settings, client);
075            this.clusterName = clusterName;
076            this.knapsackHelper = new KnapsackHelper(client, clusterService);
077    
078            controller.registerHandler(POST, "/_import", this);
079            controller.registerHandler(POST, "/{index}/_import", this);
080            controller.registerHandler(POST, "/{index}/{type}/_import", this);
081    
082            controller.registerHandler(POST, "/_import/{mode}", this);
083            controller.registerHandler(POST, "/{index}/_import/{mode}", this);
084            controller.registerHandler(POST, "/{index}/{type}/_import/{mode}", this);
085    
086            controller.registerHandler(GET, "/_import/state", new RestKnapsackImportState());
087        }
088    
089        @Override
090        public void handleRequest(final RestRequest request, RestChannel channel) {
091            try {
092                ConnectionService<Session<Packet<String>>> service = ConnectionService.getInstance();
093                final String newIndex = request.param("index", "_all");
094                final String newType = request.param("type");
095                final String defaultSpec = newIndex + (newType != null ? "_" + newType : "") + ".tar.gz";
096                final String path = request.param("path", defaultSpec);
097                URI pathURI = URI.create("file:" + path);
098                if ("s3".equals(request.param("mode")) && request.param("s3") != null && !new File(path).exists()) {
099                    // fetch file from S3
100                    URI s3 = URI.create(request.param("s3")
101                            + "?bucketName=" + request.param("bucketName")
102                            + "&key=" + request.param("key")
103                            + "&path=" + path);
104                    logger.info("trying to transfer file from s3 to: {}", s3);
105                    ConnectionFactory<Session<Packet<String>>> factory = service.getConnectionFactory(s3);
106                    final Connection<Session<Packet<String>>> connection = factory.getConnection(s3);
107                    final Session session = connection.createSession();
108                    session.open(Session.Mode.READ);
109                    session.read(); // this will copy the file from S3 to path
110                    session.close();
111                    logger.info("file from s3 succesfully transferred");
112                }
113                ConnectionFactory<Session<Packet<String>>> factory = service.getConnectionFactory(pathURI);
114                final Connection<Session<Packet<String>>> connection = factory.getConnection(pathURI);
115                final Session<Packet<String>> session = connection.createSession();
116                session.open(Session.Mode.READ);
117                XContentBuilder builder = restContentBuilder(request);
118                if (session.isOpen()) {
119                    EsExecutors.daemonThreadFactory(settings, "knapsack-import-[" + pathURI + "]")
120                            .newThread(new ImportThread(request, connection, session)).start();
121                    builder.startObject()
122                            .field("running", true)
123                            .field("mode", request.param("mode") != null ? request.param("mode") : "import")
124                            .field("type", factory.getName())
125                            .field("path", connection.getURI())
126                            .endObject();
127                } else {
128                    builder.startObject()
129                            .field("running", false)
130                            .endObject();
131                }
132                channel.sendResponse(new XContentRestResponse(request, OK, builder));
133            } catch (Throwable ex) {
134                try {
135                    logger.error(ex.getMessage(), ex);
136                    channel.sendResponse(new XContentThrowableRestResponse(request, ex));
137                } catch (Exception ex2) {
138                    logger.error(ex2.getMessage(), ex2);
139                }
140            }
141        }

143        private class ImportThread extends Thread {
144    
145            private final RestRequest request;
146    
147            private final Map<String, CreateIndexRequest> indexRequestMap = newHashMap();
148    
149            private final Connection<Session<Packet<String>>> connection;
150    
151            private final Session<Packet<String>> session;
152    
153            private TimeValue timeout;
154    
155            private BulkClient bulkClient;
156    
157            private Map<String,Object> map;
158    
159            private Map<String, Packet<String>> packets;
160    
161            ImportThread(RestRequest request, Connection<Session<Packet<String>>> connection, Session<Packet<String>> session) {
162                this.request = request;
163                this.connection = connection;
164                this.session = session;
165            }
166    
167            @Override
168            public void run() {
169                this.bulkClient = new BulkClient();
170                final Map<String, String> params = request.params();
171                logger.info("params = {}", params);
172                this.timeout = request.paramAsTime("timeout", TimeValue.timeValueSeconds(30L));
173                final int maxActionsPerBulkRequest = request.paramAsInt("maxActionsPerBulkRequest", 1000);
174                final int maxBulkConcurrency = request.paramAsInt("maxBulkConcurrency",
175                        Runtime.getRuntime().availableProcessors() * 2);
176                map = toMap(request.param("map"));
177                logger.info("index/type map = {}", map);
178                URI destination = KnapsackHelper.getAddress(client, clusterName, request.param("host"), request.paramAsInt("port", 0), request.param("cluster"));
179                String path = connection.getURI().getSchemeSpecificPart();
180                final boolean copy = "copy".equals(request.param("mode"));
181                final boolean s3 = "s3".equals(request.param("mode"));
182                final KnapsackStatus status = new KnapsackStatus()
183                        .setTimestamp(new Date())
184                        .setMap(map)
185                        .setPath(path)
186                        .setURI(destination)
187                        .setCopy(copy)
188                        .setS3(s3);
189                try {
190                    logger.info("starting import: {}", status);
191                    knapsackHelper.addImport(status);
192                    bulkClient.flushInterval(TimeValue.timeValueSeconds(5))
193                        .maxActionsPerBulkRequest(maxActionsPerBulkRequest)
194                        .maxConcurrentBulkRequests(maxBulkConcurrency)
195                        .newClient(destination, KnapsackHelper.clientSettings(destination));
196                    logger.info("waiting for healthy cluster...");
197                    bulkClient.waitForCluster(ClusterHealthStatus.YELLOW, timeout);
198                    logger.info("... cluster is ready");
199                    packets = newLinkedHashMap();
200                    Packet<String> packet;
201                    String lastCoord = null;
202                    while ((packet = session.read()) != null) {
203                        String[] entry = KnapsackPacket.decodeName(packet.name());
204                        if (entry.length != 4) {
205                            throw new ElasticsearchIllegalStateException("archive entry too short, can't import");
206                        }
207                        String index = entry[0];
208                        String type = entry[1];
209                        String id = entry[2];
210                        String field = entry[3];
211                        if ("_settings".equals(type)) {
212                            index = mapIndex(index);
213                            String settings;
214                            // override settings by user request
215                            if (params.containsKey(index + "_settings")) {
216                                InputStreamReader reader = new InputStreamReader(new FileInputStream(params.get(index + "_settings")), "UTF-8");
217                                settings = Streams.copyToString(reader);
218                                reader.close();
219                            } else {
220                                settings = packet.packet();
221                            }
222                            if (!"_all".equals(index)) {
223                                logger.info("index {}: got settings {}", index, settings);
224                                CreateIndexRequest createIndexRequest = indexRequestMap.get(index);
225                                if (createIndexRequest == null) {
226                                    createIndexRequest = createIndexRequest(index);
227                                    indexRequestMap.put(index, createIndexRequest);
228                                }
229                                createIndexRequest.settings(settings);
230                            }
231                        }
232                        else if ("_mapping".equals(id)) {
233                            type = mapType(index, type);
234                            index = mapIndex(index);
235                            String mapping;
236                            // override mappings by user request
237                            if (params.containsKey(index + "_" + type + "_mapping")) {
238                                InputStreamReader reader = new InputStreamReader(new FileInputStream(params.get(index + "_" + type + "_mapping")), "UTF-8");
239                                mapping = Streams.copyToString(reader);
240                                reader.close();
241                            } else {
242                                mapping = packet.packet();
243                            }
244                            if (!"_all".equals(index)) {
245                                logger.info("index {}: got mapping {}", index, mapping);
246                                CreateIndexRequest createIndexRequest = indexRequestMap.get(index);
247                                if (createIndexRequest == null) {
248                                    createIndexRequest = createIndexRequest(index);
249                                    indexRequestMap.put(index, createIndexRequest);
250                                }
251                                createIndexRequest.mapping(type, mapping);
252                            }
253                        } else {
254                            // normal document fields
255                            String coord = index + File.separator + type + File.separator + id;
256                            if (!coord.equals(lastCoord) && !packets.isEmpty()) {
257                                indexPackets(packets);
258                                packets.clear();
259                            }
260                            packets.put(field, packet);
261                            lastCoord = coord;
262                        }
263                    }
264                    if (!packets.isEmpty()) {
265                        indexPackets(packets);
266                    }
267                    logger.info("end of import: {}", status);
268                    session.close();
269                    bulkClient.refresh().shutdown();
270                    bulkClient = null;
271                } catch (Exception e) {
272                    logger.error(e.getMessage(), e);
273                } finally {
274                    try {
275                        knapsackHelper.removeImport(status);
276                    } catch (IOException ex) {
277                        logger.error(ex.getMessage(), ex);
278                    }
279                }
280            }
281    
282            private void indexPackets(Map<String, Packet<String>> packets) throws IOException {
283                String entryName = packets.values().iterator().next().name();
284                String[] entry = KnapsackPacket.decodeName(entryName);
285                if (entry.length < 3) {
286                    throw new ElasticsearchIllegalStateException("entry too short: " + entryName);
287                }
288                String index = entry[0];
289                String type = entry[1];
290                String id = entry[2];
291                if (indexRequestMap.containsKey(index)) {
292                    CreateIndexRequest createIndexRequest = indexRequestMap.remove(index);
293                    if (request.paramAsBoolean("createIndex", true)) {
294                        logger.info("creating index {}", index);
295                        createIndexRequest.timeout(timeout);
296                        try {
297                            CreateIndexResponse response = bulkClient.client().admin().indices()
298                                    .create(createIndexRequest).actionGet();
299                            if (!response.isAcknowledged()) {
300                                logger.warn("index creation was not acknowledged");
301                            }
302                        } catch (IndexAlreadyExistsException e) {
303                            if (request.paramAsBoolean("createIndex", true)) {
304                                throw e;
305                            } else {
306                                logger.warn("index already exists: {}", index);
307                            }
308                        }
309                    }
310                }
311                IndexRequest indexRequest = new IndexRequest(mapIndex(index),
312                        mapType(index, type), id);
313                for (String f : packets.keySet()) {
314                    if (f == null) {
315                        continue;
316                    }
317                    if ("_parent".equals(f)) {
318                        indexRequest.parent(packets.get(f).packet());
319                    } else if ("_routing".equals(f)) {
320                        indexRequest.routing(packets.get(f).packet());
321                    } else if ("_timestamp".equals(f)) {
322                        indexRequest.timestamp(packets.get(f).packet());
323                    } else if ("_version".equals(f)) {
324                        indexRequest.versionType(VersionType.EXTERNAL)
325                                .version(Long.parseLong(packets.get(f).packet()));
326                    } else if ("_source".equals(f)) {
327                        indexRequest.source(packets.get(f).packet());
328                    } else {
329                        indexRequest.source(f, packets.get(f).packet());
330                    }
331                }
332                bulkClient.index(indexRequest);
333            }

335            private Map<String, Object> toMap(String s) {
336                if (s == null) {
337                    logger.warn("no map defined");
338                    return newHashMap();
339                }
340                BytesArray ref = new BytesArray(s);
341                try {
342                    byte[] b = ref.hasArray() ? ref.array() : ref.toBytes();
343                    XContentParser parser = XContentFactory.xContent(XContentType.JSON).createParser(b, 0, b.length);
344                    return parser.mapAndClose();
345                } catch (IOException e) {
346                    logger.warn("parse error while reading map", e);
347                    return newHashMap();
348                }
349            }
350    
351            private String mapIndex(String index) {
352                return map.containsKey(index) ? map.get(index).toString() : index;
353            }
354    
355            private String mapType(String index, String type) {
356                String s = index + "/" + type;
357                return map.containsKey(s) ? map.get(s).toString() : type;
358            }
359        }

361        private class RestKnapsackImportState implements RestHandler {
362    
363            @Override
364            public void handleRequest(RestRequest request, RestChannel channel) {
365                try {
366                    XContentBuilder builder = restContentBuilder(request);
367                    builder.startObject()
368                            .field("imports")
369                            .startArray();
370                    for (KnapsackStatus export : knapsackHelper.getImports()) {
371                        export.toXContent(builder, EMPTY_PARAMS);
372                    }
373                    builder.endArray()
374                            .endObject();
375                    channel.sendResponse(new XContentRestResponse(request, OK, builder));
376                } catch (IOException ioe) {
377                    try {
378                        channel.sendResponse(new XContentThrowableRestResponse(request, ioe));
379                    } catch (IOException e) {
380                        logger.error("unable to send response to client");
381                    }
382                }
383            }
384        }
