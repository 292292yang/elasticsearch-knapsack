019import org.elasticsearch.ElasticsearchException;
020import org.elasticsearch.cluster.ClusterService;
021import org.elasticsearch.cluster.ClusterState;
022import org.elasticsearch.cluster.ClusterStateUpdateTask;
023import org.elasticsearch.cluster.metadata.MetaData;
024import org.elasticsearch.common.component.AbstractLifecycleComponent;
025import org.elasticsearch.common.inject.Inject;
026import org.elasticsearch.common.inject.Injector;
027import org.elasticsearch.common.logging.ESLogger;
028import org.elasticsearch.common.logging.ESLoggerFactory;
029import org.elasticsearch.common.settings.Settings;
030import org.elasticsearch.common.xcontent.XContentBuilder;
031import org.elasticsearch.common.xcontent.XContentFactory;
032import org.elasticsearch.common.xcontent.XContentParser;
033
034import java.io.IOException;
035import java.util.ArrayList;
036import java.util.Iterator;
037import java.util.List;
038import java.util.concurrent.ExecutorService;
039import java.util.concurrent.Executors;
040import java.util.concurrent.Future;
041import java.util.concurrent.TimeUnit;
042
043import static org.elasticsearch.common.xcontent.ToXContent.EMPTY_PARAMS;
044import static org.elasticsearch.common.xcontent.XContentFactory.xContent;
045import static org.elasticsearch.common.xcontent.XContentParser.Token.END_ARRAY;
046import static org.elasticsearch.common.xcontent.XContentType.JSON;
047
048public class KnapsackService extends AbstractLifecycleComponent<KnapsackService> {
049
050    private final static ESLogger logger = ESLoggerFactory.getLogger(KnapsackService.class.getSimpleName());
051
052    public static final String EXPORT_STATE_SETTING_NAME = "plugin.knapsack.export.state";
053
054    public static final String IMPORT_STATE_SETTING_NAME = "plugin.knapsack.import.state";
055
056    private final Injector injector;
057
058    private ExecutorService executor;
059
060    private List<Future<?>> tasks;
061
062    @Inject
063    public KnapsackService(Settings settings, Injector injector) {
064        super(settings);
065        this.injector = injector;
066    }
067
068    @Override
069    protected void doStart() throws ElasticsearchException {
070        this.tasks = new ArrayList<>();
071        this.executor = newExecutorService();
072    }
073
074    @Override
075    protected void doStop() throws ElasticsearchException {
076    }
077
078    @Override
079    protected void doClose() throws ElasticsearchException {
080        int size = tasks.size();
081        if (size > 0) {
082            for (Future<?> f : tasks) {
083                if (!f.isDone()) {
084                    logger.info("aborting knapsack task {}", f);
085                    boolean b = f.cancel(true);
086                    if (!b) {
087                        logger.error("knapsack task {} could not be cancelled", f);
088                    }
089                }
090            }
091            tasks.clear();
092        }
093        logger.info("knapsack shutdown...");
094        executor.shutdown();
095        try {
096            this.executor.awaitTermination(5, TimeUnit.SECONDS);
097        } catch (InterruptedException e) {
098            throw new ElasticsearchException(e.getMessage());
099        }
100        if (!executor.isShutdown()) {
101            logger.info("knapsack shutdown now");
102            executor.shutdownNow();
103            try {
104                this.executor.awaitTermination(5, TimeUnit.SECONDS);
105            } catch (InterruptedException e) {
106                throw new ElasticsearchException(e.getMessage());
107            }
108        }
109        logger.info("knapsack shutdown complete");
110
111    }
112
113    protected ExecutorService newExecutorService() {
114        return Executors.newFixedThreadPool(4);
115    }
116

118        return get(IMPORT_STATE_SETTING_NAME);
119    }

124
125    public void removeImport(KnapsackState targetImport) throws IOException {
126        remove(IMPORT_STATE_SETTING_NAME, getImports(), targetImport);
127    }
128
129    public void updateImport(KnapsackState targetImport) throws IOException {
130        update(IMPORT_STATE_SETTING_NAME, getImports(), targetImport);
131    }
132
133    public List<KnapsackState> getExports() throws IOException {
134        return get(EXPORT_STATE_SETTING_NAME);
135    }
136
137    public void addExport(KnapsackState newExport) throws IOException {
138        add(EXPORT_STATE_SETTING_NAME, getExports(), newExport);
139    }
140
141    public void removeExport(KnapsackState targetExport) throws IOException {
142        remove(EXPORT_STATE_SETTING_NAME, getExports(), targetExport);
143    }
144
145    public void updateExport(KnapsackState targetExport) throws IOException {
146        update(EXPORT_STATE_SETTING_NAME, getExports(), targetExport);
147    }
148
149    private List<KnapsackState> get(String name) throws IOException {
150        return parseStates(getClusterSetting(name));
151    }
152
153    private void add(String name, List<KnapsackState> values, KnapsackState targetValue) throws IOException {
154        logger.debug("add: {} -> {}", name, values);
155        updateClusterSettings(name, generateSetting(ImmutableList.<KnapsackState>builder()
156                .addAll(values).add(targetValue).build()));
157    }
158
159    private void remove(String name, List<KnapsackState> values, KnapsackState targetValue) throws IOException {
160        logger.debug("remove: {} -> {}", name, values);
161        ImmutableList.Builder<KnapsackState> updatedValues = ImmutableList.builder();
162        for (KnapsackState value : values) {
163            if (!value.equals(targetValue)) {
164                updatedValues.add(value);
165            }
166        }
167        updateClusterSettings(name, generateSetting(updatedValues.build()));
168    }
169
170    private void update(String name, List<KnapsackState> values, KnapsackState targetValue) throws IOException {
171        ImmutableList.Builder<KnapsackState> updatedValues = ImmutableList.builder();
172        for (KnapsackState value : values) {
173            if (value.equals(targetValue)) {
174                updatedValues.add(targetValue);
175            } else {
176                updatedValues.add(value);
177            }
178        }
179        updateClusterSettings(name, generateSetting(updatedValues.build()));
180    }
181
182    private String getClusterSetting(String name) {
183        return getClusterSettings().get(name, "[]");
184    }
185
186    private Settings getClusterSettings() {
187        ClusterService clusterService = injector.getInstance(ClusterService.class);
188        return clusterService.state().getMetaData().transientSettings();
189    }
190
191    private void updateClusterSettings(final String name, final String value) {
192        logger.debug("update cluster settings: {} -> {}", name, value);
193        final ClusterService clusterService = injector.getInstance(ClusterService.class);
194        try {
195            clusterService.submitStateUpdateTask("knapsack", new ClusterStateUpdateTask() {
196                @Override
197                public ClusterState execute(ClusterState currentState) throws Exception {
198                    Settings oldSettings = clusterService.state().getMetaData().transientSettings();
199                    Settings newSettings = Settings.settingsBuilder()
200                            .put(oldSettings)
201                            .put(name, value)
202                            .build();
203                    MetaData.Builder mdBuilder = MetaData.builder(currentState.metaData());
204                    mdBuilder.transientSettings(newSettings);
205                    logger.debug("cluster transient settings update done: {} -> {}",
206                            oldSettings.getAsMap(), newSettings.getAsMap());
207                    return ClusterState.builder(currentState).metaData(mdBuilder).build();
208                }
209
210                @Override
211                public void onFailure(String source, Throwable t) {
212                    logger.error("submitStateUpdateTask failure", t);
213                }
214            });
215        } catch (Throwable t) {
216            logger.error("submitStateUpdateTask failure", t);
217        }
218
219    }
220
221    private static List<KnapsackState> parseStates(String value) throws IOException {
222        XContentParser parser = xContent(JSON).createParser(value);
223        ImmutableList.Builder<KnapsackState> builder = ImmutableList.builder();
224        parser.nextToken();
225        while (parser.nextToken() != END_ARRAY) {
226            KnapsackState state = new KnapsackState();
227            builder.add(state.fromXContent(parser));
228        }
229        return builder.build();
230    }
231
232    private static String generateSetting(List<KnapsackState> values) throws IOException {
233        XContentBuilder builder = XContentFactory.jsonBuilder();
234        builder.startArray();
235        for (KnapsackState value : values) {
236            value.toXContent(builder, EMPTY_PARAMS);
237        }
238        builder.endArray();
239        return builder.string();
240    }
241
242    public void submit(Runnable runnable) {
243        Iterator<Future<?>> it = tasks.iterator();
244        while (it.hasNext()) {
245            Future<?> f = it.next();
246            if (f.isDone()) {
247                it.remove();
248            }
249        }
250        Future<?> f = executor.submit(runnable);
251        tasks.add(f);
252    }
253
254    public void abort() {
255        doClose();
256        this.executor = newExecutorService();
257    }
