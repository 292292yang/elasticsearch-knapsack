001package org.xbib.elasticsearch.knapsack;
002
003import static org.elasticsearch.common.base.Objects.firstNonNull;
004import static org.elasticsearch.common.collect.Lists.newArrayList;
005import static org.elasticsearch.common.xcontent.XContentParser.Token.END_ARRAY;
006import static org.elasticsearch.common.xcontent.XContentParser.Token.END_OBJECT;
007import static org.elasticsearch.common.xcontent.XContentParser.Token.FIELD_NAME;
008import static org.elasticsearch.common.xcontent.XContentParser.Token.START_ARRAY;
009import static org.elasticsearch.common.xcontent.XContentParser.Token.VALUE_NULL;
010
011import java.io.IOException;
012import java.util.Date;
013import java.util.List;
014import java.util.concurrent.TimeUnit;
015
016import org.elasticsearch.common.joda.DateMathParser;
017import org.elasticsearch.common.joda.Joda;
018import org.elasticsearch.common.xcontent.ToXContent;
019import org.elasticsearch.common.xcontent.XContentBuilder;
020import org.elasticsearch.common.xcontent.XContentParser;
021import org.elasticsearch.common.xcontent.XContentParser.Token;
022
023public class KnapsackStatus implements ToXContent {
024
025    private Date timestamp;
026
027    private String[] indices;
028
029    private String[] types;
030
031    private String target;
032
033    protected KnapsackStatus() {
034    }
035
036    public KnapsackStatus(Date timestamp, String[] indices, String[] types, String target) {
037        this.timestamp = timestamp;
038        this.indices = firstNonNull(indices, new String[] {});
039        this.types = firstNonNull(types, new String[] {});
040        this.target = target;
041    }
042
043    public KnapsackStatus(String[] indices, String[] types, String target) {
044        this(new Date(), indices, types, target);
045    }
046
047    public KnapsackStatus fromXContent(XContentParser parser) throws IOException {
048        DateMathParser dateParser = new DateMathParser(Joda.forPattern("dateOptionalTime"), TimeUnit.MILLISECONDS);
049        Long startTimestamp = null;
050        String[] indices = null;
051        String[] types = null;
052        String target = null;
053        String currentFieldName = null;
054        Token token;
055        while ((token = parser.nextToken()) != END_OBJECT) {
056            if (token == FIELD_NAME) {
057                currentFieldName = parser.currentName();
058            } else if (token.isValue() || token == VALUE_NULL) {
059                if ("started".equals(currentFieldName)) {
060                    startTimestamp =  dateParser.parse(parser.text(), 0);
061                } else if ("target".equals(currentFieldName)) {
062                    target = parser.text();
063                }
064            } else if (token == START_ARRAY) {
065                List<String> values = newArrayList();
066                while ((parser.nextToken()) != END_ARRAY) {
067                    values.add(parser.text());
068                }
069                if ("indices".equals(currentFieldName)) {
070                    indices = values.toArray(new String[values.size()]);
071                } else if ("types".equals(currentFieldName)) {
072                    types = values.toArray(new String[values.size()]);
073                }
074            }
075        }
076        return new KnapsackStatus(new Date(startTimestamp), indices, types, target);
077    }
078
079    @Override
080    public XContentBuilder toXContent(XContentBuilder builder, Params params) throws IOException {
081        return builder.startObject()
082                .field("started", timestamp)
083                .field("indices", indices)
084                .field("types", types)
085                .field("target", target)
086                .endObject();
087    }
088
089    public String id() {
090        StringBuilder sb = new StringBuilder();
091        sb.append(timestamp.getTime()).append(target);
092        if (indices != null) {
093            for (String index : indices) {
094                sb.append(index);
095            }
096        }
097        if (types != null) {
098            for (String type: types) {
099                sb.append(type);
100            }
101        }
102        return sb.toString();
103    }
104
105    @Override
106    public int hashCode() {
107        return id().hashCode();
108    }
109
110    @Override
111    public boolean equals(Object obj) {
112        if (this == obj) {
113            return true;
114        }
115        if (obj == null) {
116            return false;
117        }
118        if (getClass() != obj.getClass()) {
119            return false;
120        }
121        KnapsackStatus other = (KnapsackStatus) obj;
122        return id().equals(other.id());
123    }
124
125}
