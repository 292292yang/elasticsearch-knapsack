027import java.io.IOException;

029
030public class ClientHelper {
031
032    public static void updateIndexSetting(ElasticsearchClient client, String index, String key, Object value) throws IOException {
033        if (client == null) {
034            throw new IOException("no client");
035        }
036        if (index == null) {
037            throw new IOException("no index name given");
038        }
039        if (key == null) {
040            throw new IOException("no key given");
041        }
042        if (value == null) {
043            throw new IOException("no value given");
044        }
045        Settings.Builder settingsBuilder = Settings.settingsBuilder();
046        settingsBuilder.put(key, value.toString());
047        UpdateSettingsRequest updateSettingsRequest = new UpdateSettingsRequest(index)
048                .settings(settingsBuilder);
049        client.execute(UpdateSettingsAction.INSTANCE, updateSettingsRequest).actionGet();
050    }
051
052    public static void waitForRecovery(ElasticsearchClient client) throws IOException {
053        RecoveryResponse response = client.execute(RecoveryAction.INSTANCE, new RecoveryRequest()).actionGet();
054    }
055
056    public static int waitForRecovery(ElasticsearchClient client, String index) throws IOException {
057        if (index == null) {
058            throw new IOException("unable to waitfor recovery, index not set");
059        }
060        RecoveryResponse response = client.execute(RecoveryAction.INSTANCE, new RecoveryRequest(index)).actionGet();
061        int shards = response.getTotalShards();
062        client.execute(ClusterHealthAction.INSTANCE, new ClusterHealthRequest(index).waitForActiveShards(shards)).actionGet();
063        return shards;
064    }
065
066    public static void waitForCluster(ElasticsearchClient client, ClusterHealthStatus status, TimeValue timeout) throws IOException {
067        try {
068            ClusterHealthResponse healthResponse =
069                    client.execute(ClusterHealthAction.INSTANCE, new ClusterHealthRequest().waitForStatus(status).timeout(timeout)).actionGet();
070            if (healthResponse != null && healthResponse.isTimedOut()) {
071                throw new IOException("cluster state is " + healthResponse.getStatus().name()
072                        + " and not " + status.name()
073                        + ", from here on, everything will fail!");
074            }
075        } catch (ElasticsearchTimeoutException e) {
076            throw new IOException("timeout, cluster does not respond to health request, cowardly refusing to continue with operations");
077        }
078    }
079
080    public static String clusterName(ElasticsearchClient client) {
081        try {
082            ClusterStateRequestBuilder clusterStateRequestBuilder =
083                    new ClusterStateRequestBuilder(client, ClusterStateAction.INSTANCE).all();
084            ClusterStateResponse clusterStateResponse = clusterStateRequestBuilder.execute().actionGet();
085            String name = clusterStateResponse.getClusterName().value();
086            int nodeCount = clusterStateResponse.getState().getNodes().size();
087            return name + " (" + nodeCount + " nodes connected)";
088        } catch (ElasticsearchTimeoutException e) {
089            return "TIMEOUT";
090        } catch (NoNodeAvailableException e) {
091            return "DISCONNECTED";
092        } catch (Throwable t) {
093            return "[" + t.getMessage() + "]";
094        }
095    }
096
097    public static String healthColor(ElasticsearchClient client) {
098        try {
099            ClusterHealthResponse healthResponse =
100                    client.execute(ClusterHealthAction.INSTANCE, new ClusterHealthRequest().timeout(TimeValue.timeValueSeconds(30))).actionGet();
101            ClusterHealthStatus status = healthResponse.getStatus();
102            return status.name();
103        } catch (ElasticsearchTimeoutException e) {
104            return "TIMEOUT";
105        } catch (NoNodeAvailableException e) {
106            return "DISCONNECTED";
107        } catch (Throwable t) {
108            return "[" + t.getMessage() + "]";
109        }
110    }
111
112    public static int updateReplicaLevel(ElasticsearchClient client, String index, int level) throws IOException {
113        waitForCluster(client, ClusterHealthStatus.YELLOW, TimeValue.timeValueSeconds(30));
114        updateIndexSetting(client, index, "number_of_replicas", level);
115        return waitForRecovery(client, index);
116    }
117
118    public static void flushIndex(ElasticsearchClient client, String index) {
119        if (client != null && index != null) {
120            client.execute(FlushAction.INSTANCE, new FlushRequest(index)).actionGet();
121        }
122    }
123
124    public static void refreshIndex(ElasticsearchClient client, String index) {
125        if (client != null && index != null) {
126            client.execute(RefreshAction.INSTANCE, new RefreshRequest(index)).actionGet();
127        }
128    }
129
130    public static void putMapping(ElasticsearchClient client, ConfigHelper configHelper, String index) {
131        if (!configHelper.mappings().isEmpty()) {
132            for (Map.Entry<String, String> me : configHelper.mappings().entrySet()) {
133                client.execute(PutMappingAction.INSTANCE,
134                        new PutMappingRequest(index).type(me.getKey()).source(me.getValue())).actionGet();
135            }
136        }
137    }
