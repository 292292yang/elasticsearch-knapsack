037
038import java.io.File;
039import java.nio.file.Path;
040
041import static org.elasticsearch.rest.RestRequest.Method.POST;
042
043/**
044 * The REST knapsack export action performs a scan/scroll action over a user defined query
045 * and stores the result in an archive file
046 */
047public class RestKnapsackExportAction extends BaseRestHandler implements KnapsackParameter {
048
049    private final static ESLogger logger = ESLoggerFactory.getLogger(RestKnapsackExportAction.class.getSimpleName());
050
051    @Inject
052    public RestKnapsackExportAction(Settings settings, Client client, RestController controller) {
053        super(settings, controller, client);
054
055        controller.registerHandler(POST, "/_export", this);
056        controller.registerHandler(POST, "/{index}/_export", this);
057        controller.registerHandler(POST, "/{index}/{type}/_export", this);
058    }
059
060    @Override
061    public void handleRequest(final RestRequest request, final RestChannel channel, final Client client) {
062        try {
063            final String index = request.param(INDEX_PARAM, "_all");
064            final String type = request.param(TYPE_PARAM);
065            final String defaultSpec = index + (type != null ? "_" + type : "") + ".tar.gz";
066            File file = new File(request.param(PATH_PARAM, defaultSpec));
067            final Path path = file.toPath();
068            KnapsackExportRequest exportRequest = new KnapsackExportRequest()
069                    .setIndex(index)
070                    .setType(type)
071                    .setPath(path)
072                    .setOverwriteAllowed(request.paramAsBoolean(OVERWRITE_PARAM, false))
073                    .withMetadata(request.paramAsBoolean(WITH_METADATA_PARAM, true))
074                    .withAliases(request.paramAsBoolean(WITH_ALIASES, true))
075                    .setIndexTypeNames(KnapsackHelper.toMap(request.param(MAP_PARAM), logger))
076                    .setSearchRequest(toSearchRequest(request))
077                    .setBytesToTransfer(request.paramAsSize(BYTES_PARAM, ByteSizeValue.parseBytesSizeValue("0", "")));
078            client.admin().indices().execute(KnapsackExportAction.INSTANCE, exportRequest,
079                    new RestToXContentListener<KnapsackExportResponse>(channel));
080        } catch (Throwable ex) {
081            try {
082                logger.error(ex.getMessage(), ex);
083                channel.sendResponse(new BytesRestResponse(channel, ex));
084            } catch (Exception ex2) {
085                logger.error(ex2.getMessage(), ex2);
086            }
087        }
088    }
089
090    private SearchRequest toSearchRequest(RestRequest request) {
091        SearchRequest searchRequest;
092        // override search action "size" (default = 10) by bulk request size. The size is per shard!
093        request.params().put("size", request.param(MAX_BULK_ACTIONS_PER_REQUEST_PARAM, "1000"));
094        searchRequest = RestSearchAction.parseSearchRequest(request, parseFieldMatcher);
095        return searchRequest;
096    }
097
098}
