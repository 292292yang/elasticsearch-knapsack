037
038import java.util.Map;
039
040import static org.elasticsearch.rest.RestRequest.Method.POST;
041
042/**
043 * The REST knapsack pull action transfers a remote index into an Elasticsearch cluster
044 */
045public class RestKnapsackPullAction extends BaseRestHandler implements KnapsackParameter {
046
047    private final static ESLogger logger = ESLoggerFactory.getLogger(RestKnapsackPullAction.class.getSimpleName());
048
049    @Inject
050    public RestKnapsackPullAction(Settings settings, Client client, RestController controller) {
051        super(settings, controller, client);
052
053        controller.registerHandler(POST, "/_pull", this);
054        controller.registerHandler(POST, "/{index}/_pull", this);
055        controller.registerHandler(POST, "/{index}/{type}/_pull", this);
056    }
057
058    @Override
059    public void handleRequest(final RestRequest request, RestChannel channel, Client client) {
060        try {
061            final String index = request.param(INDEX_PARAM, "_all");
062            final String type = request.param(TYPE_PARAM);
063            KnapsackPullRequest pullRequest = new KnapsackPullRequest()
064                    .setIndex(index)
065                    .setType(type)
066                    .setHost(request.param(HOST_PARAM, "localhost"))
067                    .setPort(request.paramAsInt(PORT_PARAM, 9300))
068                    .setCluster(request.param(CLUSTER_PARAM))
069                    .setSniff(request.paramAsBoolean(SNIFF_PARAM, true))
070                    .setMaxActionsPerBulkRequest(request.paramAsInt(MAX_BULK_ACTIONS_PER_REQUEST_PARAM, 1000))
071                    .setMaxBulkConcurrency(request.paramAsInt(MAX_BULK_CONCURRENCY_PARAM,
072                            Runtime.getRuntime().availableProcessors() * 2))
073                    .setTimeout(request.paramAsTime(TIMEOUT_PARAM, TimeValue.timeValueSeconds(30)))
074                    .withMetadata(request.paramAsBoolean(WITH_METADATA_PARAM, true))
075                    .setIndexTypeNames(KnapsackHelper.toMap(request.param(MAP_PARAM), logger))
076                    .setSearchRequest(toSearchRequest(request));
077            // add user-defined settings and mappings
078            for (Map.Entry<String, String> e : request.params().entrySet()) {
079                if (e.getKey().endsWith("_settings")) {
080                    pullRequest.addIndexSettings(e.getKey(), e.getValue());
081                } else if (e.getKey().endsWith("_mapping")) {
082                    pullRequest.addIndexTypeMapping(e.getKey(), e.getValue());
083                }
084            }
085            client.admin().indices().execute(KnapsackPullAction.INSTANCE, pullRequest,
086                    new RestToXContentListener<KnapsackPullResponse>(channel));
087        } catch (Throwable ex) {
088            try {
089                logger.error(ex.getMessage(), ex);
090                channel.sendResponse(new BytesRestResponse(channel, ex));
091            } catch (Exception ex2) {
092                logger.error(ex2.getMessage(), ex2);
093            }
094        }
095    }
096
097    private SearchRequest toSearchRequest(RestRequest request) {
098        SearchRequest searchRequest;
099        // override search action "size" (default = 10) by bulk request size. The size is per shard!
100        request.params().put("size", request.param(MAX_BULK_ACTIONS_PER_REQUEST_PARAM, "1000"));
101        searchRequest = RestSearchAction.parseSearchRequest(request, parseFieldMatcher);
102        return searchRequest;
103    }
104}
