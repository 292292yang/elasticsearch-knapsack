002package org.xbib.elasticsearch.plugin.knapsack;
003
004import org.xbib.io.Packet;
005import org.xbib.io.URIUtil;
006
007import java.io.File;
008import java.util.StringTokenizer;
009
010public class KnapsackPacket implements Packet<String> {
011
012    private String name;
013
014    private String packet;
015
016    public KnapsackPacket(String index, String type, String id, String field, String packet) {
017        this.name = encodeName(new String[]{index, type, id, field});
018        this.packet = packet;
019    }
020
021    @Override
022    public KnapsackPacket name(String name) {
023        this.name = name;
024        return this;
025    }
026
027    @Override
028    public String name() {
029        return name;
030    }
031
032    @Override
033    public KnapsackPacket packet(String packet) {
034        this.packet = packet;
035        return this;
036    }
037
038    @Override
039    public String packet() {
040        return packet;
041    }
042
043    @Override
044    public String toString() {
045        return packet;
046    }
047
048    public static String encodeName(String[] components) {
049        StringBuilder sb = new StringBuilder();
050        for (int i = 0; i < components.length; i++) {
051            if (i > 0) {
052                sb.append(File.separator);
053            }
054            sb.append(URIUtil.encode(components[i] != null ? components[i] : "", URIUtil.UTF8));
055        }
056        return sb.toString();
057    }
058
059    public static String[] decodeName(String component) {
060        String[] components = split(component, File.separator);
061        for (int i = 0; i < components.length; i++) {
062            components[i] = URIUtil.decode(components[i] != null ? components[i] : "", URIUtil.UTF8);
063        }
064        return components;
065    }
066
067    /**
068     * Split "str" into tokens by delimiters and optionally remove white spaces
069     * from the splitted tokens.
070     */
071    private static String[] split(String str, String delims) {
072        StringTokenizer tokenizer = new StringTokenizer(str, delims);
073        int n = tokenizer.countTokens();
074        String[] list = new String[n];
075        for (int i = 0; i < n; i++) {
076            list[i] = tokenizer.nextToken();
077        }
078        return list;
079    }
080
081}
