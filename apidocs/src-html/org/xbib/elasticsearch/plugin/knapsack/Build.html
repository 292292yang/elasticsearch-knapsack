002    package org.xbib.elasticsearch.plugin.knapsack;
003    
004    import java.io.ByteArrayOutputStream;
005    import java.io.IOException;
006    import java.io.InputStream;
007    import java.io.OutputStream;
008    import java.io.StringReader;
009    import java.util.Properties;
010    
011    public class Build {
012    
013        private static final Build INSTANCE;
014    
015        static {
016            String version = "NA";
017            String hash = "NA";
018            String hashShort = "NA";
019            String timestamp = "NA";
020            String date = "NA";
021    
022            try {
023                InputStream in = Build.class.getResourceAsStream("/es-plugin.properties");
024                if (in == null) {
025                    System.err.println("no es-plugin.properties in class path");
026                } else {
027                    ByteArrayOutputStream out = new ByteArrayOutputStream();
028                    copy(in, out);
029                    Properties props = new Properties();
030                    props.load(new StringReader(new String(out.toByteArray())));
031                    version = props.getProperty("version");
032                    hash = props.getProperty("hash");
033                    if (!"NA".equals(hash)) {
034                        hashShort = hash.substring(0, 7);
035                    }
036                    timestamp = props.getProperty("timestamp");
037                    date = props.getProperty("date");
038                }
039            } catch (Throwable e) {
040                // just ignore...
041            }
042            INSTANCE = new Build(version, hash, hashShort, timestamp, date);
043        }
044    
045        public static void copy(InputStream in, OutputStream out) throws IOException {
046            byte[] buffer = new byte[1024];
047            int len;
048            while ((len = in.read(buffer)) != -1) {
049                out.write(buffer, 0, len);
050            }
051        }
052    
053        private String version;
054    
055        private String hash;
056    
057        private String hashShort;
058    
059        private String timestamp;
060    
061        private String date;
062    
063        Build(String version, String hash, String hashShort, String timestamp, String date) {
064            this.version = version;
065            this.hash = hash;
066            this.hashShort = hashShort;
067            this.timestamp = timestamp;
068            this.date = date;
069        }
070    
071        public static Build getInstance() {
072            return INSTANCE;
073        }
074    
075        public String getVersion() {
076            return version;
077        }
078    
079        public String getHash() {
080            return hash;
081        }
082    
083        public String getShortHash() {
084            return hashShort;
085        }
086    
087        public String getTimestamp() {
088            return timestamp;
089        }
090    
091        public String getDate() {
092            return date;
093        }
094    }
