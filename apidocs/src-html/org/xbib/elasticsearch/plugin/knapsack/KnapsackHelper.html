033public class KnapsackHelper {
034
035    public static final String EXPORT_STATE_SETTING_NAME = "plugin.knapsack.export.state";
036
037    public static final String IMPORT_STATE_SETTING_NAME = "plugin.knapsack.import.state";
038
039    private final Client client;
040
041    private final ClusterService clusterService;
042
043    public KnapsackHelper(Client client, ClusterService clusterService) {
044        this.client = client;
045        this.clusterService = clusterService;
046    }
047
048    public List<KnapsackStatus> getImports() throws IOException {
049        return get(IMPORT_STATE_SETTING_NAME);
050    }
051
052    public void addImport(KnapsackStatus newImport) throws IOException {
053        add(IMPORT_STATE_SETTING_NAME, getImports(), newImport);
054    }
055
056    public void removeImport(KnapsackStatus targetImport) throws IOException {
057        remove(IMPORT_STATE_SETTING_NAME, getImports(), targetImport);
058    }
059
060    public void updateImport(KnapsackStatus targetImport) throws IOException {
061        update(IMPORT_STATE_SETTING_NAME, getImports(), targetImport);
062    }
063
064    public List<KnapsackStatus> getExports() throws IOException {
065        return get(EXPORT_STATE_SETTING_NAME);
066    }
067
068    public void addExport(KnapsackStatus newExport) throws IOException {
069        add(EXPORT_STATE_SETTING_NAME, getExports(), newExport);
070    }
071
072    public void removeExport(KnapsackStatus targetExport) throws IOException {
073        remove(EXPORT_STATE_SETTING_NAME, getExports(), targetExport);
074    }
075
076    public void updateExport(KnapsackStatus targetExport) throws IOException {
077        update(EXPORT_STATE_SETTING_NAME, getExports(), targetExport);
078    }
079
080    private List<KnapsackStatus> get(String name) throws IOException {
081        String value = getSetting(name);
082
083        return parseSetting(value);
084    }
085
086    private void add(String name, List<KnapsackStatus> values, KnapsackStatus newValue) throws IOException {
087        String value = generateSetting(ImmutableList.<KnapsackStatus>builder()
088                .addAll(values)
089                .add(newValue)
090                .build());
091        updateSetting(name, value);
092    }
093
094    private void remove(String name, List<KnapsackStatus> values, KnapsackStatus targetValue) throws IOException {
095        Builder<KnapsackStatus> updatedValues = ImmutableList.builder();
096        for (KnapsackStatus value : values) {
097            if (!value.equals(targetValue)) {
098                updatedValues.add(value);
099            }
100        }
101        String value = generateSetting(updatedValues.build());
102        updateSetting(name, value);
103    }
104
105    private void update(String name, List<KnapsackStatus> values, KnapsackStatus targetValue) throws IOException {
106        Builder<KnapsackStatus> updatedValues = ImmutableList.builder();
107        for (KnapsackStatus value : values) {
108            if (value.equals(targetValue)) {
109                updatedValues.add(targetValue);
110            } else {
111                updatedValues.add(value);
112            }
113        }
114
115        String value = generateSetting(updatedValues.build());
116        updateSetting(name, value);
117    }
118
119    private Settings getSettings() {
120        return clusterService.state().getMetaData().transientSettings();
121    }
122
123    private String getSetting(String name) {
124        return getSettings().get(name, "[]");
125    }
126
127    private void updateSetting(String name, String value) {
128        client.admin().cluster().prepareUpdateSettings()
129                .setTransientSettings(ImmutableSettings.builder()
130                        .put(getSettings())
131                        .put(name, value)
132                        .build())
133                .execute()
134                .actionGet();
135    }
136
137    private static List<KnapsackStatus> parseSetting(String value) throws IOException {
138        XContentParser parser = xContent(JSON).createParser(value);
139        Builder<KnapsackStatus> builder = ImmutableList.builder();
140        parser.nextToken();
141        while (parser.nextToken() != END_ARRAY) {
142            KnapsackStatus status = new KnapsackStatus();
143            builder.add(status.fromXContent(parser));
144        }
145
146        return builder.build();
147    }
148
149    private static String generateSetting(List<KnapsackStatus> values) throws IOException {
150        XContentBuilder builder = XContentFactory.jsonBuilder();
151        builder.startArray();
152        for (KnapsackStatus value : values) {
153            value.toXContent(builder, EMPTY_PARAMS);
154        }
155        builder.endArray();
156        return builder.string();
157    }
158
159    public static URI getAddress(Client client, ClusterName clusterName, String host, int port, String cluster) {
160        NodesInfoResponse response = client.admin().cluster()
161                .nodesInfo(new NodesInfoRequest().transport(true)).actionGet();
162        InetSocketTransportAddress address = (InetSocketTransportAddress)response.iterator().next()
163                .getTransport().getAddress().publishAddress();
164        if (host == null) {
165            host = address.address().getAddress().getHostAddress();
166        }
167        if (port == 0) {
168            port = address.address().getPort();
169        }
170        if (cluster == null) {
171            cluster = clusterName.value();
172        }
173        return URI.create("es://" + host + ":" + port + "?es.cluster.name=" + cluster);
174    }
175
176    public static Settings clientSettings(URI uri) {
177        return settingsBuilder()
178                .put("network.server", false)
179                .put("node.client", true)
180                .put("cluster.name", URIUtil.parseQueryString(uri).get("es.cluster.name"))
181                .put("client.transport.sniff", false)
182                .put("client.transport.ignore_cluster_name", false)
183                .put("client.transport.ping_timeout", "30s")
184                .put("client.transport.nodes_sampler_interval", "30s")
185                .build();
186    }
