036public class KnapsackHelper {
037
038    private final static ESLogger logger = ESLoggerFactory.getLogger(KnapsackHelper.class.getSimpleName());
039
040    public static final String EXPORT_STATE_SETTING_NAME = "plugin.knapsack.export.state";
041
042    public static final String IMPORT_STATE_SETTING_NAME = "plugin.knapsack.import.state";
043
044    private final Client client;
045
046    private final ClusterService clusterService;
047
048    public KnapsackHelper(Client client, ClusterService clusterService) {
049        this.client = client;
050        this.clusterService = clusterService;
051    }
052
053    public List<KnapsackStatus> getImports() throws IOException {
054        return get(IMPORT_STATE_SETTING_NAME);
055    }
056
057    public void addImport(KnapsackStatus newImport) throws IOException {
058        add(IMPORT_STATE_SETTING_NAME, getImports(), newImport);
059    }
060
061    public void removeImport(KnapsackStatus targetImport) throws IOException {
062        remove(IMPORT_STATE_SETTING_NAME, getImports(), targetImport);
063    }
064
065    public void updateImport(KnapsackStatus targetImport) throws IOException {
066        update(IMPORT_STATE_SETTING_NAME, getImports(), targetImport);
067    }
068
069    public List<KnapsackStatus> getExports() throws IOException {
070        return get(EXPORT_STATE_SETTING_NAME);
071    }
072
073    public void addExport(KnapsackStatus newExport) throws IOException {
074        add(EXPORT_STATE_SETTING_NAME, getExports(), newExport);
075    }
076
077    public void removeExport(KnapsackStatus targetExport) throws IOException {
078        remove(EXPORT_STATE_SETTING_NAME, getExports(), targetExport);
079    }
080
081    public void updateExport(KnapsackStatus targetExport) throws IOException {
082        update(EXPORT_STATE_SETTING_NAME, getExports(), targetExport);
083    }
084
085    private List<KnapsackStatus> get(String name) throws IOException {
086        String value = getSetting(name);
087
088        return parseSetting(value);
089    }
090
091    private void add(String name, List<KnapsackStatus> values, KnapsackStatus newValue) throws IOException {
092        String value = generateSetting(ImmutableList.<KnapsackStatus>builder()
093                .addAll(values)
094                .add(newValue)
095                .build());
096        updateSetting(name, value);
097    }
098
099    private void remove(String name, List<KnapsackStatus> values, KnapsackStatus targetValue) throws IOException {
100        Builder<KnapsackStatus> updatedValues = ImmutableList.builder();
101        for (KnapsackStatus value : values) {
102            if (!value.equals(targetValue)) {
103                updatedValues.add(value);
104            }
105        }
106        String value = generateSetting(updatedValues.build());
107        updateSetting(name, value);
108    }
109
110    private void update(String name, List<KnapsackStatus> values, KnapsackStatus targetValue) throws IOException {
111        Builder<KnapsackStatus> updatedValues = ImmutableList.builder();
112        for (KnapsackStatus value : values) {
113            if (value.equals(targetValue)) {
114                updatedValues.add(targetValue);
115            } else {
116                updatedValues.add(value);
117            }
118        }
119
120        String value = generateSetting(updatedValues.build());
121        updateSetting(name, value);
122    }
123
124    private Settings getSettings() {
125        return clusterService.state().getMetaData().transientSettings();
126    }
127
128    private String getSetting(String name) {
129        return getSettings().get(name, "[]");
130    }
131
132    private void updateSetting(String name, String value) {
133        client.admin().cluster().prepareUpdateSettings()
134                .setTransientSettings(ImmutableSettings.settingsBuilder()
135                        .put(getSettings())
136                        .put(name, value)
137                        .build())
138                .execute()
139                .actionGet();
140    }
141
142    private static List<KnapsackStatus> parseSetting(String value) throws IOException {
143        XContentParser parser = xContent(JSON).createParser(value);
144        Builder<KnapsackStatus> builder = ImmutableList.builder();
145        parser.nextToken();
146        while (parser.nextToken() != END_ARRAY) {
147            KnapsackStatus status = new KnapsackStatus();
148            builder.add(status.fromXContent(parser));
149        }
150
151        return builder.build();
152    }
153
154    private static String generateSetting(List<KnapsackStatus> values) throws IOException {
155        XContentBuilder builder = XContentFactory.jsonBuilder();
156        builder.startArray();
157        for (KnapsackStatus value : values) {
158            value.toXContent(builder, EMPTY_PARAMS);
159        }
160        builder.endArray();
161        return builder.string();
162    }
163
164    public static URI getAddress(Client client, ClusterName clusterName, String host, int port, String cluster) {
165        NodesInfoResponse response = client.admin().cluster()
166                .nodesInfo(new NodesInfoRequest().transport(true)).actionGet();
167        InetSocketTransportAddress address = (InetSocketTransportAddress)response.iterator().next()
168                .getTransport().getAddress().publishAddress();
169        if (host == null) {
170            host = address.address().getAddress().getHostAddress();
171        }
172        if (port == 0) {
173            port = address.address().getPort();
174        }
175        if (cluster == null) {
176            cluster = clusterName.value();
177        }
178        return URI.create("es://" + host + ":" + port + "?es.cluster.name=" + cluster);
179    }
180
181    public static Settings clientSettings(Environment environment, URI uri) {
182        return settingsBuilder()
183                .put("network.server", false)
184                .put("node.client", true)
185                .put("cluster.name", URIUtil.parseQueryString(uri).get("es.cluster.name"))
186                .put("client.transport.sniff", false)
187                .put("client.transport.ignore_cluster_name", false)
188                .put("client.transport.ping_timeout", "30s")
189                .put("client.transport.nodes_sampler_interval", "30s")
190                .put("path.plugins", ".dontexist") // this disables site plugins
191                .classLoader(getClassLoader(environment)) // this disables jvm plugins
192                .build();
193    }
194
195    /**
196     * Filter out all jvm plugins
197     * @param environment the environment
198     * @return a custom class loader with our dependencies
199     */
200    private static ClassLoader getClassLoader(Environment environment) {
201        URIClassLoader classLoader = new URIClassLoader();
202        File[] libs = new File(environment.homeFile() + "/lib").listFiles();
203        if (libs != null) {
204            for (File file : libs) {
205                if (file.getName().toLowerCase().endsWith(".jar")) {
206                    classLoader.addURI(file.toURI());
207                }
208            }
209        }
210        return classLoader;
211    }
