002package org.xbib.elasticsearch.plugin.knapsack;
003
004import org.elasticsearch.common.joda.DateMathParser;
005import org.elasticsearch.common.joda.Joda;
006import org.elasticsearch.common.xcontent.ToXContent;
007import org.elasticsearch.common.xcontent.XContentBuilder;
008import org.elasticsearch.common.xcontent.XContentParser;
009import org.elasticsearch.common.xcontent.XContentParser.Token;
010
011import java.io.IOException;
012import java.net.URI;
013import java.util.Date;
014import java.util.Map;
015import java.util.concurrent.TimeUnit;
016
017import static org.elasticsearch.common.xcontent.XContentFactory.jsonBuilder;
018import static org.elasticsearch.common.xcontent.XContentParser.Token.END_OBJECT;
019import static org.elasticsearch.common.xcontent.XContentParser.Token.FIELD_NAME;
020import static org.elasticsearch.common.xcontent.XContentParser.Token.START_OBJECT;
021import static org.elasticsearch.common.xcontent.XContentParser.Token.VALUE_NULL;
022
023public class KnapsackStatus implements ToXContent {
024
025    /**
026     * The time stamp of this Knapsack operation
027     */
028    private Date timestamp;
029
030    /**
031     * The index/type map of this Knapsack operation
032     */
033    private Map<String,Object> map;
034
035    /**
036     * The path of the archive involved in this Knapsack operation
037     */
038    private String path;
039
040    /**
041     * The URI of the Elasticsearch cluster
042     */
043    private URI uri;
044
045    /**
046     * direct copy flag
047     */
048    private boolean copy;
049
050    /**
051     * S3 flag
052     */
053    private boolean s3;
054
055    public KnapsackStatus() {
056    }
057
058    public KnapsackStatus setTimestamp(Date timestamp) {
059        this.timestamp = timestamp;
060        return this;
061    }
062
063    public KnapsackStatus setMap(Map<String,Object> map) {
064        this.map = map;
065        return this;
066    }
067
068    public KnapsackStatus setPath(String path) {
069        this.path = path;
070        return this;
071    }
072
073    public KnapsackStatus setURI(URI uri) {
074        this.uri = uri;
075        return this;
076    }
077
078    public KnapsackStatus setCopy(boolean copy) {
079        this.copy = copy;
080        return this;
081    }
082
083    public KnapsackStatus setS3(boolean s3) {
084        this.s3 = s3;
085        return this;
086    }
087
088    public KnapsackStatus fromXContent(XContentParser parser) throws IOException {
089        DateMathParser dateParser = new DateMathParser(Joda.forPattern("dateOptionalTime"), TimeUnit.MILLISECONDS);
090        Long startTimestamp = new Date().getTime();
091        Map<String,Object> map = null;
092        String path = null;
093        URI uri = null;
094        String currentFieldName = null;
095        Token token;
096        while ((token = parser.nextToken()) != END_OBJECT) {
097            if (token == FIELD_NAME) {
098                currentFieldName = parser.currentName();
099            } else if (token.isValue() || token == VALUE_NULL) {
100                if ("started".equals(currentFieldName)) {
101                    startTimestamp = dateParser.parse(parser.text(), 0);
102                } else if ("path".equals(currentFieldName)) {
103                    path = parser.text();
104                } else if ("uri".equals(currentFieldName)) {
105                    uri = URI.create(parser.text());
106                } else if ("copy".equals(currentFieldName)) {
107                    copy = parser.booleanValue();
108                } else if ("s3".equals(currentFieldName)) {
109                    s3 = parser.booleanValue();
110                }
111            } else if (token == START_OBJECT) {
112                map = parser.map();
113            }
114        }
115        return new KnapsackStatus()
116             .setTimestamp(new Date(startTimestamp))
117             .setMap(map)
118             .setPath(path)
119             .setURI(uri)
120             .setCopy(copy)
121             .setS3(s3);
122    }
123
124    @Override
125    public XContentBuilder toXContent(XContentBuilder builder, Params params) throws IOException {
126        return builder.startObject()
127                .field("started", timestamp)
128                .field("path", path)
129                .field("map", map)
130                .field("uri", uri)
131                .field("copy", copy)
132                .field("s3", s3)
133                .endObject();
134    }
135
136    public String id() {
137        StringBuilder sb = new StringBuilder();
138        sb.append(timestamp.getTime())
139            .append(path)
140            .append(map)
141            .append(uri)
142            .append(copy)
143            .append(s3);
144        return sb.toString();
145    }
146
147    @Override
148    public int hashCode() {
149        return id().hashCode();
150    }
151
152    @Override
153    public boolean equals(Object obj) {
154        if (this == obj) {
155            return true;
156        }
157        if (obj == null) {
158            return false;
159        }
160        if (getClass() != obj.getClass()) {
161            return false;
162        }
163        KnapsackStatus other = (KnapsackStatus) obj;
164        return id().equals(other.id());
165    }
166
167    public String toString() {
168        try {
169            return toXContent(jsonBuilder(), EMPTY_PARAMS).string();
170        } catch (IOException e) {
171            // ignore
172        }
173        return "";
174    }
175
176}
