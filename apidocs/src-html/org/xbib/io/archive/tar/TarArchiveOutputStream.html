135        this.currRecIdx = 0;

169    /**
170     * Ends the TAR archive without closing the underlying OutputStream.
171     * An archive consists of a series of file entries terminated by an
172     * end-of-archive entry, which consists of two 512 blocks of zero bytes.
173     * POSIX.1 requires two EOF records, like some other implementations.
174     *
175     * @throws IOException on error
176     */
177    @Override
178    public void finish() throws IOException {
179        if (finished) {
180            throw new IOException("This archive has already been finished");
181        }
182
183        if (haveUnclosedEntry) {
184            throw new IOException("This archives contains unclosed entries.");
185        }
186        writeEOFRecord();
187        writeEOFRecord();
188        flushBlock();
189        finished = true;
190    }
191
192    /**
193     * Closes the underlying OutputStream.
194     *
195     * @throws IOException on error
196     */
197    @Override
198    public void close() throws IOException {
199        if (!finished) {
200            finish();
201        }
202        if (!closed) {
203            if (outStream != null) {
204                flushBlock();
205                if (outStream != System.out
206                        && outStream != System.err) {
207                    outStream.close();
208                }
209            }
210            closed = true;
211        }
212    }
213
214    private void flushBlock() throws IOException {
215        if (outStream == null) {
216            throw new IOException("writing to an input buffer");
217        }
218        if (currRecIdx > 0) {
219            writeBlock();
220        }
221    }
222
223    private void writeBlock() throws IOException {
224        if (outStream == null) {
225            throw new IOException("writing to an input buffer");
226        }
227        outStream.write(blockBuffer, 0, blockSize);
228        outStream.flush();
229        currRecIdx = 0;
230        Arrays.fill(blockBuffer, (byte) 0);
231    }

248    /**
249     * Put an entry on the output stream. This writes the entry's
250     * header record and positions the output stream for writing
251     * the contents of the entry. Once this method is called, the
252     * stream is ready for calls to write() to write the entry's
253     * contents. Once the contents are written, closeArchiveEntry()
254     * <B>MUST</B> be called to ensure that all buffered data
255     * is completely written to the output stream.
256     *
257     * @param archiveEntry The TarEntry to be written to the archive.
258     * @throws IOException on error
259     * @throws ClassCastException  if archiveEntry is not an instance of TarArchiveEntry
260     */
261    @Override
262    public void putArchiveEntry(TarArchiveOutputEntry archiveEntry) throws IOException {
263        if (finished) {
264            throw new IOException("Stream has already been finished");
265        }
266        Map<String, String> paxHeaders = new HashMap<String, String>();
267        final String entryName = archiveEntry.getName();
268        final byte[] nameBytes = encoding.encode(entryName).array();
269        boolean paxHeaderContainsPath = false;
270        if (nameBytes.length >= NAMELEN) {
271            if (longFileMode == LONGFILE_POSIX) {
272                paxHeaders.put("path", entryName);
273                paxHeaderContainsPath = true;
274            } else if (longFileMode == LONGFILE_GNU) {
275                // create a TarEntry for the LongLink, the contents
276                // of which are the entry's name
277                TarArchiveOutputEntry longLinkEntry = new TarArchiveOutputEntry(GNU_LONGLINK, LF_GNUTYPE_LONGNAME);
278                longLinkEntry.setEntrySize(nameBytes.length + 1); // +1 for NUL
279                putArchiveEntry(longLinkEntry);
280                write(nameBytes);
281                write(0); // NUL terminator
282                closeArchiveEntry();
283            } else if (longFileMode != LONGFILE_TRUNCATE) {
284                throw new RuntimeException("file name '" + entryName
285                        + "' is too long ( > "
286                        + NAMELEN + " bytes)");
287            }
288        }
289        if (bigNumberMode == BIGNUMBER_POSIX) {
290            addPaxHeadersForBigNumbers(paxHeaders, archiveEntry);
291        } else if (bigNumberMode != BIGNUMBER_STAR) {
292            failForBigNumbers(archiveEntry);
293        }
294        if (addPaxHeadersForNonAsciiNames && !paxHeaderContainsPath
295                && !ASCII.canEncode(entryName)) {
296            paxHeaders.put("path", entryName);
297        }
298        if (addPaxHeadersForNonAsciiNames
299                && (archiveEntry.isLink() || archiveEntry.isSymbolicLink())
300                && !ASCII.canEncode(archiveEntry.getLinkName())) {
301            paxHeaders.put("linkpath", archiveEntry.getLinkName());
302        }
303        if (paxHeaders.size() > 0) {
304            writePaxHeaders(entryName, paxHeaders);
305        }
306        archiveEntry.writeEntryHeader(recordBuf, encoding, bigNumberMode == BIGNUMBER_STAR);
307        writeRecord(recordBuf);
308        currBytes = 0;
309        if (archiveEntry.isDirectory()) {
310            currSize = 0;
311        } else {
312            currSize = archiveEntry.getEntrySize();
313        }
314        currName = entryName;
315        haveUnclosedEntry = true;
316    }

318    /**
319     * Close an entry. This method MUST be called for all file
320     * entries that contain data. The reason is that we must
321     * buffer data written to the stream in order to satisfy
322     * the buffer's record based writes. Thus, there may be
323     * data fragments still being assembled that must be written
324     * to the output stream before this entry is closed and the
325     * next entry written.
326     *
327     * @throws IOException on error
328     */
329    @Override
330    public void closeArchiveEntry() throws IOException {
331        if (finished) {
332            throw new IOException("stream has already been finished");
333        }
334        if (!haveUnclosedEntry) {
335            throw new IOException("no current entry to close");
336        }
337        if (assemLen > 0) {
338            for (int i = assemLen; i < assemBuf.length; ++i) {
339                assemBuf[i] = 0;
340            }
341
342            writeRecord(assemBuf);
343
344            currBytes += assemLen;
345            assemLen = 0;
346        }
347
348        if (currBytes < currSize) {
349            throw new IOException("entry '" + currName + "' closed at '"
350                    + currBytes
351                    + "' before the '" + currSize
352                    + "' bytes specified in the header were written");
353        }
354        haveUnclosedEntry = false;
355    }

357    /**
358     * Writes bytes to the current tar archive entry. This method
359     * is aware of the current entry and will throw an exception if
360     * you attempt to write bytes past the length specified for the
361     * current entry. The method is also (painfully) aware of the
362     * record buffering required by TarBuffer, and manages buffers
363     * that are not a multiple of recordsize in length, including
364     * assembling records from small buffers.
365     *
366     * @param wBuf       The buffer to write to the archive.
367     * @param wOffset    The offset in the buffer from which to get bytes.
368     * @param numToWrite The number of bytes to write.
369     * @throws IOException on error
370     */
371    @Override
372    public void write(byte[] wBuf, int wOffset, int numToWrite) throws IOException {
373        if ((currBytes + numToWrite) > currSize) {
374            throw new IOException("request to write '" + numToWrite
375                    + "' bytes exceeds size in header of '"
376                    + currSize + "' bytes for entry '"
377                    + currName + "'");
378
379            //
380            // We have to deal with assembly!!!
381            // The programmer can be writing little 32 byte chunks for all
382            // we know, and we must assemble complete records for writing.
383            // REVIEW Maybe this should be in TarBuffer? Could that help to
384            // eliminate some of the buffer copying.
385            //
386        }
387
388        if (assemLen > 0) {
389            if ((assemLen + numToWrite) >= recordBuf.length) {
390                int aLen = recordBuf.length - assemLen;
391
392                System.arraycopy(assemBuf, 0, recordBuf, 0,
393                        assemLen);
394                System.arraycopy(wBuf, wOffset, recordBuf,
395                        assemLen, aLen);
396                writeRecord(recordBuf);
397
398                currBytes += recordBuf.length;
399                wOffset += aLen;
400                numToWrite -= aLen;
401                assemLen = 0;
402            } else {
403                System.arraycopy(wBuf, wOffset, assemBuf, assemLen,
404                        numToWrite);
405
406                wOffset += numToWrite;
407                assemLen += numToWrite;
408                numToWrite = 0;
409            }
410        }
411
412        //
413        // When we get here we have EITHER:
414        // o An empty "assemble" buffer.
415        // o No bytes to write (numToWrite == 0)
416        //
417        while (numToWrite > 0) {
418            if (numToWrite < recordBuf.length) {
419                System.arraycopy(wBuf, wOffset, assemBuf, assemLen,
420                        numToWrite);
421
422                assemLen += numToWrite;
423
424                break;
425            }
426
427            writeRecord(wBuf, wOffset);
428
429            int num = recordBuf.length;
430
431            currBytes += num;
432            numToWrite -= num;
433            wOffset += num;
434        }
435    }

437    /**
438     * Writes a PAX extended header with the given map as contents.
439     */
440    void writePaxHeaders(String entryName, Map<String, String> headers) throws IOException {
441        String name = "./PaxHeaders.X/" + stripTo7Bits(entryName);
442        if (name.length() >= NAMELEN) {
443            name = name.substring(0, NAMELEN - 1);
444        }
445        TarArchiveOutputEntry pex = new TarArchiveOutputEntry(name, LF_PAX_EXTENDED_HEADER_LC);
446        StringWriter w = new StringWriter();
447        for (Map.Entry<String, String> h : headers.entrySet()) {
448            String key = h.getKey();
449            String value = h.getValue();
450            int len = key.length() + value.length()
451                    + 3 /* blank, equals and newline */
452                    + 2 /* guess 9 < actual length < 100 */;
453            String line = len + " " + key + "=" + value + "\n";
454            int actualLength = line.getBytes(Charset.forName("UTF-8")).length;
455            while (len != actualLength) {
456                // Adjust for cases where length < 10 or > 100
457                // or where UTF-8 encoding isn't a single octet
458                // per character.
459                // Must be in loop as size may go from 99 to 100 in
460                // first pass so we'd need a second.
461                len = actualLength;
462                line = len + " " + key + "=" + value + "\n";
463                actualLength = line.getBytes(Charset.forName("UTF-8")).length;
464            }
465            w.write(line);
466        }
467        byte[] data = w.toString().getBytes(Charset.forName("UTF-8"));
468        pex.setEntrySize(data.length);
469        putArchiveEntry(pex);
470        write(data);
471        closeArchiveEntry();
472    }

474    private String stripTo7Bits(String name) {
475        final int length = name.length();
476        StringBuilder result = new StringBuilder(length);
477        for (int i = 0; i < length; i++) {
478            char stripped = (char) (name.charAt(i) & 0x7F);
479            if (stripped != 0) { // would be read as Trailing null
480                result.append(stripped);
481            }
482        }
483        return result.toString();
484    }

486    /**
487     * Write an EOF (end of archive) record to the tar archive.
488     * An EOF record consists of a record of all zeros.
489     */
490    private void writeEOFRecord() throws IOException {
491        for (int i = 0; i < recordBuf.length; ++i) {
492            recordBuf[i] = 0;
493        }
494
495        writeRecord(recordBuf);
496    }

503    private void addPaxHeadersForBigNumbers(Map<String, String> paxHeaders,
504                                            TarArchiveOutputEntry entry) {
505        addPaxHeaderForBigNumber(paxHeaders, "size", entry.getEntrySize(), MAXSIZE);
506        addPaxHeaderForBigNumber(paxHeaders, "gid", entry.getGroupId(), MAXID);
507        addPaxHeaderForBigNumber(paxHeaders, "mtime", entry.getLastModified().getTime() / 1000, MAXSIZE);
508        addPaxHeaderForBigNumber(paxHeaders, "uid", entry.getUserId(), MAXID);
509        // star extensions by J\u00f6rg Schilling
510        addPaxHeaderForBigNumber(paxHeaders, "SCHILY.devmajor", entry.getDevMajor(), MAXID);
511        addPaxHeaderForBigNumber(paxHeaders, "SCHILY.devminor", entry.getDevMinor(), MAXID);
512        // there is no PAX header for file mode
513        failForBigNumber("mode", entry.getMode(), MAXID);
514    }
515
516    private void addPaxHeaderForBigNumber(Map<String, String> paxHeaders,
517                                          String header, long value,
518                                          long maxValue) {
519        if (value < 0 || value > maxValue) {
520            paxHeaders.put(header, String.valueOf(value));
521        }
522    }
523
524    private void failForBigNumbers(TarArchiveOutputEntry entry) {
525        failForBigNumber("entry size", entry.getEntrySize(), MAXSIZE);
526        failForBigNumber("group id", entry.getGroupId(), MAXID);
527        failForBigNumber("last modification time", entry.getLastModified().getTime() / 1000, MAXSIZE);
528        failForBigNumber("user id", entry.getUserId(), MAXID);
529        failForBigNumber("mode", entry.getMode(), MAXID);
530        failForBigNumber("major device number", entry.getDevMajor(), MAXID);
531        failForBigNumber("minor device number", entry.getDevMinor(), MAXID);
532    }
533
534    private void failForBigNumber(String field, long value, long maxValue) {
535        if (value < 0 || value > maxValue) {
536            throw new RuntimeException(field + " '" + value
537                    + "' is too big ( > "
538                    + maxValue + " )");
539        }
540    }

542    private void writeRecord(byte[] record) throws IOException {
543        if (outStream == null) {
544            throw new IOException("Output buffer is closed");
545        }
546        if (record.length != recordSize) {
547            throw new IOException("record to write has length '"
548                    + record.length
549                    + "' which is not the record size of '"
550                    + recordSize + "'");
551        }
552        if (currRecIdx >= recsPerBlock) {
553            writeBlock();
554        }
555        System.arraycopy(record, 0, blockBuffer, (currRecIdx * recordSize), recordSize);
556        currRecIdx++;
557    }
558
559    private void writeRecord(byte[] buf, int offset) throws IOException {
560        if (outStream == null) {
561            throw new IOException("Output buffer is closed");
562        }
563        if ((offset + recordSize) > buf.length) {
564            throw new IOException("record has length '" + buf.length
565                    + "' with offset '" + offset
566                    + "' which is less than the record size of '"
567                    + recordSize + "'");
568        }
569        if (currRecIdx >= recsPerBlock) {
570            writeBlock();
571        }
572        System.arraycopy(buf, offset, blockBuffer, (currRecIdx * recordSize), recordSize);
573        currRecIdx++;
574    }
