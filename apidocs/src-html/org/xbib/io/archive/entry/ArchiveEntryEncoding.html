016package org.xbib.io.archive.entry;
017
018import java.io.IOException;
019import java.nio.ByteBuffer;
020
021/**
022 * An interface for encoders that do a pretty encoding of archive
023 * filenames.
024 *
025 * <p>There are mostly two implementations, one that uses java.nio
026 * {@link java.nio.charset.Charset Charset} and one implementation,
027 * which copes with simple 8 bit charsets, because java-1.4 did not
028 * support Cp437 in java.nio.</p>
029 *
030 * <p>The main reason for defining an own encoding layer comes from
031 * the problems with {@link String#getBytes(String)
032 * String.getBytes}, which encodes unknown characters as ASCII
033 * quotation marks ('?'). Quotation marks are per definition an
034 * invalid filename on some operating systems  like Windows, which
035 * leads to ignored ZIP entries.</p>
036 *
037 * <p>All implementations should implement this interface in a
038 * reentrant way.</p>
039 */
040public interface ArchiveEntryEncoding {
041    /**
042     * Check, whether the given string may be losslessly encoded using this
043     * encoding.
044     *
045     * @param name A filename or ZIP comment.
046     * @return Whether the given name may be encoded with out any losses.
047     */
048    boolean canEncode(String name);
049
050    /**
051     * Encode a filename or a comment to a byte array suitable for
052     * storing it to a zip entry.
053     *
054     * <p>Examples for CP 437 (in pseudo-notation, right hand side is
055     * C-style notation):</p>
056     * <pre>
057     *  encode("\u20AC_for_Dollar.txt") = "%U20AC_for_Dollar.txt"
058     *  encode("\u00D6lf\u00E4sser.txt") = "\231lf\204sser.txt"
059     * </pre>
060     *
061     * @param name A filename or ZIP comment.
062     * @return A byte buffer with a backing array containing the
063     * encoded name.  Unmappable characters or malformed
064     * character sequences are mapped to a sequence of utf-16
065     * words encoded in the format <code>%Uxxxx</code>.  It is
066     * assumed, that the byte buffer is positioned at the
067     * beginning of the encoded result, the byte buffer has a
068     * backing array and the limit of the byte buffer points
069     * to the end of the encoded result.
070     * @throws java.io.IOException
071     */
072    ByteBuffer encode(String name) throws IOException;
073
074    /**
075     * @param data The byte values to decode.
076     * @return The decoded string.
077     * @throws java.io.IOException
078     */
079    String decode(byte[] data) throws IOException;
080}
