016package org.xbib.io.archive;
017
018import org.xbib.io.BytesProgressWatcher;
019
020import java.io.IOException;
021import java.io.InputStream;
022
023/**
024 * Archive input streams <b>MUST</b> override the
025 * {@link #read(byte[], int, int)} - or {@link #read()} -
026 * method so that reading from the stream generates EOF for the end of
027 * data in each entry as well as at the end of the file proper.
028 * The {@link #getNextEntry()} method is used to reset the input stream
029 * ready for reading the data from the next entry.
030 */
031public abstract class ArchiveInputStream<E extends ArchiveEntry> extends InputStream {
032
033    private BytesProgressWatcher watcher;
034
035    public ArchiveInputStream setWatcher(BytesProgressWatcher watcher) {
036        this.watcher = watcher;
037        return this;
038    }
039
040    public BytesProgressWatcher getWatcher() {
041        return watcher;
042    }
043
044    /**
045     * Returns the next Archive Entry in this Stream.
046     *
047     * @return the next entry,
048     * or {@code null} if there are no more entries
049     * @throws java.io.IOException if the next entry could not be read
050     */
051    public abstract E getNextEntry() throws IOException;
052
053    /**
054     * Reads a byte of data. This method will block until enough input is
055     * available.
056     * <p>
057     * Simply calls the {@link #read(byte[], int, int)} method.
058     * <p>
059     * MUST be overridden if the {@link #read(byte[], int, int)} method
060     * is not overridden; may be overridden otherwise.
061     *
062     * @return the byte read, or -1 if end of input is reached
063     * @throws java.io.IOException if an I/O error has occurred
064     */
065    @Override
066    public int read() throws IOException {
067        byte[] b = new byte[1];
068        int num = read(b, 0, 1);
069        return num == -1 ? -1 : b[0] & 0xFF;
070    }
071
072}
