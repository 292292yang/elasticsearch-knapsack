002package org.xbib.io.archive;
003
004import org.xbib.io.BytesProgressWatcher;
005import org.xbib.io.compress.CompressCodecService;
006
007import java.nio.file.Path;
008import java.util.Map;
009import java.util.ServiceLoader;
010import java.util.Set;
011import java.util.WeakHashMap;
012
013public class ArchiveService {
014
015    private final static Map<String, ArchiveCodec> codecs = new WeakHashMap<String, ArchiveCodec>();
016
017    private final static ArchiveService instance = new ArchiveService();
018
019    private ArchiveService() {
020        ServiceLoader<ArchiveCodec> loader = ServiceLoader.load(ArchiveCodec.class);
021        for (ArchiveCodec codec : loader) {
022            if (!codecs.containsKey(codec.getName())) {
023                codecs.put(codec.getName(), codec);
024            }
025        }
026    }
027
028    public static ArchiveService getInstance() {
029        return instance;
030    }
031
032    public ArchiveCodec getCodec(String name) {
033        if (codecs.containsKey(name)) {
034            return codecs.get(name);
035        }
036        throw new IllegalArgumentException("archive codec for " + name + " not found in " + codecs);
037    }
038
039    public static Set<String> getCodecs() {
040        return codecs.keySet();
041    }
042
043    @SuppressWarnings("unchecked")
044    public static <I extends ArchiveInputStream, O extends ArchiveOutputStream> ArchiveSession<I, O> newSession(Path path, BytesProgressWatcher watcher) {
045        for (String archiverName : getCodecs()) {
046            if (canOpen(archiverName, path)) {
047                return codecs.get(archiverName).newSession(watcher);
048            }
049        }
050        throw new IllegalArgumentException("no archive session implementation found for path " + path);
051    }
052
053    private static boolean canOpen(String suffix, Path path) {
054        String pathStr = path.toString();
055        if (pathStr.endsWith("." + suffix.toLowerCase()) || pathStr.endsWith("." + suffix.toUpperCase())) {
056            return true;
057        }
058        Set<String> codecs = CompressCodecService.getCodecs();
059        for (String codec : codecs) {
060            String s = "." + suffix + "." + codec;
061            if (pathStr.endsWith(s) || pathStr.endsWith(s.toLowerCase()) || pathStr.endsWith(s.toUpperCase())) {
062                return true;
063            }
064        }
065        return false;
066    }
067
068}
