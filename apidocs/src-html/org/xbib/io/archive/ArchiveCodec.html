016package org.xbib.io.archive;
017
018import org.xbib.io.BytesProgressWatcher;
019
020import java.io.IOException;
021import java.io.InputStream;
022import java.io.OutputStream;
023
024/**
025 * An archive codec defines the session and the input or output stream that are
026 * used for reading or writing to an archive.
027 *
028 * @param <S> the archive session type
029 * @param <I> the archive input stream type
030 * @param <O> the archive output type
031 */
032public interface ArchiveCodec<S extends ArchiveSession, I extends ArchiveInputStream, O extends ArchiveOutputStream> {
033
034    /**
035     * Returns the name of this archive codec ("cpio", "tar", "zip")
036     *
037     * @return the name
038     */
039    String getName();
040
041    /**
042     * Creates a new archive session with a progress watcher
043     *
044     * @param watcher the progress watcher
045     * @return the new archive session
046     */
047    S newSession(BytesProgressWatcher watcher);
048
049    /**
050     * Creates a new archive input stream
051     *
052     * @param in the input stream for the archive input stream
053     * @return the archive input stream
054     * @throws IOException if archive input stream can not be created
055     */
056    I createArchiveInputStream(InputStream in) throws IOException;
057
058    /**
059     * Creates a new archive output stream
060     *
061     * @param out the output stream for the archive output stream
062     * @return the archive output stream
063     * @throws IOException if archive output stream can not be created
064     */
065    O createArchiveOutputStream(OutputStream out) throws IOException;
066
067}
