042    /**
043     * Strips Windows' drive letter as well as any leading slashes,
044     * turns path separators into forward slahes.
045     */
046    public static String normalizeFileName(String fileName, boolean preserveLeadingSlashes) {
047        String osname = System.getProperty("os.name").toLowerCase(Locale.ENGLISH);
048        if (osname.startsWith("windows")) {
049            if (fileName.length() > 2) {
050                char ch1 = fileName.charAt(0);
051                char ch2 = fileName.charAt(1);
052                if (ch2 == ':' && ((ch1 >= 'a' && ch1 <= 'z') || (ch1 >= 'A' && ch1 <= 'Z'))) {
053                    fileName = fileName.substring(2);
054                }
055            }
056        } else if (osname.contains("netware")) {
057            int colon = fileName.indexOf(':');
058            if (colon != -1) {
059                fileName = fileName.substring(colon + 1);
060            }
061        }
062        fileName = fileName.replace(File.separatorChar, '/');
063        // No absolute pathnames. Windows paths can start with "\\NetworkDrive\",  so we loop on starting /'s.
064        while (!preserveLeadingSlashes && fileName.startsWith("/")) {
065            fileName = fileName.substring(1);
066        }
067        return fileName;
068    }

070    public static final ArchiveEntryEncoding DEFAULT_ENCODING = ArchiveEntryEncodingHelper.getEncoding(null);
071
072    public static final ArchiveEntryEncoding FALLBACK_ENCODING = new ArchiveEntryEncoding() {
073        public boolean canEncode(String name) {
074            return true;
075        }
076
077        public ByteBuffer encode(String name) {
078            final int length = name.length();
079            byte[] buf = new byte[length];
080            for (int i = 0; i < length; ++i) {
081                buf[i] = (byte) name.charAt(i);
082            }
083            return ByteBuffer.wrap(buf);
084        }
085
086        public String decode(byte[] buffer) {
087            final int length = buffer.length;
088            StringBuilder result = new StringBuilder(length);
089            for (byte b : buffer) {
090                if (b == 0) {
091                    break;
092                }
093                result.append((char) (b & 0xFF));
094            }
095            return result.toString();
096        }
097    };

099    /**
100     * Copy a name into a buffer.
101     * Copies characters from the name into the buffer
102     * starting at the specified offset.
103     * If the buffer is longer than the name, the buffer
104     * is filled with trailing NULs.
105     * If the name is longer than the buffer,
106     * the output is truncated.
107     *
108     * @param name   The header name from which to copy the characters.
109     * @param buf    The buffer where the name is to be stored.
110     * @param offset The starting offset into the buffer
111     * @param length The maximum number of header bytes to copy.
112     * @return The updated offset, i.e. offset + length
113     */
114    public static int formatNameBytes(String name, byte[] buf, final int offset, final int length) {
115        try {
116            return formatNameBytes(name, buf, offset, length, DEFAULT_ENCODING);
117        } catch (IOException ex) {
118            try {
119                return formatNameBytes(name, buf, offset, length, ArchiveUtils.FALLBACK_ENCODING);
120            } catch (IOException ex2) {
121                // impossible
122                throw new RuntimeException(ex2);
123            }
124        }
125    }

127    /**
128     * Copy a name into a buffer.
129     * Copies characters from the name into the buffer
130     * starting at the specified offset.
131     * If the buffer is longer than the name, the buffer
132     * is filled with trailing NULs.
133     * If the name is longer than the buffer,
134     * the output is truncated.
135     *
136     * @param name     The header name from which to copy the characters.
137     * @param buf      The buffer where the name is to be stored.
138     * @param offset   The starting offset into the buffer
139     * @param length   The maximum number of header bytes to copy.
140     * @param encoding name of the encoding to use for file names
141     * @return The updated offset, i.e. offset + length
142     */
143    public static int formatNameBytes(String name, byte[] buf, final int offset,
144                                      final int length,
145                                      final ArchiveEntryEncoding encoding)
146            throws IOException {
147        int len = name.length();
148        ByteBuffer b = encoding.encode(name);
149        while (b.limit() > length && len > 0) {
150            b = encoding.encode(name.substring(0, --len));
151        }
152        final int limit = b.limit();
153        System.arraycopy(b.array(), b.arrayOffset(), buf, offset, limit);
154
155        // Pad any remaining output bytes with NUL
156        for (int i = limit; i < length; ++i) {
157            buf[offset + i] = 0;
158        }
159
160        return offset + length;
161    }

164    /**
165     * Generates a string containing the name, isDirectory setting and size of an entry.
166     * <p>
167     * For example:
168     * <tt>-    2000 main.c</tt>
169     * <tt>d     100 testfiles</tt>
170     *
171     * @return the representation of the entry
172     */
173    public static String toString(ArchiveEntry entry) {
174        StringBuilder sb = new StringBuilder();
175        sb.append(entry.isDirectory() ? 'd' : '-');// c.f. "ls -l" output
176        String size = Long.toString((entry.getEntrySize()));
177        sb.append(' ');
178        // Pad output to 7 places, leading spaces
179        for (int i = 7; i > size.length(); i--) {
180            sb.append(' ');
181        }
182        sb.append(size);
183        sb.append(' ').append(entry.getName());
184        return sb.toString();
185    }

189     *
190     * @param expected the expected string
191     * @param buffer the buffer
192     * @param offset the offset
193     * @param length the length
194     * @return {@code true} if buffer is the same as the expected string

196    public static boolean matchAsciiBuffer(String expected, byte[] buffer, int offset, int length) {
197        byte[] buffer1;
198        try {
199            buffer1 = expected.getBytes("ASCII");
200        } catch (UnsupportedEncodingException e) {
201            throw new RuntimeException(e); // Should not happen
202        }
203        return isEqual(buffer1, 0, buffer1.length, buffer, offset, length, false);
204    }

251    /**
252     * Compare byte buffers, optionally ignoring trailing nulls
253     *
254     * @param buffer1             first buffer
255     * @param offset1             offset of first buffer
256     * @param length1             length of first buffer
257     * @param buffer2             second buffer
258     * @param offset2             offset of second buffer
259     * @param length2             length of second buffer
260     * @param ignoreTrailingNulls ignore trailing null if true
261     * @return {@code true} if buffer1 and buffer2 have same contents, having regard to trailing nulls
262     */
263    public static boolean isEqual(final byte[] buffer1, final int offset1, final int length1,
264                                  final byte[] buffer2, final int offset2, final int length2,
265                                  boolean ignoreTrailingNulls) {
266        int minLen = length1 < length2 ? length1 : length2;
267        for (int i = 0; i < minLen; i++) {
268            if (buffer1[offset1 + i] != buffer2[offset2 + i]) {
269                return false;
270            }
271        }
272        if (length1 == length2) {
273            return true;
274        }
275        if (ignoreTrailingNulls) {
276            if (length1 > length2) {
277                for (int i = length2; i < length1; i++) {
278                    if (buffer1[offset1 + i] != 0) {
279                        return false;
280                    }
281                }
282            } else {
283                for (int i = length1; i < length2; i++) {
284                    if (buffer2[offset2 + i] != 0) {
285                        return false;
286                    }
287                }
288            }
289            return true;
290        }
291        return false;
292    }

300    /**
301     * Encode archive entry name. Ensure there is no '/' File.separator at the end of the name, otherwise 'tar' will
302     * recognize it as directory entry.
303     *
304     * @param packet the packet
305     * @return teh entry name
306     */
307    public static String encodeArchiveEntryName(StringPacket packet) {
308        StringBuilder sb = new StringBuilder();
309        for (int i = 0; i < keys.length; i++) {
310            if (i > 0) {
311                sb.append(File.separator);
312            }
313            Object o = packet.meta().get(keys[i]);
314            if (o == null) {
315                o = EMPTY; // writing "null" here avoids File.separator at the end of the name
316            }
317            sb.append(encode(o.toString(), UTF8));
318        }
319        return sb.toString();
320    }

326     * @param archiveEntryName the entry name
327     */
328    public static void decodeArchiveEntryName(Packet packet, String archiveEntryName) {
329        String[] components = split(archiveEntryName, File.separator);
330        for (int i = 0; i < components.length; i++) {
331            packet.meta(keys[i], decode(components[i], UTF8));

335    /**
336     * Split "str" into tokens by delimiters and optionally remove white spaces
337     * from the splitted tokens.
338     */
339    private static String[] split(String str, String delims) {
340        StringTokenizer tokenizer = new StringTokenizer(str, delims);
341        int n = tokenizer.countTokens();
342        String[] list = new String[n];
343        for (int i = 0; i < n; i++) {
344            list[i] = tokenizer.nextToken();
345        }
346        return list;
347    }

349    /**
350     * Decodes an octet according to RFC 2396. According to this spec,
351     * any characters outside the range 0x20 - 0x7E must be escaped because
352     * they are not printable characters, except for any characters in the
353     * fragment identifier. This method will translate any escaped characters
354     * back to the original.
355     *
356     * @param s        the URI to decode
357     * @param encoding the encoding to decode into
358     * @return The decoded URI
359     */
360    public static String decode(String s, Charset encoding) {
361        if (s == null || s.isEmpty()) {
362            return null;
363        }
364        StringBuilder sb = new StringBuilder();
365        boolean fragment = false;
366        for (int i = 0; i < s.length(); i++) {
367            char ch = s.charAt(i);
368            switch (ch) {
369                case '+':
370                    sb.append(' ');
371                    break;
372                case '#':
373                    sb.append(ch);
374                    fragment = true;
375                    break;
376                case '%':
377                    if (!fragment) {
378                        // fast hex decode
379                        sb.append((char) ((Character.digit(s.charAt(++i), 16) << 4)
380                                | Character.digit(s.charAt(++i), 16)));
381                    } else {
382                        sb.append(ch);
383                    }
384                    break;
385                default:
386                    sb.append(ch);
387                    break;
388            }
389        }
390        return new String(sb.toString().getBytes(LATIN1), encoding);
391    }

393    /**
394     * <p>Escape a string into URI syntax</p>
395     * <p>This function applies the URI escaping rules defined in
396     * section 2 of [RFC 2396], as amended by [RFC 2732], to the string
397     * supplied as the first argument, which typically represents all or part
398     * of a URI, URI reference or IRI. The effect of the function is to
399     * replace any special character in the string by an escape sequence of
400     * the form %xx%yy..., where xxyy... is the hexadecimal representation of
401     * the octets used to represent the character in US-ASCII for characters
402     * in the ASCII repertoire, and a different character encoding for
403     * non-ASCII characters.</p>
404     * <p>If the second argument is true, all characters are escaped
405     * other than lower case letters a-z, upper case letters A-Z, digits 0-9,
406     * and the characters referred to in [RFC 2396] as "marks": specifically,
407     * "-" | "_" | "." | "!" | "~" | "" | "'" | "(" | ")". The "%" character
408     * itself is escaped only if it is not followed by two hexadecimal digits
409     * (that is, 0-9, a-f, and A-F).</p>
410     * <p>[RFC 2396] does not define whether escaped URIs should use
411     * lower case or upper case for hexadecimal digits. To ensure that escaped
412     * URIs can be compared using string comparison functions, this function
413     * must always use the upper-case letters A-F.</p>
414     * <p>The character encoding used as the basis for determining the
415     * octets depends on the setting of the second argument.</p>
416     *
417     * @param s        the String to convert
418     * @param encoding The encoding to use for unsafe characters
419     * @return The converted String
420     */
421    public static String encode(String s, Charset encoding) {
422        if (s == null || s.isEmpty()) {
423            return null;
424        }
425        int length = s.length();
426        int start = 0;
427        int i = 0;
428        StringBuilder result = new StringBuilder(length);
429        while (true) {
430            while ((i < length) && isSafe(s.charAt(i))) {
431                i++;
432            }
433            // Safe character can just be added
434            result.append(s.substring(start, i));
435            // Are we done?
436            if (i >= length) {
437                return result.toString();
438            } else if (s.charAt(i) == ' ') {
439                result.append('+'); // Replace space char with plus symbol.
440                i++;
441            } else {
442                // Get all unsafe characters
443                start = i;
444                char c;
445                while ((i < length) && ((c = s.charAt(i)) != ' ') && !isSafe(c)) {
446                    i++;
447                }
448                // Convert them to %XY encoded strings
449                String unsafe = s.substring(start, i);
450                byte[] bytes = unsafe.getBytes(encoding);
451                for (byte aByte : bytes) {
452                    result.append('%');
453                    result.append(hex.charAt(((int) aByte & 0xf0) >> 4));
454                    result.append(hex.charAt((int) aByte & 0x0f));
455                }
456            }
457            start = i;
458        }
459    }

461    /**
462     * Returns true if the given char is
463     * either a uppercase or lowercase letter from 'a' till 'z', or a digit
464     * froim '0' till '9', or one of the characters '-', '_', '.' or ''. Such
465     * 'safe' character don't have to be url encoded.
466     *
467     * @param c the character
468     * @return true or false
469     */
470    private static boolean isSafe(char c) {
471        return (((c >= 'a') && (c <= 'z')) || ((c >= 'A') && (c <= 'Z'))
472                || ((c >= '0') && (c <= '9')) || (c == '-') || (c == '_') || (c == '.') || (c == '*'));
473    }
