002    package org.xbib.io;
003    
004    import java.util.Map;
005    import java.util.ServiceLoader;
006    import java.util.Set;
007    import java.util.WeakHashMap;
008    
009    public class StreamCodecService {
010    
011        private final static Map<String, StreamCodec> codecs = new WeakHashMap<String, StreamCodec>();
012    
013        private final static StreamCodecService instance = new StreamCodecService();
014    
015        private StreamCodecService() {
016            ServiceLoader<StreamCodec> loader = ServiceLoader.load(StreamCodec.class);
017            for (StreamCodec codec : loader) {
018                if (!codecs.containsKey(codec.getName())) {
019                    codecs.put(codec.getName(), codec);
020                }
021            }
022        }
023    
024        public static StreamCodecService getInstance() {
025            return instance;
026        }
027    
028        public StreamCodec getCodec(String suffix) {
029            if (codecs.containsKey(suffix)) {
030                return codecs.get(suffix);
031            }
032            throw new IllegalArgumentException("Stream codec for " + suffix + " not found in " + codecs);
033        }
034    
035        public static Set<String> getCodecs() {
036            return codecs.keySet();
037        }
038    }
