001package org.xbib.io.compress;
002
003import java.io.IOException;
004
005/**
006 * Interface used by {@link Uncompressor} implementations: receives
007 * uncompressed data and processes it appropriately.
008 *
009 * @since 0.9.4
010 */
011public interface DataHandler
012{
013    /**
014     * Method called with uncompressed data as it becomes available.
015     */
016    public void handleData(byte[] buffer, int offset, int len) throws IOException;
017
018    /**
019     * Method called after last call to {@link #handleData}, for successful
020     * operation, if and when caller is informed about end of content
021     * Note that if an exception thrown by {@link #handleData} has caused processing
022     * to be aborted, this method might not get called.
023     * Implementation may choose to free resources, flush state, or perform
024     * validation at this point.
025     */
026    public void allDataHandled() throws IOException;
027}
