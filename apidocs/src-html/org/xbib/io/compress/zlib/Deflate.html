349    // Restore the heap property by moving down the tree starting at node k,
350    // exchanging a node with the smallest of its two sons if necessary, stopping
351    // when the heap property is re-established (each father smaller than its
352    // two sons).
353    protected void pqdownheap(short[] tree, // the tree to restore
354            int kk // node to move down
355            ) {
356        int k = kk;
357        int v = heap[k];
358        int j = k << 1;  // left son of k
359        while (j <= heapLen) {
360            // Set j to the smallest of the two sons:
361            if (j < heapLen
362                    && smaller(tree, heap[j + 1], heap[j], depth)) {
363                j++;
364            }
365            // Exit if v is smaller than both sons
366            if (smaller(tree, v, heap[j], depth)) {
367                break;
368            }
369
370            // Exchange v with the smallest son
371            heap[k] = heap[j];
372            k = j;
373            // And continue down the tree, setting j to the left son of k
374            j <<= 1;
375        }
376        heap[k] = v;
377    }

379    private static boolean smaller(short[] tree, int n, int m, byte[] depth) {
380        short tn2 = tree[n * 2];
381        short tm2 = tree[m * 2];
382        return (tn2 < tm2
383                || (tn2 == tm2 && depth[n] <= depth[m]));
384    }

386    // Scan a literal or distance tree to determine the frequencies of the codes
387    // in the bit length tree.
388    private void scanTree(short[] tree,// the tree to be scanned
389            int maxcode // and its largest code of non zero frequency
390            ) {
391        int n;                     // iterates over all tree elements
392        int prevlen = -1;          // last emitted length
393        int curlen;                // length of current code
394        int nextlen = tree[0 * 2 + 1]; // length of next code
395        int count = 0;             // repeat count of the current code
396        int maxcount = 7;         // max repeat count
397        int mincount = 4;         // min repeat count
398
399        if (nextlen == 0) {
400            maxcount = 138;
401            mincount = 3;
402        }
403        tree[(maxcode + 1) * 2 + 1] = (short) 0xffff; // guard
404
405        for (n = 0; n <= maxcode; n++) {
406            curlen = nextlen;
407            nextlen = tree[(n + 1) * 2 + 1];
408            if (++count < maxcount && curlen == nextlen) {
409                continue;
410            } else if (count < mincount) {
411                blTree[curlen * 2] += count;
412            } else if (curlen != 0) {
413                if (curlen != prevlen) {
414                    blTree[curlen * 2]++;
415                }
416                blTree[REP_3_6 * 2]++;
417            } else if (count <= 10) {
418                blTree[REPZ_3_10 * 2]++;
419            } else {
420                blTree[REPZ_11_138 * 2]++;
421            }
422            count = 0;
423            prevlen = curlen;
424            if (nextlen == 0) {
425                maxcount = 138;
426                mincount = 3;
427            } else if (curlen == nextlen) {
428                maxcount = 6;
429                mincount = 3;
430            } else {
431                maxcount = 7;
432                mincount = 4;
433            }
434        }
435    }

437    // Construct the Huffman tree for the bit lengths and return the index in
438    // BL_ORDER of the last bit length code to send.
439    private int buildBLTree() {
440        int maxblindex;  // index of last bit length code of non zero freq
441
442        // Determine the bit length frequencies for literal and distance trees
443        scanTree(dynLtree, lDesc.maxCode);
444        scanTree(dynDtree, dDesc.maxCode);
445
446        // Build the bit length tree:
447        blDesc.buildTree(this);
448        // opt_len now includes the length of the tree representations, except
449        // the lengths of the bit lengths codes and the 5+5+4 bits for the counts.
450
451        // Determine the number of bit length codes to send. The pkzip format
452        // requires that at least 4 bit length codes be sent. (appnote.txt says
453        // 3 but the actual value used is 4.)
454        for (maxblindex = BL_CODES - 1; maxblindex >= 3; maxblindex--) {
455            if (blTree[Tree.BL_ORDER[maxblindex] * 2 + 1] != 0) {
456                break;
457            }
458        }
459        // Update opt_len to include the bit length tree and counts
460        optLen += 3 * (maxblindex + 1) + 5 + 5 + 4;
461
462        return maxblindex;
463    }

481    // Send a literal or distance tree in compressed form, using the codes in
482    // bl_tree.
483    private void sendTree(short[] tree,// the tree to be sent
484            int max_code // and its largest code of non zero frequency
485            ) {
486        int n;                     // iterates over all tree elements
487        int prevlen = -1;          // last emitted length
488        int curlen;                // length of current code
489        int nextlen = tree[0 * 2 + 1]; // length of next code
490        int count = 0;             // repeat count of the current code
491        int maxcount = 7;         // max repeat count
492        int mincount = 4;         // min repeat count
493
494        if (nextlen == 0) {
495            maxcount = 138;
496            mincount = 3;
497        }
498
499        for (n = 0; n <= max_code; n++) {
500            curlen = nextlen;
501            nextlen = tree[(n + 1) * 2 + 1];
502            if (++count < maxcount && curlen == nextlen) {
503                continue;
504            } else if (count < mincount) {
505                do {
506                    sendCode(curlen, blTree);
507                } while (--count != 0);
508            } else if (curlen != 0) {
509                if (curlen != prevlen) {
510                    sendCode(curlen, blTree);
511                    count--;
512                }
513                sendCode(REP_3_6, blTree);
514                sendBits(count - 3, 2);
515            } else if (count <= 10) {
516                sendCode(REPZ_3_10, blTree);
517                sendBits(count - 3, 3);
518            } else {
519                sendCode(REPZ_11_138, blTree);
520                sendBits(count - 11, 7);
521            }
522            count = 0;
523            prevlen = curlen;
524            if (nextlen == 0) {
525                maxcount = 138;
526                mincount = 3;
527            } else if (curlen == nextlen) {
528                maxcount = 6;
529                mincount = 3;
530            } else {
531                maxcount = 7;
532                mincount = 4;
533            }
534        }
535    }

563    private void sendBits(int value, int length) {
564        int len = length;
565        if (biValid > BUF_SIZE - len) {
566            int val = value;
567//      bi_buf |= (val << bi_valid);
568            biBuf |= ((val << biValid) & 0xffff);
569            putShort(biBuf);
570            biBuf = (short) (val >>> (BUF_SIZE - biValid));
571            biValid += len - BUF_SIZE;
572        } else {
573//      bi_buf |= (value) << bi_valid;
574            biBuf |= (((value) << biValid) & 0xffff);
575            biValid += len;
576        }
577    }

579    // Send one empty static block to give enough lookahead for inflate.
580    // This takes 10 bits, of which 7 may remain in the bit buffer.
581    // The current inflate code requires 9 bits of lookahead. If the
582    // last two codes for the previous block (real code plus EOB) were coded
583    // on 5 bits or less, inflate may have only 5+3 bits of lookahead to decode
584    // the last real code. In this case we send two empty static blocks instead
585    // of one. (There are no problems if the previous block is stored or fixed.)
586    // To simplify the code, we assume the worst case of last real code encoded
587    // on one bit only.
588    private void trAlign() {
589        sendBits(STATIC_TREES << 1, 3);
590        sendCode(END_BLOCK, StaticTree.STATIC_LTREE);
591
592        biFlush();
593
594        // Of the 10 bits for the empty block, we have already sent
595        // (10 - bi_valid) bits. The lookahead for the last real code (before
596        // the EOB of the previous block) was thus at least one plus the length
597        // of the EOB plus what we have just sent of the empty static block.
598        if (1 + lastEobLen + 10 - biValid < 9) {
599            sendBits(STATIC_TREES << 1, 3);
600            sendCode(END_BLOCK, StaticTree.STATIC_LTREE);
601            biFlush();
602        }
603        lastEobLen = 7;
604    }

606    // Save the match info and tally the frequency counts. Return true if
607    // the current block must be flushed.
608    private boolean trTally(int dist, // distance of matched string
609            int lc // match length-MIN_MATCH or unmatched char (if dist==0)
610            ) {
611
612        pendingBuf[dBuf + lastLit * 2] = (byte) (dist >>> 8);
613        pendingBuf[dBuf + lastLit * 2 + 1] = (byte) dist;
614
615        pendingBuf[lBuf + lastLit] = (byte) lc;
616        lastLit++;
617
618        if (dist == 0) {
619            // lc is the unmatched char
620            dynLtree[lc * 2]++;
621        } else {
622            matches++;
623            // Here, lc is the match length - MIN_MATCH
624            dist--;             // dist = match distance - 1
625            dynLtree[(Tree.LENGTH_CODE[lc] + LITERALS + 1) * 2]++;
626            dynDtree[Tree.distanceCode(dist) * 2]++;
627        }
628
629        if ((lastLit & 0x1fff) == 0 && level > 2) {
630            // Compute an upper bound for the compressed length
631            int out_length = lastLit * 8;
632            int in_length = strstart - blockStart;
633            int dcode;
634            for (dcode = 0; dcode < D_CODES; dcode++) {
635                out_length += (int) dynDtree[dcode * 2]
636                        * (5L + Tree.EXTRA_DBITS[dcode]);
637            }
638            out_length >>>= 3;
639            if ((matches < (lastLit / 2)) && out_length < in_length / 2) {
640                return true;
641            }
642        }
643
644        return (lastLit == litBufsize - 1);
645        // We avoid equality with lit_bufsize because of wraparound at 64K
646        // on 16 bit machines and because stored blocks are restricted to
647        // 64K-1 bytes.
648    }

650    // Send the block data compressed using the given Huffman trees
651    private void compressBlock(short[] ltree, short[] dtree) {
652        int dist;      // distance of matched string
653        int lc;         // match length or unmatched char (if dist == 0)
654        int lx = 0;     // running index in l_buf
655        int code;       // the code to send
656        int extra;      // number of extra bits to send
657
658        if (lastLit != 0) {
659            do {
660                dist = ((pendingBuf[dBuf + lx * 2] << 8) & 0xff00)
661                        | (pendingBuf[dBuf + lx * 2 + 1] & 0xff);
662                lc = (pendingBuf[lBuf + lx]) & 0xff;
663                lx++;
664
665                if (dist == 0) {
666                    sendCode(lc, ltree); // send a literal byte
667                } else {
668                    // Here, lc is the match length - MIN_MATCH
669                    code = Tree.LENGTH_CODE[lc];
670
671                    sendCode(code + LITERALS + 1, ltree); // send the length code
672                    extra = Tree.EXTRA_LBITS[code];
673                    if (extra != 0) {
674                        lc -= Tree.BASE_LENGTH[code];
675                        sendBits(lc, extra);       // send the extra length bits
676                    }
677                    dist--; // dist is now the match distance - 1
678                    code = Tree.distanceCode(dist);
679
680                    sendCode(code, dtree);       // send the distance code
681                    extra = Tree.EXTRA_DBITS[code];
682                    if (extra != 0) {
683                        dist -= Tree.BASE_DIST[code];
684                        sendBits(dist, extra);   // send the extra distance bits
685                    }
686                } // literal or match pair ?
687
688                // Check that the overlay between pending_buf and d_buf+l_buf is ok:
689            } while (lx < lastLit);
690        }
691
692        sendCode(END_BLOCK, ltree);
693        lastEobLen = ltree[END_BLOCK * 2 + 1];
694    }

696    // Set the data type to ASCII or BINARY, using a crude approximation:
697    // binary if more than 20% of the bytes are <= 6 or >= 128, ascii otherwise.
698    // IN assertion: the fields freq of dyn_ltree are set and the total of all
699    // frequencies does not exceed 64K (to fit in an int on 16 bit machines).
700    private void setDataType() {
701        int n = 0;
702        int asciifreq = 0;
703        int binfreq = 0;
704        while (n < 7) {
705            binfreq += dynLtree[n * 2];
706            n++;
707        }
708        while (n < 128) {
709            asciifreq += dynLtree[n * 2];
710            n++;
711        }
712        while (n < LITERALS) {
713            binfreq += dynLtree[n * 2];
714            n++;
715        }
716        dataType = (byte) (binfreq > (asciifreq >>> 2) ? Z_BINARY : Z_ASCII);
717    }

719    // Flush the bit buffer, keeping at most 7 bits in it.
720    private void biFlush() {
721        if (biValid == 16) {
722            putShort(biBuf);
723            biBuf = 0;
724            biValid = 0;
725        } else if (biValid >= 8) {
726            putByte((byte) biBuf);
727            biBuf >>>= 8;
728            biValid -= 8;
729        }
730    }

732    // Flush the bit buffer and align the output on a byte boundary
733    private void biWindup() {
734        if (biValid > 8) {
735            putShort(biBuf);
736        } else if (biValid > 0) {
737            putByte((byte) biBuf);
738        }
739        biBuf = 0;
740        biValid = 0;
741    }

767    // Copy without compression as much as possible from the input stream, return
768    // the current block state.
769    // This function does not insert new strings in the dictionary since
770    // uncompressible data is probably not useful. This function is used
771    // only for the level=0 compression option.
772    // NOTE: this function should be optimized to avoid extra copying from
773    // window to pending_buf.
774    private int deflateStored(int flush) {
775        // Stored blocks are limited to 0xffff bytes, pending_buf is limited
776        // to pending_buf_size, and each stored block has a 5 byte header:
777
778        int maxblocksize = 0xffff;
779        int maxstart;
780
781        if (maxblocksize > pendingBufSize - 5) {
782            maxblocksize = pendingBufSize - 5;
783        }
784
785        // Copy as much as possible from input to output:
786        while (true) {
787            // Fill the window as much as possible:
788            if (lookahead <= 1) {
789                fillWindow();
790                if (lookahead == 0 && flush == Z_NO_FLUSH) {
791                    return NEED_MORE;
792                }
793                if (lookahead == 0) {
794                    break; // flush the current block
795                }
796            }
797
798            strstart += lookahead;
799            lookahead = 0;
800
801            // Emit a stored block if pending_buf will be full:
802            maxstart = blockStart + maxblocksize;
803            if (strstart == 0 || strstart >= maxstart) {
804                // strstart == 0 is possible when wraparound on 16-bit machine
805                lookahead = (strstart - maxstart);
806                strstart = maxstart;
807
808                flushBlockOnly(false);
809                if (strm.availout == 0) {
810                    return NEED_MORE;
811                }
812
813            }
814
815            // Flush if we may have to slide, otherwise block_start may become
816            // negative and the data will be gone:
817            if (strstart - blockStart >= wSize - MIN_LOOKAHEAD) {
818                flushBlockOnly(false);
819                if (strm.availout == 0) {
820                    return NEED_MORE;
821                }
822            }
823        }
824
825        flushBlockOnly(flush == Z_FINISH);
826        if (strm.availout == 0) {
827            return (flush == Z_FINISH) ? FINISH_STARTED : NEED_MORE;
828        }
829
830        return flush == Z_FINISH ? FINISH_DONE : BLOCK_DONE;
831    }

842    // Determine the best encoding for the current block: dynamic trees, static
843    // trees or store, and output the encoded block to the zip file.
844    private void trFlushBlock(int buf, // input block, or NULL if too old
845            int storedlen, // length of input block
846            boolean eof // true if this is the last block for a file
847            ) {
848        int optlenb, staticlenb;// opt_len and static_len in bytes
849        int maxblindex = 0;      // index of last bit length code of non zero freq
850
851        // Build the Huffman trees unless a stored block is forced
852        if (level > 0) {
853            // Check if the file is ascii or binary
854            if (dataType == Z_UNKNOWN) {
855                setDataType();
856            }
857
858            // Construct the literal and distance trees
859            lDesc.buildTree(this);
860
861            dDesc.buildTree(this);
862
863            // At this point, opt_len and static_len are the total bit lengths of
864            // the compressed block data, excluding the tree representations.
865
866            // Build the bit length tree for the above two trees, and get the index
867            // in BL_ORDER of the last bit length code to send.
868            maxblindex = buildBLTree();
869
870            // Determine the best encoding. Compute first the block length in bytes
871            optlenb = (optLen + 3 + 7) >>> 3;
872            staticlenb = (staticLen + 3 + 7) >>> 3;
873
874            if (staticlenb <= optlenb) {
875                optlenb = staticlenb;
876            }
877        } else {
878            optlenb = staticlenb = storedlen + 5; // force a stored block
879        }
880
881        if (storedlen + 4 <= optlenb && buf != -1) {
882            // 4: two words for the lengths
883            // The test buf != NULL is only necessary if LIT_BUFSIZE > WSIZE.
884            // Otherwise we can't have processed more than WSIZE input bytes since
885            // the last block flush, because compression would have been
886            // successful. If LIT_BUFSIZE <= WSIZE, it is never too late to
887            // transform a block into a stored block.
888            trStoredBlock(buf, storedlen, eof);
889        } else if (staticlenb == optlenb) {
890            sendBits((STATIC_TREES << 1) + (eof ? 1 : 0), 3);
891            compressBlock(StaticTree.STATIC_LTREE, StaticTree.STATIC_DTREE);
892        } else {
893            sendBits((DYN_TREES << 1) + (eof ? 1 : 0), 3);
894            sendAllTree(lDesc.maxCode + 1, dDesc.maxCode + 1, maxblindex + 1);
895            compressBlock(dynLtree, dynDtree);
896        }
897
898        // The above check is made mod 2^32, for files larger than 512 MB
899        // and uLong implemented on 32 bits.
900
901        initBlock();
902
903        if (eof) {
904            biWindup();
905        }
906    }

908    // Fill the window when the lookahead becomes insufficient.
909    // Updates strstart and lookahead.
910    //
911    // IN assertion: lookahead < MIN_LOOKAHEAD
912    // OUT assertions: strstart <= window_size-MIN_LOOKAHEAD
913    //    At least one byte has been read, or avail_in == 0; reads are
914    //    performed for at least two bytes (required for the zip translate_eol
915    //    option -- not supported here).
916    private void fillWindow() {
917        int n, m;
918        int p;
919        int more;    // Amount of free space at the end of the window.
920
921        do {
922            more = (windowSize - lookahead - strstart);
923
924            // Deal with !@#$% 64K limit:
925            if (more == 0 && strstart == 0 && lookahead == 0) {
926                more = wSize;
927            } else if (more == -1) {
928                // Very unlikely, but possible on 16 bit machine if strstart == 0
929                // and lookahead == 1 (input done one byte at time)
930                more--;
931
932                // If the window is almost full and there is insufficient lookahead,
933                // move the upper half to the lower one to make room in the upper half.
934            } else if (strstart >= wSize + wSize - MIN_LOOKAHEAD) {
935                System.arraycopy(window, wSize, window, 0, wSize);
936                matchStart -= wSize;
937                strstart -= wSize; // we now have strstart >= MAX_DIST
938                blockStart -= wSize;
939
940                // Slide the hash table (could be avoided with 32 bit values
941                // at the expense of memory usage). We slide even when level == 0
942                // to keep the hash table consistent if we switch back to level > 0
943                // later. (Using level 0 permanently is not an optimal usage of
944                // zlib, so we don't care about this pathological case.)
945
946                n = hashSize;
947                p = n;
948                do {
949                    m = (head[--p] & 0xffff);
950                    head[p] = (m >= wSize ? (short) (m - wSize) : 0);
951                } while (--n != 0);
952
953                n = wSize;
954                p = n;
955                do {
956                    m = (prev[--p] & 0xffff);
957                    prev[p] = (m >= wSize ? (short) (m - wSize) : 0);
958                    // If n is not on any hash chain, prev[n] is garbage but
959                    // its value will never be used.
960                } while (--n != 0);
961                more += wSize;
962            }
963
964            if (strm.availin == 0) {
965                return;
966            }
967
968            // If there was no sliding:
969            //    strstart <= WSIZE+MAX_DIST-1 && lookahead <= MIN_LOOKAHEAD - 1 &&
970            //    more == window_size - lookahead - strstart
971            // => more >= window_size - (MIN_LOOKAHEAD-1 + WSIZE + MAX_DIST-1)
972            // => more >= window_size - 2*WSIZE + 2
973            // In the BIG_MEM or MMAP case (not yet supported),
974            //   window_size == input_size + MIN_LOOKAHEAD  &&
975            //   strstart + s->lookahead <= input_size => more >= MIN_LOOKAHEAD.
976            // Otherwise, window_size == 2*WSIZE so more >= 2.
977            // If there was sliding, more >= WSIZE. So in all cases, more >= 2.
978
979            n = strm.readBuf(window, strstart + lookahead, more);
980            lookahead += n;
981
982            // Initialize the hash value now that we have some input:
983            if (lookahead >= MIN_MATCH) {
984                insh = window[strstart] & 0xff;
985                insh = (((insh) << hashShift) ^ (window[strstart + 1] & 0xff)) & hashMask;
986            }
987            // If the whole input has less than MIN_MATCH bytes, ins_h is garbage,
988            // but this is not important since only literal bytes will be emitted.
989        } while (lookahead < MIN_LOOKAHEAD && strm.availin != 0);
990    }

992    // Compress as much as possible from the input stream, return the current
993    // block state.
994    // This function does not perform lazy evaluation of matches and inserts
995    // new strings in the dictionary only for unmatched strings or for short
996    // matches. It is used only for the fast compression options.
997    private int deflateFast(int flush) {
998//    short hash_head = 0; // head of the hash chain
999        int hashhead = 0; // head of the hash chain
1000        boolean bflush;      // set if current block must be flushed
1001
1002        while (true) {
1003            // Make sure that we always have enough lookahead, except
1004            // at the end of the input file. We need MAX_MATCH bytes
1005            // for the next match, plus MIN_MATCH bytes to insert the
1006            // string following the next match.
1007            if (lookahead < MIN_LOOKAHEAD) {
1008                fillWindow();
1009                if (lookahead < MIN_LOOKAHEAD && flush == Z_NO_FLUSH) {
1010                    return NEED_MORE;
1011                }
1012                if (lookahead == 0) {
1013                    break; // flush the current block
1014                }
1015            }
1016
1017            // Insert the string window[strstart .. strstart+2] in the
1018            // dictionary, and set hash_head to the head of the hash chain:
1019            if (lookahead >= MIN_MATCH) {
1020                insh = (((insh) << hashShift) ^ (window[(strstart) + (MIN_MATCH - 1)] & 0xff)) & hashMask;
1021
1022//      prev[strstart&w_mask]=hash_head=head[ins_h];
1023                hashhead = (head[insh] & 0xffff);
1024                prev[strstart & wMask] = head[insh];
1025                head[insh] = (short) strstart;
1026            }
1027
1028            // Find the longest match, discarding those <= prev_length.
1029            // At this point we have always match_length < MIN_MATCH
1030
1031            // To simplify the code, we prevent matches with the string
1032            // of window index 0 (in particular we have to avoid a match
1033            // of the string with itself at the start of the input file).
1034            if (hashhead != 0L
1035                    && ((strstart - hashhead) & 0xffff) <= wSize - MIN_LOOKAHEAD
1036                    && (strategy != Z_HUFFMAN_ONLY)) {
1037                matchLength = longestMatch(hashhead);
1038            }
1039            // longest_match() sets match_start
1040
1041            if (matchLength >= MIN_MATCH) {
1042                //        check_match(strstart, match_start, match_length);
1043
1044                bflush = trTally(strstart - matchStart, matchLength - MIN_MATCH);
1045
1046                lookahead -= matchLength;
1047
1048                // Insert new strings in the hash table only if the match length
1049                // is not too large. This saves time but degrades compression.
1050                if (matchLength <= maxLazyMatch
1051                        && lookahead >= MIN_MATCH) {
1052                    matchLength--; // string at strstart already in hash table
1053                    do {
1054                        strstart++;
1055
1056                        insh = ((insh << hashShift) ^ (window[(strstart) + (MIN_MATCH - 1)] & 0xff)) & hashMask;
1057//          prev[strstart&w_mask]=hash_head=head[ins_h];
1058                        hashhead = (head[insh] & 0xffff);
1059                        prev[strstart & wMask] = head[insh];
1060                        head[insh] = (short) strstart;
1061
1062                        // strstart never exceeds WSIZE-MAX_MATCH, so there are
1063                        // always MIN_MATCH bytes ahead.
1064                    } while (--matchLength != 0);
1065                    strstart++;
1066                } else {
1067                    strstart += matchLength;
1068                    matchLength = 0;
1069                    insh = window[strstart] & 0xff;
1070
1071                    insh = (((insh) << hashShift) ^ (window[strstart + 1] & 0xff)) & hashMask;
1072                    // If lookahead < MIN_MATCH, ins_h is garbage, but it does not
1073                    // matter since it will be recomputed at next deflate call.
1074                }
1075            } else {
1076                // No match, output a literal byte
1077
1078                bflush = trTally(0, window[strstart] & 0xff);
1079                lookahead--;
1080                strstart++;
1081            }
1082            if (bflush) {
1083
1084                flushBlockOnly(false);
1085                if (strm.availout == 0) {
1086                    return NEED_MORE;
1087                }
1088            }
1089        }
1090
1091        flushBlockOnly(flush == Z_FINISH);
1092        if (strm.availout == 0) {
1093            if (flush == Z_FINISH) {
1094                return FINISH_STARTED;
1095            } else {
1096                return NEED_MORE;
1097            }
1098        }
1099        return flush == Z_FINISH ? FINISH_DONE : BLOCK_DONE;
1100    }

1102    // Same as above, but achieves better compression. We use a lazy
1103    // evaluation for matches: a match is finally adopted only if there is
1104    // no better match at the next window position.
1105    private int deflateSlow(int flush) {
1106//    short hash_head = 0;    // head of hash chain
1107        int hashhead = 0;    // head of hash chain
1108        boolean bflush;         // set if current block must be flushed
1109
1110        // Process the input block.
1111        while (true) {
1112            // Make sure that we always have enough lookahead, except
1113            // at the end of the input file. We need MAX_MATCH bytes
1114            // for the next match, plus MIN_MATCH bytes to insert the
1115            // string following the next match.
1116
1117            if (lookahead < MIN_LOOKAHEAD) {
1118                fillWindow();
1119                if (lookahead < MIN_LOOKAHEAD && flush == Z_NO_FLUSH) {
1120                    return NEED_MORE;
1121                }
1122                if (lookahead == 0) {
1123                    break; // flush the current block
1124                }
1125            }
1126
1127            // Insert the string window[strstart .. strstart+2] in the
1128            // dictionary, and set hash_head to the head of the hash chain:
1129
1130            if (lookahead >= MIN_MATCH) {
1131                insh = (((insh) << hashShift) ^ (window[(strstart) + (MIN_MATCH - 1)] & 0xff)) & hashMask;
1132//      prev[strstart&w_mask]=hash_head=head[ins_h];
1133                hashhead = (head[insh] & 0xffff);
1134                prev[strstart & wMask] = head[insh];
1135                head[insh] = (short) strstart;
1136            }
1137
1138            // Find the longest match, discarding those <= prev_length.
1139            prevLength = matchLength;
1140            prevMatch = matchStart;
1141            matchLength = MIN_MATCH - 1;
1142
1143            if (hashhead != 0 && prevLength < maxLazyMatch
1144                    && ((strstart - hashhead) & 0xffff) <= wSize - MIN_LOOKAHEAD) {
1145                // To simplify the code, we prevent matches with the string
1146                // of window index 0 (in particular we have to avoid a match
1147                // of the string with itself at the start of the input file).
1148
1149                if (strategy != Z_HUFFMAN_ONLY) {
1150                    matchLength = longestMatch(hashhead);
1151                }
1152                // longest_match() sets match_start
1153
1154                if (matchLength <= 5 && (strategy == Z_FILTERED
1155                        || (matchLength == MIN_MATCH
1156                        && strstart - matchStart > 4096))) {
1157
1158                    // If prev_match is also MIN_MATCH, match_start is garbage
1159                    // but we will ignore the current match anyway.
1160                    matchLength = MIN_MATCH - 1;
1161                }
1162            }
1163
1164            // If there was a match at the previous step and the current
1165            // match is not better, output the previous match:
1166            if (prevLength >= MIN_MATCH && matchLength <= prevLength) {
1167                int max_insert = strstart + lookahead - MIN_MATCH;
1168                // Do not insert strings in hash table beyond this.
1169
1170                //          check_match(strstart-1, prev_match, prev_length);
1171
1172                bflush = trTally(strstart - 1 - prevMatch, prevLength - MIN_MATCH);
1173
1174                // Insert in hash table all strings up to the end of the match.
1175                // strstart-1 and strstart are already inserted. If there is not
1176                // enough lookahead, the last two strings are not inserted in
1177                // the hash table.
1178                lookahead -= prevLength - 1;
1179                prevLength -= 2;
1180                do {
1181                    if (++strstart <= max_insert) {
1182                        insh = (((insh) << hashShift) ^ (window[(strstart) + (MIN_MATCH - 1)] & 0xff)) & hashMask;
1183                        //prev[strstart&w_mask]=hash_head=head[ins_h];
1184                        hashhead = (head[insh] & 0xffff);
1185                        prev[strstart & wMask] = head[insh];
1186                        head[insh] = (short) strstart;
1187                    }
1188                } while (--prevLength != 0);
1189                matchAvailable = 0;
1190                matchLength = MIN_MATCH - 1;
1191                strstart++;
1192
1193                if (bflush) {
1194                    flushBlockOnly(false);
1195                    if (strm.availout == 0) {
1196                        return NEED_MORE;
1197                    }
1198                }
1199            } else if (matchAvailable != 0) {
1200
1201                // If there was no match at the previous position, output a
1202                // single literal. If there was a match but the current match
1203                // is longer, truncate the previous match to a single literal.
1204
1205                bflush = trTally(0, window[strstart - 1] & 0xff);
1206
1207                if (bflush) {
1208                    flushBlockOnly(false);
1209                }
1210                strstart++;
1211                lookahead--;
1212                if (strm.availout == 0) {
1213                    return NEED_MORE;
1214                }
1215            } else {
1216                // There is no previous match to compare with, wait for
1217                // the next step to decide.
1218
1219                matchAvailable = 1;
1220                strstart++;
1221                lookahead--;
1222            }
1223        }
1224
1225        if (matchAvailable != 0) {
1226            bflush = trTally(0, window[strstart - 1] & 0xff);
1227            matchAvailable = 0;
1228        }
1229        flushBlockOnly(flush == Z_FINISH);
1230
1231        if (strm.availout == 0) {
1232            if (flush == Z_FINISH) {
1233                return FINISH_STARTED;
1234            } else {
1235                return NEED_MORE;
1236            }
1237        }
1238
1239        return flush == Z_FINISH ? FINISH_DONE : BLOCK_DONE;
1240    }

1242    private int longestMatch(int cmatch) {
1243        int curmatch = cmatch;
1244        int chainlength = maxChainLength; // max hash chain length
1245        int scan = strstart;                 // current string
1246        int match;                           // matched string
1247        int len;                             // length of current match
1248        int bestlen = prevLength;          // best match length so far
1249        int limit = strstart > (wSize - MIN_LOOKAHEAD) ? strstart - (wSize - MIN_LOOKAHEAD) : 0;
1250        int nmatch = this.niceMatch;
1251
1252        // Stop when cur_match becomes <= limit. To simplify the code,
1253        // we prevent matches with the string of window index 0.
1254
1255        int wmask = wMask;
1256
1257        int strend = strstart + MAX_MATCH;
1258        byte scanend1 = window[scan + bestlen - 1];
1259        byte scanend = window[scan + bestlen];
1260
1261        // The code is optimized for HASH_BITS >= 8 and MAX_MATCH-2 multiple of 16.
1262        // It is easy to get rid of this optimization if necessary.
1263
1264        // Do not waste too much time if we already have a good match:
1265        if (prevLength >= goodMatch) {
1266            chainlength >>= 2;
1267        }
1268
1269        // Do not look for matches beyond the end of the input. This is necessary
1270        // to make deflate deterministic.
1271        if (nmatch > lookahead) {
1272            nmatch = lookahead;
1273        }
1274
1275        do {
1276            match = curmatch;
1277
1278            // Skip to next match if the match length cannot increase
1279            // or if the match length is less than 2:
1280            if (window[match + bestlen] != scanend
1281                    || window[match + bestlen - 1] != scanend1
1282                    || window[match] != window[scan]
1283                    || window[++match] != window[scan + 1]) {
1284                continue;
1285            }
1286
1287            // The check at best_len-1 can be removed because it will be made
1288            // again later. (This heuristic is not always a win.)
1289            // It is not necessary to compare scan[2] and match[2] since they
1290            // are always equal when the other bytes match, given that
1291            // the hash keys are equal and that HASH_BITS >= 8.
1292            scan += 2;
1293            match++;
1294
1295            // We check for insufficient lookahead only every 8th comparison;
1296            // the 256th check will be made at strstart+258.
1297            do {
1298            } while (window[++scan] == window[++match]
1299                    && window[++scan] == window[++match]
1300                    && window[++scan] == window[++match]
1301                    && window[++scan] == window[++match]
1302                    && window[++scan] == window[++match]
1303                    && window[++scan] == window[++match]
1304                    && window[++scan] == window[++match]
1305                    && window[++scan] == window[++match]
1306                    && scan < strend);
1307
1308            len = MAX_MATCH - (strend - scan);
1309            scan = strend - MAX_MATCH;
1310
1311            if (len > bestlen) {
1312                matchStart = curmatch;
1313                bestlen = len;
1314                if (len >= nmatch) {
1315                    break;
1316                }
1317                scanend1 = window[scan + bestlen - 1];
1318                scanend = window[scan + bestlen];
1319            }
1320
1321        } while ((curmatch = (prev[curmatch & wmask] & 0xffff)) > limit && --chainlength != 0);
1322
1323        if (bestlen <= lookahead) {
1324            return bestlen;
1325        }
1326        return lookahead;
1327    }

1338    private int deflateInit2(ZStream strm, int level, int method, int windowBits,
1339            int memLevel, int strategy) {
1340        int nheader = 0;
1341
1342        strm.msg = null;
1343
1344        if (level == Z_DEFAULT_COMPRESSION) {
1345            level = 6;
1346        }
1347
1348        if (windowBits < 0) { // undocumented feature: suppress zlib header
1349            nheader = 1;
1350            windowBits = -windowBits;
1351        }
1352
1353        if (memLevel < 1 || memLevel > MAX_MEM_LEVEL
1354                || method != Z_DEFLATED
1355                || windowBits < 9 || windowBits > 15 || level < 0 || level > 9
1356                || strategy < 0 || strategy > Z_HUFFMAN_ONLY) {
1357            return Z_STREAM_ERROR;
1358        }
1359
1360        strm.dstate = this;
1361
1362        this.noheader = nheader;
1363        wBits = windowBits;
1364        wSize = 1 << wBits;
1365        wMask = wSize - 1;
1366
1367        hashBits = memLevel + 7;
1368        hashSize = 1 << hashBits;
1369        hashMask = hashSize - 1;
1370        hashShift = ((hashBits + MIN_MATCH - 1) / MIN_MATCH);
1371
1372        window = new byte[wSize * 2];
1373        prev = new short[wSize];
1374        head = new short[hashSize];
1375
1376        litBufsize = 1 << (memLevel + 6); // 16K elements by default
1377
1378        // We overlay pending_buf and d_buf+l_buf. This works since the average
1379        // output size for (length,distance) codes is <= 24 bits.
1380        pendingBuf = new byte[litBufsize * 4];
1381        pendingBufSize = litBufsize * 4;
1382
1383        dBuf = litBufsize / 2;
1384        lBuf = (1 + 2) * litBufsize;
1385
1386        this.level = level;
1387
1388        this.strategy = strategy;
1389        return deflateReset(strm);
1390    }

1455    public int deflateSetDictionary(ZStream strm, byte[] dictionary, int dictLength) {
1456        int length = dictLength;
1457        int index = 0;
1458
1459        if (dictionary == null || status != INIT_STATE) {
1460            return Z_STREAM_ERROR;
1461        }
1462
1463        strm.adler = Adler32.adler32(strm.adler, dictionary, 0, dictLength);
1464
1465        if (length < MIN_MATCH) {
1466            return Z_OK;
1467        }
1468        if (length > wSize - MIN_LOOKAHEAD) {
1469            length = wSize - MIN_LOOKAHEAD;
1470            index = dictLength - length; // use the tail of the dictionary
1471        }
1472        System.arraycopy(dictionary, index, window, 0, length);
1473        strstart = length;
1474        blockStart = length;
1475
1476        // Insert all strings in the hash table (except for the last two bytes).
1477        // s->lookahead stays null, so s->ins_h will be recomputed at the next
1478        // call of fill_window.
1479
1480        insh = window[0] & 0xff;
1481        insh = (((insh) << hashShift) ^ (window[1] & 0xff)) & hashMask;
1482
1483        for (int n = 0; n <= length - MIN_MATCH; n++) {
1484            insh = (((insh) << hashShift) ^ (window[(n) + (MIN_MATCH - 1)] & 0xff)) & hashMask;
1485            prev[n & wMask] = head[insh];
1486            head[insh] = (short) n;
1487        }
1488        return Z_OK;
1489    }

1491    public int deflate(ZStream strm, int flush) {
1492        int oldflush;
1493
1494        if (flush > Z_FINISH || flush < 0) {
1495            return Z_STREAM_ERROR;
1496        }
1497
1498        if (strm.nextout == null
1499                || (strm.nextin == null && strm.availin != 0)
1500                || (status == FINISH_STATE && flush != Z_FINISH)) {
1501            strm.msg = zerrmsg[Z_NEED_DICT - (Z_STREAM_ERROR)];
1502            return Z_STREAM_ERROR;
1503        }
1504        if (strm.availout == 0) {
1505            strm.msg = zerrmsg[Z_NEED_DICT - (Z_BUF_ERROR)];
1506            return Z_BUF_ERROR;
1507        }
1508
1509        this.strm = strm; // just in case
1510        oldflush = lastFlush;
1511        lastFlush = flush;
1512
1513        // Write the zlib header
1514        if (status == INIT_STATE) {
1515            int header = (Z_DEFLATED + ((wBits - 8) << 4)) << 8;
1516            int level_flags = ((level - 1) & 0xff) >> 1;
1517
1518            if (level_flags > 3) {
1519                level_flags = 3;
1520            }
1521            header |= (level_flags << 6);
1522            if (strstart != 0) {
1523                header |= PRESET_DICT;
1524            }
1525            header += 31 - (header % 31);
1526
1527            status = BUSY_STATE;
1528            putShortMSB(header);
1529
1530
1531            // Save the adler32 of the preset dictionary:
1532            if (strstart != 0) {
1533                putShortMSB((int) (strm.adler >>> 16));
1534                putShortMSB((int) (strm.adler & 0xffff));
1535            }
1536            strm.adler = Adler32.adler32(0, null, 0, 0);
1537        }
1538
1539        // Flush as much pending output as possible
1540        if (pending != 0) {
1541            strm.flushPending();
1542            if (strm.availout == 0) {
1543                // Since avail_out is 0, deflate will be called again with
1544                // more output space, but possibly with both pending and
1545                // avail_in equal to zero. There won't be anything to do,
1546                // but this is not an error situation so make sure we
1547                // return OK instead of BUF_ERROR at next call of deflate:
1548                lastFlush = -1;
1549                return Z_OK;
1550            }
1551
1552            // Make sure there is something to do and avoid duplicate consecutive
1553            // flushes. For repeated and useless calls with Z_FINISH, we keep
1554            // returning Z_STREAM_END instead of Z_BUFF_ERROR.
1555        } else if (strm.availin == 0 && flush <= oldflush
1556                && flush != Z_FINISH) {
1557            strm.msg = zerrmsg[Z_NEED_DICT - (Z_BUF_ERROR)];
1558            return Z_BUF_ERROR;
1559        }
1560
1561        // User must not provide more input after the first FINISH:
1562        if (status == FINISH_STATE && strm.availin != 0) {
1563            strm.msg = zerrmsg[Z_NEED_DICT - (Z_BUF_ERROR)];
1564            return Z_BUF_ERROR;
1565        }
1566
1567        // Start a new block or continue the current one.
1568        if (strm.availin != 0 || lookahead != 0
1569                || (flush != Z_NO_FLUSH && status != FINISH_STATE)) {
1570            int bstate = -1;
1571            switch (CONFIG_TABLE[level].func) {
1572                case STORED:
1573                    bstate = deflateStored(flush);
1574                    break;
1575                case FAST:
1576                    bstate = deflateFast(flush);
1577                    break;
1578                case SLOW:
1579                    bstate = deflateSlow(flush);
1580                    break;
1581                default:
1582            }
1583
1584            if (bstate == FINISH_STARTED || bstate == FINISH_DONE) {
1585                status = FINISH_STATE;
1586            }
1587            if (bstate == NEED_MORE || bstate == FINISH_STARTED) {
1588                if (strm.availout == 0) {
1589                    lastFlush = -1; // avoid BUF_ERROR next call, see above
1590                }
1591                return Z_OK;
1592                // If flush != Z_NO_FLUSH && avail_out == 0, the next call
1593                // of deflate should use the same flush parameter to make sure
1594                // that the flush is complete. So we don't have to output an
1595                // empty block here, this will be done at next call. This also
1596                // ensures that for a very small output buffer, we emit at most
1597                // one empty block.
1598            }
1599
1600            if (bstate == BLOCK_DONE) {
1601                if (flush == Z_PARTIAL_FLUSH) {
1602                    trAlign();
1603                } else { // FULL_FLUSH or SYNC_FLUSH
1604                    trStoredBlock(0, 0, false);
1605                    // For a full flush, this empty block will be recognized
1606                    // as a special marker by inflate_sync().
1607                    if (flush == Z_FULL_FLUSH) {
1608                        //state.head[s.hash_size-1]=0;
1609                        for (int i = 0; i < hashSize/*-1*/; i++) // forget history
1610                        {
1611                            head[i] = 0;
1612                        }
1613                    }
1614                }
1615                strm.flushPending();
1616                if (strm.availout == 0) {
1617                    lastFlush = -1; // avoid BUF_ERROR at next call, see above
1618                    return Z_OK;
1619                }
1620            }
1621        }
1622
1623        if (flush != Z_FINISH) {
1624            return Z_OK;
1625        }
1626        if (noheader != 0) {
1627            return Z_STREAM_END;
1628        }
1629
1630        // Write the zlib trailer (adler32)
1631        putShortMSB((int) (strm.adler >>> 16));
1632        putShortMSB((int) (strm.adler & 0xffff));
1633        strm.flushPending();
1634
1635        // If avail_out is zero, the application will call deflate again
1636        // to flush the rest.
1637        noheader = -1; // write the trailer only once!
1638        return pending != 0 ? Z_OK : Z_STREAM_END;
1639    }
