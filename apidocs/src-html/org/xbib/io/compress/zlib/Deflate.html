326        // Restore the heap property by moving down the tree starting at node k,
327        // exchanging a node with the smallest of its two sons if necessary, stopping
328        // when the heap property is re-established (each father smaller than its
329        // two sons).
330        protected void pqdownheap(short[] tree, // the tree to restore
331                                  int kk // node to move down
332        ) {
333            int k = kk;
334            int v = heap[k];
335            int j = k << 1;  // left son of k
336            while (j <= heapLen) {
337                // Set j to the smallest of the two sons:
338                if (j < heapLen
339                        && smaller(tree, heap[j + 1], heap[j], depth)) {
340                    j++;
341                }
342                // Exit if v is smaller than both sons
343                if (smaller(tree, v, heap[j], depth)) {
344                    break;
345                }
346    
347                // Exchange v with the smallest son
348                heap[k] = heap[j];
349                k = j;
350                // And continue down the tree, setting j to the left son of k
351                j <<= 1;
352            }
353            heap[k] = v;
354        }

356        private static boolean smaller(short[] tree, int n, int m, byte[] depth) {
357            short tn2 = tree[n * 2];
358            short tm2 = tree[m * 2];
359            return (tn2 < tm2
360                    || (tn2 == tm2 && depth[n] <= depth[m]));
361        }

363        // Scan a literal or distance tree to determine the frequencies of the codes
364        // in the bit length tree.
365        private void scanTree(short[] tree,// the tree to be scanned
366                              int maxcode // and its largest code of non zero frequency
367        ) {
368            int n;                     // iterates over all tree elements
369            int prevlen = -1;          // last emitted length
370            int curlen;                // length of current code
371            int nextlen = tree[0 * 2 + 1]; // length of next code
372            int count = 0;             // repeat count of the current code
373            int maxcount = 7;         // max repeat count
374            int mincount = 4;         // min repeat count
375    
376            if (nextlen == 0) {
377                maxcount = 138;
378                mincount = 3;
379            }
380            tree[(maxcode + 1) * 2 + 1] = (short) 0xffff; // guard
381    
382            for (n = 0; n <= maxcode; n++) {
383                curlen = nextlen;
384                nextlen = tree[(n + 1) * 2 + 1];
385                if (++count < maxcount && curlen == nextlen) {
386                    continue;
387                } else if (count < mincount) {
388                    blTree[curlen * 2] += count;
389                } else if (curlen != 0) {
390                    if (curlen != prevlen) {
391                        blTree[curlen * 2]++;
392                    }
393                    blTree[REP_3_6 * 2]++;
394                } else if (count <= 10) {
395                    blTree[REPZ_3_10 * 2]++;
396                } else {
397                    blTree[REPZ_11_138 * 2]++;
398                }
399                count = 0;
400                prevlen = curlen;
401                if (nextlen == 0) {
402                    maxcount = 138;
403                    mincount = 3;
404                } else if (curlen == nextlen) {
405                    maxcount = 6;
406                    mincount = 3;
407                } else {
408                    maxcount = 7;
409                    mincount = 4;
410                }
411            }
412        }

414        // Construct the Huffman tree for the bit lengths and return the index in
415        // BL_ORDER of the last bit length code to send.
416        private int buildBLTree() {
417            int maxblindex;  // index of last bit length code of non zero freq
418    
419            // Determine the bit length frequencies for literal and distance trees
420            scanTree(dynLtree, lDesc.maxCode);
421            scanTree(dynDtree, dDesc.maxCode);
422    
423            // Build the bit length tree:
424            blDesc.buildTree(this);
425            // opt_len now includes the length of the tree representations, except
426            // the lengths of the bit lengths codes and the 5+5+4 bits for the counts.
427    
428            // Determine the number of bit length codes to send. The pkzip format
429            // requires that at least 4 bit length codes be sent. (appnote.txt says
430            // 3 but the actual value used is 4.)
431            for (maxblindex = BL_CODES - 1; maxblindex >= 3; maxblindex--) {
432                if (blTree[Tree.BL_ORDER[maxblindex] * 2 + 1] != 0) {
433                    break;
434                }
435            }
436            // Update opt_len to include the bit length tree and counts
437            optLen += 3 * (maxblindex + 1) + 5 + 5 + 4;
438    
439            return maxblindex;
440        }

445        private void sendAllTree(int lcodes, int dcodes, int blcodes) {
446            int rank;                    // index in BL_ORDER
447    
448            sendBits(lcodes - 257, 5); // not +255 as stated in appnote.txt
449            sendBits(dcodes - 1, 5);
450            sendBits(blcodes - 4, 4); // not -3 as stated in appnote.txt

454            sendTree(dynLtree, lcodes - 1); // literal tree
455            sendTree(dynDtree, dcodes - 1); // distance tree
456        }

458        // Send a literal or distance tree in compressed form, using the codes in
459        // bl_tree.
460        private void sendTree(short[] tree,// the tree to be sent
461                              int max_code // and its largest code of non zero frequency
462        ) {
463            int n;                     // iterates over all tree elements
464            int prevlen = -1;          // last emitted length
465            int curlen;                // length of current code
466            int nextlen = tree[0 * 2 + 1]; // length of next code
467            int count = 0;             // repeat count of the current code
468            int maxcount = 7;         // max repeat count
469            int mincount = 4;         // min repeat count
470    
471            if (nextlen == 0) {
472                maxcount = 138;
473                mincount = 3;
474            }
475    
476            for (n = 0; n <= max_code; n++) {
477                curlen = nextlen;
478                nextlen = tree[(n + 1) * 2 + 1];
479                if (++count < maxcount && curlen == nextlen) {
480                    continue;
481                } else if (count < mincount) {
482                    do {
483                        sendCode(curlen, blTree);
484                    } while (--count != 0);
485                } else if (curlen != 0) {
486                    if (curlen != prevlen) {
487                        sendCode(curlen, blTree);
488                        count--;
489                    }
490                    sendCode(REP_3_6, blTree);
491                    sendBits(count - 3, 2);
492                } else if (count <= 10) {
493                    sendCode(REPZ_3_10, blTree);
494                    sendBits(count - 3, 3);
495                } else {
496                    sendCode(REPZ_11_138, blTree);
497                    sendBits(count - 11, 7);
498                }
499                count = 0;
500                prevlen = curlen;
501                if (nextlen == 0) {
502                    maxcount = 138;
503                    mincount = 3;
504                } else if (curlen == nextlen) {
505                    maxcount = 6;
506                    mincount = 3;
507                } else {
508                    maxcount = 7;
509                    mincount = 4;
510                }
511            }
512        }

540        private void sendBits(int value, int length) {
541            int len = length;
542            if (biValid > BUF_SIZE - len) {
543                int val = value;
544    //      bi_buf |= (val << bi_valid);
545                biBuf |= ((val << biValid) & 0xffff);
546                putShort(biBuf);
547                biBuf = (short) (val >>> (BUF_SIZE - biValid));
548                biValid += len - BUF_SIZE;
549            } else {
550    //      bi_buf |= (value) << bi_valid;
551                biBuf |= (((value) << biValid) & 0xffff);
552                biValid += len;
553            }
554        }

556        // Send one empty static block to give enough lookahead for inflate.
557        // This takes 10 bits, of which 7 may remain in the bit buffer.
558        // The current inflate code requires 9 bits of lookahead. If the
559        // last two codes for the previous block (real code plus EOB) were coded
560        // on 5 bits or less, inflate may have only 5+3 bits of lookahead to decode
561        // the last real code. In this case we send two empty static blocks instead
562        // of one. (There are no problems if the previous block is stored or fixed.)
563        // To simplify the code, we assume the worst case of last real code encoded
564        // on one bit only.
565        private void trAlign() {
566            sendBits(STATIC_TREES << 1, 3);
567            sendCode(END_BLOCK, StaticTree.STATIC_LTREE);
568    
569            biFlush();
570    
571            // Of the 10 bits for the empty block, we have already sent
572            // (10 - bi_valid) bits. The lookahead for the last real code (before
573            // the EOB of the previous block) was thus at least one plus the length
574            // of the EOB plus what we have just sent of the empty static block.
575            if (1 + lastEobLen + 10 - biValid < 9) {
576                sendBits(STATIC_TREES << 1, 3);
577                sendCode(END_BLOCK, StaticTree.STATIC_LTREE);
578                biFlush();
579            }
580            lastEobLen = 7;
581        }

583        // Save the match info and tally the frequency counts. Return true if
584        // the current block must be flushed.
585        private boolean trTally(int dist, // distance of matched string
586                                int lc // match length-MIN_MATCH or unmatched char (if dist==0)
587        ) {
588    
589            pendingBuf[dBuf + lastLit * 2] = (byte) (dist >>> 8);
590            pendingBuf[dBuf + lastLit * 2 + 1] = (byte) dist;
591    
592            pendingBuf[lBuf + lastLit] = (byte) lc;
593            lastLit++;
594    
595            if (dist == 0) {
596                // lc is the unmatched char
597                dynLtree[lc * 2]++;
598            } else {
599                matches++;
600                // Here, lc is the match length - MIN_MATCH
601                dist--;             // dist = match distance - 1
602                dynLtree[(Tree.LENGTH_CODE[lc] + LITERALS + 1) * 2]++;
603                dynDtree[Tree.distanceCode(dist) * 2]++;
604            }
605    
606            if ((lastLit & 0x1fff) == 0 && level > 2) {
607                // Compute an upper bound for the compressed length
608                int out_length = lastLit * 8;
609                int in_length = strstart - blockStart;
610                int dcode;
611                for (dcode = 0; dcode < D_CODES; dcode++) {
612                    out_length += (int) dynDtree[dcode * 2]
613                            * (5L + Tree.EXTRA_DBITS[dcode]);
614                }
615                out_length >>>= 3;
616                if ((matches < (lastLit / 2)) && out_length < in_length / 2) {
617                    return true;
618                }
619            }
620    
621            return (lastLit == litBufsize - 1);
622            // We avoid equality with lit_bufsize because of wraparound at 64K
623            // on 16 bit machines and because stored blocks are restricted to
624            // 64K-1 bytes.
625        }

627        // Send the block data compressed using the given Huffman trees
628        private void compressBlock(short[] ltree, short[] dtree) {
629            int dist;      // distance of matched string
630            int lc;         // match length or unmatched char (if dist == 0)
631            int lx = 0;     // running index in l_buf
632            int code;       // the code to send
633            int extra;      // number of extra bits to send
634    
635            if (lastLit != 0) {
636                do {
637                    dist = ((pendingBuf[dBuf + lx * 2] << 8) & 0xff00)
638                            | (pendingBuf[dBuf + lx * 2 + 1] & 0xff);
639                    lc = (pendingBuf[lBuf + lx]) & 0xff;
640                    lx++;
641    
642                    if (dist == 0) {
643                        sendCode(lc, ltree); // send a literal byte
644                    } else {
645                        // Here, lc is the match length - MIN_MATCH
646                        code = Tree.LENGTH_CODE[lc];
647    
648                        sendCode(code + LITERALS + 1, ltree); // send the length code
649                        extra = Tree.EXTRA_LBITS[code];
650                        if (extra != 0) {
651                            lc -= Tree.BASE_LENGTH[code];
652                            sendBits(lc, extra);       // send the extra length bits
653                        }
654                        dist--; // dist is now the match distance - 1
655                        code = Tree.distanceCode(dist);
656    
657                        sendCode(code, dtree);       // send the distance code
658                        extra = Tree.EXTRA_DBITS[code];
659                        if (extra != 0) {
660                            dist -= Tree.BASE_DIST[code];
661                            sendBits(dist, extra);   // send the extra distance bits
662                        }
663                    } // literal or match pair ?
664    
665                    // Check that the overlay between pending_buf and d_buf+l_buf is ok:
666                } while (lx < lastLit);
667            }
668    
669            sendCode(END_BLOCK, ltree);
670            lastEobLen = ltree[END_BLOCK * 2 + 1];
671        }

673        // Set the data type to ASCII or BINARY, using a crude approximation:
674        // binary if more than 20% of the bytes are <= 6 or >= 128, ascii otherwise.
675        // IN assertion: the fields freq of dyn_ltree are set and the total of all
676        // frequencies does not exceed 64K (to fit in an int on 16 bit machines).
677        private void setDataType() {
678            int n = 0;
679            int asciifreq = 0;
680            int binfreq = 0;
681            while (n < 7) {
682                binfreq += dynLtree[n * 2];
683                n++;
684            }
685            while (n < 128) {
686                asciifreq += dynLtree[n * 2];
687                n++;
688            }
689            while (n < LITERALS) {
690                binfreq += dynLtree[n * 2];
691                n++;
692            }
693            dataType = (byte) (binfreq > (asciifreq >>> 2) ? Z_BINARY : Z_ASCII);
694        }

696        // Flush the bit buffer, keeping at most 7 bits in it.
697        private void biFlush() {
698            if (biValid == 16) {
699                putShort(biBuf);
700                biBuf = 0;
701                biValid = 0;
702            } else if (biValid >= 8) {
703                putByte((byte) biBuf);
704                biBuf >>>= 8;
705                biValid -= 8;
706            }
707        }

709        // Flush the bit buffer and align the output on a byte boundary
710        private void biWindup() {
711            if (biValid > 8) {
712                putShort(biBuf);
713            } else if (biValid > 0) {
714                putByte((byte) biBuf);
715            }
716            biBuf = 0;
717            biValid = 0;
718        }

721        // one's complement if requested.

723                               int len, // its length

744        // Copy without compression as much as possible from the input stream, return
745        // the current block state.
746        // This function does not insert new strings in the dictionary since
747        // uncompressible data is probably not useful. This function is used
748        // only for the level=0 compression option.
749        // NOTE: this function should be optimized to avoid extra copying from
750        // window to pending_buf.
751        private int deflateStored(int flush) {
752            // Stored blocks are limited to 0xffff bytes, pending_buf is limited
753            // to pending_buf_size, and each stored block has a 5 byte header:
754    
755            int maxblocksize = 0xffff;
756            int maxstart;
757    
758            if (maxblocksize > pendingBufSize - 5) {
759                maxblocksize = pendingBufSize - 5;
760            }
761    
762            // Copy as much as possible from input to output:
763            while (true) {
764                // Fill the window as much as possible:
765                if (lookahead <= 1) {
766                    fillWindow();
767                    if (lookahead == 0 && flush == Z_NO_FLUSH) {
768                        return NEED_MORE;
769                    }
770                    if (lookahead == 0) {
771                        break; // flush the current block
772                    }
773                }
774    
775                strstart += lookahead;
776                lookahead = 0;
777    
778                // Emit a stored block if pending_buf will be full:
779                maxstart = blockStart + maxblocksize;
780                if (strstart == 0 || strstart >= maxstart) {
781                    // strstart == 0 is possible when wraparound on 16-bit machine
782                    lookahead = (strstart - maxstart);
783                    strstart = maxstart;
784    
785                    flushBlockOnly(false);
786                    if (strm.availout == 0) {
787                        return NEED_MORE;
788                    }
789    
790                }
791    
792                // Flush if we may have to slide, otherwise block_start may become
793                // negative and the data will be gone:
794                if (strstart - blockStart >= wSize - MIN_LOOKAHEAD) {
795                    flushBlockOnly(false);
796                    if (strm.availout == 0) {
797                        return NEED_MORE;
798                    }
799                }
800            }
801    
802            flushBlockOnly(flush == Z_FINISH);
803            if (strm.availout == 0) {
804                return (flush == Z_FINISH) ? FINISH_STARTED : NEED_MORE;
805            }
806    
807            return flush == Z_FINISH ? FINISH_DONE : BLOCK_DONE;
808        }

819        // Determine the best encoding for the current block: dynamic trees, static
820        // trees or store, and output the encoded block to the zip file.
821        private void trFlushBlock(int buf, // input block, or NULL if too old
822                                  int storedlen, // length of input block
823                                  boolean eof // true if this is the last block for a file
824        ) {
825            int optlenb, staticlenb;// opt_len and static_len in bytes
826            int maxblindex = 0;      // index of last bit length code of non zero freq
827    
828            // Build the Huffman trees unless a stored block is forced
829            if (level > 0) {
830                // Check if the file is ascii or binary
831                if (dataType == Z_UNKNOWN) {
832                    setDataType();
833                }
834    
835                // Construct the literal and distance trees
836                lDesc.buildTree(this);
837    
838                dDesc.buildTree(this);
839    
840                // At this point, opt_len and static_len are the total bit lengths of
841                // the compressed block data, excluding the tree representations.
842    
843                // Build the bit length tree for the above two trees, and get the index
844                // in BL_ORDER of the last bit length code to send.
845                maxblindex = buildBLTree();
846    
847                // Determine the best encoding. Compute first the block length in bytes
848                optlenb = (optLen + 3 + 7) >>> 3;
849                staticlenb = (staticLen + 3 + 7) >>> 3;
850    
851                if (staticlenb <= optlenb) {
852                    optlenb = staticlenb;
853                }
854            } else {
855                optlenb = staticlenb = storedlen + 5; // force a stored block
856            }
857    
858            if (storedlen + 4 <= optlenb && buf != -1) {
859                // 4: two words for the lengths
860                // The test buf != NULL is only necessary if LIT_BUFSIZE > WSIZE.
861                // Otherwise we can't have processed more than WSIZE input bytes since
862                // the last block flush, because compression would have been
863                // successful. If LIT_BUFSIZE <= WSIZE, it is never too late to
864                // transform a block into a stored block.
865                trStoredBlock(buf, storedlen, eof);
866            } else if (staticlenb == optlenb) {
867                sendBits((STATIC_TREES << 1) + (eof ? 1 : 0), 3);
868                compressBlock(StaticTree.STATIC_LTREE, StaticTree.STATIC_DTREE);
869            } else {
870                sendBits((DYN_TREES << 1) + (eof ? 1 : 0), 3);
871                sendAllTree(lDesc.maxCode + 1, dDesc.maxCode + 1, maxblindex + 1);
872                compressBlock(dynLtree, dynDtree);
873            }
874    
875            // The above check is made mod 2^32, for files larger than 512 MB
876            // and uLong implemented on 32 bits.
877    
878            initBlock();
879    
880            if (eof) {
881                biWindup();
882            }
883        }

885        // Fill the window when the lookahead becomes insufficient.
886        // Updates strstart and lookahead.
887        //
888        // IN assertion: lookahead < MIN_LOOKAHEAD
889        // OUT assertions: strstart <= window_size-MIN_LOOKAHEAD
890        //    At least one byte has been read, or avail_in == 0; reads are
891        //    performed for at least two bytes (required for the zip translate_eol
892        //    option -- not supported here).
893        private void fillWindow() {
894            int n, m;
895            int p;
896            int more;    // Amount of free space at the end of the window.
897    
898            do {
899                more = (windowSize - lookahead - strstart);
900    
901                // Deal with !@#$% 64K limit:
902                if (more == 0 && strstart == 0 && lookahead == 0) {
903                    more = wSize;
904                } else if (more == -1) {
905                    // Very unlikely, but possible on 16 bit machine if strstart == 0
906                    // and lookahead == 1 (input done one byte at time)
907                    more--;
908    
909                    // If the window is almost full and there is insufficient lookahead,
910                    // move the upper half to the lower one to make room in the upper half.
911                } else if (strstart >= wSize + wSize - MIN_LOOKAHEAD) {
912                    System.arraycopy(window, wSize, window, 0, wSize);
913                    matchStart -= wSize;
914                    strstart -= wSize; // we now have strstart >= MAX_DIST
915                    blockStart -= wSize;
916    
917                    // Slide the hash table (could be avoided with 32 bit values
918                    // at the expense of memory usage). We slide even when level == 0
919                    // to keep the hash table consistent if we switch back to level > 0
920                    // later. (Using level 0 permanently is not an optimal usage of
921                    // zlib, so we don't care about this pathological case.)
922    
923                    n = hashSize;
924                    p = n;
925                    do {
926                        m = (head[--p] & 0xffff);
927                        head[p] = (m >= wSize ? (short) (m - wSize) : 0);
928                    } while (--n != 0);
929    
930                    n = wSize;
931                    p = n;
932                    do {
933                        m = (prev[--p] & 0xffff);
934                        prev[p] = (m >= wSize ? (short) (m - wSize) : 0);
935                        // If n is not on any hash chain, prev[n] is garbage but
936                        // its value will never be used.
937                    } while (--n != 0);
938                    more += wSize;
939                }
940    
941                if (strm.availin == 0) {
942                    return;
943                }
944    
945                // If there was no sliding:
946                //    strstart <= WSIZE+MAX_DIST-1 && lookahead <= MIN_LOOKAHEAD - 1 &&
947                //    more == window_size - lookahead - strstart
948                // => more >= window_size - (MIN_LOOKAHEAD-1 + WSIZE + MAX_DIST-1)
949                // => more >= window_size - 2*WSIZE + 2
950                // In the BIG_MEM or MMAP case (not yet supported),
951                //   window_size == input_size + MIN_LOOKAHEAD  &&
952                //   strstart + s->lookahead <= input_size => more >= MIN_LOOKAHEAD.
953                // Otherwise, window_size == 2*WSIZE so more >= 2.
954                // If there was sliding, more >= WSIZE. So in all cases, more >= 2.
955    
956                n = strm.readBuf(window, strstart + lookahead, more);
957                lookahead += n;
958    
959                // Initialize the hash value now that we have some input:
960                if (lookahead >= MIN_MATCH) {
961                    insh = window[strstart] & 0xff;
962                    insh = (((insh) << hashShift) ^ (window[strstart + 1] & 0xff)) & hashMask;
963                }
964                // If the whole input has less than MIN_MATCH bytes, ins_h is garbage,
965                // but this is not important since only literal bytes will be emitted.
966            } while (lookahead < MIN_LOOKAHEAD && strm.availin != 0);
967        }

969        // Compress as much as possible from the input stream, return the current
970        // block state.
971        // This function does not perform lazy evaluation of matches and inserts
972        // new strings in the dictionary only for unmatched strings or for short
973        // matches. It is used only for the fast compression options.
974        private int deflateFast(int flush) {
975    //    short hash_head = 0; // head of the hash chain
976            int hashhead = 0; // head of the hash chain
977            boolean bflush;      // set if current block must be flushed
978    
979            while (true) {
980                // Make sure that we always have enough lookahead, except
981                // at the end of the input file. We need MAX_MATCH bytes
982                // for the next match, plus MIN_MATCH bytes to insert the
983                // string following the next match.
984                if (lookahead < MIN_LOOKAHEAD) {
985                    fillWindow();
986                    if (lookahead < MIN_LOOKAHEAD && flush == Z_NO_FLUSH) {
987                        return NEED_MORE;
988                    }
989                    if (lookahead == 0) {
990                        break; // flush the current block
991                    }
992                }
993    
994                // Insert the string window[strstart .. strstart+2] in the
995                // dictionary, and set hash_head to the head of the hash chain:
996                if (lookahead >= MIN_MATCH) {
997                    insh = (((insh) << hashShift) ^ (window[(strstart) + (MIN_MATCH - 1)] & 0xff)) & hashMask;
998    
999    //      prev[strstart&w_mask]=hash_head=head[ins_h];
1000                    hashhead = (head[insh] & 0xffff);
1001                    prev[strstart & wMask] = head[insh];
1002                    head[insh] = (short) strstart;
1003                }
1004    
1005                // Find the longest match, discarding those <= prev_length.
1006                // At this point we have always match_length < MIN_MATCH
1007    
1008                // To simplify the code, we prevent matches with the string
1009                // of window index 0 (in particular we have to avoid a match
1010                // of the string with itself at the start of the input file).
1011                if (hashhead != 0L
1012                        && ((strstart - hashhead) & 0xffff) <= wSize - MIN_LOOKAHEAD
1013                        && (strategy != Z_HUFFMAN_ONLY)) {
1014                    matchLength = longestMatch(hashhead);
1015                }
1016                // longest_match() sets match_start
1017    
1018                if (matchLength >= MIN_MATCH) {
1019                    //        check_match(strstart, match_start, match_length);
1020    
1021                    bflush = trTally(strstart - matchStart, matchLength - MIN_MATCH);
1022    
1023                    lookahead -= matchLength;
1024    
1025                    // Insert new strings in the hash table only if the match length
1026                    // is not too large. This saves time but degrades compression.
1027                    if (matchLength <= maxLazyMatch
1028                            && lookahead >= MIN_MATCH) {
1029                        matchLength--; // string at strstart already in hash table
1030                        do {
1031                            strstart++;
1032    
1033                            insh = ((insh << hashShift) ^ (window[(strstart) + (MIN_MATCH - 1)] & 0xff)) & hashMask;
1034    //          prev[strstart&w_mask]=hash_head=head[ins_h];
1035                            hashhead = (head[insh] & 0xffff);
1036                            prev[strstart & wMask] = head[insh];
1037                            head[insh] = (short) strstart;
1038    
1039                            // strstart never exceeds WSIZE-MAX_MATCH, so there are
1040                            // always MIN_MATCH bytes ahead.
1041                        } while (--matchLength != 0);
1042                        strstart++;
1043                    } else {
1044                        strstart += matchLength;
1045                        matchLength = 0;
1046                        insh = window[strstart] & 0xff;
1047    
1048                        insh = (((insh) << hashShift) ^ (window[strstart + 1] & 0xff)) & hashMask;
1049                        // If lookahead < MIN_MATCH, ins_h is garbage, but it does not
1050                        // matter since it will be recomputed at next deflate call.
1051                    }
1052                } else {
1053                    // No match, output a literal byte
1054    
1055                    bflush = trTally(0, window[strstart] & 0xff);
1056                    lookahead--;
1057                    strstart++;
1058                }
1059                if (bflush) {
1060    
1061                    flushBlockOnly(false);
1062                    if (strm.availout == 0) {
1063                        return NEED_MORE;
1064                    }
1065                }
1066            }
1067    
1068            flushBlockOnly(flush == Z_FINISH);
1069            if (strm.availout == 0) {
1070                if (flush == Z_FINISH) {
1071                    return FINISH_STARTED;
1072                } else {
1073                    return NEED_MORE;
1074                }
1075            }
1076            return flush == Z_FINISH ? FINISH_DONE : BLOCK_DONE;
1077        }

1079        // Same as above, but achieves better compression. We use a lazy
1080        // evaluation for matches: a match is finally adopted only if there is
1081        // no better match at the next window position.
1082        private int deflateSlow(int flush) {
1083    //    short hash_head = 0;    // head of hash chain
1084            int hashhead = 0;    // head of hash chain
1085            boolean bflush;         // set if current block must be flushed
1086    
1087            // Process the input block.
1088            while (true) {
1089                // Make sure that we always have enough lookahead, except
1090                // at the end of the input file. We need MAX_MATCH bytes
1091                // for the next match, plus MIN_MATCH bytes to insert the
1092                // string following the next match.
1093    
1094                if (lookahead < MIN_LOOKAHEAD) {
1095                    fillWindow();
1096                    if (lookahead < MIN_LOOKAHEAD && flush == Z_NO_FLUSH) {
1097                        return NEED_MORE;
1098                    }
1099                    if (lookahead == 0) {
1100                        break; // flush the current block
1101                    }
1102                }
1103    
1104                // Insert the string window[strstart .. strstart+2] in the
1105                // dictionary, and set hash_head to the head of the hash chain:
1106    
1107                if (lookahead >= MIN_MATCH) {
1108                    insh = (((insh) << hashShift) ^ (window[(strstart) + (MIN_MATCH - 1)] & 0xff)) & hashMask;
1109    //      prev[strstart&w_mask]=hash_head=head[ins_h];
1110                    hashhead = (head[insh] & 0xffff);
1111                    prev[strstart & wMask] = head[insh];
1112                    head[insh] = (short) strstart;
1113                }
1114    
1115                // Find the longest match, discarding those <= prev_length.
1116                prevLength = matchLength;
1117                prevMatch = matchStart;
1118                matchLength = MIN_MATCH - 1;
1119    
1120                if (hashhead != 0 && prevLength < maxLazyMatch
1121                        && ((strstart - hashhead) & 0xffff) <= wSize - MIN_LOOKAHEAD) {
1122                    // To simplify the code, we prevent matches with the string
1123                    // of window index 0 (in particular we have to avoid a match
1124                    // of the string with itself at the start of the input file).
1125    
1126                    if (strategy != Z_HUFFMAN_ONLY) {
1127                        matchLength = longestMatch(hashhead);
1128                    }
1129                    // longest_match() sets match_start
1130    
1131                    if (matchLength <= 5 && (strategy == Z_FILTERED
1132                            || (matchLength == MIN_MATCH
1133                            && strstart - matchStart > 4096))) {
1134    
1135                        // If prev_match is also MIN_MATCH, match_start is garbage
1136                        // but we will ignore the current match anyway.
1137                        matchLength = MIN_MATCH - 1;
1138                    }
1139                }
1140    
1141                // If there was a match at the previous step and the current
1142                // match is not better, output the previous match:
1143                if (prevLength >= MIN_MATCH && matchLength <= prevLength) {
1144                    int max_insert = strstart + lookahead - MIN_MATCH;
1145                    // Do not insert strings in hash table beyond this.
1146    
1147                    //          check_match(strstart-1, prev_match, prev_length);
1148    
1149                    bflush = trTally(strstart - 1 - prevMatch, prevLength - MIN_MATCH);
1150    
1151                    // Insert in hash table all strings up to the end of the match.
1152                    // strstart-1 and strstart are already inserted. If there is not
1153                    // enough lookahead, the last two strings are not inserted in
1154                    // the hash table.
1155                    lookahead -= prevLength - 1;
1156                    prevLength -= 2;
1157                    do {
1158                        if (++strstart <= max_insert) {
1159                            insh = (((insh) << hashShift) ^ (window[(strstart) + (MIN_MATCH - 1)] & 0xff)) & hashMask;
1160                            //prev[strstart&w_mask]=hash_head=head[ins_h];
1161                            hashhead = (head[insh] & 0xffff);
1162                            prev[strstart & wMask] = head[insh];
1163                            head[insh] = (short) strstart;
1164                        }
1165                    } while (--prevLength != 0);
1166                    matchAvailable = 0;
1167                    matchLength = MIN_MATCH - 1;
1168                    strstart++;
1169    
1170                    if (bflush) {
1171                        flushBlockOnly(false);
1172                        if (strm.availout == 0) {
1173                            return NEED_MORE;
1174                        }
1175                    }
1176                } else if (matchAvailable != 0) {
1177    
1178                    // If there was no match at the previous position, output a
1179                    // single literal. If there was a match but the current match
1180                    // is longer, truncate the previous match to a single literal.
1181    
1182                    bflush = trTally(0, window[strstart - 1] & 0xff);
1183    
1184                    if (bflush) {
1185                        flushBlockOnly(false);
1186                    }
1187                    strstart++;
1188                    lookahead--;
1189                    if (strm.availout == 0) {
1190                        return NEED_MORE;
1191                    }
1192                } else {
1193                    // There is no previous match to compare with, wait for
1194                    // the next step to decide.
1195    
1196                    matchAvailable = 1;
1197                    strstart++;
1198                    lookahead--;
1199                }
1200            }
1201    
1202            if (matchAvailable != 0) {
1203                bflush = trTally(0, window[strstart - 1] & 0xff);
1204                matchAvailable = 0;
1205            }
1206            flushBlockOnly(flush == Z_FINISH);
1207    
1208            if (strm.availout == 0) {
1209                if (flush == Z_FINISH) {
1210                    return FINISH_STARTED;
1211                } else {
1212                    return NEED_MORE;
1213                }
1214            }
1215    
1216            return flush == Z_FINISH ? FINISH_DONE : BLOCK_DONE;
1217        }

1219        private int longestMatch(int cmatch) {
1220            int curmatch = cmatch;
1221            int chainlength = maxChainLength; // max hash chain length
1222            int scan = strstart;                 // current string
1223            int match;                           // matched string
1224            int len;                             // length of current match
1225            int bestlen = prevLength;          // best match length so far
1226            int limit = strstart > (wSize - MIN_LOOKAHEAD) ? strstart - (wSize - MIN_LOOKAHEAD) : 0;
1227            int nmatch = this.niceMatch;
1228    
1229            // Stop when cur_match becomes <= limit. To simplify the code,
1230            // we prevent matches with the string of window index 0.
1231    
1232            int wmask = wMask;
1233    
1234            int strend = strstart + MAX_MATCH;
1235            byte scanend1 = window[scan + bestlen - 1];
1236            byte scanend = window[scan + bestlen];
1237    
1238            // The code is optimized for HASH_BITS >= 8 and MAX_MATCH-2 multiple of 16.
1239            // It is easy to get rid of this optimization if necessary.
1240    
1241            // Do not waste too much time if we already have a good match:
1242            if (prevLength >= goodMatch) {
1243                chainlength >>= 2;
1244            }
1245    
1246            // Do not look for matches beyond the end of the input. This is necessary
1247            // to make deflate deterministic.
1248            if (nmatch > lookahead) {
1249                nmatch = lookahead;
1250            }
1251    
1252            do {
1253                match = curmatch;
1254    
1255                // Skip to next match if the match length cannot increase
1256                // or if the match length is less than 2:
1257                if (window[match + bestlen] != scanend
1258                        || window[match + bestlen - 1] != scanend1
1259                        || window[match] != window[scan]
1260                        || window[++match] != window[scan + 1]) {
1261                    continue;
1262                }
1263    
1264                // The check at best_len-1 can be removed because it will be made
1265                // again later. (This heuristic is not always a win.)
1266                // It is not necessary to compare scan[2] and match[2] since they
1267                // are always equal when the other bytes match, given that
1268                // the hash keys are equal and that HASH_BITS >= 8.
1269                scan += 2;
1270                match++;
1271    
1272                // We check for insufficient lookahead only every 8th comparison;
1273                // the 256th check will be made at strstart+258.
1274                do {
1275                } while (window[++scan] == window[++match]
1276                        && window[++scan] == window[++match]
1277                        && window[++scan] == window[++match]
1278                        && window[++scan] == window[++match]
1279                        && window[++scan] == window[++match]
1280                        && window[++scan] == window[++match]
1281                        && window[++scan] == window[++match]
1282                        && window[++scan] == window[++match]
1283                        && scan < strend);
1284    
1285                len = MAX_MATCH - (strend - scan);
1286                scan = strend - MAX_MATCH;
1287    
1288                if (len > bestlen) {
1289                    matchStart = curmatch;
1290                    bestlen = len;
1291                    if (len >= nmatch) {
1292                        break;
1293                    }
1294                    scanend1 = window[scan + bestlen - 1];
1295                    scanend = window[scan + bestlen];
1296                }
1297    
1298            } while ((curmatch = (prev[curmatch & wmask] & 0xffff)) > limit && --chainlength != 0);
1299    
1300            if (bestlen <= lookahead) {
1301                return bestlen;
1302            }
1303            return lookahead;
1304        }

1315        private int deflateInit2(ZStream strm, int level, int method, int windowBits,
1316                                 int memLevel, int strategy) {
1317            int nheader = 0;
1318    
1319            strm.msg = null;
1320    
1321            if (level == Z_DEFAULT_COMPRESSION) {
1322                level = 6;
1323            }
1324    
1325            if (windowBits < 0) { // undocumented feature: suppress zlib header
1326                nheader = 1;
1327                windowBits = -windowBits;
1328            }
1329    
1330            if (memLevel < 1 || memLevel > MAX_MEM_LEVEL
1331                    || method != Z_DEFLATED
1332                    || windowBits < 9 || windowBits > 15 || level < 0 || level > 9
1333                    || strategy < 0 || strategy > Z_HUFFMAN_ONLY) {
1334                return Z_STREAM_ERROR;
1335            }
1336    
1337            strm.dstate = this;
1338    
1339            this.noheader = nheader;
1340            wBits = windowBits;
1341            wSize = 1 << wBits;
1342            wMask = wSize - 1;
1343    
1344            hashBits = memLevel + 7;
1345            hashSize = 1 << hashBits;
1346            hashMask = hashSize - 1;
1347            hashShift = ((hashBits + MIN_MATCH - 1) / MIN_MATCH);
1348    
1349            window = new byte[wSize * 2];
1350            prev = new short[wSize];
1351            head = new short[hashSize];
1352    
1353            litBufsize = 1 << (memLevel + 6); // 16K elements by default
1354    
1355            // We overlay pending_buf and d_buf+l_buf. This works since the average
1356            // output size for (length,distance) codes is <= 24 bits.
1357            pendingBuf = new byte[litBufsize * 4];
1358            pendingBufSize = litBufsize * 4;
1359    
1360            dBuf = litBufsize / 2;
1361            lBuf = (1 + 2) * litBufsize;
1362    
1363            this.level = level;
1364    
1365            this.strategy = strategy;
1366            return deflateReset(strm);
1367        }

1432        public int deflateSetDictionary(ZStream strm, byte[] dictionary, int dictLength) {
1433            int length = dictLength;
1434            int index = 0;
1435    
1436            if (dictionary == null || status != INIT_STATE) {
1437                return Z_STREAM_ERROR;
1438            }
1439    
1440            strm.adler = Adler32.adler32(strm.adler, dictionary, 0, dictLength);
1441    
1442            if (length < MIN_MATCH) {
1443                return Z_OK;
1444            }
1445            if (length > wSize - MIN_LOOKAHEAD) {
1446                length = wSize - MIN_LOOKAHEAD;
1447                index = dictLength - length; // use the tail of the dictionary
1448            }
1449            System.arraycopy(dictionary, index, window, 0, length);
1450            strstart = length;
1451            blockStart = length;
1452    
1453            // Insert all strings in the hash table (except for the last two bytes).
1454            // s->lookahead stays null, so s->ins_h will be recomputed at the next
1455            // call of fill_window.
1456    
1457            insh = window[0] & 0xff;
1458            insh = (((insh) << hashShift) ^ (window[1] & 0xff)) & hashMask;
1459    
1460            for (int n = 0; n <= length - MIN_MATCH; n++) {
1461                insh = (((insh) << hashShift) ^ (window[(n) + (MIN_MATCH - 1)] & 0xff)) & hashMask;
1462                prev[n & wMask] = head[insh];
1463                head[insh] = (short) n;
1464            }
1465            return Z_OK;
1466        }

1468        public int deflate(ZStream strm, int flush) {
1469            int oldflush;
1470    
1471            if (flush > Z_FINISH || flush < 0) {
1472                return Z_STREAM_ERROR;
1473            }
1474    
1475            if (strm.nextout == null
1476                    || (strm.nextin == null && strm.availin != 0)
1477                    || (status == FINISH_STATE && flush != Z_FINISH)) {
1478                strm.msg = zerrmsg[Z_NEED_DICT - (Z_STREAM_ERROR)];
1479                return Z_STREAM_ERROR;
1480            }
1481            if (strm.availout == 0) {
1482                strm.msg = zerrmsg[Z_NEED_DICT - (Z_BUF_ERROR)];
1483                return Z_BUF_ERROR;
1484            }
1485    
1486            this.strm = strm; // just in case
1487            oldflush = lastFlush;
1488            lastFlush = flush;
1489    
1490            // Write the zlib header
1491            if (status == INIT_STATE) {
1492                int header = (Z_DEFLATED + ((wBits - 8) << 4)) << 8;
1493                int level_flags = ((level - 1) & 0xff) >> 1;
1494    
1495                if (level_flags > 3) {
1496                    level_flags = 3;
1497                }
1498                header |= (level_flags << 6);
1499                if (strstart != 0) {
1500                    header |= PRESET_DICT;
1501                }
1502                header += 31 - (header % 31);
1503    
1504                status = BUSY_STATE;
1505                putShortMSB(header);
1506    
1507    
1508                // Save the adler32 of the preset dictionary:
1509                if (strstart != 0) {
1510                    putShortMSB((int) (strm.adler >>> 16));
1511                    putShortMSB((int) (strm.adler & 0xffff));
1512                }
1513                strm.adler = Adler32.adler32(0, null, 0, 0);
1514            }
1515    
1516            // Flush as much pending output as possible
1517            if (pending != 0) {
1518                strm.flushPending();
1519                if (strm.availout == 0) {
1520                    // Since avail_out is 0, deflate will be called again with
1521                    // more output space, but possibly with both pending and
1522                    // avail_in equal to zero. There won't be anything to do,
1523                    // but this is not an error situation so make sure we
1524                    // return OK instead of BUF_ERROR at next call of deflate:
1525                    lastFlush = -1;
1526                    return Z_OK;
1527                }
1528    
1529                // Make sure there is something to do and avoid duplicate consecutive
1530                // flushes. For repeated and useless calls with Z_FINISH, we keep
1531                // returning Z_STREAM_END instead of Z_BUFF_ERROR.
1532            } else if (strm.availin == 0 && flush <= oldflush
1533                    && flush != Z_FINISH) {
1534                strm.msg = zerrmsg[Z_NEED_DICT - (Z_BUF_ERROR)];
1535                return Z_BUF_ERROR;
1536            }
1537    
1538            // User must not provide more input after the first FINISH:
1539            if (status == FINISH_STATE && strm.availin != 0) {
1540                strm.msg = zerrmsg[Z_NEED_DICT - (Z_BUF_ERROR)];
1541                return Z_BUF_ERROR;
1542            }
1543    
1544            // Start a new block or continue the current one.
1545            if (strm.availin != 0 || lookahead != 0
1546                    || (flush != Z_NO_FLUSH && status != FINISH_STATE)) {
1547                int bstate = -1;
1548                switch (CONFIG_TABLE[level].func) {
1549                    case STORED:
1550                        bstate = deflateStored(flush);
1551                        break;
1552                    case FAST:
1553                        bstate = deflateFast(flush);
1554                        break;
1555                    case SLOW:
1556                        bstate = deflateSlow(flush);
1557                        break;
1558                    default:
1559                }
1560    
1561                if (bstate == FINISH_STARTED || bstate == FINISH_DONE) {
1562                    status = FINISH_STATE;
1563                }
1564                if (bstate == NEED_MORE || bstate == FINISH_STARTED) {
1565                    if (strm.availout == 0) {
1566                        lastFlush = -1; // avoid BUF_ERROR next call, see above
1567                    }
1568                    return Z_OK;
1569                    // If flush != Z_NO_FLUSH && avail_out == 0, the next call
1570                    // of deflate should use the same flush parameter to make sure
1571                    // that the flush is complete. So we don't have to output an
1572                    // empty block here, this will be done at next call. This also
1573                    // ensures that for a very small output buffer, we emit at most
1574                    // one empty block.
1575                }
1576    
1577                if (bstate == BLOCK_DONE) {
1578                    if (flush == Z_PARTIAL_FLUSH) {
1579                        trAlign();
1580                    } else { // FULL_FLUSH or SYNC_FLUSH
1581                        trStoredBlock(0, 0, false);
1582                        // For a full flush, this empty block will be recognized
1583                        // as a special marker by inflate_sync().
1584                        if (flush == Z_FULL_FLUSH) {
1585                            //state.head[s.hash_size-1]=0;
1586                            for (int i = 0; i < hashSize/*-1*/; i++) // forget history
1587                            {
1588                                head[i] = 0;
1589                            }
1590                        }
1591                    }
1592                    strm.flushPending();
1593                    if (strm.availout == 0) {
1594                        lastFlush = -1; // avoid BUF_ERROR at next call, see above
1595                        return Z_OK;
1596                    }
1597                }
1598            }
1599    
1600            if (flush != Z_FINISH) {
1601                return Z_OK;
1602            }
1603            if (noheader != 0) {
1604                return Z_STREAM_END;
1605            }
1606    
1607            // Write the zlib trailer (adler32)
1608            putShortMSB((int) (strm.adler >>> 16));
1609            putShortMSB((int) (strm.adler & 0xffff));
1610            strm.flushPending();
1611    
1612            // If avail_out is zero, the application will call deflate again
1613            // to flush the rest.
1614            noheader = -1; // write the trailer only once!
1615            return pending != 0 ? Z_OK : Z_STREAM_END;
1616        }
