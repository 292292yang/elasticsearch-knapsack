002    package org.xbib.io.compress.zlib;
003    
004    public final class Inflate {
005    
006        // preset dictionary flag in zlib header
007        private static final int PRESET_DICT = 0x20;
008        private static final int Z_FINISH = 4;
009        private static final int Z_DEFLATED = 8;
010        private static final int Z_OK = 0;
011        private static final int Z_STREAM_END = 1;
012        private static final int Z_NEED_DICT = 2;
013        private static final int Z_STREAM_ERROR = -2;
014        private static final int Z_DATA_ERROR = -3;
015        private static final int Z_BUF_ERROR = -5;
016        private static final int METHOD = 0;   // waiting for method byte
017        private static final int FLAG = 1;     // waiting for flag byte
018        private static final int DICT4 = 2;    // four dictionary check bytes to go
019        private static final int DICT3 = 3;    // three dictionary check bytes to go
020        private static final int DICT2 = 4;    // two dictionary check bytes to go
021        private static final int DICT1 = 5;    // one dictionary check byte to go
022        private static final int DICT0 = 6;    // waiting for inflateSetDictionary
023        private static final int BLOCKS = 7;   // decompressing blocks
024        private static final int CHECK4 = 8;   // four check bytes to go
025        private static final int CHECK3 = 9;   // three check bytes to go
026        private static final int CHECK2 = 10;  // two check bytes to go
027        private static final int CHECK1 = 11;  // one check byte to go
028        private static final int DONE = 12;    // finished check, done
029        private static final int BAD = 13;     // got an error--stay here
030        private int mode;                            // current inflate mode
031        // mode dependent information
032        private int method;        // if FLAGS, method byte
033        // if CHECK, check values to compare
034        private long[] was = new long[1]; // computed check value
035        private long need;               // stream check value
036        // if BAD, inflateSync's marker bytes count
037        private int marker;
038        // mode independent information
039        private int nowrap;          // flag for no wrapper
040        private int wbits;            // log2(window size)  (8..15, defaults to 15)
041        private InfBlocks blocks;     // current inflate_blocks state
042    
043        protected int inflateReset(ZStream z) {
044            if (z == null || z.istate == null) {
045                return Z_STREAM_ERROR;
046            }
047    
048            z.totalin = z.totalout = 0;
049            z.msg = null;
050            z.istate.mode = z.istate.nowrap != 0 ? BLOCKS : METHOD;
051            z.istate.blocks.reset(z, null);
052            return Z_OK;
053        }
054    
055        protected int inflateEnd(ZStream z) {
056            if (blocks != null) {
057                blocks.free(z);
058            }
059            blocks = null;
060            //    ZFREE(z, z->state);
061            return Z_OK;
062        }
063    
064        protected int inflateInit(ZStream z, int w) {
065            z.msg = null;
066            blocks = null;
067    
068            // handle undocumented nowrap option (no zlib header or check)
069            nowrap = 0;
070            if (w < 0) {
071                w = -w;
072                nowrap = 1;
073            }
074    
075            // set window size
076            if (w < 8 || w > 15) {
077                inflateEnd(z);
078                return Z_STREAM_ERROR;
079            }
080            wbits = w;
081    
082            z.istate.blocks = new InfBlocks(z,
083                    z.istate.nowrap != 0 ? null : this,
084                    1 << w);
085    
086            // reset state
087            inflateReset(z);
088            return Z_OK;
089        }
090    
091        protected int inflate(ZStream z, int f) {
092            int r;
093            int b;
094    
095            if (z == null || z.istate == null || z.nextin == null) {
096                return Z_STREAM_ERROR;
097            }
098            f = f == Z_FINISH ? Z_BUF_ERROR : Z_OK;
099            r = Z_BUF_ERROR;
100            while (true) {
101                switch (z.istate.mode) {
102                    case METHOD:
103                        if (z.availin == 0) {
104                            return r;
105                        }
106                        r = f;
107                        z.availin--;
108                        z.totalin++;
109                        if (((z.istate.method = z.nextin[z.nextinindex++]) & 0xf) != Z_DEFLATED) {
110                            z.istate.mode = BAD;
111                            z.msg = "unknown compression method";
112                            z.istate.marker = 5;       // can't try inflateSync
113                            break;
114                        }
115                        if ((z.istate.method >> 4) + 8 > z.istate.wbits) {
116                            z.istate.mode = BAD;
117                            z.msg = "invalid window size";
118                            z.istate.marker = 5;       // can't try inflateSync
119                            break;
120                        }
121                        z.istate.mode = FLAG;
122                        break;
123                    case FLAG:
124                        if (z.availin == 0) {
125                            return r;
126                        }
127                        r = f;
128                        z.availin--;
129                        z.totalin++;
130                        b = (z.nextin[z.nextinindex++]) & 0xff;
131                        if ((((z.istate.method << 8) + b) % 31) != 0) {
132                            z.istate.mode = BAD;
133                            z.msg = "incorrect header check";
134                            z.istate.marker = 5;       // can't try inflateSync
135                            break;
136                        }
137    
138                        if ((b & PRESET_DICT) == 0) {
139                            z.istate.mode = BLOCKS;
140                            break;
141                        }
142                        z.istate.mode = DICT4;
143                        break;
144                    case DICT4:
145                        if (z.availin == 0) {
146                            return r;
147                        }
148                        r = f;
149                        z.availin--;
150                        z.totalin++;
151                        z.istate.need = ((z.nextin[z.nextinindex++] & 0xff) << 24) & 0xff000000L;
152                        z.istate.mode = DICT3;
153                        break;
154                    case DICT3:
155                        if (z.availin == 0) {
156                            return r;
157                        }
158                        r = f;
159                        z.availin--;
160                        z.totalin++;
161                        z.istate.need += ((z.nextin[z.nextinindex++] & 0xff) << 16) & 0xff0000L;
162                        z.istate.mode = DICT2;
163                        break;
164                    case DICT2:
165                        if (z.availin == 0) {
166                            return r;
167                        }
168                        r = f;
169                        z.availin--;
170                        z.totalin++;
171                        z.istate.need += ((z.nextin[z.nextinindex++] & 0xff) << 8) & 0xff00L;
172                        z.istate.mode = DICT1;
173                        break;
174                    case DICT1:
175                        if (z.availin == 0) {
176                            return r;
177                        }
178                        r = f;
179                        z.availin--;
180                        z.totalin++;
181                        z.istate.need += (z.nextin[z.nextinindex++] & 0xffL);
182                        z.adler = z.istate.need;
183                        z.istate.mode = DICT0;
184                        return Z_NEED_DICT;
185                    case DICT0:
186                        z.istate.mode = BAD;
187                        z.msg = "need dictionary";
188                        z.istate.marker = 0;       // can try inflateSync
189                        return Z_STREAM_ERROR;
190                    case BLOCKS:
191                        r = z.istate.blocks.proc(z, r);
192                        if (r == Z_DATA_ERROR) {
193                            z.istate.mode = BAD;
194                            z.istate.marker = 0;     // can try inflateSync
195                            break;
196                        }
197                        if (r == Z_OK) {
198                            r = f;
199                        }
200                        if (r != Z_STREAM_END) {
201                            return r;
202                        }
203                        r = f;
204                        z.istate.blocks.reset(z, z.istate.was);
205                        if (z.istate.nowrap != 0) {
206                            z.istate.mode = DONE;
207                            break;
208                        }
209                        z.istate.mode = CHECK4;
210                        break;
211                    case CHECK4:
212                        if (z.availin == 0) {
213                            return r;
214                        }
215                        r = f;
216                        z.availin--;
217                        z.totalin++;
218                        z.istate.need = ((z.nextin[z.nextinindex++] & 0xff) << 24) & 0xff000000L;
219                        z.istate.mode = CHECK3;
220                        break;
221                    case CHECK3:
222                        if (z.availin == 0) {
223                            return r;
224                        }
225                        r = f;
226                        z.availin--;
227                        z.totalin++;
228                        z.istate.need += ((z.nextin[z.nextinindex++] & 0xff) << 16) & 0xff0000L;
229                        z.istate.mode = CHECK2;
230                        break;
231                    case CHECK2:
232                        if (z.availin == 0) {
233                            return r;
234                        }
235                        r = f;
236                        z.availin--;
237                        z.totalin++;
238                        z.istate.need += ((z.nextin[z.nextinindex++] & 0xff) << 8) & 0xff00L;
239                        z.istate.mode = CHECK1;
240                        break;
241                    case CHECK1:
242                        if (z.availin == 0) {
243                            return r;
244                        }
245                        r = f;
246                        z.availin--;
247                        z.totalin++;
248                        z.istate.need += (z.nextin[z.nextinindex++] & 0xffL);
249                        if (((int) (z.istate.was[0])) != ((int) (z.istate.need))) {
250                            z.istate.mode = BAD;
251                            z.msg = "incorrect data check";
252                            z.istate.marker = 5;       // can't try inflateSync
253                            break;
254                        }
255                        z.istate.mode = DONE;
256                        break;
257                    case DONE:
258                        return Z_STREAM_END;
259                    case BAD:
260                        return Z_DATA_ERROR;
261                    default:
262                        return Z_STREAM_ERROR;
263                }
264            }
265        }

267        protected int inflateSetDictionary(ZStream z, byte[] dictionary, int dictLength) {
268            int index = 0;
269            int length = dictLength;
270            if (z == null || z.istate == null || z.istate.mode != DICT0) {
271                return Z_STREAM_ERROR;
272            }
273    
274            if (Adler32.adler32(1L, dictionary, 0, dictLength) != z.adler) {
275                return Z_DATA_ERROR;
276            }
277    
278            z.adler = Adler32.adler32(0, null, 0, 0);
279    
280            if (length >= (1 << z.istate.wbits)) {
281                length = (1 << z.istate.wbits) - 1;
282                index = dictLength - length;
283            }
284            z.istate.blocks.setDictionary(dictionary, index, length);
285            z.istate.mode = BLOCKS;
286            return Z_OK;
287        }

289        private static byte[] mark = {(byte) 0, (byte) 0, (byte) 0xff, (byte) 0xff};
290    
291        protected int inflateSync(ZStream z) {
292            int n;       // number of bytes to look at
293            int p;       // pointer to bytes
294            int m;       // number of marker bytes found in a row
295            long r, w;   // temporaries to save total_in and total_out
296    
297            // set up
298            if (z == null || z.istate == null) {
299                return Z_STREAM_ERROR;
300            }
301            if (z.istate.mode != BAD) {
302                z.istate.mode = BAD;
303                z.istate.marker = 0;
304            }
305            if ((n = z.availin) == 0) {
306                return Z_BUF_ERROR;
307            }
308            p = z.nextinindex;
309            m = z.istate.marker;
310    
311            // search
312            while (n != 0 && m < 4) {
313                if (z.nextin[p] == mark[m]) {
314                    m++;
315                } else if (z.nextin[p] != 0) {
316                    m = 0;
317                } else {
318                    m = 4 - m;
319                }
320                p++;
321                n--;
322            }
323    
324            // restore
325            z.totalin += p - z.nextinindex;
326            z.nextinindex = p;
327            z.availin = n;
328            z.istate.marker = m;
329    
330            // return no joy or set up to restart on a new block
331            if (m != 4) {
332                return Z_DATA_ERROR;
333            }
334            r = z.totalin;
335            w = z.totalout;
336            inflateReset(z);
337            z.totalin = r;
338            z.totalout = w;
339            z.istate.mode = BLOCKS;
340            return Z_OK;
341        }
