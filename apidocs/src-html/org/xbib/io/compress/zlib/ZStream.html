002package org.xbib.io.compress.zlib;
003
004public class ZStream {
005
006    private final static int MAX_WBITS = 15;        // 32K LZ77 window
007
008    private final static int DEF_WBITS = MAX_WBITS;
009
010    private final static int Z_STREAM_ERROR = -2;
011
012    protected byte[] nextin;     // next input byte
013
014    protected int nextinindex;
015
016    protected int availin;       // number of bytes available at next_in
017
018    protected long totalin;      // total nb of input bytes read so far
019
020    protected byte[] nextout;    // next output byte should be put there
021
022    protected int nextoutindex;
023
024    protected int availout;      // remaining free space at next_out
025
026    protected long totalout;     // total nb of bytes output so far
027
028    protected String msg;
029
030    protected Deflate dstate;
031
032    protected Inflate istate;
033
034    protected int dataType; // best guess about the data type: ascii or binary
035
036    protected long adler;
037
038    public int inflateInit() {
039        return inflateInit(DEF_WBITS);
040    }
041
042    public int inflateInit(boolean nowrap) {
043        return inflateInit(DEF_WBITS, nowrap);
044    }
045
046    public int inflateInit(int w) {
047        return inflateInit(w, false);
048    }
049
050    public int inflateInit(int w, boolean nowrap) {
051        istate = new Inflate();
052        return istate.inflateInit(this, nowrap ? -w : w);
053    }
054
055    public int inflate(int f) {
056        if (istate == null) {
057            return Z_STREAM_ERROR;
058        }
059        return istate.inflate(this, f);
060    }
061
062    public int inflateEnd() {
063        if (istate == null) {
064            return Z_STREAM_ERROR;
065        }
066        int ret = istate.inflateEnd(this);
067        istate = null;
068        return ret;
069    }
070
071    public int inflateSync() {
072        if (istate == null) {
073            return Z_STREAM_ERROR;
074        }
075        return istate.inflateSync(this);
076    }
077
078    public int inflateSetDictionary(byte[] dictionary, int dictLength) {
079        if (istate == null) {
080            return Z_STREAM_ERROR;
081        }
082        return istate.inflateSetDictionary(this, dictionary, dictLength);
083    }
084
085    public int deflateInit(int level) {
086        return deflateInit(level, MAX_WBITS);
087    }
088
089    public int deflateInit(int level, boolean nowrap) {
090        return deflateInit(level, MAX_WBITS, nowrap);
091    }
092
093    public int deflateInit(int level, int bits) {
094        return deflateInit(level, bits, false);
095    }
096
097    public int deflateInit(int level, int bits, boolean nowrap) {
098        dstate = new Deflate();
099        return dstate.deflateInit(this, level, nowrap ? -bits : bits);
100    }
101
102    final public int deflate(int flush) {
103        if (dstate == null) {
104            return Z_STREAM_ERROR;
105        }
106        return dstate.deflate(this, flush);
107    }
108
109    final public int deflateEnd() {
110        if (dstate == null) {
111            return Z_STREAM_ERROR;
112        }
113        int ret = dstate.deflateEnd();
114        dstate = null;
115        return ret;
116    }
117
118    final public int deflateParams(int level, int strategy) {
119        if (dstate == null) {
120            return Z_STREAM_ERROR;
121        }
122        return dstate.deflateParams(this, level, strategy);
123    }
124
125    final public int deflateSetDictionary(byte[] dictionary, int dictLength) {
126        if (dstate == null) {
127            return Z_STREAM_ERROR;
128        }
129        return dstate.deflateSetDictionary(this, dictionary, dictLength);
130    }
131
132    // Flush as much pending output as possible. All deflate() output goes
133    // through this function so some applications may wish to modify it
134    // to avoid allocating a large strm->next_out buffer and copying into it.
135    // (See also read_buf()).
136    protected void flushPending() {
137        int len = dstate.getPending();
138
139        if (len > availout) {
140            len = availout;
141        }
142        if (len == 0) {
143            return;
144        }
145
146        System.arraycopy(dstate.getPendingBuf(), dstate.getPendingOut(),
147                nextout, nextoutindex, len);
148
149        nextoutindex += len;
150        dstate.setPendingOut(dstate.getPendingOut() + len);
151        totalout += len;
152        availout -= len;
153        dstate.setPending(dstate.getPending() - len);
154        if (dstate.getPending() == 0) {
155            dstate.setPendingOut(0);
156        }
157    }
158
159    // Read a new buffer from the current input stream, update the adler32
160    // and total number of bytes read.  All deflate() input goes through
161    // this function so some applications may wish to modify it to avoid
162    // allocating a large strm->next_in buffer and copying from it.
163    // (See also flush_pending()).
164    protected int readBuf(byte[] buf, int start, int size) {
165        int len = availin;
166
167        if (len > size) {
168            len = size;
169        }
170        if (len == 0) {
171            return 0;
172        }
173
174        availin -= len;
175
176        if (dstate.getNoHeader() == 0) {
177            adler = Adler32.adler32(adler, nextin, nextinindex, len);
178        }
179        System.arraycopy(nextin, nextinindex, buf, start, len);
180        nextinindex += len;
181        totalin += len;
182        return len;
183    }
184
185    public void free() {
186        nextin = null;
187        nextout = null;
188        msg = null;
189    }
190}
