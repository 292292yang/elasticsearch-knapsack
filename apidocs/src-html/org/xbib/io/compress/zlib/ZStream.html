033package org.xbib.io.compress.zlib;
034
035public class ZStream {
036
037    private final static int MAX_WBITS = 15;        // 32K LZ77 window
038    private final static int DEF_WBITS = MAX_WBITS;
039    private final static int Z_STREAM_ERROR = -2;
040    protected byte[] nextin;     // next input byte
041    protected int nextinindex;
042    protected int availin;       // number of bytes available at next_in
043    protected long totalin;      // total nb of input bytes read so far
044    protected byte[] nextout;    // next output byte should be put there
045    protected int nextoutindex;
046    protected int availout;      // remaining free space at next_out
047    protected long totalout;     // total nb of bytes output so far
048    protected String msg;
049    protected Deflate dstate;
050    protected Inflate istate;
051    protected int dataType; // best guess about the data type: ascii or binary
052    protected long adler;
053
054    final public int inflateInit() {
055        return inflateInit(DEF_WBITS);
056    }
057
058    final public int inflateInit(boolean nowrap) {
059        return inflateInit(DEF_WBITS, nowrap);
060    }
061
062    final public int inflateInit(int w) {
063        return inflateInit(w, false);
064    }
065
066    final public int inflateInit(int w, boolean nowrap) {
067        istate = new Inflate();
068        return istate.inflateInit(this, nowrap ? -w : w);
069    }
070
071    final public int inflate(int f) {
072        if (istate == null) {
073            return Z_STREAM_ERROR;
074        }
075        return istate.inflate(this, f);
076    }
077
078    final public int inflateEnd() {
079        if (istate == null) {
080            return Z_STREAM_ERROR;
081        }
082        int ret = istate.inflateEnd(this);
083        istate = null;
084        return ret;
085    }
086
087    final public int inflateSync() {
088        if (istate == null) {
089            return Z_STREAM_ERROR;
090        }
091        return istate.inflateSync(this);
092    }
093
094    final public int inflateSetDictionary(byte[] dictionary, int dictLength) {
095        if (istate == null) {
096            return Z_STREAM_ERROR;
097        }
098        return istate.inflateSetDictionary(this, dictionary, dictLength);
099    }
100
101    final public int deflateInit(int level) {
102        return deflateInit(level, MAX_WBITS);
103    }
104
105    final public int deflateInit(int level, boolean nowrap) {
106        return deflateInit(level, MAX_WBITS, nowrap);
107    }
108
109    final public int deflateInit(int level, int bits) {
110        return deflateInit(level, bits, false);
111    }
112
113    final public int deflateInit(int level, int bits, boolean nowrap) {
114        dstate = new Deflate();
115        return dstate.deflateInit(this, level, nowrap ? -bits : bits);
116    }
117
118    final public int deflate(int flush) {
119        if (dstate == null) {
120            return Z_STREAM_ERROR;
121        }
122        return dstate.deflate(this, flush);
123    }
124
125    final public int deflateEnd() {
126        if (dstate == null) {
127            return Z_STREAM_ERROR;
128        }
129        int ret = dstate.deflateEnd();
130        dstate = null;
131        return ret;
132    }
133
134    final public int deflateParams(int level, int strategy) {
135        if (dstate == null) {
136            return Z_STREAM_ERROR;
137        }
138        return dstate.deflateParams(this, level, strategy);
139    }
140
141    final public int deflateSetDictionary(byte[] dictionary, int dictLength) {
142        if (dstate == null) {
143            return Z_STREAM_ERROR;
144        }
145        return dstate.deflateSetDictionary(this, dictionary, dictLength);
146    }
147
148    // Flush as much pending output as possible. All deflate() output goes
149    // through this function so some applications may wish to modify it
150    // to avoid allocating a large strm->next_out buffer and copying into it.
151    // (See also read_buf()).
152    protected void flushPending() {
153        int len = dstate.getPending();
154
155        if (len > availout) {
156            len = availout;
157        }
158        if (len == 0) {
159            return;
160        }
161
162        System.arraycopy(dstate.getPendingBuf(), dstate.getPendingOut(),
163                nextout, nextoutindex, len);
164
165        nextoutindex += len;
166        dstate.setPendingOut(dstate.getPendingOut() + len);
167        totalout += len;
168        availout -= len;
169        dstate.setPending(dstate.getPending() - len);
170        if (dstate.getPending() == 0) {
171            dstate.setPendingOut(0);
172        }
173    }
174
175    // Read a new buffer from the current input stream, update the adler32
176    // and total number of bytes read.  All deflate() input goes through
177    // this function so some applications may wish to modify it to avoid
178    // allocating a large strm->next_in buffer and copying from it.
179    // (See also flush_pending()).
180    protected int readBuf(byte[] buf, int start, int size) {
181        int len = availin;
182
183        if (len > size) {
184            len = size;
185        }
186        if (len == 0) {
187            return 0;
188        }
189
190        availin -= len;
191
192        if (dstate.getNoHeader() == 0) {
193            adler = Adler32.adler32(adler, nextin, nextinindex, len);
194        }
195        System.arraycopy(nextin, nextinindex, buf, start, len);
196        nextinindex += len;
197        totalin += len;
198        return len;
199    }
200
201    public void free() {
202        nextin = null;
203        nextout = null;
204        msg = null;
205    }
206}
