002package org.xbib.io.compress.zlib;
003
004public class Adler32 {
005
006    // largest prime smaller than 65536
007    private static final int BASE = 65521;
008    // NMAX is the largest n such that 255n(n+1)/2 + (n+1)(BASE-1) <= 2^32-1
009    private static final int NMAX = 5552;
010
011    private Adler32() {
012    }
013
014    public static long adler32(long adler, byte[] buf, int ind, int len) {
015        if (buf == null) {
016            return 1L;
017        }
018        int index = ind;
019        long s1 = adler & 0xffff;
020        long s2 = (adler >> 16) & 0xffff;
021        int k;
022
023        while (len > 0) {
024            k = len < NMAX ? len : NMAX;
025            len -= k;
026            while (k >= 16) {
027                s1 += buf[index++] & 0xff;
028                s2 += s1;
029                s1 += buf[index++] & 0xff;
030                s2 += s1;
031                s1 += buf[index++] & 0xff;
032                s2 += s1;
033                s1 += buf[index++] & 0xff;
034                s2 += s1;
035                s1 += buf[index++] & 0xff;
036                s2 += s1;
037                s1 += buf[index++] & 0xff;
038                s2 += s1;
039                s1 += buf[index++] & 0xff;
040                s2 += s1;
041                s1 += buf[index++] & 0xff;
042                s2 += s1;
043                s1 += buf[index++] & 0xff;
044                s2 += s1;
045                s1 += buf[index++] & 0xff;
046                s2 += s1;
047                s1 += buf[index++] & 0xff;
048                s2 += s1;
049                s1 += buf[index++] & 0xff;
050                s2 += s1;
051                s1 += buf[index++] & 0xff;
052                s2 += s1;
053                s1 += buf[index++] & 0xff;
054                s2 += s1;
055                s1 += buf[index++] & 0xff;
056                s2 += s1;
057                s1 += buf[index++] & 0xff;
058                s2 += s1;
059                k -= 16;
060            }
061            if (k != 0) {
062                do {
063                    s1 += buf[index++] & 0xff;
064                    s2 += s1;
065                } while (--k != 0);
066            }
067            s1 %= BASE;
068            s2 %= BASE;
069        }
070        return (s2 << 16) | s1;
071    }
072}
