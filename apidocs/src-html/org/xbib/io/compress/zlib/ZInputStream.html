033package org.xbib.io.compress.zlib;
034
035import java.io.FilterInputStream;
036import java.io.IOException;
037import java.io.InputStream;
038
039public class ZInputStream extends FilterInputStream {
040
041    protected ZStream z = new ZStream();
042    protected int bufsize = 512;
043    protected int flush = ZConstants.Z_NO_FLUSH;
044    protected byte[] buf = new byte[bufsize],  buf1 = new byte[1];
045    protected boolean compress;
046
047    public ZInputStream(InputStream in) {
048        this(in, false);
049    }
050
051    public ZInputStream(InputStream in, boolean nowrap) {
052        super(in);
053        z.inflateInit(nowrap);
054        compress = false;
055        z.nextin = buf;
056        z.nextinindex = 0;
057        z.availin = 0;
058    }
059
060    public ZInputStream(InputStream in, int level) {
061        super(in);
062        z.deflateInit(level);
063        compress = true;
064        z.nextin = buf;
065        z.nextinindex = 0;
066        z.availin = 0;
067    }
068
069    /*public int available() throws IOException {
070    return inf.finished() ? 0 : 1;
071    }*/
072    @Override
073    public int read() throws IOException {
074        if (read(buf1, 0, 1) == -1) {
075            return (-1);
076        }
077        return (buf1[0] & 0xFF);
078    }
079    private boolean nomoreinput = false;
080
081    @Override
082    public int read(byte[] b, int off, int len) throws IOException {
083        if (len == 0) {
084            return (0);
085        }
086        int err;
087        z.nextout = b;
088        z.nextoutindex = off;
089        z.availout = len;
090        do {
091            if ((z.availin == 0) && (!nomoreinput)) { // if buffer is empty and more input is avaiable, refill it
092                z.nextinindex = 0;
093                z.availin = in.read(buf, 0, bufsize);//(bufsize<z.avail_out ? bufsize : z.avail_out));
094                if (z.availin == -1) {
095                    z.availin = 0;
096                    nomoreinput = true;
097                }
098            }
099            if (compress) {
100                err = z.deflate(flush);
101            } else {
102                err = z.inflate(flush);
103            }
104            if (nomoreinput && (err == ZConstants.Z_BUF_ERROR)) {
105                return (-1);
106            }
107            if (err != ZConstants.Z_OK && err != ZConstants.Z_STREAM_END) {
108                throw new IOException((compress ? "de" : "in") + "flating: " + z.msg);
109            }
110            if ((nomoreinput || err == ZConstants.Z_STREAM_END) && (z.availout == len)) {
111                return (-1);
112            }
113        } while (z.availout == len && err == ZConstants.Z_OK);
114        //System.err.print("("+(len-z.avail_out)+")");
115        return (len - z.availout);
116    }
117
118    public long skip(long n) throws IOException {
119        int len = 512;
120        if (n < len) {
121            len = (int) n;
122        }
123        byte[] tmp = new byte[len];
124        return ((long) read(tmp));
125    }
126
127    public int getFlushMode() {
128        return (flush);
129    }
130
131    public void setFlushMode(int flush) {
132        this.flush = flush;
133    }
134
135    /**
136     * Returns the total number of bytes input so far.
137     */
138    public long getTotalIn() {
139        return z.totalin;
140    }
141
142    /**
143     * Returns the total number of bytes output so far.
144     */
145    public long getTotalOut() {
146        return z.totalout;
147    }
148
149    public void close() throws IOException {
150        in.close();
151    }
152}
