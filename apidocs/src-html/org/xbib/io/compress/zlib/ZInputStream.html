002package org.xbib.io.compress.zlib;
003
004import java.io.FilterInputStream;
005import java.io.IOException;
006import java.io.InputStream;
007
008public class ZInputStream extends FilterInputStream {
009
010    protected ZStream z = new ZStream();
011    protected int flush = ZConstants.Z_NO_FLUSH;
012    protected byte[] buf;
013    protected byte[] buf1 = new byte[1];
014    protected boolean compress;
015    private int bufsize;
016
017    public ZInputStream(InputStream in) {
018        this(in, false);
019    }
020
021    public ZInputStream(InputStream in, boolean nowrap) {
022        super(in);
023        this.bufsize = 512;
024        this.buf = new byte[bufsize];
025        z.inflateInit(nowrap);
026        compress = false;
027        z.nextin = buf;
028        z.nextinindex = 0;
029        z.availin = 0;
030    }
031
032    public ZInputStream(InputStream in, int bufsize) {
033        super(in);
034        this.bufsize = bufsize;
035        this.buf = new byte[bufsize];
036        z.inflateInit(false);
037        compress = false;
038        z.nextin = buf;
039        z.nextinindex = 0;
040        z.availin = 0;
041    }
042
043    public void level(int level) {
044        z.deflateInit(level);
045    }
046
047    @Override
048    public int read() throws IOException {
049        if (read(buf1, 0, 1) == -1) {
050            return (-1);
051        }
052        return (buf1[0] & 0xFF);
053    }
054
055    private boolean nomoreinput = false;
056
057    @Override
058    public int read(byte[] b, int off, int len) throws IOException {
059        if (len == 0) {
060            return (0);
061        }
062        int err;
063        z.nextout = b;
064        z.nextoutindex = off;
065        z.availout = len;
066        do {
067            if ((z.availin == 0) && (!nomoreinput)) { // if buffer is empty and more input is avaiable, refill it
068                z.nextinindex = 0;
069                z.availin = in.read(buf, 0, bufsize);//(bufsize<z.avail_out ? bufsize : z.avail_out));
070                if (z.availin == -1) {
071                    z.availin = 0;
072                    nomoreinput = true;
073                }
074            }
075            if (compress) {
076                err = z.deflate(flush);
077            } else {
078                err = z.inflate(flush);
079            }
080            if (nomoreinput && (err == ZConstants.Z_BUF_ERROR)) {
081                return (-1);
082            }
083            if (err != ZConstants.Z_OK && err != ZConstants.Z_STREAM_END) {
084                throw new IOException((compress ? "de" : "in") + "flating: " + z.msg);
085            }
086            if ((nomoreinput || err == ZConstants.Z_STREAM_END) && (z.availout == len)) {
087                return (-1);
088            }
089        } while (z.availout == len && err == ZConstants.Z_OK);
090        //System.err.print("("+(len-z.avail_out)+")");
091        return (len - z.availout);
092    }
093
094    public long skip(long n) throws IOException {
095        int len = 512;
096        if (n < len) {
097            len = (int) n;
098        }
099        byte[] tmp = new byte[len];
100        return ((long) read(tmp));
101    }
102
103    public int getFlushMode() {
104        return (flush);
105    }
106
107    public void setFlushMode(int flush) {
108        this.flush = flush;
109    }
110
111    /**
112     * Returns the total number of bytes input so far.
113     */
114    public long getTotalIn() {
115        return z.totalin;
116    }
117
118    /**
119     * Returns the total number of bytes output so far.
120     */
121    public long getTotalOut() {
122        return z.totalout;
123    }
124
125    public void close() throws IOException {
126        in.close();
127    }
128}
