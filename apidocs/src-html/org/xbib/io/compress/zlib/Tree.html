078    // Mapping from a distance to a distance code. dist is the distance - 1 and
079    // must not have side effects. DIST_CODE[256] and DIST_CODE[257] are never
080    // used.
081    protected static int distanceCode(int dist) {
082        return ((dist) < 256 ? DIST_CODE[dist] : DIST_CODE[256 + ((dist) >>> 7)]);
083    }

089    // Compute the optimal bit lengths for a tree and update the total bit length
090    // for the current block.
091    // IN assertion: the fields freq and dad are set, heap[heap_max] and
092    //    above are the tree nodes sorted by increasing frequency.
093    // OUT assertions: the field len is set to the optimal bit length, the
094    //     array bl_count contains the frequencies for each bit length.
095    //     The length opt_len is updated; static_len is also updated if stree is
096    //     not null.
097    private void genBitLen(Deflate s) {
098        short[] tree = dynTree;
099        short[] stree = statDesc.statictree;
100        int[] extra = statDesc.extrabits;
101        int base = statDesc.extrabase;
102        int max_length = statDesc.maxlength;
103        int h;              // heap index
104        int n, m;           // iterate over the tree elements
105        int bits;           // bit length
106        int xbits;          // extra bits
107        short f;            // frequency
108        int overflow = 0;   // number of elements with bit length too large
109
110        for (bits = 0; bits <= MAX_BITS; bits++) {
111            s.blCount[bits] = 0;
112        }
113
114        // In a first pass, compute the optimal bit lengths (which may
115        // overflow in the case of the bit length tree).
116        tree[s.heap[s.heapMax] * 2 + 1] = 0; // root of the heap
117
118        for (h = s.heapMax + 1; h < HEAP_SIZE; h++) {
119            n = s.heap[h];
120            bits = tree[tree[n * 2 + 1] * 2 + 1] + 1;
121            if (bits > max_length) {
122                bits = max_length;
123                overflow++;
124            }
125            tree[n * 2 + 1] = (short) bits;
126            // We overwrite tree[n*2+1] which is no longer needed
127
128            if (n > maxCode) {
129                continue;  // not a leaf node
130            }
131            s.blCount[bits]++;
132            xbits = 0;
133            if (n >= base) {
134                xbits = extra[n - base];
135            }
136            f = tree[n * 2];
137            s.optLen += f * (bits + xbits);
138            if (stree != null) {
139                s.staticLen += f * (stree[n * 2 + 1] + xbits);
140            }
141        }
142        if (overflow == 0) {
143            return;
144        }
145
146        // This happens for example on obj2 and pic of the Calgary corpus
147        // Find the first bit length which could increase:
148        do {
149            bits = max_length - 1;
150            while (s.blCount[bits] == 0) {
151                bits--;
152            }
153            s.blCount[bits]--;      // move one leaf down the tree
154            s.blCount[bits + 1] += 2;   // move one overflow item as its brother
155            s.blCount[max_length]--;
156            // The brother of the overflow item also moves one step up,
157            // but this does not affect bl_count[max_length]
158            overflow -= 2;
159        } while (overflow > 0);
160
161        for (bits = max_length; bits != 0; bits--) {
162            n = s.blCount[bits];
163            while (n != 0) {
164                m = s.heap[--h];
165                if (m > maxCode) {
166                    continue;
167                }
168                if (tree[m * 2 + 1] != bits) {
169                    s.optLen += ((long) bits - (long) tree[m * 2 + 1]) * (long) tree[m * 2];
170                    tree[m * 2 + 1] = (short) bits;
171                }
172                n--;
173            }
174        }
175    }

177    // Construct one Huffman tree and assigns the code bit strings and lengths.
178    // Update the total bit length for the current block.
179    // IN assertion: the field freq is set for all tree elements.
180    // OUT assertions: the fields len and code are set to the optimal bit length
181    //     and corresponding code. The length opt_len is updated; static_len is
182    //     also updated if stree is not null. The field max_code is set.
183    protected void buildTree(Deflate s) {
184        short[] tree = dynTree;
185        short[] stree = statDesc.statictree;
186        int elems = statDesc.elems;
187        int n, m;          // iterate over heap elements
188        int maxCode = -1;   // largest code with non zero frequency
189        int node;          // new node being created
190
191        // Construct the initial heap, with least frequent element in
192        // heap[1]. The sons of heap[n] are heap[2*n] and heap[2*n+1].
193        // heap[0] is not used.
194        s.heapLen = 0;
195        s.heapMax = HEAP_SIZE;
196
197        for (n = 0; n < elems; n++) {
198            if (tree[n * 2] != 0) {
199                s.heap[++s.heapLen] = maxCode = n;
200                s.depth[n] = 0;
201            } else {
202                tree[n * 2 + 1] = 0;
203            }
204        }
205
206        // The pkzip format requires that at least one distance code exists,
207        // and that at least one bit should be sent even if there is only one
208        // possible code. So to avoid special checks later on we force at least
209        // two codes of non zero frequency.
210        while (s.heapLen < 2) {
211            node = s.heap[++s.heapLen] = (maxCode < 2 ? ++maxCode : 0);
212            tree[node * 2] = 1;
213            s.depth[node] = 0;
214            s.optLen--;
215            if (stree != null) {
216                s.staticLen -= stree[node * 2 + 1];
217            }
218            // node is 0 or 1 so it does not have extra bits
219        }
220        this.maxCode = maxCode;
221
222        // The elements heap[heap_len/2+1 .. heap_len] are leaves of the tree,
223        // establish sub-heaps of increasing lengths:
224
225        for (n = s.heapLen / 2; n >= 1; n--) {
226            s.pqdownheap(tree, n);
227        }
228
229        // Construct the Huffman tree by repeatedly combining the least two
230        // frequent nodes.
231
232        node = elems;                 // next internal node of the tree
233        do {
234            // n = node of least frequency
235            n = s.heap[1];
236            s.heap[1] = s.heap[s.heapLen--];
237            s.pqdownheap(tree, 1);
238            m = s.heap[1];                // m = node of next least frequency
239
240            s.heap[--s.heapMax] = n; // keep the nodes sorted by frequency
241            s.heap[--s.heapMax] = m;
242
243            // Create a new node father of n and m
244            tree[node * 2] = (short) (tree[n * 2] + tree[m * 2]);
245            s.depth[node] = (byte) (Math.max(s.depth[n], s.depth[m]) + 1);
246            tree[n * 2 + 1] = tree[m * 2 + 1] = (short) node;
247
248            // and insert the new node in the heap
249            s.heap[1] = node++;
250            s.pqdownheap(tree, 1);
251        } while (s.heapLen >= 2);
252
253        s.heap[--s.heapMax] = s.heap[1];
254
255        // At this point, the fields freq and dad are set. We can now
256        // generate the bit lengths.
257
258        genBitLen(s);
259
260        // The field len is now set, we can generate the bit codes
261        genCodes(tree, maxCode, s.blCount);
262    }
263
264    // Generate the codes for a given tree and bit counts (which need not be
265    // optimal).
266    // IN assertion: the array bl_count contains the bit length statistics for
267    // the given tree and the field len is set for all tree elements.
268    // OUT assertion: the field code is set for all tree elements of non
269    //     zero code length.
270    private void genCodes(short[] tree, // the tree to decorate
271                          int max_code, // largest code with non zero frequency
272                          short[] bl_count // number of codes at each bit length
273    ) {
274        short[] next_code = new short[MAX_BITS + 1]; // next code value for each bit length
275        short code = 0;            // running code value
276        int bits;                  // bit index
277        int n;                     // code index
278
279        // The distribution counts are first used to generate the code values
280        // without bit reversal.
281        for (bits = 1; bits <= MAX_BITS; bits++) {
282            next_code[bits] = code = (short) ((code + bl_count[bits - 1]) << 1);
283        }
284
285        // Check that the bit counts in bl_count are consistent. The last code
286        // must be all ones.
287        //Assert (code + bl_count[MAX_BITS]-1 == (1<<MAX_BITS)-1,
288        //        "inconsistent bit counts");
289        //Tracev((stderr,"\ngen_codes: max_code %d ", max_code));
290
291        for (n = 0; n <= max_code; n++) {
292            int len = tree[n * 2 + 1];
293            if (len == 0) {
294                continue;
295            }
296            // Now reverse the bits
297            tree[n * 2] = (short) (biReverse(next_code[len]++, len));
298        }
299    }
300
301    // Reverse the first len bits of a code, using straightforward code (a faster
302    // method would use a table)
303    // IN assertion: 1 <= len <= 15
304    private int biReverse(int code, // the value to invert
305                          int len // its bit length
306    ) {
307        int res = 0;
308        do {
309            res |= code & 1;
310            code >>>= 1;
311            res <<= 1;
312        } while (--len > 0);
313        return res >>> 1;
314    }
