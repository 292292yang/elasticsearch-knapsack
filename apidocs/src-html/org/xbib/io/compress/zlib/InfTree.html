217    private int huft_build(int[] b, // code lengths in bits (all assumed <= BMAX)
218            int bindex,
219            int n, // number of codes (assumed <= 288)
220            int s, // number of simple-valued codes (0..s-1)
221            int[] d, // list of base values for non-simple codes
222            int[] e, // list of extra bits for non-simple codes
223            int[] t, // result: starting table
224            int[] m, // maximum lookup bits, returns actual
225            int[] hp,// space for trees
226            int[] hn,// hufts used in space
227            int[] v // working area: values in order of bit length
228            ) {
229        // Given a list of code lengths and a maximum table size, make a set of
230        // tables to decode that set of codes.  Return Z_OK on success, Z_BUF_ERROR
231        // if the given code set is incomplete (the tables are still built in this
232        // case), Z_DATA_ERROR if the input is invalid (an over-subscribed set of
233        // lengths), or Z_MEM_ERROR if not enough memory.
234
235        int a;                       // counter for codes of length k
236        int f;                       // i repeats in table every f entries
237        int g;                       // maximum code length
238        int h;                       // table level
239        int i;                       // counter, current code
240        int j;                       // counter
241        int k;                       // number of bits in current code
242        int l;                       // bits per table (returned in m)
243        int mask;                    // (1 << w) - 1, to avoid cc -O bug on HP
244        int p;                       // pointer into c[], b[], or v[]
245        int q;                       // points to current table
246        int w;                       // bits before this table == (l * h)
247        int xp;                      // pointer into x
248        int y;                       // number of dummy codes added
249        int z;                       // number of entries in current table
250
251        // Generate counts for each bit length
252
253        p = 0;
254        i = n;
255        do {
256            c[b[bindex + p]]++;
257            p++;
258            i--;   // assume all entries <= BMAX
259        } while (i != 0);
260
261        if (c[0] == n) {                // null input--all zero length codes
262            t[0] = -1;
263            m[0] = 0;
264            return Z_OK;
265        }
266
267        // Find minimum and maximum length, bound *m by those
268        l = m[0];
269        for (j = 1; j <= BMAX; j++) {
270            if (c[j] != 0) {
271                break;
272            }
273        }
274        k = j;                        // minimum code length
275        if (l < j) {
276            l = j;
277        }
278        for (i = BMAX; i != 0; i--) {
279            if (c[i] != 0) {
280                break;
281            }
282        }
283        g = i;                        // maximum code length
284        if (l > i) {
285            l = i;
286        }
287        m[0] = l;
288
289        // Adjust last length count to fill out codes, if needed
290        for (y = 1 << j; j < i; j++, y <<= 1) {
291            if ((y -= c[j]) < 0) {
292                return Z_DATA_ERROR;
293            }
294        }
295        if ((y -= c[i]) < 0) {
296            return Z_DATA_ERROR;
297        }
298        c[i] += y;
299
300        // Generate starting offsets into the value table for each length
301        x[1] = j = 0;
302        p = 1;
303        xp = 2;
304        while (--i != 0) {                 // note that i == g from above
305            x[xp] = (j += c[p]);
306            xp++;
307            p++;
308        }
309
310        // Make a table of values in order of bit lengths
311        i = 0;
312        p = 0;
313        do {
314            if ((j = b[bindex + p]) != 0) {
315                v[x[j]++] = i;
316            }
317            p++;
318        } while (++i < n);
319        n = x[g];                     // set n to length of v
320
321        // Generate the Huffman codes and for each, make the table entries
322        x[0] = i = 0;                 // first Huffman code is zero
323        p = 0;                        // grab values in bit order
324        h = -1;                       // no tables yet--level -1
325        w = -l;                       // bits decoded == (l * h)
326        u[0] = 0;                     // just to keep compilers happy
327        q = 0;                        // ditto
328        z = 0;                        // ditto
329
330        // go through the bit lengths (k already is bits in shortest code)
331        for (; k <= g; k++) {
332            a = c[k];
333            while (a-- != 0) {
334                // here i is the Huffman code of length k bits for value *p
335                // make tables up to required level
336                while (k > w + l) {
337                    h++;
338                    w += l;                 // previous table always l bits
339                    // compute minimum size table less than or equal to l bits
340                    z = g - w;
341                    z = (z > l) ? l : z;        // table size upper limit
342                    if ((f = 1 << (j = k - w)) > a + 1) {     // try a k-w bit table
343                        // too few codes for k-w bit table
344                        f -= a + 1;               // deduct codes from patterns left
345                        xp = k;
346                        if (j < z) {
347                            while (++j < z) {        // try smaller tables up to z bits
348                                if ((f <<= 1) <= c[++xp]) {
349                                    break;              // enough codes to use up j bits
350                                }
351                                f -= c[xp];           // else deduct codes from patterns
352                            }
353                        }
354                    }
355                    z = 1 << j;                 // table entries for j-bit table
356
357                    // allocate new table
358                    if (hn[0] + z > MANY) {       // (note: doesn't matter for fixed)
359                        return Z_DATA_ERROR;       // overflow of MANY
360                    }
361                    u[h] = q = /*hp+*/ hn[0];   // DEBUG
362                    hn[0] += z;
363
364                    // connect to last table, if there is one
365                    if (h != 0) {
366                        x[h] = i;           // save pattern for backing up
367                        r[0] = (byte) j;     // bits in this table
368                        r[1] = (byte) l;     // bits to dump before this table
369                        j = i >>> (w - l);
370                        r[2] = (q - u[h - 1] - j);               // offset to this table
371                        System.arraycopy(r, 0, hp, (u[h - 1] + j) * 3, 3); // connect to last table
372                    } else {
373                        t[0] = q;               // first table is returned result
374                    }
375                }
376
377                // set up table entry in r
378                r[1] = (byte) (k - w);
379                if (p >= n) {
380                    r[0] = 128 + 64;      // out of values--invalid code
381                } else if (v[p] < s) {
382                    r[0] = (byte) (v[p] < 256 ? 0 : 32 + 64);  // 256 is end-of-block
383                    r[2] = v[p++];          // simple code is just the value
384                } else {
385                    r[0] = (byte) (e[v[p] - s] + 16 + 64); // non-simple--look up in lists
386                    r[2] = d[v[p++] - s];
387                }
388
389                // fill code-like entries with r
390                f = 1 << (k - w);
391                for (j = i >>> w; j < z; j += f) {
392                    System.arraycopy(r, 0, hp, (q + j) * 3, 3);
393                }
394
395                // backwards increment the k-bit code i
396                for (j = 1 << (k - 1); (i & j) != 0; j >>>= 1) {
397                    i ^= j;
398                }
399                i ^= j;
400
401                // backup over finished tables
402                mask = (1 << w) - 1;      // needed on HP, cc -O bug
403                while ((i & mask) != x[h]) {
404                    h--;                    // don't need to update q
405                    w -= l;
406                    mask = (1 << w) - 1;
407                }
408            }
409        }
410        // Return Z_BUF_ERROR if we were given an incomplete table
411        return y != 0 && g != 1 ? Z_BUF_ERROR : Z_OK;
412    }

414    protected int inflateTreesBits(int[] c, // 19 code lengths
415            int[] bb, // bits tree desired/actual depth
416            int[] tb, // bits tree result
417            int[] hp, // space for trees
418            ZStream z // for messages
419            ) {
420        int result;
421        initWorkArea(19);
422        hn[0] = 0;
423        result = huft_build(c, 0, 19, 19, null, null, tb, bb, hp, hn, v);
424
425        if (result == Z_DATA_ERROR) {
426            z.msg = "oversubscribed dynamic bit lengths tree";
427        } else if (result == Z_BUF_ERROR || bb[0] == 0) {
428            z.msg = "incomplete dynamic bit lengths tree";
429            result = Z_DATA_ERROR;
430        }
431        return result;
432    }

434    int inflate_trees_dynamic(int nl, // number of literal/length codes
435            int nd, // number of distance codes
436            int[] c, // that many (total) code lengths
437            int[] bl, // literal desired/actual bit depth
438            int[] bd, // distance desired/actual bit depth
439            int[] tl, // literal/length tree result
440            int[] td, // distance tree result
441            int[] hp, // space for trees
442            ZStream z // for messages
443            ) {
444        int result;
445
446        // build literal/length tree
447        initWorkArea(288);
448        hn[0] = 0;
449        result = huft_build(c, 0, nl, 257, cplens, cplext, tl, bl, hp, hn, v);
450        if (result != Z_OK || bl[0] == 0) {
451            if (result == Z_DATA_ERROR) {
452                z.msg = "oversubscribed literal/length tree";
453            } else if (result != Z_MEM_ERROR) {
454                z.msg = "incomplete literal/length tree";
455                result = Z_DATA_ERROR;
456            }
457            return result;
458        }
459
460        // build distance tree
461        initWorkArea(288);
462        result = huft_build(c, nl, nd, 0, cpdist, cpdext, td, bd, hp, hn, v);
463
464        if (result != Z_OK || (bd[0] == 0 && nl > 257)) {
465            if (result == Z_DATA_ERROR) {
466                z.msg = "oversubscribed distance tree";
467            } else if (result == Z_BUF_ERROR) {
468                z.msg = "incomplete distance tree";
469                result = Z_DATA_ERROR;
470            } else if (result != Z_MEM_ERROR) {
471                z.msg = "empty distance tree with lengths";
472                result = Z_DATA_ERROR;
473            }
474            return result;
475        }
476
477        return Z_OK;
478    }
