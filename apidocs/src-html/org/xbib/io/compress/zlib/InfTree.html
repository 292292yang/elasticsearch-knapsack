186        private int huft_build(int[] b, // code lengths in bits (all assumed <= BMAX)
187                               int bindex,
188                               int n, // number of codes (assumed <= 288)
189                               int s, // number of simple-valued codes (0..s-1)
190                               int[] d, // list of base values for non-simple codes
191                               int[] e, // list of extra bits for non-simple codes
192                               int[] t, // result: starting table
193                               int[] m, // maximum lookup bits, returns actual
194                               int[] hp,// space for trees
195                               int[] hn,// hufts used in space
196                               int[] v // working area: values in order of bit length
197        ) {
198            // Given a list of code lengths and a maximum table size, make a set of
199            // tables to decode that set of codes.  Return Z_OK on success, Z_BUF_ERROR
200            // if the given code set is incomplete (the tables are still built in this
201            // case), Z_DATA_ERROR if the input is invalid (an over-subscribed set of
202            // lengths), or Z_MEM_ERROR if not enough memory.
203    
204            int a;                       // counter for codes of length k
205            int f;                       // i repeats in table every f entries
206            int g;                       // maximum code length
207            int h;                       // table level
208            int i;                       // counter, current code
209            int j;                       // counter
210            int k;                       // number of bits in current code
211            int l;                       // bits per table (returned in m)
212            int mask;                    // (1 << w) - 1, to avoid cc -O bug on HP
213            int p;                       // pointer into c[], b[], or v[]
214            int q;                       // points to current table
215            int w;                       // bits before this table == (l * h)
216            int xp;                      // pointer into x
217            int y;                       // number of dummy codes added
218            int z;                       // number of entries in current table
219    
220            // Generate counts for each bit length
221    
222            p = 0;
223            i = n;
224            do {
225                c[b[bindex + p]]++;
226                p++;
227                i--;   // assume all entries <= BMAX
228            } while (i != 0);
229    
230            if (c[0] == n) {                // null input--all zero length codes
231                t[0] = -1;
232                m[0] = 0;
233                return Z_OK;
234            }
235    
236            // Find minimum and maximum length, bound *m by those
237            l = m[0];
238            for (j = 1; j <= BMAX; j++) {
239                if (c[j] != 0) {
240                    break;
241                }
242            }
243            k = j;                        // minimum code length
244            if (l < j) {
245                l = j;
246            }
247            for (i = BMAX; i != 0; i--) {
248                if (c[i] != 0) {
249                    break;
250                }
251            }
252            g = i;                        // maximum code length
253            if (l > i) {
254                l = i;
255            }
256            m[0] = l;
257    
258            // Adjust last length count to fill out codes, if needed
259            for (y = 1 << j; j < i; j++, y <<= 1) {
260                if ((y -= c[j]) < 0) {
261                    return Z_DATA_ERROR;
262                }
263            }
264            if ((y -= c[i]) < 0) {
265                return Z_DATA_ERROR;
266            }
267            c[i] += y;
268    
269            // Generate starting offsets into the value table for each length
270            x[1] = j = 0;
271            p = 1;
272            xp = 2;
273            while (--i != 0) {                 // note that i == g from above
274                x[xp] = (j += c[p]);
275                xp++;
276                p++;
277            }
278    
279            // Make a table of values in order of bit lengths
280            i = 0;
281            p = 0;
282            do {
283                if ((j = b[bindex + p]) != 0) {
284                    v[x[j]++] = i;
285                }
286                p++;
287            } while (++i < n);
288            n = x[g];                     // set n to length of v
289    
290            // Generate the Huffman codes and for each, make the table entries
291            x[0] = i = 0;                 // first Huffman code is zero
292            p = 0;                        // grab values in bit order
293            h = -1;                       // no tables yet--level -1
294            w = -l;                       // bits decoded == (l * h)
295            u[0] = 0;                     // just to keep compilers happy
296            q = 0;                        // ditto
297            z = 0;                        // ditto
298    
299            // go through the bit lengths (k already is bits in shortest code)
300            for (; k <= g; k++) {
301                a = c[k];
302                while (a-- != 0) {
303                    // here i is the Huffman code of length k bits for value *p
304                    // make tables up to required level
305                    while (k > w + l) {
306                        h++;
307                        w += l;                 // previous table always l bits
308                        // compute minimum size table less than or equal to l bits
309                        z = g - w;
310                        z = (z > l) ? l : z;        // table size upper limit
311                        if ((f = 1 << (j = k - w)) > a + 1) {     // try a k-w bit table
312                            // too few codes for k-w bit table
313                            f -= a + 1;               // deduct codes from patterns left
314                            xp = k;
315                            if (j < z) {
316                                while (++j < z) {        // try smaller tables up to z bits
317                                    if ((f <<= 1) <= c[++xp]) {
318                                        break;              // enough codes to use up j bits
319                                    }
320                                    f -= c[xp];           // else deduct codes from patterns
321                                }
322                            }
323                        }
324                        z = 1 << j;                 // table entries for j-bit table
325    
326                        // allocate new table
327                        if (hn[0] + z > MANY) {       // (note: doesn't matter for fixed)
328                            return Z_DATA_ERROR;       // overflow of MANY
329                        }
330                        u[h] = q = /*hp+*/ hn[0];   // DEBUG
331                        hn[0] += z;
332    
333                        // connect to last table, if there is one
334                        if (h != 0) {
335                            x[h] = i;           // save pattern for backing up
336                            r[0] = (byte) j;     // bits in this table
337                            r[1] = (byte) l;     // bits to dump before this table
338                            j = i >>> (w - l);
339                            r[2] = (q - u[h - 1] - j);               // offset to this table
340                            System.arraycopy(r, 0, hp, (u[h - 1] + j) * 3, 3); // connect to last table
341                        } else {
342                            t[0] = q;               // first table is returned result
343                        }
344                    }
345    
346                    // set up table entry in r
347                    r[1] = (byte) (k - w);
348                    if (p >= n) {
349                        r[0] = 128 + 64;      // out of values--invalid code
350                    } else if (v[p] < s) {
351                        r[0] = (byte) (v[p] < 256 ? 0 : 32 + 64);  // 256 is end-of-block
352                        r[2] = v[p++];          // simple code is just the value
353                    } else {
354                        r[0] = (byte) (e[v[p] - s] + 16 + 64); // non-simple--look up in lists
355                        r[2] = d[v[p++] - s];
356                    }
357    
358                    // fill code-like entries with r
359                    f = 1 << (k - w);
360                    for (j = i >>> w; j < z; j += f) {
361                        System.arraycopy(r, 0, hp, (q + j) * 3, 3);
362                    }
363    
364                    // backwards increment the k-bit code i
365                    for (j = 1 << (k - 1); (i & j) != 0; j >>>= 1) {
366                        i ^= j;
367                    }
368                    i ^= j;
369    
370                    // backup over finished tables
371                    mask = (1 << w) - 1;      // needed on HP, cc -O bug
372                    while ((i & mask) != x[h]) {
373                        h--;                    // don't need to update q
374                        w -= l;
375                        mask = (1 << w) - 1;
376                    }
377                }
378            }
379            // Return Z_BUF_ERROR if we were given an incomplete table
380            return y != 0 && g != 1 ? Z_BUF_ERROR : Z_OK;
381        }

383        protected int inflateTreesBits(int[] c, // 19 code lengths
384                                       int[] bb, // bits tree desired/actual depth
385                                       int[] tb, // bits tree result
386                                       int[] hp, // space for trees
387                                       ZStream z // for messages
388        ) {
389            int result;
390            initWorkArea(19);
391            hn[0] = 0;
392            result = huft_build(c, 0, 19, 19, null, null, tb, bb, hp, hn, v);
393    
394            if (result == Z_DATA_ERROR) {
395                z.msg = "oversubscribed dynamic bit lengths tree";
396            } else if (result == Z_BUF_ERROR || bb[0] == 0) {
397                z.msg = "incomplete dynamic bit lengths tree";
398                result = Z_DATA_ERROR;
399            }
400            return result;
401        }

403        int inflate_trees_dynamic(int nl, // number of literal/length codes
404                                  int nd, // number of distance codes
405                                  int[] c, // that many (total) code lengths
406                                  int[] bl, // literal desired/actual bit depth
407                                  int[] bd, // distance desired/actual bit depth
408                                  int[] tl, // literal/length tree result
409                                  int[] td, // distance tree result
410                                  int[] hp, // space for trees
411                                  ZStream z // for messages
412        ) {
413            int result;
414    
415            // build literal/length tree
416            initWorkArea(288);
417            hn[0] = 0;
418            result = huft_build(c, 0, nl, 257, cplens, cplext, tl, bl, hp, hn, v);
419            if (result != Z_OK || bl[0] == 0) {
420                if (result == Z_DATA_ERROR) {
421                    z.msg = "oversubscribed literal/length tree";
422                } else if (result != Z_MEM_ERROR) {
423                    z.msg = "incomplete literal/length tree";
424                    result = Z_DATA_ERROR;
425                }
426                return result;
427            }
428    
429            // build distance tree
430            initWorkArea(288);
431            result = huft_build(c, nl, nd, 0, cpdist, cpdext, td, bd, hp, hn, v);
432    
433            if (result != Z_OK || (bd[0] == 0 && nl > 257)) {
434                if (result == Z_DATA_ERROR) {
435                    z.msg = "oversubscribed distance tree";
436                } else if (result == Z_BUF_ERROR) {
437                    z.msg = "incomplete distance tree";
438                    result = Z_DATA_ERROR;
439                } else if (result != Z_MEM_ERROR) {
440                    z.msg = "empty distance tree with lengths";
441                    result = Z_DATA_ERROR;
442                }
443                return result;
444            }
445    
446            return Z_OK;
447        }
