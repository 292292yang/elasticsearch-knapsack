033package org.xbib.io.compress.zlib;
034
035import java.io.IOException;
036import java.io.OutputStream;
037
038public class ZOutputStream extends OutputStream {
039
040    protected ZStream z = new ZStream();
041    protected int bufsize = 512;
042    protected int flush = ZConstants.Z_NO_FLUSH;
043    protected byte[] buf = new byte[bufsize],  buf1 = new byte[1];
044    protected boolean compress;
045    protected OutputStream out;
046
047    public ZOutputStream(OutputStream out) {
048        super();
049        this.out = out;
050        z.inflateInit();
051        compress = false;
052    }
053
054    public ZOutputStream(OutputStream out, int level) {
055        this(out, level, false);
056    }
057
058    public ZOutputStream(OutputStream out, int level, boolean nowrap) {
059        super();
060        this.out = out;
061        z.deflateInit(level, nowrap);
062        compress = true;
063    }
064
065    public void write(int b) throws IOException {
066        buf1[0] = (byte) b;
067        write(buf1, 0, 1);
068    }
069
070    @Override
071    public void write(byte b[], int off, int len) throws IOException {
072        if (len == 0) {
073            return;
074        }
075        int err;
076        z.nextin = b;
077        z.nextinindex = off;
078        z.availin = len;
079        do {
080            z.nextout = buf;
081            z.nextoutindex = 0;
082            z.availout = bufsize;
083            if (compress) {
084                err = z.deflate(flush);
085            } else {
086                err = z.inflate(flush);
087            }
088            if (err != ZConstants.Z_OK) {
089                throw new IOException((compress ? "de" : "in") + "flating: " + z.msg);
090            }
091            out.write(buf, 0, bufsize - z.availout);
092        } while (z.availin > 0 || z.availout == 0);
093    }
094
095    public int getFlushMode() {
096        return (flush);
097    }
098
099    public void setFlushMode(int flush) {
100        this.flush = flush;
101    }
102
103    public void finish() throws IOException {
104        int err;
105        do {
106            z.nextout = buf;
107            z.nextoutindex = 0;
108            z.availout = bufsize;
109            if (compress) {
110                err = z.deflate(ZConstants.Z_FINISH);
111            } else {
112                err = z.inflate(ZConstants.Z_FINISH);
113            }
114            if (err != ZConstants.Z_STREAM_END && err != ZConstants.Z_OK) {
115                throw new IOException((compress ? "de" : "in") + "flating: " + z.msg);
116            }
117            if (bufsize - z.availout > 0) {
118                out.write(buf, 0, bufsize - z.availout);
119            }
120        } while (z.availin > 0 || z.availout == 0);
121        flush();
122    }
123
124    public void end() {
125        if (z == null) {
126            return;
127        }
128        if (compress) {
129            z.deflateEnd();
130        } else {
131            z.inflateEnd();
132        }
133        z.free();
134        z = null;
135    }
136
137    @Override
138    public void close() throws IOException {
139        try {
140            try {
141                finish();
142            } catch (IOException ignored) {
143            }
144        } finally {
145            end();
146            out.close();
147            out = null;
148        }
149    }
150
151    /**
152     * Returns the total number of bytes input so far.
153     */
154    public long getTotalIn() {
155        return z.totalin;
156    }
157
158    /**
159     * Returns the total number of bytes output so far.
160     */
161    public long getTotalOut() {
162        return z.totalout;
163    }
164
165    @Override
166    public void flush() throws IOException {
167        out.flush();
168    }
169}
