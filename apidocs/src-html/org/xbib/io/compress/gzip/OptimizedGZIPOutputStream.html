001package org.xbib.io.compress.gzip;
002
003import java.io.IOException;
004import java.io.OutputStream;
005import java.util.zip.CRC32;
006import java.util.zip.Deflater;
007import java.util.zip.DeflaterOutputStream;
008
009/**
010 * Optimized variant of {@link java.util.zip.GZIPOutputStream} that reuses
011 * underlying {@link java.util.zip.Deflater} instance}.
012 */
013public class OptimizedGZIPOutputStream extends OutputStream {
014
015    /**
016     * GZIP header magic number; written out LSB like most everything else (i.e.
017     * as 0x1f 0x8b)
018     */
019    private final static int GZIP_MAGIC = 0x8b1f;
020    /**
021     * For now, static header seems fine, since JDK default gzip writer does it
022     * too:
023     */
024    final static byte[] DEFAULT_HEADER = new byte[]{
025        (byte) GZIP_MAGIC, // Magic number (short)
026        (byte) (GZIP_MAGIC >> 8), // Magic number (short)
027        Deflater.DEFLATED, // Compression method (CM)
028        0, // Flags (FLG)
029        0, // Modification time MTIME (int)
030        0, // Modification time MTIME (int)
031        0, // Modification time MTIME (int)
032        0, // Modification time MTIME (int)
033        0, // Extra flags (XFLG)
034        (byte) 0xff // Operating system (OS), UNKNOWN
035    };
036
037    /*
038     * Helper objects
039     */
040    protected Deflater _deflater;
041    protected final GZIPRecycler _gzipRecycler;
042    protected final byte[] _eightByteBuffer = new byte[8];
043
044    /***
045     * Underlying output stream that header, compressed content and footer go to
046     */
047    protected OutputStream _rawOut;
048    // TODO: write this out, not strictly needed...
049    protected DeflaterOutputStream _deflaterOut;
050    protected CRC32 _crc;
051
052    /*
053     * Construction
054     */
055    public OptimizedGZIPOutputStream(OutputStream out) throws IOException {
056        super();
057        _gzipRecycler = GZIPRecycler.instance();
058        _rawOut = out;
059        // write header:
060        _rawOut.write(DEFAULT_HEADER);
061        _deflater = _gzipRecycler.allocDeflater();
062        _deflaterOut = new DeflaterOutputStream(_rawOut, _deflater, 512);
063        _crc = new CRC32();
064    }
065
066    /*
067     * OutputStream implementation
068     */
069    @Override
070    public void close() throws IOException {
071        _deflaterOut.finish();
072        _writeTrailer(_rawOut);
073        _rawOut.close();
074        Deflater d = _deflater;
075        if (d != null) {
076            _deflater = null;
077            _gzipRecycler.releaseDeflater(d);
078        }
079    }
080
081    @Override
082    public void flush() throws IOException {
083        _deflaterOut.flush();
084    }
085
086    @Override
087    public final void write(byte[] buf) throws IOException {
088        write(buf, 0, buf.length);
089    }
090
091    @Override
092    public final void write(int c) throws IOException {
093        _eightByteBuffer[0] = (byte) c;
094        write(_eightByteBuffer, 0, 1);
095    }
096
097    @Override
098    public void write(byte[] buf, int off, int len) throws IOException {
099        _deflaterOut.write(buf, off, len);
100        _crc.update(buf, off, len);
101    }
102
103    /*
104     ///////////////////////////////////////////////////////////////////////
105     // Internal methods
106     ///////////////////////////////////////////////////////////////////////
107     */
108    private void _writeTrailer(OutputStream out) throws IOException {
109        _putInt(_eightByteBuffer, 0, (int) _crc.getValue());
110        _putInt(_eightByteBuffer, 4, _deflater.getTotalIn());
111        out.write(_eightByteBuffer, 0, 8);
112    }
113
114    /**
115     * Stupid GZIP, writes stuff in wrong order (not network, but x86)
116     */
117    private final static void _putInt(byte[] buf, int offset, int value) {
118        buf[offset++] = (byte) (value);
119        buf[offset++] = (byte) (value >> 8);
120        buf[offset++] = (byte) (value >> 16);
121        buf[offset] = (byte) (value >> 24);
122    }
123}
