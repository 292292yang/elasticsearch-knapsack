001package org.xbib.io.compress.gzip;
002
003import java.lang.ref.SoftReference;
004import java.util.zip.Deflater;
005import java.util.zip.Inflater;
006
007/**
008 * GZIP-codec-specific "extension" to {@link com.ning.compress.BufferRecycler},
009 * used for recycling expensive objects.
010 *
011 * @author Tatu Saloranta (tatu.saloranta@iki.fi)
012 */
013public final class GZIPRecycler {
014
015    protected final static ThreadLocal<SoftReference<GZIPRecycler>> _recyclerRef = new ThreadLocal();
016    protected Inflater _inflater;
017    protected Deflater _deflater;
018
019    /**
020     * Accessor to get thread-local recycler instance
021     */
022    public static GZIPRecycler instance() {
023        SoftReference<GZIPRecycler> ref = _recyclerRef.get();
024        GZIPRecycler br = (ref == null) ? null : ref.get();
025        if (br == null) {
026            br = new GZIPRecycler();
027            _recyclerRef.set(new SoftReference(br));
028        }
029        return br;
030    }
031
032    /*
033     * API
034     */
035    public Deflater allocDeflater() {
036        Deflater d = _deflater;
037        if (d == null) { // important: true means 'dont add zlib header'; gzip has its own
038            d = new Deflater(Deflater.DEFAULT_COMPRESSION, true);
039        } else {
040            _deflater = null;
041        }
042        return d;
043    }
044
045    public void releaseDeflater(Deflater d) {
046        if (d != null) {
047            d.reset();
048            _deflater = d;
049        }
050    }
051
052    public Inflater allocInflater() {
053        Inflater i = _inflater;
054        if (i == null) { // important: true means 'dont add zlib header'; gzip has its own
055            i = new Inflater(true);
056        } else {
057            _inflater = null;
058        }
059        return i;
060    }
061
062    public void releaseInflater(Inflater i) {
063        if (i != null) {
064            i.reset();
065            _inflater = i;
066        }
067    }
068}
