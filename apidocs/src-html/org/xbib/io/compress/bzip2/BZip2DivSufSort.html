084        final int[] bucketA = new int[BZip2DivSufSort.BUCKET_A_SIZE];

086        if (n == 0) {
087            return 0;
088        } else if (n == 1) {
089            SA[0] = T[0];

092        int m = sortTypeBstar(bucketA, bucketB);
093        if (0 < m) {
094            return constructBWT(bucketA, bucketB);
095        }
096        return 0;

105    private int ssCompare(final int p1, final int p2, final int depth) {
106        final int[] SA = this.SA;
107        final byte[] T = this.T;
108        final int U1n, U2n; // pointers within T
109        int U1, U2;
110        for (U1 = depth + SA[p1], U2 = depth + SA[p2], U1n = SA[p1 + 1] + 2, U2n = SA[p2 + 1] + 2;
111             (U1 < U1n) && (U2 < U2n) && (T[U1] == T[U2]);
112             ++U1, ++U2
113                ) {
114            ;
115        }
116        return U1 < U1n ?
117                (U2 < U2n ? (T[U1] & 0xff) - (T[U2] & 0xff) : 1)
118                : (U2 < U2n ? -1 : 0);
119    }

121    private int ssCompareLast(int PA, int p1, int p2, int depth, int size) {
122        final int[] SA = this.SA;
123        final byte[] T = this.T;
124        int U1, U2, U1n, U2n;
125        for (U1 = depth + SA[p1], U2 = depth + SA[p2], U1n = size, U2n = SA[(p2 + 1)] + 2;
126             (U1 < U1n) && (U2 < U2n) && (T[U1] == T[U2]);
127             ++U1, ++U2
128                ) {
129            ;
130        }
131        if (U1 < U1n) {
132            return (U2 < U2n) ? (T[U1] & 0xff) - (T[U2] & 0xff) : 1;
133        } else if (U2 == U2n) {
134            return 1;
135        }
136        for (U1 = U1 % size, U1n = SA[PA] + 2;
137             (U1 < U1n) && (U2 < U2n) && (T[U1] == T[U2]);
138             ++U1, ++U2
139                ) {
140            ;
141        }
142        return U1 < U1n ?
143                (U2 < U2n ? (T[U1] & 0xff) - (T[U2] & 0xff) : 1)
144                : (U2 < U2n ? -1 : 0);
145    }

147    private void ssInsertionSort(int PA, int first, int last, int depth) {
148        final int[] SA = this.SA;
149        int i, j; // pointer within SA
150        int t;
151        int r;
152        for (i = last - 2; first <= i; --i) {
153            for (t = SA[i], j = i + 1; 0 < (r = ssCompare(PA + t, PA + SA[j], depth)); ) {
154                do {
155                    SA[j - 1] = SA[j];
156                } while ((++j < last) && (SA[j] < 0));
157                if (last <= j) {
158                    break;
159                }
160            }
161            if (r == 0) {
162                SA[j] = ~SA[j];
163            }
164            SA[j - 1] = t;
165        }
166    }

168    private void ssFixdown(int Td, int PA, int sa, int i, int size) {
169        final int[] SA = this.SA;
170        final byte[] T = this.T;
171        int j, k;
172        int v;
173        int c, d, e;
174        for (v = SA[sa + i], c = (T[Td + SA[PA + v]]) & 0xff; (j = 2 * i + 1) < size; SA[sa + i] = SA[sa + k], i = k) {
175            d = T[Td + SA[PA + SA[sa + (k = j++)]]] & 0xff;
176            if (d < (e = T[Td + SA[PA + SA[sa + j]]] & 0xff)) {
177                k = j;
178                d = e;
179            }
180            if (d <= c) {
181                break;
182            }
183        }
184        SA[sa + i] = v;
185    }

187    private void ssHeapSort(int Td, int PA, int sa, int size) {
188        final int[] SA = this.SA;
189        final byte[] T = this.T;
190        int i, m;
191        int t;
192        m = size;
193        if ((size % 2) == 0) {
194            m--;
195            if ((T[Td + SA[PA + SA[sa + (m / 2)]]] & 0xff) < (T[Td + SA[PA + SA[sa + m]]] & 0xff)) {
196                swapElements(SA, sa + m, SA, sa + (m / 2));
197            }
198        }
199        for (i = m / 2 - 1; 0 <= i; --i) {
200            ssFixdown(Td, PA, sa, i, m);
201        }
202        if ((size % 2) == 0) {
203            swapElements(SA, sa, SA, sa + m);
204            ssFixdown(Td, PA, sa, 0, m);
205        }
206        for (i = m - 1; 0 < i; --i) {
207            t = SA[sa];
208            SA[sa] = SA[sa + i];
209            ssFixdown(Td, PA, sa, 0, i);
210            SA[sa + i] = t;
211        }
212    }

214    private int ssMedian3(final int Td, final int PA, int v1, int v2, int v3) {
215        final int[] SA = this.SA;
216        final byte[] T = this.T;
217        int T_v1 = T[Td + SA[PA + SA[v1]]] & 0xff;
218        int T_v2 = T[Td + SA[PA + SA[v2]]] & 0xff;
219        int T_v3 = T[Td + SA[PA + SA[v3]]] & 0xff;
220        if (T_v1 > T_v2) {
221            final int temp = v1;
222            v1 = v2;
223            v2 = temp;
224            final int T_vtemp = T_v1;
225            T_v1 = T_v2;
226            T_v2 = T_vtemp;
227        }
228        if (T_v2 > T_v3) {
229            if (T_v1 > T_v3) {
230                return v1;
231            }
232            return v3;
233        }
234        return v2;
235    }

305    private int ssPivot(final int Td, final int PA, final int first, final int last) {
306        int middle;
307        int t;
308        t = last - first;
309        middle = first + t / 2;
310        if (t <= 512) {
311            if (t <= 32) {
312                return ssMedian3(Td, PA, first, middle, last - 1);
313            }
314            t >>= 2;
315            return ssMedian5(Td, PA, first, first + t, middle, last - 1 - t, last - 1);
316        }
317        t >>= 3;
318        return ssMedian3(
319                Td, PA,
320                ssMedian3(Td, PA, first, first + t, first + (t << 1)),
321                ssMedian3(Td, PA, middle - t, middle, middle + t),
322                ssMedian3(Td, PA, last - 1 - (t << 1), last - 1 - t, last - 1)
323        );
324    }

326    private int ssLog(final int n) {
327        return ((n & 0xff00) != 0) ?
328                8 + log2table[(n >> 8) & 0xff]
329                : log2table[n & 0xff];

332    private int ssSubstringPartition(final int PA, final int first, final int last, final int depth) {
333        final int[] SA = this.SA;
334        int a, b;
335        int t;
336        for (a = first - 1, b = last; ; ) {
337            for (; (++a < b) && ((SA[PA + SA[a]] + depth) >= (SA[PA + SA[a] + 1] + 1)); ) {
338                SA[a] = ~SA[a];
339            }
340            for (; (a < --b) && ((SA[PA + SA[b]] + depth) < (SA[PA + SA[b] + 1] + 1)); ) {
341                ;
342            }
343            if (b <= a) {
344                break;
345            }
346            t = ~SA[b];
347            SA[b] = SA[a];
348            SA[a] = t;
349        }
350        if (first < a) {
351            SA[first] = ~SA[first];
352        }
353        return a;
354    }

372    private void ssMultiKeyIntroSort(final int PA, int first, int last, int depth) {
373        final int[] SA = this.SA;
374        final byte[] T = this.T;
375        final StackEntry[] stack = new StackEntry[STACK_SIZE];
376        int Td = 0;
377        int a = 0, b = 0, c = 0, d = 0, e = 0, f = 0;
378        int s = 0, t = 0;
379        int ssize;
380        int limit;
381        int v = 0, x = 0;
382        for (ssize = 0, limit = ssLog(last - first); ; ) {
383            if ((last - first) <= INSERTIONSORT_THRESHOLD) {
384                if (1 < (last - first)) {
385                    ssInsertionSort(PA, first, last, depth);
386                }
387                if (ssize == 0) {
388                    return;
389                }
390                StackEntry entry = stack[--ssize];
391                first = entry.a;
392                last = entry.b;
393                depth = entry.c;
394                limit = entry.d;
395                continue;
396            }
397            Td = depth;
398            if (limit-- == 0) {
399                ssHeapSort(Td, PA, first, last - first);
400            }
401            if (limit < 0) {
402                for (a = first + 1, v = T[Td + SA[PA + SA[first]]] & 0xff; a < last; ++a) {
403                    if ((x = (T[Td + SA[PA + SA[a]]] & 0xff)) != v) {
404                        if (1 < (a - first)) {
405                            break;
406                        }
407                        v = x;
408                        first = a;
409                    }
410                }
411                if ((T[Td + SA[PA + SA[first]] - 1] & 0xff) < v) {
412                    first = ssSubstringPartition(PA, first, a, depth);
413                }
414                if ((a - first) <= (last - a)) {
415                    if (1 < (a - first)) {
416                        stack[ssize++] = new StackEntry(a, last, depth, -1);
417                        last = a;
418                        depth += 1;
419                        limit = ssLog(a - first);
420                    } else {
421                        first = a;
422                        limit = -1;
423                    }
424                } else {
425                    if (1 < (last - a)) {
426                        stack[ssize++] = new StackEntry(first, a, depth + 1, ssLog(a - first));
427                        first = a;
428                        limit = -1;
429                    } else {
430                        last = a;
431                        depth += 1;
432                        limit = ssLog(a - first);
433                    }
434                }
435                continue;
436            }
437            a = ssPivot(Td, PA, first, last);
438            v = T[Td + SA[PA + SA[a]]] & 0xff;
439            swapElements(SA, first, SA, a);
440            for (b = first; (++b < last) && ((x = (T[Td + SA[PA + SA[b]]] & 0xff)) == v); ) {
441                ;
442            }
443            if (((a = b) < last) && (x < v)) {
444                for (; (++b < last) && ((x = (T[Td + SA[PA + SA[b]]] & 0xff)) <= v); ) {
445                    if (x == v) {
446                        swapElements(SA, b, SA, a);
447                        ++a;
448                    }
449                }
450            }
451            for (c = last; (b < --c) && ((x = (T[Td + SA[PA + SA[c]]] & 0xff)) == v); ) {
452                ;
453            }
454            if ((b < (d = c)) && (x > v)) {
455                for (; (b < --c) && ((x = (T[Td + SA[PA + SA[c]]] & 0xff)) >= v); ) {
456                    if (x == v) {
457                        swapElements(SA, c, SA, d);
458                        --d;
459                    }
460                }
461            }
462            for (; b < c; ) {
463                swapElements(SA, b, SA, c);
464                for (; (++b < c) && ((x = (T[Td + SA[PA + SA[b]]] & 0xff)) <= v); ) {
465                    if (x == v) {
466                        swapElements(SA, b, SA, a);
467                        ++a;
468                    }
469                }
470                for (; (b < --c) && ((x = (T[Td + SA[PA + SA[c]]] & 0xff)) >= v); ) {
471                    if (x == v) {
472                        swapElements(SA, c, SA, d);
473                        --d;
474                    }
475                }
476            }
477            if (a <= d) {
478                c = b - 1;
479                if ((s = a - first) > (t = b - a)) {
480                    s = t;
481                }
482                for (e = first, f = b - s; 0 < s; --s, ++e, ++f) {
483                    swapElements(SA, e, SA, f);
484                }
485                if ((s = d - c) > (t = last - d - 1)) {
486                    s = t;
487                }
488                for (e = b, f = last - s; 0 < s; --s, ++e, ++f) {
489                    swapElements(SA, e, SA, f);
490                }
491                a = first + (b - a);
492                c = last - (d - c);
493                b = (v <= (T[Td + SA[PA + SA[a]] - 1] & 0xff)) ? a : ssSubstringPartition(PA, a, c, depth);
494
495                if ((a - first) <= (last - c)) {
496                    if ((last - c) <= (c - b)) {
497                        stack[ssize++] = new StackEntry(b, c, depth + 1, ssLog(c - b));
498                        stack[ssize++] = new StackEntry(c, last, depth, limit);
499                        last = a;
500                    } else if ((a - first) <= (c - b)) {
501                        stack[ssize++] = new StackEntry(c, last, depth, limit);
502                        stack[ssize++] = new StackEntry(b, c, depth + 1, ssLog(c - b));
503                        last = a;
504                    } else {
505                        stack[ssize++] = new StackEntry(c, last, depth, limit);
506                        stack[ssize++] = new StackEntry(first, a, depth, limit);
507                        first = b;
508                        last = c;
509                        depth += 1;
510                        limit = ssLog(c - b);
511                    }
512                } else {
513                    if ((a - first) <= (c - b)) {
514                        stack[ssize++] = new StackEntry(b, c, depth + 1, ssLog(c - b));
515                        stack[ssize++] = new StackEntry(first, a, depth, limit);
516                        first = c;
517                    } else if ((last - c) <= (c - b)) {
518                        stack[ssize++] = new StackEntry(first, a, depth, limit);
519                        stack[ssize++] = new StackEntry(b, c, depth + 1, ssLog(c - b));
520                        first = c;
521                    } else {
522                        stack[ssize++] = new StackEntry(first, a, depth, limit);
523                        stack[ssize++] = new StackEntry(c, last, depth, limit);
524                        first = b;
525                        last = c;
526                        depth += 1;
527                        limit = ssLog(c - b);
528                    }
529                }
530            } else {
531                limit += 1;
532                if ((T[Td + SA[PA + SA[first]] - 1] & 0xff) < v) {
533                    first = ssSubstringPartition(PA, first, last, depth);
534                    limit = ssLog(last - first);
535                }
536                depth += 1;
537            }
538        }
539    }

549    private void ssMergeForward(final int PA, int[] buf, final int bufoffset, final int first, final int middle, final int last, final int depth) {
550        final int[] SA = this.SA;
551        int bufend;
552        int i, j, k;
553        int t;
554        int r;
555        bufend = bufoffset + (middle - first) - 1;
556        ssBlockSwap(buf, bufoffset, SA, first, middle - first);
557        for (t = SA[first], i = first, j = bufoffset, k = middle; ; ) {
558            r = ssCompare(PA + buf[j], PA + SA[k], depth);
559            if (r < 0) {
560                do {
561                    SA[i++] = buf[j];
562                    if (bufend <= j) {
563                        buf[j] = t;
564                        return;
565                    }
566                    buf[j++] = SA[i];
567                } while (buf[j] < 0);
568            } else if (r > 0) {
569                do {
570                    SA[i++] = SA[k];
571                    SA[k++] = SA[i];
572                    if (last <= k) {
573                        while (j < bufend) {
574                            SA[i++] = buf[j];
575                            buf[j++] = SA[i];
576                        }
577                        SA[i] = buf[j];
578                        buf[j] = t;
579                        return;
580                    }
581                } while (SA[k] < 0);
582            } else {
583                SA[k] = ~SA[k];
584                do {
585                    SA[i++] = buf[j];
586                    if (bufend <= j) {
587                        buf[j] = t;
588                        return;
589                    }
590                    buf[j++] = SA[i];
591                } while (buf[j] < 0);
592
593                do {
594                    SA[i++] = SA[k];
595                    SA[k++] = SA[i];
596                    if (last <= k) {
597                        while (j < bufend) {
598                            SA[i++] = buf[j];
599                            buf[j++] = SA[i];
600                        }

745    private void ssMerge(final int PA, int first, int middle, int last, int[] buf, final int bufoffset, final int bufsize, final int depth) {
746        final int[] SA = this.SA;
747        final StackEntry[] stack = new StackEntry[STACK_SIZE];
748        int i, j;
749        int m, len, half;
750        int ssize;
751        int check, next;
752        for (check = 0, ssize = 0; ; ) {
753            if ((last - middle) <= bufsize) {
754                if ((first < middle) && (middle < last)) {
755                    ssMergeBackward(PA, buf, bufoffset, first, middle, last, depth);
756                }
757                if ((check & 1) != 0) {
758                    ssMergeCheckEqual(PA, depth, first);
759                }
760                if ((check & 2) != 0) {
761                    ssMergeCheckEqual(PA, depth, last);
762                }
763                if (ssize == 0) {
764                    return;
765                }
766                StackEntry entry = stack[--ssize];
767                first = entry.a;
768                middle = entry.b;
769                last = entry.c;
770                check = entry.d;
771                continue;
772            }
773            if ((middle - first) <= bufsize) {
774                if (first < middle) {
775                    ssMergeForward(PA, buf, bufoffset, first, middle, last, depth);
776                }
777                if ((check & 1) != 0) {
778                    ssMergeCheckEqual(PA, depth, first);
779                }
780                if ((check & 2) != 0) {
781                    ssMergeCheckEqual(PA, depth, last);
782                }
783                if (ssize == 0) {
784                    return;
785                }
786                StackEntry entry = stack[--ssize];
787                first = entry.a;
788                middle = entry.b;
789                last = entry.c;
790                check = entry.d;
791                continue;
792            }
793            for (m = 0, len = Math.min(middle - first, last - middle), half = len >> 1;
794                 0 < len;
795                 len = half, half >>= 1) {
796                if (ssCompare(PA + getIDX(SA[middle + m + half]),
797                        PA + getIDX(SA[middle - m - half - 1]), depth) < 0) {
798                    m += half + 1;
799                    half -= (len & 1) ^ 1;
800                }
801            }
802            if (0 < m) {
803                ssBlockSwap(SA, middle - m, SA, middle, m);
804                i = j = middle;
805                next = 0;
806                if ((middle + m) < last) {
807                    if (SA[middle + m] < 0) {
808                        for (; SA[i - 1] < 0; --i) {
809                            ;
810                        }
811                        SA[middle + m] = ~SA[middle + m];
812                    }
813                    for (j = middle; SA[j] < 0; ++j) {
814                        ;
815                    }
816                    next = 1;
817                }
818                if ((i - first) <= (last - j)) {
819                    stack[ssize++] = new StackEntry(j, middle + m, last, (check & 2) | (next & 1));
820                    middle -= m;
821                    last = i;
822                    check = (check & 1);
823                } else {
824                    if ((i == middle) && (middle == j)) {
825                        next <<= 1;
826                    }
827                    stack[ssize++] = new StackEntry(first, middle - m, i, (check & 1) | (next & 2));
828                    first = j;
829                    middle += m;
830                    check = (check & 2) | (next & 1);
831                }
832            } else {
833                if ((check & 1) != 0) {
834                    ssMergeCheckEqual(PA, depth, first);
835                }
836                ssMergeCheckEqual(PA, depth, middle);
837                if ((check & 2) != 0) {
838                    ssMergeCheckEqual(PA, depth, last);
839                }
840                if (ssize == 0) {
841                    return;
842                }
843                StackEntry entry = stack[--ssize];
844                first = entry.a;
845                middle = entry.b;
846                last = entry.c;
847                check = entry.d;
848            }
849        }
850    }
