013        private void makeMaps() {
014            final boolean[] inUse = this.data.inUse;
015            final byte[] seqToUnseq = this.data.seqToUnseq;
016    
017            int nInUseShadow = 0;
018    
019            for (int i = 0; i < 256; i++) {
020                if (inUse[i]) {
021                    seqToUnseq[nInUseShadow++] = (byte) i;
022                }
023            }
024    
025            this.nInUse = nInUseShadow;
026        }

084         * throw an exception. </p>

087         * @throws NullPointerException if <tt>in == null</tt>
088         */

090            super();
091            this.in = in;

101        @Override
102        public final int read() throws IOException {
103            if (this.in != null) {
104                return read0();
105            } else {
106                throw new IOException("stream closed");
107            }
108        }

110        @Override
111        public final int read(final byte[] dest, final int offs, final int len)
112                throws IOException {
113            if (offs < 0) {
114                throw new IndexOutOfBoundsException("offs(" + offs + ") < 0.");
115            }
116            if (len < 0) {
117                throw new IndexOutOfBoundsException("len(" + len + ") < 0.");
118            }
119            if (offs + len > dest.length) {
120                throw new IndexOutOfBoundsException("offs(" + offs + ") + len(" + len + ") > dest.length(" + dest.length + ").");
121            }
122            if (this.in == null) {
123                throw new IOException("stream closed");
124            }
125    
126            final int hi = offs + len;
127            int destOffs = offs;
128            for (int b; (destOffs < hi) && ((b = read0()) >= 0); ) {
129                dest[destOffs++] = (byte) b;
130            }
131    
132            return (destOffs == offs) ? -1 : (destOffs - offs);
133        }

135        private int read0() throws IOException {
136            final int retChar = this.currentChar;
137    
138            switch (this.currentState) {
139                case EOF:
140                    return -1;
141    
142                case START_BLOCK_STATE:
143                    throw new IllegalStateException();
144    
145                case RAND_PART_A_STATE:
146                    throw new IllegalStateException();
147    
148                case RAND_PART_B_STATE:
149                    setupRandPartB();
150                    break;
151    
152                case RAND_PART_C_STATE:
153                    setupRandPartC();
154                    break;
155    
156                case NO_RAND_PART_A_STATE:
157                    throw new IllegalStateException();
158    
159                case NO_RAND_PART_B_STATE:
160                    setupNoRandPartB();
161                    break;
162    
163                case NO_RAND_PART_C_STATE:
164                    setupNoRandPartC();
165                    break;
166    
167                default:
168                    throw new IllegalStateException();
169            }
170    
171            return retChar;
172        }

174        private void init() throws IOException {
175            if (null == in) {
176                throw new IOException("no input stream");
177            }
178            if (in.available() == 0) {
179                throw new IOException("empty input stream");
180            }
181            // skip "BZ" marker
182            in.read();
183            in.read();
184            int magic2 = this.in.read();
185            if (magic2 != 'h') {
186                throw new IOException("stream is not bzip2: expected 'h'" + " as first byte but got '" + (char) magic2 + "'");
187            }
188    
189            int blockSize = this.in.read();
190            if ((blockSize < '1') || (blockSize > '9')) {
191                throw new IOException("stream is not bzip2: illegal " + "blocksize " + (char) blockSize);
192            }
193    
194            this.blockSize100k = blockSize - '0';
195    
196            initBlock();
197            setupBlock();
198        }
199    
200        private void initBlock() throws IOException {
201            char magic0 = bsGetUByte();
202            char magic1 = bsGetUByte();
203            char magic2 = bsGetUByte();
204            char magic3 = bsGetUByte();
205            char magic4 = bsGetUByte();
206            char magic5 = bsGetUByte();
207    
208            if (magic0 == 0x17 &&
209                    magic1 == 0x72 &&
210                    magic2 == 0x45 &&
211                    magic3 == 0x38 &&
212                    magic4 == 0x50 &&
213                    magic5 == 0x90) {
214                complete(); // end of file
215            } else if (magic0 != 0x31 || // '1'
216                    magic1 != 0x41 || // ')'
217                    magic2 != 0x59 || // 'Y'
218                    magic3 != 0x26 || // '&'
219                    magic4 != 0x53 || // 'S'
220                    magic5 != 0x59 // 'Y'
221                    ) {
222                this.currentState = EOF;
223                throw new IOException("bad block header");
224            } else {
225                this.storedBlockCRC = bsGetInt();
226                this.blockRandomised = bsR(1) == 1;
227    
228                /**
229                 * Allocate data here instead in constructor, so we do not
230                 * allocate it if the input file is empty.
231                 */
232                if (this.data == null) {
233                    this.data = new Data(this.blockSize100k);
234                }
235    
236                // currBlockNo++;
237                getAndMoveToFrontDecode();
238    
239                this.crc.initialiseCRC();
240                this.currentState = START_BLOCK_STATE;
241            }
242        }
243    
244        private void endBlock() throws IOException {
245            this.computedBlockCRC = this.crc.getFinalCRC();
246    
247            // A bad CRC is considered a fatal error.
248            if (this.storedBlockCRC != this.computedBlockCRC) {
249                // make next blocks readable without error
250                // (repair feature, not yet documented, not tested)
251                this.computedCombinedCRC = (this.storedCombinedCRC << 1) | (this.storedCombinedCRC >>> 31);
252                this.computedCombinedCRC ^= this.storedBlockCRC;
253    
254                throw new IOException("CRC error");
255            }
256    
257            this.computedCombinedCRC = (this.computedCombinedCRC << 1) | (this.computedCombinedCRC >>> 31);
258            this.computedCombinedCRC ^= this.computedBlockCRC;
259        }
260    
261        private void complete() throws IOException {
262            this.storedCombinedCRC = bsGetInt();
263            this.currentState = EOF;
264            this.data = null;
265    
266            if (this.storedCombinedCRC != this.computedCombinedCRC) {
267                throw new IOException("CRC error");
268            }
269        }

271        @Override
272        public final void close() throws IOException {
273            InputStream inShadow = this.in;
274            if (inShadow != null) {
275                try {
276                    if (inShadow != System.in) {
277                        inShadow.close();
278                    }
279                } finally {
280                    this.data = null;
281                    this.in = null;
282                }
283            }
284        }

286        private int bsR(final int n) throws IOException {
287            int bsLiveShadow = this.bsLive;
288            int bsBuffShadow = this.bsBuff;
289    
290            if (bsLiveShadow < n) {
291                final InputStream inShadow = this.in;
292                do {
293                    int thech = inShadow.read();
294    
295                    if (thech < 0) {
296                        throw new IOException("unexpected end of stream");
297                    }
298    
299                    bsBuffShadow = (bsBuffShadow << 8) | thech;
300                    bsLiveShadow += 8;
301                } while (bsLiveShadow < n);
302    
303                this.bsBuff = bsBuffShadow;
304            }
305    
306            this.bsLive = bsLiveShadow - n;
307            return (bsBuffShadow >> (bsLiveShadow - n)) & ((1 << n) - 1);
308        }

310        private boolean bsGetBit() throws IOException {
311            int bsLiveShadow = this.bsLive;
312            int bsBuffShadow = this.bsBuff;
313    
314            if (bsLiveShadow < 1) {
315                int thech = this.in.read();
316    
317                if (thech < 0) {
318                    throw new IOException("unexpected end of stream");
319                }
320    
321                bsBuffShadow = (bsBuffShadow << 8) | thech;
322                bsLiveShadow += 8;
323                this.bsBuff = bsBuffShadow;
324            }
325    
326            this.bsLive = bsLiveShadow - 1;
327            return ((bsBuffShadow >> (bsLiveShadow - 1)) & 1) != 0;
328        }

330        private char bsGetUByte() throws IOException {
331            return (char) bsR(8);
332        }

334        private int bsGetInt() throws IOException {
335            return (((((bsR(8) << 8) | bsR(8)) << 8) | bsR(8)) << 8) | bsR(8);
336        }

338        /**
339         * Called by createHuffmanDecodingTables() exclusively.
340         */
341        private static void hbCreateDecodeTables(final int[] limit,
342                                                 final int[] base,
343                                                 final int[] perm,
344                                                 final char[] length,
345                                                 final int minLen,
346                                                 final int maxLen,
347                                                 final int alphaSize) {
348            for (int i = minLen, pp = 0; i <= maxLen; i++) {
349                for (int j = 0; j < alphaSize; j++) {
350                    if (length[j] == i) {
351                        perm[pp++] = j;
352                    }
353                }
354            }
355    
356            for (int i = MAX_CODE_LEN; --i > 0; ) {
357                base[i] = 0;
358                limit[i] = 0;
359            }
360    
361            for (int i = 0; i < alphaSize; i++) {
362                base[length[i] + 1]++;
363            }
364    
365            for (int i = 1, b = base[0]; i < MAX_CODE_LEN; i++) {
366                b += base[i];
367                base[i] = b;
368            }
369    
370            for (int i = minLen, vec = 0, b = base[i]; i <= maxLen; i++) {
371                final int nb = base[i + 1];
372                vec += nb - b;
373                b = nb;
374                limit[i] = vec - 1;
375                vec <<= 1;
376            }
377    
378            for (int i = minLen + 1; i <= maxLen; i++) {
379                base[i] = ((limit[i - 1] + 1) << 1) - base[i];
380            }
381        }

383        private void recvDecodingTables() throws IOException {
384            final Data dataShadow = this.data;
385            final boolean[] inUse = dataShadow.inUse;
386            final byte[] pos = dataShadow.recvDecodingTables_pos;
387            final byte[] selector = dataShadow.selector;
388            final byte[] selectorMtf = dataShadow.selectorMtf;
389    
390            int inUse16 = 0;
391    
392            /* Receive the mapping table */
393            for (int i = 0; i < 16; i++) {
394                if (bsGetBit()) {
395                    inUse16 |= 1 << i;
396                }
397            }
398    
399            for (int i = 256; --i >= 0; ) {
400                inUse[i] = false;
401            }
402    
403            for (int i = 0; i < 16; i++) {
404                if ((inUse16 & (1 << i)) != 0) {
405                    final int i16 = i << 4;
406                    for (int j = 0; j < 16; j++) {
407                        if (bsGetBit()) {
408                            inUse[i16 + j] = true;
409                        }
410                    }
411                }
412            }
413    
414            makeMaps();
415            final int alphaSize = this.nInUse + 2;
416    
417            /* Now the selectors */
418            final int nGroups = bsR(3);
419            final int nSelectors = bsR(15);
420    
421            for (int i = 0; i < nSelectors; i++) {
422                int j = 0;
423                while (bsGetBit()) {
424                    j++;
425                }
426                selectorMtf[i] = (byte) j;
427            }
428    
429            /* Undo the MTF values for the selectors. */
430            for (int v = nGroups; --v >= 0; ) {
431                pos[v] = (byte) v;
432            }
433    
434            for (int i = 0; i < nSelectors; i++) {
435                int v = selectorMtf[i] & 0xff;
436                final byte tmp = pos[v];
437                while (v > 0) {
438                    // nearly all times v is zero, 4 in most other cases
439                    pos[v] = pos[v - 1];
440                    v--;
441                }
442                pos[0] = tmp;
443                selector[i] = tmp;
444            }
445    
446            final char[][] len = dataShadow.temp_charArray2d;
447    
448            /* Now the coding tables */
449            for (int t = 0; t < nGroups; t++) {
450                int curr = bsR(5);
451                final char[] len_t = len[t];
452                for (int i = 0; i < alphaSize; i++) {
453                    while (bsGetBit()) {
454                        curr += bsGetBit() ? -1 : 1;
455                    }
456                    len_t[i] = (char) curr;
457                }
458            }
459    
460            // finally create the Huffman tables
461            createHuffmanDecodingTables(alphaSize, nGroups);
462        }

464        /**
465         * Called by recvDecodingTables() exclusively.
466         */
467        private void createHuffmanDecodingTables(final int alphaSize,
468                                                 final int nGroups) {
469            final Data dataShadow = this.data;
470            final char[][] len = dataShadow.temp_charArray2d;
471            final int[] minLens = dataShadow.minLens;
472            final int[][] limit = dataShadow.limit;
473            final int[][] base = dataShadow.base;
474            final int[][] perm = dataShadow.perm;
475    
476            for (int t = 0; t < nGroups; t++) {
477                int minLen = 32;
478                int maxLen = 0;
479                final char[] len_t = len[t];
480                for (int i = alphaSize; --i >= 0; ) {
481                    final char lent = len_t[i];
482                    if (lent > maxLen) {
483                        maxLen = lent;
484                    }
485                    if (lent < minLen) {
486                        minLen = lent;
487                    }
488                }
489                hbCreateDecodeTables(limit[t], base[t], perm[t], len[t], minLen,
490                        maxLen, alphaSize);
491                minLens[t] = minLen;
492            }
493        }

495        private void getAndMoveToFrontDecode() throws IOException {
496            this.origPtr = bsR(24);
497            recvDecodingTables();
498    
499            final InputStream inShadow = this.in;
500            final Data dataShadow = this.data;
501            final byte[] ll8 = dataShadow.ll8;
502            final int[] unzftab = dataShadow.unzftab;
503            final byte[] selector = dataShadow.selector;
504            final byte[] seqToUnseq = dataShadow.seqToUnseq;
505            final char[] yy = dataShadow.getAndMoveToFrontDecodeyy;
506            final int[] minLens = dataShadow.minLens;
507            final int[][] limit = dataShadow.limit;
508            final int[][] base = dataShadow.base;
509            final int[][] perm = dataShadow.perm;
510            final int limitLast = this.blockSize100k * 100000;
511    
512            /*
513            Setting up the unzftab entries here is not strictly
514            necessary, but it does save having to do it later
515            in a separate pass, and so saves a block's worth of
516            cache misses.
517             */
518            for (int i = 256; --i >= 0; ) {
519                yy[i] = (char) i;
520                unzftab[i] = 0;
521            }
522    
523            int groupNo = 0;
524            int groupPos = G_SIZE - 1;
525            final int eob = this.nInUse + 1;
526            int nextSym = getAndMoveToFrontDecode0(0);
527            int bsBuffShadow = this.bsBuff;
528            int bsLiveShadow = this.bsLive;
529            int lastShadow = -1;
530            int zt = selector[groupNo] & 0xff;
531            int[] basezt = base[zt];
532            int[] limitzt = limit[zt];
533            int[] permzt = perm[zt];
534            int minLenszt = minLens[zt];
535    
536            while (nextSym != eob) {
537                if ((nextSym == RUNA) || (nextSym == RUNB)) {
538                    int s = -1;
539    
540                    for (int n = 1; true; n <<= 1) {
541                        if (nextSym == RUNA) {
542                            s += n;
543                        } else if (nextSym == RUNB) {
544                            s += n << 1;
545                        } else {
546                            break;
547                        }
548    
549                        if (groupPos == 0) {
550                            groupPos = G_SIZE - 1;
551                            zt = selector[++groupNo] & 0xff;
552                            basezt = base[zt];
553                            limitzt = limit[zt];
554                            permzt = perm[zt];
555                            minLenszt = minLens[zt];
556                        } else {
557                            groupPos--;
558                        }
559    
560                        int zn = minLenszt;
561    
562                        // Inlined:
563                        // int zvec = bsR(zn);
564                        while (bsLiveShadow < zn) {
565                            final int thech = inShadow.read();
566                            if (thech >= 0) {
567                                bsBuffShadow = (bsBuffShadow << 8) | thech;
568                                bsLiveShadow += 8;
569                                continue;
570                            } else {
571                                throw new IOException("unexpected end of stream");
572                            }
573                        }
574                        int zvec = (bsBuffShadow >> (bsLiveShadow - zn)) & ((1 << zn) - 1);
575                        bsLiveShadow -= zn;
576    
577                        while (zvec > limitzt[zn]) {
578                            zn++;
579                            while (bsLiveShadow < 1) {
580                                final int thech = inShadow.read();
581                                if (thech >= 0) {
582                                    bsBuffShadow = (bsBuffShadow << 8) | thech;
583                                    bsLiveShadow += 8;
584                                    continue;
585                                } else {
586                                    throw new IOException("unexpected end of stream");
587                                }
588                            }
589                            bsLiveShadow--;
590                            zvec = (zvec << 1) | ((bsBuffShadow >> bsLiveShadow) & 1);
591                        }
592                        nextSym = permzt[zvec - basezt[zn]];
593                    }
594    
595                    final byte ch = seqToUnseq[yy[0]];
596                    unzftab[ch & 0xff] += s + 1;
597    
598                    while (s-- >= 0) {
599                        ll8[++lastShadow] = ch;
600                    }
601    
602                    if (lastShadow >= limitLast) {
603                        throw new IOException("block overrun");
604                    }
605                } else {
606                    if (++lastShadow >= limitLast) {
607                        throw new IOException("block overrun");
608                    }
609    
610                    final char tmp = yy[nextSym - 1];
611                    unzftab[seqToUnseq[tmp] & 0xff]++;
612                    ll8[lastShadow] = seqToUnseq[tmp];
613    
614                    /*
615                    This loop is hammered during decompression,
616                    hence avoid native method call overhead of
617                    System.arraycopy for very small ranges to copy.
618                     */
619                    if (nextSym <= 16) {
620                        for (int j = nextSym - 1; j > 0; ) {
621                            yy[j] = yy[--j];
622                        }
623                    } else {
624                        System.arraycopy(yy, 0, yy, 1, nextSym - 1);
625                    }
626    
627                    yy[0] = tmp;
628    
629                    if (groupPos == 0) {
630                        groupPos = G_SIZE - 1;
631                        zt = selector[++groupNo] & 0xff;
632                        basezt = base[zt];
633                        limitzt = limit[zt];
634                        permzt = perm[zt];
635                        minLenszt = minLens[zt];
636                    } else {
637                        groupPos--;
638                    }
639    
640                    int zn = minLenszt;
641    
642                    // Inlined:
643                    // int zvec = bsR(zn);
644                    while (bsLiveShadow < zn) {
645                        final int thech = inShadow.read();
646                        if (thech >= 0) {
647                            bsBuffShadow = (bsBuffShadow << 8) | thech;
648                            bsLiveShadow += 8;
649                            continue;
650                        } else {
651                            throw new IOException("unexpected end of stream");
652                        }
653                    }
654                    int zvec = (bsBuffShadow >> (bsLiveShadow - zn)) & ((1 << zn) - 1);
655                    bsLiveShadow -= zn;
656    
657                    while (zvec > limitzt[zn]) {
658                        zn++;
659                        while (bsLiveShadow < 1) {
660                            final int thech = inShadow.read();
661                            if (thech >= 0) {
662                                bsBuffShadow = (bsBuffShadow << 8) | thech;
663                                bsLiveShadow += 8;
664                                continue;
665                            } else {
666                                throw new IOException("unexpected end of stream");
667                            }
668                        }
669                        bsLiveShadow--;
670                        zvec = (zvec << 1) | ((bsBuffShadow >> bsLiveShadow) & 1);
671                    }
672                    nextSym = permzt[zvec - basezt[zn]];
673                }
674            }
675    
676            this.last = lastShadow;
677            this.bsLive = bsLiveShadow;
678            this.bsBuff = bsBuffShadow;
679        }

681        private int getAndMoveToFrontDecode0(final int groupNo)
682                throws IOException {
683            final InputStream inShadow = this.in;
684            final Data dataShadow = this.data;
685            final int zt = dataShadow.selector[groupNo] & 0xff;
686            final int[] limitzt = dataShadow.limit[zt];
687            int zn = dataShadow.minLens[zt];
688            int zvec = bsR(zn);
689            int bsLiveShadow = this.bsLive;
690            int bsBuffShadow = this.bsBuff;
691    
692            while (zvec > limitzt[zn]) {
693                zn++;
694                while (bsLiveShadow < 1) {
695                    final int thech = inShadow.read();
696    
697                    if (thech >= 0) {
698                        bsBuffShadow = (bsBuffShadow << 8) | thech;
699                        bsLiveShadow += 8;
700                        continue;
701                    } else {
702                        throw new IOException("unexpected end of stream");
703                    }
704                }
705                bsLiveShadow--;
706                zvec = (zvec << 1) | ((bsBuffShadow >> bsLiveShadow) & 1);
707            }
708    
709            this.bsLive = bsLiveShadow;
710            this.bsBuff = bsBuffShadow;
711    
712            return dataShadow.perm[zt][zvec - dataShadow.base[zt][zn]];
713        }

715        private void setupBlock() throws IOException {
716            if (this.data == null) {
717                return;
718            }
719    
720            final int[] cftab = this.data.cftab;
721            final int[] tt = this.data.initTT(this.last + 1);
722            final byte[] ll8 = this.data.ll8;
723            cftab[0] = 0;
724            System.arraycopy(this.data.unzftab, 0, cftab, 1, 256);
725    
726            for (int i = 1, c = cftab[0]; i <= 256; i++) {
727                c += cftab[i];
728                cftab[i] = c;
729            }
730    
731            for (int i = 0, lastShadow = this.last; i <= lastShadow; i++) {
732                tt[cftab[ll8[i] & 0xff]++] = i;
733            }
734    
735            if ((this.origPtr < 0) || (this.origPtr >= tt.length)) {
736                throw new IOException("stream corrupted");
737            }
738    
739            this.setuptPos = tt[this.origPtr];
740            this.setupcount = 0;
741            this.setupi2 = 0;
742            this.setupch2 = 256;   /* not a char and not EOF */
743    
744            if (this.blockRandomised) {
745                this.setuprNToGo = 0;
746                this.setuprTPos = 0;
747                setupRandPartA();
748            } else {
749                setupNoRandPartA();
750            }
751        }
752    
753        private void setupRandPartA() throws IOException {
754            if (this.setupi2 <= this.last) {
755                this.setupchPrev = this.setupch2;
756                int setupch2Shadow = this.data.ll8[this.setuptPos] & 0xff;
757                this.setuptPos = this.data.tt[this.setuptPos];
758                if (this.setuprNToGo == 0) {
759                    this.setuprNToGo = NUMS[this.setuprTPos] - 1;
760                    if (++this.setuprTPos == 512) {
761                        this.setuprTPos = 0;
762                    }
763                } else {
764                    this.setuprNToGo--;
765                }
766                this.setupch2 = setupch2Shadow ^= (this.setuprNToGo == 1) ? 1 : 0;
767                this.setupi2++;
768                this.currentChar = setupch2Shadow;
769                this.currentState = RAND_PART_B_STATE;
770                this.crc.updateCRC(setupch2Shadow);
771            } else {
772                endBlock();
773                initBlock();
774                setupBlock();
775            }
776        }
777    
778        private void setupNoRandPartA() throws IOException {
779            if (this.setupi2 <= this.last) {
780                this.setupchPrev = this.setupch2;
781                int setupch2Shadow = this.data.ll8[this.setuptPos] & 0xff;
782                this.setupch2 = setupch2Shadow;
783                this.setuptPos = this.data.tt[this.setuptPos];
784                this.setupi2++;
785                this.currentChar = setupch2Shadow;
786                this.currentState = NO_RAND_PART_B_STATE;
787                this.crc.updateCRC(setupch2Shadow);
788            } else {
789                this.currentState = NO_RAND_PART_A_STATE;
790                endBlock();
791                initBlock();
792                setupBlock();
793            }
794        }
795    
796        private void setupRandPartB() throws IOException {
797            if (this.setupch2 != this.setupchPrev) {
798                this.currentState = RAND_PART_A_STATE;
799                this.setupcount = 1;
800                setupRandPartA();
801            } else if (++this.setupcount >= 4) {
802                this.setupz = (char) (this.data.ll8[this.setuptPos] & 0xff);
803                this.setuptPos = this.data.tt[this.setuptPos];
804                if (this.setuprNToGo == 0) {
805                    this.setuprNToGo = NUMS[this.setuprTPos] - 1;
806                    if (++this.setuprTPos == 512) {
807                        this.setuprTPos = 0;
808                    }
809                } else {
810                    this.setuprNToGo--;
811                }
812                this.setupj2 = 0;
813                this.currentState = RAND_PART_C_STATE;
814                if (this.setuprNToGo == 1) {
815                    this.setupz ^= 1;
816                }
817                setupRandPartC();
818            } else {
819                this.currentState = RAND_PART_A_STATE;
820                setupRandPartA();
821            }
822        }
823    
824        private void setupRandPartC() throws IOException {
825            if (this.setupj2 < this.setupz) {
826                this.currentChar = this.setupch2;
827                this.crc.updateCRC(this.setupch2);
828                this.setupj2++;
829            } else {
830                this.currentState = RAND_PART_A_STATE;
831                this.setupi2++;
832                this.setupcount = 0;
833                setupRandPartA();
834            }
835        }
836    
837        private void setupNoRandPartB() throws IOException {
838            if (this.setupch2 != this.setupchPrev) {
839                this.setupcount = 1;
840                setupNoRandPartA();
841            } else if (++this.setupcount >= 4) {
842                this.setupz = (char) (this.data.ll8[this.setuptPos] & 0xff);
843                this.setuptPos = this.data.tt[this.setuptPos];
844                this.setupj2 = 0;
845                setupNoRandPartC();
846            } else {
847                setupNoRandPartA();
848            }
849        }
850    
851        private void setupNoRandPartC() throws IOException {
852            if (this.setupj2 < this.setupz) {
853                int setupch2Shadow = this.setupch2;
854                this.currentChar = setupch2Shadow;
855                this.crc.updateCRC(setupch2Shadow);
856                this.setupj2++;
857                this.currentState = NO_RAND_PART_C_STATE;
858            } else {
859                this.setupi2++;
860                this.setupcount = 0;
861                setupNoRandPartA();
862            }
863        }
