023package org.xbib.io.compress.bzip2;
024
025import java.io.IOException;
026import java.io.InputStream;
027
028
029/**
030 * <p>An InputStream wrapper that decompresses BZip2 data</p>
031 *
032 * <p>A BZip2 stream consists of one or more blocks of compressed data. This decompressor reads a
033 * whole block at a time, then progressively returns decompressed output.</p>
034 *
035 * <p>On encountering any error decoding the compressed stream, an IOException is thrown, and
036 * further reads will return {@code -1}</p>
037 *
038 * <p><b>Note:</b> Each BZip2 compressed block contains a CRC code which is verified after the block
039 * has been read completely. If verification fails, an exception is thrown on the final read from
040 * the block, <b>potentially after corrupt data has already been returned</b>. The compressed stream
041 * also contains a final CRC code which is verified once the end of the stream has been reached.
042 * <b>This check may fail even if every individual block in the stream passes CRC verification</b>.
043 * If this possibility is of concern, you should read and store the entire decompressed stream
044 * before further processing.</p>
045 *
046 * <p>Instances of this class are not threadsafe.</p>
047 */
048public class BZip2InputStream extends InputStream {
049
050    /**
051     * The stream from which compressed BZip2 data is read and decoded
052     */
053    private InputStream inputStream;
054
055    /**
056     * An InputStream wrapper that provides bit-level reads
057     */
058    private BZip2BitInputStream bitInputStream;
059
060    /**
061     * If {@code true}, the caller is assumed to have read away the stream's leading "BZ" identifier
062     * bytes
063     */
064    private final boolean headerless;
065
066    /**
067     * (@code true} if the end of the compressed stream has been reached, otherwise {@code false}
068     */
069    private boolean streamComplete = false;
070
071    /**
072     * The declared block size of the stream (before final run-length decoding). The final block
073     * will usually be smaller, but no block in the stream has to be exactly this large, and an
074     * encoder could in theory choose to mix blocks of any size up to this value. Its function is
075     * therefore as a hint to the decompressor as to how much working space is sufficient to
076     * decompress blocks in a given stream
077     */
078    private int streamBlockSize;
079
080    /**
081     * The merged CRC of all blocks decompressed so far
082     */
083    private int streamCRC = 0;
084
085    /**
086     * The decompressor for the current block
087     */
088    private BZip2BlockDecompressor blockDecompressor = null;
089
090    public BZip2InputStream(final InputStream inputStream) {
091        this(inputStream, false);
092    }
093
094    public BZip2InputStream(final InputStream inputStream, int bufferSize) {
095        this(inputStream, false);
096    }
097
098    /**
099     * @param inputStream The InputStream to wrap
100     * @param headerless  If {@code true}, the caller is assumed to have read away the stream's
101     *                    leading "BZ" identifier bytes
102     */
103    public BZip2InputStream(final InputStream inputStream, final boolean headerless) {
104
105        if (inputStream == null) {
106            throw new IllegalArgumentException("Null input stream");
107        }
108
109        this.inputStream = inputStream;
110        this.bitInputStream = new BZip2BitInputStream(inputStream);
111        this.headerless = headerless;
112
113    }
114
115    /* (non-Javadoc)
116     * @see java.io.InputStream#read()
117     */
118    @Override
119    public int read() throws IOException {
120
121        int nextByte = -1;
122        if (this.blockDecompressor == null) {
123            initialiseStream();
124        } else {
125            nextByte = this.blockDecompressor.read();
126        }
127
128        if (nextByte == -1) {
129            if (initialiseNextBlock()) {
130                nextByte = this.blockDecompressor.read();
131            }
132        }
133
134        return nextByte;
135
136    }
137
138
139    /* (non-Javadoc)
140     * @see java.io.InputStream#read(byte[], int, int)
141     */
142    @Override
143    public int read(final byte[] destination, final int offset, final int length) throws IOException {
144
145        int bytesRead = -1;
146        if (this.blockDecompressor == null) {
147            initialiseStream();
148        } else {
149            bytesRead = this.blockDecompressor.read(destination, offset, length);
150        }
151
152        if (bytesRead == -1) {
153            if (initialiseNextBlock()) {
154                bytesRead = this.blockDecompressor.read(destination, offset, length);
155            }
156        }
157
158        return bytesRead;
159
160    }
161
162
163    /* (non-Javadoc)
164     * @see java.io.InputStream#close()
165     */
166    @Override
167    public void close() throws IOException {
168
169        if (this.bitInputStream != null) {
170            this.streamComplete = true;
171            this.blockDecompressor = null;
172            this.bitInputStream = null;
173
174            try {
175                this.inputStream.close();
176            } finally {
177                this.inputStream = null;
178            }
179        }
180
181    }
182
183
184    /**
185     * Reads the stream header and checks that the data appears to be a valid BZip2 stream
186     *
187     * @throws java.io.IOException if the stream header is not valid
188     */
189    private void initialiseStream() throws IOException {
190
191                /* If the stream has been explicitly closed, throw an exception */
192        if (this.bitInputStream == null) {
193            throw new BZip2Exception("Stream closed");
194        }
195
196                /* If we're already at the end of the stream, do nothing */
197        if (this.streamComplete) {
198            return;
199        }
200
201                /* Read the stream header */
202        try {
203            int marker1 = this.headerless ? 0 : this.bitInputStream.readBits(16);
204            int marker2 = this.bitInputStream.readBits(8);
205            int blockSize = (this.bitInputStream.readBits(8) - '0');
206
207            if (
208                    (!this.headerless && (marker1 != BZip2Constants.STREAM_START_MARKER_1))
209                            || (marker2 != BZip2Constants.STREAM_START_MARKER_2)
210                            || (blockSize < 1) || (blockSize > 9)) {
211                throw new BZip2Exception("Invalid BZip2 header");
212            }
213
214            this.streamBlockSize = blockSize * 100000;
215        } catch (IOException e) {
216            // If the stream header was not valid, stop trying to read more data
217            this.streamComplete = true;
218            throw e;
219        }
220
221
222    }
223
224
225    /**
226     * Prepares a new block for decompression if any remain in the stream. If a previous block has
227     * completed, its CRC is checked and merged into the stream CRC. If the previous block was the
228     * final block in the stream, the stream CRC is validated
229     *
230     * @return {@code true} if a block was successfully initialised, or {@code false} if the end of
231     * file marker was encountered
232     * @throws java.io.IOException if either the block or stream CRC check failed, if the following data is
233     *                             not a valid block-header or end-of-file marker, or if the following
234     *                             block could not be decoded
235     */
236    private boolean initialiseNextBlock() throws IOException {
237
238                /* If we're already at the end of the stream, do nothing */
239        if (this.streamComplete) {
240            return false;
241        }
242
243                /* If a block is complete, check the block CRC and integrate it into the stream CRC */
244        if (this.blockDecompressor != null) {
245            int blockCRC = this.blockDecompressor.checkCRC();
246            this.streamCRC = ((this.streamCRC << 1) | (this.streamCRC >>> 31)) ^ blockCRC;
247        }
248
249                /* Read block-header or end-of-stream marker */
250        final int marker1 = this.bitInputStream.readBits(24);
251        final int marker2 = this.bitInputStream.readBits(24);
252
253        if (marker1 == BZip2Constants.BLOCK_HEADER_MARKER_1 && marker2 == BZip2Constants.BLOCK_HEADER_MARKER_2) {
254            // Initialise a new block
255            try {
256                this.blockDecompressor = new BZip2BlockDecompressor(this.bitInputStream, this.streamBlockSize);
257            } catch (IOException e) {
258                // If the block could not be decoded, stop trying to read more data
259                this.streamComplete = true;
260                throw e;
261            }
262            return true;
263        } else if (marker1 == BZip2Constants.STREAM_END_MARKER_1 && marker2 == BZip2Constants.STREAM_END_MARKER_2) {
264            // Read and verify the end-of-stream CRC
265            this.streamComplete = true;
266            final int storedCombinedCRC = this.bitInputStream.readInteger();
267            if (storedCombinedCRC != this.streamCRC) {
268                throw new BZip2Exception("BZip2 stream CRC error");
269            }
270            return false;
271        }
272
273                /* If what was read is not a valid block-header or end-of-stream marker, the stream is broken */
274        this.streamComplete = true;
275        throw new BZip2Exception("BZip2 stream format error");
276
277    }
