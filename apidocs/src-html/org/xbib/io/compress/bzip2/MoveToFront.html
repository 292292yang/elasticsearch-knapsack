061    /**
062     * Moves a value to the head of the MTF list (forward Move To Front transform)
063     *
064     * @param value The value to move
065     * @return The position the value moved from
066     */
067    public int valueToFront(final byte value) {
068
069        final byte[] mtf = this.mtf;
070
071        int index = 0;
072        byte temp = mtf[0];
073        if (value != temp) {
074            mtf[0] = value;
075            while (value != temp) {
076                index++;
077                final byte temp2 = temp;
078                temp = mtf[index];
079                mtf[index] = temp2;
080            }
081        }
082
083        return index;
084
085    }

088    /**
089     * Gets the value from a given index and moves it to the front of the MTF list (inverse Move To
090     * Front transform)
091     *
092     * @param index The index to move
093     * @return The value at the given index
094     */
095    public byte indexToFront(final int index) {
096
097        final byte[] mtf = this.mtf;
098
099        final byte value = mtf[index];
100        System.arraycopy(mtf, 0, mtf, 1, index);
101        mtf[0] = value;
102
103        return value;
104
105    }
