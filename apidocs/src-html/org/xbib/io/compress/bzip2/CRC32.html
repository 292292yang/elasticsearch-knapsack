072    /**
073     * The current CRC
074     */
075    private int crc = 0xffffffff;
076
077
078    /**
079     * @return The current CRC
080     */
081    public int getCRC() {
082
083        return ~this.crc;
084
085    }
086
087
088    /**
089     * Update the CRC with a single byte
090     *
091     * @param value The value to update the CRC with
092     */
093    public void updateCRC(final int value) {
094
095        final int crc = this.crc;
096
097        this.crc = (crc << 8) ^ crc32Lookup[((crc >> 24) ^ value) & 0xff];
098
099    }
100
101
102    /**
103     * Update the CRC with a sequence of identical bytes
104     *
105     * @param value The value to update the CRC with
106     * @param count The number of bytes
107     */
108    public void updateCRC(final int value, int count) {
109
110        int crc = this.crc;
111
112        while (count-- > 0) {
113            crc = (crc << 8) ^ crc32Lookup[((crc >> 24) ^ value) & 0xff];
114        }
115
116        this.crc = crc;
117
118    }
