023package org.xbib.io.compress.bzip2;

026/**
027 * An in-place, length restricted Canonical Huffman code length allocator
028 * <p>
029 * Based on the algorithm proposed by R. L. Milidi√∫, A. A. Pessoa and E. S. Laber in "In-place
030 * Length-Restricted Prefix Coding" (see: http://www-di.inf.puc-rio.br/~laber/public/spire98.ps)
031 * and incorporating additional ideas from the implementation of "shcodec" by Simakov Alexander
032 * (see: http://webcenter.ru/~xander/)
033 */
034public class HuffmanAllocator {
035
036    /**
037     * FIRST() function
038     *
039     * @param array       The code length array
040     * @param i           The input position
041     * @param nodesToMove The number of internal nodes to be relocated
042     * @return The smallest {@code k} such that {@code nodesToMove <= k <= i} and
043     * {@code i <= (array[k] % array.length)}
044     */
045    private static int first(final int[] array, int i, final int nodesToMove) {
046
047        final int length = array.length;
048        final int limit = i;
049        int k = array.length - 2;
050
051        while ((i >= nodesToMove) && ((array[i] % length) > limit)) {
052            k = i;
053            i -= (limit - i + 1);
054        }
055        i = Math.max(nodesToMove - 1, i);
056
057        while (k > (i + 1)) {
058            int temp = (i + k) >> 1;
059            if ((array[temp] % length) > limit) {
060                k = temp;
061            } else {
062                i = temp;
063            }
064        }
065
066        return k;
067
068    }
069
070
071    /**
072     * Fills the code array with extended parent pointers
073     *
074     * @param array The code length array
075     */
076    private static void setExtendedParentPointers(final int[] array) {
077
078        final int length = array.length;
079
080        array[0] += array[1];
081
082        for (int headNode = 0, tailNode = 1, topNode = 2; tailNode < (length - 1); tailNode++) {
083            int temp;
084            if ((topNode >= length) || (array[headNode] < array[topNode])) {
085                temp = array[headNode];
086                array[headNode++] = tailNode;
087            } else {
088                temp = array[topNode++];
089            }
090
091            if ((topNode >= length) || ((headNode < tailNode) && (array[headNode] < array[topNode]))) {
092                temp += array[headNode];
093                array[headNode++] = tailNode + length;
094            } else {
095                temp += array[topNode++];
096            }
097
098            array[tailNode] = temp;
099        }
100
101    }
102
103
104    /**
105     * Finds the number of nodes to relocate in order to achieve a given code length limit
106     *
107     * @param array         The code length array
108     * @param maximumLength The maximum bit length for the generated codes
109     * @return The number of nodes to relocate
110     */
111    private static int findNodesToRelocate(final int[] array, final int maximumLength) {
112
113        int currentNode = array.length - 2;
114        for (int currentDepth = 1; (currentDepth < (maximumLength - 1)) && (currentNode > 1); currentDepth++) {
115            currentNode = first(array, currentNode - 1, 0);
116        }
117
118        return currentNode;
119
120    }
121
122
123    /**
124     * A final allocation pass with no code length limit
125     *
126     * @param array The code length array
127     */
128    private static void allocateNodeLengths(final int[] array) {
129
130        int firstNode = array.length - 2;
131        int nextNode = array.length - 1;
132
133        for (int currentDepth = 1, availableNodes = 2; availableNodes > 0; currentDepth++) {
134            final int lastNode = firstNode;
135            firstNode = first(array, lastNode - 1, 0);
136
137            for (int i = availableNodes - (lastNode - firstNode); i > 0; i--) {
138                array[nextNode--] = currentDepth;
139            }
140
141            availableNodes = (lastNode - firstNode) << 1;
142        }
143
144    }
145
146
147    /**
148     * A final allocation pass that relocates nodes in order to achieve a maximum code length limit
149     *
150     * @param array       The code length array
151     * @param nodesToMove The number of internal nodes to be relocated
152     * @param insertDepth The depth at which to insert relocated nodes
153     */
154    private static void allocateNodeLengthsWithRelocation(final int[] array, final int nodesToMove, final int insertDepth) {
155
156        int firstNode = array.length - 2;
157        int nextNode = array.length - 1;
158        int currentDepth = (insertDepth == 1) ? 2 : 1;
159        int nodesLeftToMove = (insertDepth == 1) ? nodesToMove - 2 : nodesToMove;
160
161        for (int availableNodes = currentDepth << 1; availableNodes > 0; currentDepth++) {
162            final int lastNode = firstNode;
163            firstNode = (firstNode <= nodesToMove) ? firstNode : first(array, lastNode - 1, nodesToMove);
164
165            int offset = 0;
166            if (currentDepth >= insertDepth) {
167                offset = Math.min(nodesLeftToMove, 1 << (currentDepth - insertDepth));
168            } else if (currentDepth == (insertDepth - 1)) {
169                offset = 1;
170                if ((array[firstNode]) == lastNode) {
171                    firstNode++;
172                }
173            }
174
175            for (int i = availableNodes - (lastNode - firstNode + offset); i > 0; i--) {
176                array[nextNode--] = currentDepth;
177            }
178
179            nodesLeftToMove -= offset;
180            availableNodes = (lastNode - firstNode + offset) << 1;
181        }
182
183    }
184
185
186    /**
187     * Allocates Canonical Huffman code lengths in place based on a sorted frequency array
188     *
189     * @param array         On input, a sorted array of symbol frequencies; On output, an array of Canonical
190     *                      Huffman code lengths
191     * @param maximumLength The maximum code length. Must be at least {@code ceil(log2(array.length))}
192     */
193    public static void allocateHuffmanCodeLengths(final int[] array, final int maximumLength) {
194
195        switch (array.length) {
196            case 2:
197                array[1] = 1;
198            case 1:
199                array[0] = 1;
200                return;
201        }
202
203                /* Pass 1 : Set extended parent pointers */
204        setExtendedParentPointers(array);
205
206                /* Pass 2 : Find number of nodes to relocate in order to achieve maximum code length */
207        int nodesToRelocate = findNodesToRelocate(array, maximumLength);
208
209                /* Pass 3 : Generate code lengths */
210        if ((array[0] % array.length) >= nodesToRelocate) {
211            allocateNodeLengths(array);
212        } else {
213            int insertDepth = maximumLength - (32 - Integer.numberOfLeadingZeros(nodesToRelocate - 1));
214            allocateNodeLengthsWithRelocation(array, nodesToRelocate, insertDepth);
215        }
216
217    }
218
219    /**
220     * Non-instantiable
221     */
222    private HuffmanAllocator() {
223    }
