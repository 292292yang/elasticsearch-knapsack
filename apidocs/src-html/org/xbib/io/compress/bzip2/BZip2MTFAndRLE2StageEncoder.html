001package org.xbib.io.compress.bzip2;
002
003/**
004 * An encoder for the BZip2 Move To Front Transform and Run-Length Encoding[2] stages<br>
005 * Although conceptually these two stages are separate, it is computationally efficient to perform
006 * them in one pass.
007 */
008public class BZip2MTFAndRLE2StageEncoder {
009
010    /**
011     * The Burrows-Wheeler transformed block
012     */
013    private final int[] bwtBlock;
014
015    /**
016     * Actual length of the data in the bwtBlock array
017     */
018    private int bwtLength;
019
020    /**
021     * At each position, {@code true} if the byte value with that index is present within the block,
022     * otherwise {@code false}
023     */
024    private final boolean[] bwtValuesInUse;
025
026    /**
027     * The output of the Move To Front Transform and Run-Length Encoding[2] stages
028     */
029    private final char[] mtfBlock;
030
031    /**
032     * The actual number of values contained in the mtfBlock array
033     */
034    private int mtfLength;
035
036    /**
037     * The global frequencies of values within the mtfBlock array
038     */
039    private final int[] mtfSymbolFrequencies = new int[BZip2Constants.HUFFMAN_MAXIMUM_ALPHABET_SIZE];
040
041    /**
042     * The encoded alphabet size
043     */
044    private int alphabetSize;
045
046
047    /**
048     * Performs the Move To Front transform and Run Length Encoding[1] stages
049     */
050    public void encode() {
051
052        final int bwtLength = this.bwtLength;
053        final boolean[] bwtValuesInUse = this.bwtValuesInUse;
054        final int[] bwtBlock = this.bwtBlock;
055        final char[] mtfBlock = this.mtfBlock;
056        final int[] mtfSymbolFrequencies = this.mtfSymbolFrequencies;
057        final byte[] huffmanSymbolMap = new byte[256];
058        final MoveToFront symbolMTF = new MoveToFront();
059
060        int totalUniqueValues = 0;
061        for (int i = 0; i < 256; i++) {
062            if (bwtValuesInUse[i]) {
063                huffmanSymbolMap[i] = (byte) totalUniqueValues++;
064            }
065        }
066
067        final int endOfBlockSymbol = totalUniqueValues + 1;
068
069        int mtfIndex = 0;
070        int repeatCount = 0;
071        int totalRunAs = 0;
072        int totalRunBs = 0;
073
074        for (int i = 0; i < bwtLength; i++) {
075
076            // Move To Front
077            final int mtfPosition = symbolMTF.valueToFront(huffmanSymbolMap[bwtBlock[i] & 0xff]);
078
079            // Run Length Encode
080            if (mtfPosition == 0) {
081                repeatCount++;
082            } else {
083                if (repeatCount > 0) {
084                    repeatCount--;
085                    while (true) {
086                        if ((repeatCount & 1) == 0) {
087                            mtfBlock[mtfIndex++] = BZip2Constants.HUFFMAN_SYMBOL_RUNA;
088                            totalRunAs++;
089                        } else {
090                            mtfBlock[mtfIndex++] = BZip2Constants.HUFFMAN_SYMBOL_RUNB;
091                            totalRunBs++;
092                        }
093
094                        if (repeatCount <= 1) {
095                            break;
096                        }
097                        repeatCount = (repeatCount - 2) >>> 1;
098                    }
099                    repeatCount = 0;
100                }
101
102                mtfBlock[mtfIndex++] = (char) (mtfPosition + 1);
103                mtfSymbolFrequencies[mtfPosition + 1]++;
104            }
105
106        }
107
108        if (repeatCount > 0) {
109            repeatCount--;
110            while (true) {
111                if ((repeatCount & 1) == 0) {
112                    mtfBlock[mtfIndex++] = BZip2Constants.HUFFMAN_SYMBOL_RUNA;
113                    totalRunAs++;
114                } else {
115                    mtfBlock[mtfIndex++] = BZip2Constants.HUFFMAN_SYMBOL_RUNB;
116                    totalRunBs++;
117                }
118
119                if (repeatCount <= 1) {
120                    break;
121                }
122                repeatCount = (repeatCount - 2) >>> 1;
123            }
124        }
125
126        mtfBlock[mtfIndex] = (char) endOfBlockSymbol;
127        mtfSymbolFrequencies[endOfBlockSymbol]++;
128        mtfSymbolFrequencies[BZip2Constants.HUFFMAN_SYMBOL_RUNA] += totalRunAs;
129        mtfSymbolFrequencies[BZip2Constants.HUFFMAN_SYMBOL_RUNB] += totalRunBs;
130
131        this.mtfLength = mtfIndex + 1;
132        this.alphabetSize = endOfBlockSymbol + 1;
133
134    }
135
136
137    /**
138     * @return The encoded MTF block
139     */
140    public char[] getMtfBlock() {
141
142        return this.mtfBlock;
143
144    }
145
146
147    /**
148     * @return The actual length of the MTF block
149     */
150    public int getMtfLength() {
151
152        return this.mtfLength;
153
154    }
155
156
157    /**
158     * @return The size of the MTF block's alphabet
159     */
160    public int getMtfAlphabetSize() {
161
162        return this.alphabetSize;
163
164    }
165
166
167    /**
168     * @return The frequencies of the MTF block's symbols
169     */
170    public int[] getMtfSymbolFrequencies() {
171
172        return this.mtfSymbolFrequencies;
173
174    }
175
176
177    /**
178     * @param bwtBlock         The Burrows Wheeler Transformed block data
179     * @param bwtLength        The actual length of the BWT data
180     * @param bwtValuesPresent The values that are present within the BWT data. For each index,
181     *                         {@code true} if that value is present within the data, otherwise {@code false}
182     */
183    public BZip2MTFAndRLE2StageEncoder(final int[] bwtBlock, final int bwtLength, final boolean[] bwtValuesPresent) {
184
185        this.bwtBlock = bwtBlock;
186        this.bwtLength = bwtLength;
187        this.bwtValuesInUse = bwtValuesPresent;
188        this.mtfBlock = new char[bwtLength + 1];
189
190    }
191
192}
