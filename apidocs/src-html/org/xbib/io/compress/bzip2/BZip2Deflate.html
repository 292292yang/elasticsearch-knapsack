142    protected static void hbMakeCodeLengths(char[] len, int[] freq,
143                                            int alphaSize, int maxLen) throws IOException {
144        /*
145        Nodes and heap entries run from 1.  Entry 0
146        for both the heap and nodes is a sentinel.
147         */
148        int nNodes, nHeap, n1, n2, i, j, k;
149        boolean tooLong;
150
151        int[] heap = new int[MAX_ALPHA_SIZE + 2];
152        int[] weight = new int[MAX_ALPHA_SIZE * 2];
153        int[] parent = new int[MAX_ALPHA_SIZE * 2];
154
155        for (i = 0; i < alphaSize; i++) {
156            weight[i + 1] = (freq[i] == 0 ? 1 : freq[i]) << 8;
157        }
158
159        while (true) {
160            nNodes = alphaSize;
161            nHeap = 0;
162
163            heap[0] = 0;
164            weight[0] = 0;
165            parent[0] = -2;
166
167            for (i = 1; i <= alphaSize; i++) {
168                parent[i] = -1;
169                nHeap++;
170                heap[nHeap] = i;
171                {
172                    int zz, tmp;
173                    zz = nHeap;
174                    tmp = heap[zz];
175                    while (weight[tmp] < weight[heap[zz >> 1]]) {
176                        heap[zz] = heap[zz >> 1];
177                        zz >>= 1;
178                    }
179                    heap[zz] = tmp;
180                }
181            }
182            if (!(nHeap < (MAX_ALPHA_SIZE + 2))) {
183                throw new IOException("panic");
184            }
185
186            while (nHeap > 1) {
187                n1 = heap[1];
188                heap[1] = heap[nHeap];
189                nHeap--;
190                {
191                    int zz = 0, yy = 0, tmp = 0;
192                    zz = 1;
193                    tmp = heap[zz];
194                    while (true) {
195                        yy = zz << 1;
196                        if (yy > nHeap) {
197                            break;
198                        }
199                        if (yy < nHeap && weight[heap[yy + 1]] < weight[heap[yy]]) {
200                            yy++;
201                        }
202                        if (weight[tmp] < weight[heap[yy]]) {
203                            break;
204                        }
205                        heap[zz] = heap[yy];
206                        zz = yy;
207                    }
208                    heap[zz] = tmp;
209                }
210                n2 = heap[1];
211                heap[1] = heap[nHeap];
212                nHeap--;
213                {
214                    int zz = 0, yy = 0, tmp = 0;
215                    zz = 1;
216                    tmp = heap[zz];
217                    while (true) {
218                        yy = zz << 1;
219                        if (yy > nHeap) {
220                            break;
221                        }
222                        if (yy < nHeap && weight[heap[yy + 1]] < weight[heap[yy]]) {
223                            yy++;
224                        }
225                        if (weight[tmp] < weight[heap[yy]]) {
226                            break;
227                        }
228                        heap[zz] = heap[yy];
229                        zz = yy;
230                    }
231                    heap[zz] = tmp;
232                }
233                nNodes++;
234                parent[n1] = parent[n2] = nNodes;
235
236                weight[nNodes] = ((weight[n1] & 0xffffff00) + (weight[n2] & 0xffffff00)) | (1 + (((weight[n1] & 0x000000ff) > (weight[n2] & 0x000000ff))
237                        ? (weight[n1] & 0x000000ff)
238                        : (weight[n2] & 0x000000ff)));
239
240                parent[nNodes] = -1;
241                nHeap++;
242                heap[nHeap] = nNodes;
243                {
244                    int zz = 0, tmp = 0;
245                    zz = nHeap;
246                    tmp = heap[zz];
247                    while (weight[tmp] < weight[heap[zz >> 1]]) {
248                        heap[zz] = heap[zz >> 1];
249                        zz >>= 1;
250                    }
251                    heap[zz] = tmp;
252                }
253            }
254            if (!(nNodes < (MAX_ALPHA_SIZE * 2))) {
255                throw new IOException("panic");
256            }
257
258            tooLong = false;
259            for (i = 1; i <= alphaSize; i++) {
260                j = 0;
261                k = i;
262                while (parent[k] >= 0) {
263                    k = parent[k];
264                    j++;
265                }
266                len[i - 1] = (char) j;
267                if (j > maxLen) {
268                    tooLong = true;
269                }
270            }
271
272            if (!tooLong) {
273                break;
274            }
275
276            for (i = 1; i < alphaSize; i++) {
277                j = weight[i] >> 8;
278                j = 1 + (j / 2);
279                weight[i] = j << 8;
280            }
281        }
282    }

323    public int setInput(int bv) throws IOException {
324        int b = (256 + bv) % 256;
325        if (currentChar != -1) {
326            if (currentChar == b) {
327                runLength++;
328                if (runLength > 254) {
329                    writeRun();
330                    currentChar = -1;
331                    runLength = 0;
332                }
333            } else {
334                writeRun();
335                runLength = 1;
336                currentChar = b;
337            }
338        } else {
339            currentChar = b;
340            runLength++;
341        }
342
343        this.totalIn++;
344        return this.inputBuffer.size();
345    }

381    /**
382     * Fills specified buffer with compressed data.
383     *
384     * @param b array to fill
385     * @return number of bytes added to the array
386     */
387    public int deflate(byte[] b) {
388        int rc = 0;
389
390        if (!this.wroteHeader) {
391            this.wroteHeader = true;
392
393            this.outputBuffer = new ByteArrayInputStream(new byte[]{0x42, 0x5A});
394        }
395
396        if (this.outputBuffer != null) {
397            try {
398                if (this.outputBuffer.available() > 0) {
399                    rc = this.outputBuffer.read(b);
400                }
401            } catch (IOException eIO) {
402            }
403        }
404
405        if (rc < b.length && this.inputBuffer.size() != 0) {
406            if (this.inputBuffer.size() < b.length - rc) {
407                byte[] t = this.inputBuffer.toByteArray();
408                this.inputBuffer.reset();
409
410                System.arraycopy(t, 0, b, rc, t.length);
411                rc += t.length;
412            } else {
413                byte[] t = this.inputBuffer.toByteArray();
414                this.inputBuffer.reset();
415
416                System.arraycopy(t, 0, b, rc, b.length - rc);
417                this.outputBuffer = new ByteArrayInputStream(t, b.length - rc, t.length - (b.length - rc));
418                rc = b.length;
419            }
420        }
421
422        this.totalOut += rc;
423
424        return rc;
425    }

428     * Fills specified buffer with compressed data.

431     * @param off index to start filling at
432     * @param len maximum number of bytes to add
433     * @return number of bytes added to the array

435    public int deflate(byte[] b, int off, int len) {
436        byte[] buff = new byte[len - off];
437
438        int rc = this.deflate(buff);
439
440        if (rc > 0) {
441            System.arraycopy(buff, 0, b, off, rc);
442        }
443
444        return rc;
445    }

447    private void writeRun() throws IOException {
448        if (last < allowableBlockSize) {
449            inUse[currentChar] = true;
450            for (int i = 0; i < runLength; i++) {
451                mCrc.updateCRC((char) currentChar);
452            }
453            switch (runLength) {
454                case 1:
455                    last++;
456                    block[last + 1] = (char) currentChar;
457                    break;
458                case 2:
459                    last++;
460                    block[last + 1] = (char) currentChar;
461                    last++;
462                    block[last + 1] = (char) currentChar;
463                    break;
464                case 3:
465                    last++;
466                    block[last + 1] = (char) currentChar;
467                    last++;
468                    block[last + 1] = (char) currentChar;
469                    last++;
470                    block[last + 1] = (char) currentChar;
471                    break;
472                default:
473                    inUse[runLength - 4] = true;
474                    last++;
475                    block[last + 1] = (char) currentChar;
476                    last++;
477                    block[last + 1] = (char) currentChar;
478                    last++;
479                    block[last + 1] = (char) currentChar;
480                    last++;
481                    block[last + 1] = (char) currentChar;
482                    last++;
483                    block[last + 1] = (char) (runLength - 4);
484                    break;
485            }
486        } else {
487            endBlock();
488            initBlock();
489            writeRun();
490        }
491    }

570    private void initialize() throws IOException {

584    private void initBlock() {
585        //        blockNo++;
586        mCrc.initialiseCRC();
587        last = -1;
588        //        ch = 0;
589
590        for (int i = 0; i < 256; i++) {
591            inUse[i] = false;
592        }
593
594        /* 20 is just a paranoia constant */
595        allowableBlockSize = BASE_BLOCK_SIZE * blockSize100k - 20;
596    }

598    private void endBlock() throws IOException {
599        blockCRC = mCrc.getFinalCRC();
600        combinedCRC = (combinedCRC << 1) | (combinedCRC >>> 31);
601        combinedCRC ^= blockCRC;
602
603        /* sort the block and establish posn of original string */
604        doReversibleTransformation();
605
606        /*
607        A 6-byte block header, the value chosen arbitrarily
608        as 0x314159265359 :-).  A 32 bit value does not really
609        give a strong enough guarantee that the value will not
610        appear by chance in the compressed datastream.  Worst-case
611        probability of this event, for a 900k block, is about
612        2.0e-3 for 32 bits, 1.0e-5 for 40 bits and 4.0e-8 for 48 bits.
613        For a compressed file of size 100Gb -- about 100000 blocks --
614        only a 48-bit marker will do.  NB: normal compression/
615        decompression do *not* rely on these statistical properties.
616        They are only important when trying to recover blocks from
617        damaged files.
618         */
619        bsPutUChar(0x31);
620        bsPutUChar(0x41);
621        bsPutUChar(0x59);
622        bsPutUChar(0x26);
623        bsPutUChar(0x53);
624        bsPutUChar(0x59);
625
626        /* Now the block's CRC, so it is in a known place. */
627        bsPutint(blockCRC);
628
629        /* Now a single bit indicating randomisation. */
630        if (blockRandomised) {
631            bsW(1, 1);
632        } else {
633            bsW(1, 0);
634        }
635
636        /* Finally, block's contents proper. */
637        moveToFrontCodeAndSend();
638    }
639
640    private void endCompression() throws IOException {
641        /*
642        Now another magic 48-bit number, 0x177245385090, to
643        indicate the end of the last block.  (sqrt(pi), if
644        you want to know.  I did want to use e, but it contains
645        too much repetition -- 27 18 28 18 28 46 -- for me
646        to feel statistically comfortable.  Call me paranoid.)
647         */
648        bsPutUChar(0x17);
649        bsPutUChar(0x72);
650        bsPutUChar(0x45);
651        bsPutUChar(0x38);
652        bsPutUChar(0x50);
653        bsPutUChar(0x90);
654
655        bsPutint(combinedCRC);
656
657        bsFinishedWithStream();
658    }

660    private void hbAssignCodes(int[] code, char[] length, int minLen,
661                               int maxLen, int alphaSize) {
662        int n, vec, i;
663
664        vec = 0;
665        for (n = minLen; n <= maxLen; n++) {
666            for (i = 0; i < alphaSize; i++) {
667                if (length[i] == n) {
668                    code[i] = vec;
669                    vec++;
670                }
671            }
672            vec <<= 1;
673        }
674    }

683    private void bsFinishedWithStream() throws IOException {
684        while (bsLive > 0) {
685            int ch = (bsBuff >> 24);
686
687            inputBuffer.write(ch); // write 8-bit
688
689            bsBuff <<= 8;
690            bsLive -= 8;
691            bytesOut++;
692        }
693    }
694
695    private void bsW(int n, int v) throws IOException {
696        while (bsLive >= 8) {
697            int ch = (bsBuff >> 24);
698
699            inputBuffer.write(ch); // write 8-bit
700
701            bsBuff <<= 8;
702            bsLive -= 8;
703            bytesOut++;
704        }
705        bsBuff |= (v << (32 - bsLive - n));
706        bsLive += n;
707    }

709    private void bsPutUChar(int c) throws IOException {

724    private void sendMTFValues() throws IOException {
725        char len[][] = new char[N_GROUPS][MAX_ALPHA_SIZE];
726
727        int v, t, i, j, gs, ge, totc, bt, bc, iter;
728        int nSelectors = 0, alphaSize, minLen, maxLen, selCtr;
729        int nGroups;
730
731        alphaSize = nInUse + 2;
732        for (t = 0; t < N_GROUPS; t++) {
733            for (v = 0; v < alphaSize; v++) {
734                len[t][v] = (char) GREATER_ICOST;
735            }
736        }
737
738        /* Decide how many coding tables to use */
739        if (nMTF <= 0) {
740            throw new IOException("panic");
741        }
742
743        if (nMTF < 200) {
744            nGroups = 2;
745        } else if (nMTF < 600) {
746            nGroups = 3;
747        } else if (nMTF < 1200) {
748            nGroups = 4;
749        } else if (nMTF < 2400) {
750            nGroups = 5;
751        } else {
752            nGroups = 6;
753        }
754
755        /* Generate an initial set of coding tables */
756        {
757            int nPart, remF, tFreq, aFreq;
758
759            nPart = nGroups;
760            remF = nMTF;
761            gs = 0;
762            while (nPart > 0) {
763                tFreq = remF / nPart;
764                ge = gs - 1;
765                aFreq = 0;
766                while (aFreq < tFreq && ge < alphaSize - 1) {
767                    ge++;
768                    aFreq += mtfFreq[ge];
769                }
770
771                if (ge > gs && nPart != nGroups && nPart != 1 && ((nGroups - nPart) % 2 == 1)) {
772                    aFreq -= mtfFreq[ge];
773                    ge--;
774                }
775
776                for (v = 0; v < alphaSize; v++) {
777                    if (v >= gs && v <= ge) {
778                        len[nPart - 1][v] = (char) LESSER_ICOST;
779                    } else {
780                        len[nPart - 1][v] = (char) GREATER_ICOST;
781                    }
782                }
783
784                nPart--;
785                gs = ge + 1;
786                remF -= aFreq;
787            }
788        }
789
790        int[][] rfreq = new int[N_GROUPS][MAX_ALPHA_SIZE];
791        int[] fave = new int[N_GROUPS];
792        short[] cost = new short[N_GROUPS];
793        /*
794        Iterate up to N_ITERS times to improve the tables.
795         */
796        for (iter = 0; iter < N_ITERS; iter++) {
797            for (t = 0; t < nGroups; t++) {
798                fave[t] = 0;
799            }
800
801            for (t = 0; t < nGroups; t++) {
802                for (v = 0; v < alphaSize; v++) {
803                    rfreq[t][v] = 0;
804                }
805            }
806
807            nSelectors = 0;
808            totc = 0;
809            gs = 0;
810            while (true) {
811
812                /* Set group start & end marks. */
813                if (gs >= nMTF) {
814                    break;
815                }
816                ge = gs + G_SIZE - 1;
817                if (ge >= nMTF) {
818                    ge = nMTF - 1;
819                }
820
821                /*
822                Calculate the cost of this group as coded
823                by each of the coding tables.
824                 */
825                for (t = 0; t < nGroups; t++) {
826                    cost[t] = 0;
827                }
828
829                if (nGroups == 6) {
830                    short cost0, cost1, cost2, cost3, cost4, cost5;
831                    cost0 = cost1 = cost2 = cost3 = cost4 = cost5 = 0;
832                    for (i = gs; i <= ge; i++) {
833                        short icv = szptr[i];
834                        cost0 += len[0][icv];
835                        cost1 += len[1][icv];
836                        cost2 += len[2][icv];
837                        cost3 += len[3][icv];
838                        cost4 += len[4][icv];
839                        cost5 += len[5][icv];
840                    }
841                    cost[0] = cost0;
842                    cost[1] = cost1;
843                    cost[2] = cost2;
844                    cost[3] = cost3;
845                    cost[4] = cost4;
846                    cost[5] = cost5;
847                } else {
848                    for (i = gs; i <= ge; i++) {
849                        short icv = szptr[i];
850                        for (t = 0; t < nGroups; t++) {
851                            cost[t] += len[t][icv];
852                        }
853                    }
854                }
855
856                /*
857                Find the coding table which is best for this group,
858                and record its identity in the selector table.
859                 */
860                bc = 999999999;
861                bt = -1;
862                for (t = 0; t < nGroups; t++) {
863                    if (cost[t] < bc) {
864                        bc = cost[t];
865                        bt = t;
866                    }
867                }
868                totc += bc;
869                fave[bt]++;
870                selector[nSelectors] = (char) bt;
871                nSelectors++;
872
873                /*
874                Increment the symbol frequencies for the selected table.
875                 */
876                for (i = gs; i <= ge; i++) {
877                    rfreq[bt][szptr[i]]++;
878                }
879
880                gs = ge + 1;
881            }
882
883            /*
884            Recompute the tables based on the accumulated frequencies.
885             */
886            for (t = 0; t < nGroups; t++) {
887                hbMakeCodeLengths(len[t], rfreq[t], alphaSize, 20);
888            }
889        }
890
891        rfreq = null;
892        fave = null;
893        cost = null;
894
895        if (!(nGroups < 8)) {
896            throw new IOException("panic");
897        }
898        if (!(nSelectors < 32768 && nSelectors <= (2 + (900000 / G_SIZE)))) {
899            throw new IOException("panic");
900        }
901
902
903        /* Compute MTF values for the selectors. */
904        {
905            char[] pos = new char[N_GROUPS];
906            char lli, tmp2, tmp;
907            for (i = 0; i < nGroups; i++) {
908                pos[i] = (char) i;
909            }
910            for (i = 0; i < nSelectors; i++) {
911                lli = selector[i];
912                j = 0;
913                tmp = pos[j];
914                while (lli != tmp) {
915                    j++;
916                    tmp2 = tmp;
917                    tmp = pos[j];
918                    pos[j] = tmp2;
919                }
920                pos[0] = tmp;
921                selectorMtf[i] = (char) j;
922            }
923        }
924
925        int[][] code = new int[N_GROUPS][MAX_ALPHA_SIZE];
926
927        /* Assign actual codes for the tables. */
928        for (t = 0; t < nGroups; t++) {
929            minLen = 32;
930            maxLen = 0;
931            for (i = 0; i < alphaSize; i++) {
932                if (len[t][i] > maxLen) {
933                    maxLen = len[t][i];
934                }
935                if (len[t][i] < minLen) {
936                    minLen = len[t][i];
937                }
938            }
939            if (maxLen > 20) {
940                throw new IOException("panic");
941            }
942            if (minLen < 1) {
943                throw new IOException("panic");
944            }
945            hbAssignCodes(code[t], len[t], minLen, maxLen, alphaSize);
946        }
947
948        /* Transmit the mapping table. */
949        {
950            boolean[] inUse16 = new boolean[16];
951            for (i = 0; i < 16; i++) {
952                inUse16[i] = false;
953                for (j = 0; j < 16; j++) {
954                    if (inUse[i * 16 + j]) {
955                        inUse16[i] = true;
956                    }
957                }
958            }
959
960            //int nBytes = bytesOut;
961            for (i = 0; i < 16; i++) {
962                if (inUse16[i]) {
963                    bsW(1, 1);
964                } else {
965                    bsW(1, 0);
966                }
967            }
968
969            for (i = 0; i < 16; i++) {
970                if (inUse16[i]) {
971                    for (j = 0; j < 16; j++) {
972                        if (inUse[i * 16 + j]) {
973                            bsW(1, 1);
974                        } else {
975                            bsW(1, 0);
976                        }
977                    }
978                }
979            }
980
981        }
982
983        /* Now the selectors. */
984        bsW(3, nGroups);
985        bsW(15, nSelectors);
986        for (i = 0; i < nSelectors; i++) {
987            for (j = 0; j < selectorMtf[i]; j++) {
988                bsW(1, 1);
989            }
990            bsW(1, 0);
991        }
992
993        /* Now the coding tables. */
994
995        for (t = 0; t < nGroups; t++) {
996            int curr = len[t][0];
997            bsW(5, curr);
998            for (i = 0; i < alphaSize; i++) {
999                while (curr < len[t][i]) {
1000                    bsW(2, 2);
1001                    curr++; /* 10 */
1002                }
1003                while (curr > len[t][i]) {
1004                    bsW(2, 3);
1005                    curr--; /* 11 */
1006                }
1007                bsW(1, 0);
1008            }
1009        }
1010
1011        /* And finally, the block data proper */
1012        selCtr = 0;
1013        gs = 0;
1014        while (true) {
1015            if (gs >= nMTF) {
1016                break;
1017            }
1018            ge = gs + G_SIZE - 1;
1019            if (ge >= nMTF) {
1020                ge = nMTF - 1;
1021            }
1022            for (i = gs; i <= ge; i++) {
1023                bsW(len[selector[selCtr]][szptr[i]],
1024                        code[selector[selCtr]][szptr[i]]);
1025            }
1026
1027            gs = ge + 1;
1028            selCtr++;
1029        }
1030        if (!(selCtr == nSelectors)) {
1031            throw new IOException("panic");
1032        }
1033    }

1044    private void simpleSort(int lo, int hi, int d) {
1045        int i, j, h, bigN, hp;
1046        int v;
1047
1048        bigN = hi - lo + 1;
1049        if (bigN < 2) {
1050            return;
1051        }
1052
1053        hp = 0;
1054        while (incs[hp] < bigN) {
1055            hp++;
1056        }
1057        hp--;
1058
1059        for (; hp >= 0; hp--) {
1060            h = incs[hp];
1061
1062            i = lo + h;
1063            while (true) {
1064                /* copy 1 */
1065                if (i > hi) {
1066                    break;
1067                }
1068                v = zptr[i];
1069                j = i;
1070                while (fullGtU(zptr[j - h] + d, v + d)) {
1071                    zptr[j] = zptr[j - h];
1072                    j = j - h;
1073                    if (j <= (lo + h - 1)) {
1074                        break;
1075                    }
1076                }
1077                zptr[j] = v;
1078                i++;
1079
1080                /* copy 2 */
1081                if (i > hi) {
1082                    break;
1083                }
1084                v = zptr[i];
1085                j = i;
1086                while (fullGtU(zptr[j - h] + d, v + d)) {
1087                    zptr[j] = zptr[j - h];
1088                    j = j - h;
1089                    if (j <= (lo + h - 1)) {
1090                        break;
1091                    }
1092                }
1093                zptr[j] = v;
1094                i++;
1095
1096                /* copy 3 */
1097                if (i > hi) {
1098                    break;
1099                }
1100                v = zptr[i];
1101                j = i;
1102                while (fullGtU(zptr[j - h] + d, v + d)) {
1103                    zptr[j] = zptr[j - h];
1104                    j = j - h;
1105                    if (j <= (lo + h - 1)) {
1106                        break;
1107                    }
1108                }
1109                zptr[j] = v;
1110                i++;
1111
1112                if (workDone > workLimit && firstAttempt) {
1113                    return;
1114                }
1115            }
1116        }
1117    }

1156    private void qSort3(int loSt, int hiSt, int dSt) throws IOException {
1157        int unLo, unHi, ltLo, gtHi, med, n, m;
1158        int sp, lo, hi, d;
1159        StackElem[] stack = new StackElem[QSORT_STACK_SIZE];
1160        for (int count = 0; count < QSORT_STACK_SIZE; count++) {
1161            stack[count] = new StackElem();
1162        }
1163
1164        sp = 0;
1165
1166        stack[sp].ll = loSt;
1167        stack[sp].hh = hiSt;
1168        stack[sp].dd = dSt;
1169        sp++;
1170
1171        while (sp > 0) {
1172            if (sp >= QSORT_STACK_SIZE) {
1173                throw new IOException("panic");
1174            }
1175
1176            sp--;
1177            lo = stack[sp].ll;
1178            hi = stack[sp].hh;
1179            d = stack[sp].dd;
1180
1181            if (hi - lo < SMALL_THRESH || d > DEPTH_THRESH) {
1182                simpleSort(lo, hi, d);
1183                if (workDone > workLimit && firstAttempt) {
1184                    return;
1185                }
1186                continue;
1187            }
1188
1189            med = med3(block[zptr[lo] + d + 1],
1190                    block[zptr[hi] + d + 1],
1191                    block[zptr[(lo + hi) >> 1] + d + 1]);
1192
1193            unLo = ltLo = lo;
1194            unHi = gtHi = hi;
1195
1196            while (true) {
1197                while (true) {
1198                    if (unLo > unHi) {
1199                        break;
1200                    }
1201                    n = ((int) block[zptr[unLo] + d + 1]) - med;
1202                    if (n == 0) {
1203                        int temp = 0;
1204                        temp = zptr[unLo];
1205                        zptr[unLo] = zptr[ltLo];
1206                        zptr[ltLo] = temp;
1207                        ltLo++;
1208                        unLo++;
1209                        continue;
1210                    }
1211                    if (n > 0) {
1212                        break;
1213                    }
1214                    unLo++;
1215                }
1216                while (true) {
1217                    if (unLo > unHi) {
1218                        break;
1219                    }
1220                    n = ((int) block[zptr[unHi] + d + 1]) - med;
1221                    if (n == 0) {
1222                        int temp = 0;
1223                        temp = zptr[unHi];
1224                        zptr[unHi] = zptr[gtHi];
1225                        zptr[gtHi] = temp;
1226                        gtHi--;
1227                        unHi--;
1228                        continue;
1229                    }
1230                    if (n < 0) {
1231                        break;
1232                    }
1233                    unHi--;
1234                }
1235                if (unLo > unHi) {
1236                    break;
1237                }
1238                int temp = 0;
1239                temp = zptr[unLo];
1240                zptr[unLo] = zptr[unHi];
1241                zptr[unHi] = temp;
1242                unLo++;
1243                unHi--;
1244            }
1245
1246            if (gtHi < ltLo) {
1247                stack[sp].ll = lo;
1248                stack[sp].hh = hi;
1249                stack[sp].dd = d + 1;
1250                sp++;
1251                continue;
1252            }
1253
1254            n = ((ltLo - lo) < (unLo - ltLo)) ? (ltLo - lo) : (unLo - ltLo);
1255            vswap(lo, unLo - n, n);
1256            m = ((hi - gtHi) < (gtHi - unHi)) ? (hi - gtHi) : (gtHi - unHi);
1257            vswap(unLo, hi - m + 1, m);
1258
1259            n = lo + unLo - ltLo - 1;
1260            m = hi - (gtHi - unHi) + 1;
1261
1262            stack[sp].ll = lo;
1263            stack[sp].hh = n;
1264            stack[sp].dd = d;
1265            sp++;
1266
1267            stack[sp].ll = n + 1;
1268            stack[sp].hh = m - 1;
1269            stack[sp].dd = d + 1;
1270            sp++;
1271
1272            stack[sp].ll = m;
1273            stack[sp].hh = hi;
1274            stack[sp].dd = d;
1275            sp++;
1276        }
1277    }

1279    private void mainSort() throws IOException {
1280        int i, j, ss, sb;
1281        int[] runningOrder = new int[256];
1282        int[] copy = new int[256];
1283        boolean[] bigDone = new boolean[256];
1284        int c1, c2;
1285        
1286
1287        /*
1288        In the various block-sized structures, live data runs
1289        from 0 to last+NUM_OVERSHOOT_BYTES inclusive.  First,
1290        set up the overshoot area for block.
1291         */
1292
1293        //   if (verbosity >= 4) fprintf ( stderr, "   sort initialise ...\n" );
1294
1295        // if nothing was written, last is -1, which gives div by zero
1296        if (last >= 0) {
1297            for (i = 0; i < NUM_OVERSHOOT_BYTES; i++) {
1298                block[last + i + 2] = block[(i % (last + 1)) + 1];
1299            }
1300        }
1301        for (i = 0; i <= last + NUM_OVERSHOOT_BYTES; i++) {
1302            quadrant[i] = 0;
1303        }
1304
1305        block[0] = (block[last + 1]);
1306
1307        if (last < 4000) {
1308            /*
1309            Use simpleSort(), since the full sorting mechanism
1310            has quite a large constant overhead.
1311             */
1312            for (i = 0; i <= last; i++) {
1313                zptr[i] = i;
1314            }
1315            firstAttempt = false;
1316            workDone = workLimit = 0;
1317            simpleSort(0, last, 0);
1318        } else {
1319            for (i = 0; i <= 255; i++) {
1320                bigDone[i] = false;
1321            }
1322
1323            for (i = 0; i <= 65536; i++) {
1324                ftab[i] = 0;
1325            }
1326
1327            c1 = block[0];
1328            for (i = 0; i <= last; i++) {
1329                c2 = block[i + 1];
1330                ftab[(c1 << 8) + c2]++;
1331                c1 = c2;
1332            }
1333
1334            for (i = 1; i <= 65536; i++) {
1335                ftab[i] += ftab[i - 1];
1336            }
1337
1338            c1 = block[1];
1339            for (i = 0; i < last; i++) {
1340                c2 = block[i + 2];
1341                j = (c1 << 8) + c2;
1342                c1 = c2;
1343                ftab[j]--;
1344                zptr[ftab[j]] = i;
1345            }
1346
1347            j = ((block[last + 1]) << 8) + (block[1]);
1348            ftab[j]--;
1349            zptr[ftab[j]] = last;
1350
1351            /*
1352            Now ftab contains the first loc of every small bucket.
1353            Calculate the running order, from smallest to largest
1354            big bucket.
1355             */
1356
1357            for (i = 0; i <= 255; i++) {
1358                runningOrder[i] = i;
1359            }
1360
1361            {
1362                int vv;
1363                int h = 1;
1364                do {
1365                    h = 3 * h + 1;
1366                } while (h <= 256);
1367                do {
1368                    h = h / 3;
1369                    for (i = h; i <= 255; i++) {
1370                        vv = runningOrder[i];
1371                        j = i;
1372                        while ((ftab[((runningOrder[j - h]) + 1) << 8] - ftab[(runningOrder[j - h]) << 8]) > (ftab[((vv) + 1) << 8] - ftab[(vv) << 8])) {
1373                            runningOrder[j] = runningOrder[j - h];
1374                            j = j - h;
1375                            if (j <= (h - 1)) {
1376                                break;
1377                            }
1378                        }
1379                        runningOrder[j] = vv;
1380                    }
1381                } while (h != 1);
1382            }
1383
1384            /*
1385            The main sorting loop.
1386             */
1387            for (i = 0; i <= 255; i++) {
1388
1389                /*
1390                Process big buckets, starting with the least full.
1391                 */
1392                ss = runningOrder[i];
1393
1394                /*
1395                Complete the big bucket [ss] by quicksorting
1396                any unsorted small buckets [ss, j].  Hopefully
1397                previous pointer-scanning phases have already
1398                completed many of the small buckets [ss, j], so
1399                we don't have to sort them at all.
1400                 */
1401                for (j = 0; j <= 255; j++) {
1402                    sb = (ss << 8) + j;
1403                    if (!((ftab[sb] & SETMASK) == SETMASK)) {
1404                        int lo = ftab[sb] & CLEARMASK;
1405                        int hi = (ftab[sb + 1] & CLEARMASK) - 1;
1406                        if (hi > lo) {
1407                            qSort3(lo, hi, 2);
1408                            if (workDone > workLimit && firstAttempt) {
1409                                return;
1410                            }
1411                        }
1412                        ftab[sb] |= SETMASK;
1413                    }
1414                }
1415
1416                /*
1417                The ss big bucket is now done.  Record this fact,
1418                and update the quadrant descriptors.  Remember to
1419                update quadrants in the overshoot area too, if
1420                necessary.  The "if (i < 255)" test merely skips
1421                this updating for the last bucket processed, since
1422                updating for the last bucket is pointless.
1423                 */
1424                bigDone[ss] = true;
1425
1426                if (i < 255) {
1427                    int bbStart = ftab[ss << 8] & CLEARMASK;
1428                    int bbSize = (ftab[(ss + 1) << 8] & CLEARMASK) - bbStart;
1429                    int shifts = 0;
1430
1431                    while ((bbSize >> shifts) > 65534) {
1432                        shifts++;
1433                    }
1434
1435                    for (j = 0; j < bbSize; j++) {
1436                        int a2update = zptr[bbStart + j];
1437                        int qVal = (j >> shifts);
1438                        quadrant[a2update] = qVal;
1439                        if (a2update < NUM_OVERSHOOT_BYTES) {
1440                            quadrant[a2update + last + 1] = qVal;
1441                        }
1442                    }
1443
1444                    if (!(((bbSize - 1) >> shifts) <= 65535)) {
1445                        throw new IOException("panic");
1446                    }
1447                }
1448
1449                /*
1450                Now scan this big bucket so as to synthesise the
1451                sorted order for small buckets [t, ss] for all t != ss.
1452                 */
1453                for (j = 0; j <= 255; j++) {
1454                    copy[j] = ftab[(j << 8) + ss] & CLEARMASK;
1455                }
1456
1457                for (j = ftab[ss << 8] & CLEARMASK;
1458                     j < (ftab[(ss + 1) << 8] & CLEARMASK); j++) {
1459                    c1 = block[zptr[j]];
1460                    if (!bigDone[c1]) {
1461                        zptr[copy[c1]] = zptr[j] == 0 ? last : zptr[j] - 1;
1462                        copy[c1]++;
1463                    }
1464                }
1465
1466                for (j = 0; j <= 255; j++) {
1467                    ftab[(j << 8) + ss] |= SETMASK;
1468                }
1469            }
1470        }
1471    }

1529    private boolean fullGtU(int p1, int p2) {
1530        int i1 = p1, i2 = p2;
1531        int k;
1532        char c1, c2;
1533        int s1, s2;
1534
1535        c1 = block[i1 + 1];
1536        c2 = block[i2 + 1];
1537        if (c1 != c2) {
1538            return (c1 > c2);
1539        }
1540        i1++;
1541        i2++;
1542
1543        c1 = block[i1 + 1];
1544        c2 = block[i2 + 1];
