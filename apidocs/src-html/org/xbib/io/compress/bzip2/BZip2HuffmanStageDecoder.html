023package org.xbib.io.compress.bzip2;
024
025import java.io.IOException;
026
027
028/**
029 * A decoder for the BZip2 Huffman coding stage
030 */
031public class BZip2HuffmanStageDecoder {
032
033    /**
034     * The BZip2BitInputStream from which Huffman codes are read
035     */
036    private final BZip2BitInputStream bitInputStream;
037
038    /**
039     * The Huffman table number to use for each group of 50 symbols
040     */
041    private final byte[] selectors;
042
043    /**
044     * The minimum code length for each Huffman table
045     */
046    private final int[] minimumLengths = new int[BZip2Constants.HUFFMAN_MAXIMUM_TABLES];
047
048    /**
049     * An array of values for each Huffman table that must be subtracted from the numerical value of
050     * a Huffman code of a given bit length to give its canonical code index
051     */
052    private final int[][] codeBases = new int[BZip2Constants.HUFFMAN_MAXIMUM_TABLES][BZip2Constants.HUFFMAN_DECODE_MAXIMUM_CODE_LENGTH + 2];
053
054    /**
055     * An array of values for each Huffman table that gives the highest numerical value of a Huffman
056     * code of a given bit length
057     */
058    private final int[][] codeLimits = new int[BZip2Constants.HUFFMAN_MAXIMUM_TABLES][BZip2Constants.HUFFMAN_DECODE_MAXIMUM_CODE_LENGTH + 1];
059
060    /**
061     * A mapping for each Huffman table from canonical code index to output symbol
062     */
063    private final int[][] codeSymbols = new int[BZip2Constants.HUFFMAN_MAXIMUM_TABLES][BZip2Constants.HUFFMAN_MAXIMUM_ALPHABET_SIZE];
064
065    /**
066     * The Huffman table for the current group
067     */
068    private int currentTable;
069
070    /**
071     * The index of the current group within the selectors array
072     */
073    private int groupIndex = -1;
074
075    /**
076     * The byte position within the current group. A new group is selected every 50 decoded bytes
077     */
078    private int groupPosition = -1;
079
080
081    /**
082     * @param bitInputStream   The BZip2BitInputStream from which Huffman codes are read
083     * @param alphabetSize     The total number of codes (uniform for each table)
084     * @param tableCodeLengths The Canonical Huffman code lengths for each table
085     * @param selectors        The Huffman table number to use for each group of 50 symbols
086     */
087    public BZip2HuffmanStageDecoder(final BZip2BitInputStream bitInputStream, final int alphabetSize, final byte[][] tableCodeLengths, final byte[] selectors) {
088
089        this.bitInputStream = bitInputStream;
090        this.selectors = selectors;
091        this.currentTable = this.selectors[0];
092
093        createHuffmanDecodingTables(alphabetSize, tableCodeLengths);
094
095    }
096
097
098    /**
099     * Constructs Huffman decoding tables from lists of Canonical Huffman code lengths
100     *
101     * @param alphabetSize     The total number of codes (uniform for each table)
102     * @param tableCodeLengths The Canonical Huffman code lengths for each table
103     */
104    private void createHuffmanDecodingTables(final int alphabetSize, final byte[][] tableCodeLengths) {
105
106        for (int table = 0; table < tableCodeLengths.length; table++) {
107
108            final int[] tableBases = this.codeBases[table];
109            final int[] tableLimits = this.codeLimits[table];
110            final int[] tableSymbols = this.codeSymbols[table];
111
112            final byte[] codeLengths = tableCodeLengths[table];
113            int minimumLength = BZip2Constants.HUFFMAN_DECODE_MAXIMUM_CODE_LENGTH;
114            int maximumLength = 0;
115
116            // Find the minimum and maximum code length for the table
117            for (int i = 0; i < alphabetSize; i++) {
118                maximumLength = Math.max(codeLengths[i], maximumLength);
119                minimumLength = Math.min(codeLengths[i], minimumLength);
120            }
121            this.minimumLengths[table] = minimumLength;
122
123            // Calculate the first output symbol for each code length
124            for (int i = 0; i < alphabetSize; i++) {
125                tableBases[codeLengths[i] + 1]++;
126            }
127            for (int i = 1; i < BZip2Constants.HUFFMAN_DECODE_MAXIMUM_CODE_LENGTH + 2; i++) {
128                tableBases[i] += tableBases[i - 1];
129            }
130
131            // Calculate the first and last Huffman code for each code length (codes at a given
132            // length are sequential in value)
133            int code = 0;
134            for (int i = minimumLength; i <= maximumLength; i++) {
135                int base = code;
136                code += tableBases[i + 1] - tableBases[i];
137                tableBases[i] = base - tableBases[i];
138                tableLimits[i] = code - 1;
139                code <<= 1;
140            }
141
142            // Populate the mapping from canonical code index to output symbol
143            int codeIndex = 0;
144            for (int bitLength = minimumLength; bitLength <= maximumLength; bitLength++) {
145                for (int symbol = 0; symbol < alphabetSize; symbol++) {
146                    if (codeLengths[symbol] == bitLength) {
147                        tableSymbols[codeIndex++] = symbol;
148                    }
149                }
150            }
151
152        }
153
154    }
155
156
157    /**
158     * Decodes and returns the next symbol
159     *
160     * @return The decoded symbol
161     * @throws java.io.IOException if the end of the input stream is reached while decoding
162     */
163    public int nextSymbol() throws IOException {
164
165        final BZip2BitInputStream bitInputStream = this.bitInputStream;
166
167        // Move to next group selector if required
168        if (((++this.groupPosition % BZip2Constants.HUFFMAN_GROUP_RUN_LENGTH) == 0)) {
169            this.groupIndex++;
170            if (this.groupIndex == this.selectors.length) {
171                throw new BZip2Exception("Error decoding BZip2 block");
172            }
173            this.currentTable = this.selectors[this.groupIndex] & 0xff;
174        }
175
176        final int currentTable = this.currentTable;
177        final int[] tableLimits = this.codeLimits[currentTable];
178        int codeLength = this.minimumLengths[currentTable];
179
180        // Starting with the minimum bit length for the table, read additional bits one at a time
181        // until a complete code is recognised
182        int codeBits = bitInputStream.readBits(codeLength);
183        for (; codeLength <= BZip2Constants.HUFFMAN_DECODE_MAXIMUM_CODE_LENGTH; codeLength++) {
184            if (codeBits <= tableLimits[codeLength]) {
185                // Convert the code to a symbol index and return
186                return this.codeSymbols[currentTable][codeBits - this.codeBases[currentTable][codeLength]];
187            }
188            codeBits = (codeBits << 1) | bitInputStream.readBits(1);
189        }
190
191        // A valid code was not recognised
192        throw new BZip2Exception("Error decoding BZip2 block");
193
194    }
195
196}
