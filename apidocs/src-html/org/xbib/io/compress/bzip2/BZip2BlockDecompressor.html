028/*
029 * Block decoding consists of the following stages:
030 * 1. Read block header - BZip2BlockDecompressor()
031 * 2. Read Huffman tables - readHuffmanTables()
032 * 3. Read and decode Huffman encoded data - decodeHuffmanData()
033 * 4. Run-Length Decoding[2] - decodeHuffmanData()
034 * 5. Inverse Move To Front Transform - decodeHuffmanData()
035 * 6. Inverse Burrows Wheeler Transform - initialiseInverseBWT()
036 * 7. Run-Length Decoding[1] - read()
037 * 8. Optional Block De-Randomisation - read() (through decodeNextBWTByte())
038 */
039
040/**
041 * Reads and decompresses a single BZip2 block
042 */

090     * Provides bits of input to decode
091     */
092    private final BZip2BitInputStream bitInputStream;
093
094    /**
095     * Calculates the block CRC from the fully decoded bytes of the block
096     */
097    private final CRC32 crc = new CRC32();
098
099    /**
100     * The CRC of the current block as read from the block header
101     */
102    private final int blockCRC;
103
104    /**
105     * {@code true} if the current block is randomised, otherwise {@code false}
106     */
107    private final boolean blockRandomised;
108
109        /* Huffman Decoding stage */
110
111    /**
112     * The end-of-block Huffman symbol. Decoding of the block ends when this is encountered
113     */
114    private int huffmanEndOfBlockSymbol;
115
116    /**
117     * A map from Huffman symbol index to output character. Some types of data (e.g. ASCII text)
118     * may contain only a limited number of byte values; Huffman symbols are only allocated to
119     * those values that actually occur in the uncompressed data.
120     */
121    private final byte[] huffmanSymbolMap = new byte[256];
122
123        /* Move To Front stage */
124
125    /**
126     * Counts of each byte value within the {@link bwtTransformedArray} data. Collected at the Move
127     * To Front stage, consumed by the Inverse Burrows Wheeler Transform stage
128     */
129    private final int[] bwtByteCounts = new int[256];
130
131    /**
132     * The Burrows-Wheeler Transform processed data. Read at the Move To Front stage, consumed by the
133     * Inverse Burrows Wheeler Transform stage
134     */
135    private byte[] bwtBlock;
136
137        /* Inverse Burrows-Wheeler Transform stage */
138
139    /**
140     * At each position contains the union of :-
141     * An output character (8 bits)
142     * A pointer from each position to its successor (24 bits, left shifted 8 bits)
143     * As the pointer cannot exceed the maximum block size of 900k, 24 bits is more than enough to
144     * hold it; Folding the character data into the spare bits while performing the inverse BWT,
145     * when both pieces of information are available, saves a large number of memory accesses in
146     * the final decoding stages.
147     */
148    private int[] bwtMergedPointers;
149
150    /**
151     * The current merged pointer into the Burrow-Wheeler Transform array
152     */
153    private int bwtCurrentMergedPointer;
154
155    /**
156     * The actual length in bytes of the current block at the Inverse Burrows Wheeler Transform
157     * stage (before final Run-Length Decoding)
158     */
159    private int bwtBlockLength;
160
161    /**
162     * The number of output bytes that have been decoded up to the Inverse Burrows Wheeler Transform
163     * stage
164     */
165    private int bwtBytesDecoded;
166
167        /* Run-Length Encoding and Random Perturbation stage */
168
169    /**
170     * The most recently RLE decoded byte
171     */
172    private int rleLastDecodedByte = -1;
173
174    /**
175     * The number of previous identical output bytes decoded. After 4 identical bytes, the next byte
176     * decoded is an RLE repeat count
177     */
178    private int rleAccumulator;
179
180    /**
181     * The RLE repeat count of the current decoded byte. When this reaches zero, a new byte is
182     * decoded
183     */
184    private int rleRepeat;
185
186    /**
187     * If the current block is randomised, the position within the RNUMS randomisation array
188     */
189    private int randomIndex = 0;
190
191    /**
192     * If the current block is randomised, the remaining count at the current RNUMS position
193     */
194    private int randomCount = RNUMS[0] - 1;

197    /**
198     * Read and decode the block's Huffman tables
199     *
200     * @return A decoder for the Huffman stage that uses the decoded tables
201     * @throws java.io.IOException if the input stream reaches EOF before all table data has been read
202     */
203    private BZip2HuffmanStageDecoder readHuffmanTables() throws IOException {
204
205        final BZip2BitInputStream bitInputStream = this.bitInputStream;
206        final byte[] huffmanSymbolMap = this.huffmanSymbolMap;
207        final byte[][] tableCodeLengths = new byte[BZip2Constants.HUFFMAN_MAXIMUM_TABLES][BZip2Constants.HUFFMAN_MAXIMUM_ALPHABET_SIZE];
208
209                /* Read Huffman symbol to output byte map */
210        int huffmanUsedRanges = bitInputStream.readBits(16);
211        int huffmanSymbolCount = 0;
212
213        for (int i = 0; i < 16; i++) {
214            if ((huffmanUsedRanges & ((1 << 15) >>> i)) != 0) {
215                for (int j = 0, k = i << 4; j < 16; j++, k++) {
216                    if (bitInputStream.readBoolean()) {
217                        huffmanSymbolMap[huffmanSymbolCount++] = (byte) k;
218                    }
219                }
220            }
221        }
222        int endOfBlockSymbol = huffmanSymbolCount + 1;
223        this.huffmanEndOfBlockSymbol = endOfBlockSymbol;
224
225                /* Read total number of tables and selectors*/
226        final int totalTables = bitInputStream.readBits(3);
227        final int totalSelectors = bitInputStream.readBits(15);
228        if (
229                (totalTables < BZip2Constants.HUFFMAN_MINIMUM_TABLES)
230                        || (totalTables > BZip2Constants.HUFFMAN_MAXIMUM_TABLES)
231                        || (totalSelectors < 1)
232                        || (totalSelectors > BZip2Constants.HUFFMAN_MAXIMUM_SELECTORS)
233                ) {
234            throw new BZip2Exception("BZip2 block Huffman tables invalid");
235        }
236
237                /* Read and decode MTFed Huffman selector list */
238        final MoveToFront tableMTF = new MoveToFront();
239        final byte[] selectors = new byte[totalSelectors];
240        for (int selector = 0; selector < totalSelectors; selector++) {
241            selectors[selector] = tableMTF.indexToFront(bitInputStream.readUnary());
242        }
243
244                /* Read the Canonical Huffman code lengths for each table */
245        for (int table = 0; table < totalTables; table++) {
246            int currentLength = bitInputStream.readBits(5);
247            for (int i = 0; i <= endOfBlockSymbol; i++) {
248                while (bitInputStream.readBoolean()) {
249                    currentLength += bitInputStream.readBoolean() ? -1 : 1;
250                }
251                tableCodeLengths[table][i] = (byte) currentLength;
252            }
253        }
254
255        return new BZip2HuffmanStageDecoder(bitInputStream, endOfBlockSymbol + 1, tableCodeLengths, selectors);
256
257    }

260    /**
261     * Reads the Huffman encoded data from the input stream, performs Run-Length Decoding and
262     * applies the Move To Front transform to reconstruct the Burrows-Wheeler Transform array
263     *
264     * @param huffmanDecoder The Huffman decoder through which symbols are read
265     * @throws java.io.IOException if an end-of-block symbol was not decoded within the declared block size
266     */
267    private void decodeHuffmanData(final BZip2HuffmanStageDecoder huffmanDecoder) throws IOException {
268
269        final byte[] bwtBlock = this.bwtBlock;
270        final byte[] huffmanSymbolMap = this.huffmanSymbolMap;
271        final int streamBlockSize = this.bwtBlock.length;
272        final int huffmanEndOfBlockSymbol = this.huffmanEndOfBlockSymbol;
273        final int[] bwtByteCounts = this.bwtByteCounts;
274        final MoveToFront symbolMTF = new MoveToFront();
275        int bwtBlockLength = 0;
276        int repeatCount = 0;
277        int repeatIncrement = 1;
278        int mtfValue = 0;
279
280        for (; ; ) {
281            final int nextSymbol = huffmanDecoder.nextSymbol();
282
283            if (nextSymbol == BZip2Constants.HUFFMAN_SYMBOL_RUNA) {
284                repeatCount += repeatIncrement;
285                repeatIncrement <<= 1;
286            } else if (nextSymbol == BZip2Constants.HUFFMAN_SYMBOL_RUNB) {
287                repeatCount += repeatIncrement << 1;
288                repeatIncrement <<= 1;
289            } else {
290                if (repeatCount > 0) {
291                    if (bwtBlockLength + repeatCount > streamBlockSize) {
292                        throw new BZip2Exception("BZip2 block exceeds declared block size");
293                    }
294                    final byte nextByte = huffmanSymbolMap[mtfValue];
295                    bwtByteCounts[nextByte & 0xff] += repeatCount;
296                    while (--repeatCount >= 0) {
297                        bwtBlock[bwtBlockLength++] = nextByte;
298                    }
299
300                    repeatCount = 0;
301                    repeatIncrement = 1;
302                }
303
304                if (nextSymbol == huffmanEndOfBlockSymbol) {
305                    break;
306                }
307
308                if (bwtBlockLength >= streamBlockSize) {
309                    throw new BZip2Exception("BZip2 block exceeds declared block size");
310                }
311
312                mtfValue = symbolMTF.indexToFront(nextSymbol - 1) & 0xff;
313
314                final byte nextByte = huffmanSymbolMap[mtfValue];
315                bwtByteCounts[nextByte & 0xff]++;
316                bwtBlock[bwtBlockLength++] = nextByte;
317
318            }
319        }
320
321        this.bwtBlockLength = bwtBlockLength;
322
323    }

326    /**
327     * Set up the Inverse Burrows-Wheeler Transform merged pointer array
328     *
329     * @param bwtStartPointer The start pointer into the BWT array
330     * @throws java.io.IOException if the given start pointer is invalid
331     */
332    private void initialiseInverseBWT(final int bwtStartPointer) throws IOException {
333
334        final byte[] bwtBlock = this.bwtBlock;
335        final int[] bwtMergedPointers = new int[this.bwtBlockLength];
336        final int[] characterBase = new int[256];
337
338        if ((bwtStartPointer < 0) || (bwtStartPointer >= this.bwtBlockLength)) {
339            throw new BZip2Exception("BZip2 start pointer invalid");
340        }
341
342        // Cumulatise character counts
343        System.arraycopy(this.bwtByteCounts, 0, characterBase, 1, 255);
344        for (int i = 2; i <= 255; i++) {
345            characterBase[i] += characterBase[i - 1];
346        }
347
348        // Merged-Array Inverse Burrows-Wheeler Transform
349        // Combining the output characters and forward pointers into a single array here, where we
350        // have already read both of the corresponding values, cuts down on memory accesses in the
351        // final walk through the array
352        for (int i = 0; i < this.bwtBlockLength; i++) {
353            int value = bwtBlock[i] & 0xff;
354            bwtMergedPointers[characterBase[value]++] = (i << 8) + value;
355        }
356
357        this.bwtBlock = null;
358        this.bwtMergedPointers = bwtMergedPointers;
359        this.bwtCurrentMergedPointer = bwtMergedPointers[bwtStartPointer];
360
361    }

364    /**
365     * Decodes a byte from the Burrows-Wheeler Transform stage. If the block has randomisation
366     * applied, reverses the randomisation
367     *
368     * @return The decoded byte
369     */
370    private int decodeNextBWTByte() {
371
372        int mergedPointer = this.bwtCurrentMergedPointer;
373        int nextDecodedByte = mergedPointer & 0xff;
374        this.bwtCurrentMergedPointer = this.bwtMergedPointers[mergedPointer >>> 8];
375
376        if (this.blockRandomised) {
377            if (--this.randomCount == 0) {
378                nextDecodedByte ^= 1;
379                this.randomIndex = (this.randomIndex + 1) % 512;
380                this.randomCount = RNUMS[this.randomIndex];
381            }
382        }
383
384        this.bwtBytesDecoded++;
385
386        return nextDecodedByte;
387
388    }

391    /**
392     * Decodes a byte from the final Run-Length Encoding stage, pulling a new byte from the
393     * Burrows-Wheeler Transform stage when required
394     *
395     * @return The decoded byte, or -1 if there are no more bytes
396     */
397    public int read() {
398
399        while (this.rleRepeat < 1) {
400
401            if (this.bwtBytesDecoded == this.bwtBlockLength) {
402                return -1;
403            }
404
405            int nextByte = decodeNextBWTByte();
406
407            if (nextByte != this.rleLastDecodedByte) {
408                // New byte, restart accumulation
409                this.rleLastDecodedByte = nextByte;
410                this.rleRepeat = 1;
411                this.rleAccumulator = 1;
412                this.crc.updateCRC(nextByte);
413            } else {
414                if (++this.rleAccumulator == 4) {
415                    // Accumulation complete, start repetition
416                    int rleRepeat = decodeNextBWTByte() + 1;
417                    this.rleRepeat = rleRepeat;
418                    this.rleAccumulator = 0;
419                    this.crc.updateCRC(nextByte, rleRepeat);
420                } else {
421                    this.rleRepeat = 1;
422                    this.crc.updateCRC(nextByte);
423                }
424            }
425
426        }
427
428        this.rleRepeat--;
429
430        return this.rleLastDecodedByte;
431
432    }

435    /**
436     * Decodes multiple bytes from the final Run-Length Encoding stage, pulling new bytes from the
437     * Burrows-Wheeler Transform stage when required
438     *
439     * @param destination The array to write to
440     * @param offset      The starting position within the array
441     * @param length      The number of bytes to read
442     * @return The number of bytes actually read, or -1 if there are no bytes left in the block
443     */
444    public int read(final byte[] destination, int offset, final int length) {
445
446        int i;
447        for (i = 0; i < length; i++, offset++) {
448            int decoded = read();
449            if (decoded == -1) {
450                return (i == 0) ? -1 : i;
451            }
452            destination[offset] = (byte) decoded;
453        }
454        return i;
455
456    }

459    /**
460     * Verify and return the block CRC. This method may only be called after all of the block's
461     * bytes have been read
462     *
463     * @return The block CRC
464     * @throws java.io.IOException if the CRC verification failed
465     */
466    public int checkCRC() throws IOException {
467
468        if (this.blockCRC != this.crc.getCRC()) {
469            throw new BZip2Exception("BZip2 block CRC error");
470        }
471
472        return this.crc.getCRC();
473
474    }

477    /**
478     * @param bitInputStream The BZip2BitInputStream to read from
479     * @param blockSize      The maximum decoded size of the block
480     * @throws java.io.IOException If the block could not be decoded
481     */
482    public BZip2BlockDecompressor(final BZip2BitInputStream bitInputStream, final int blockSize) throws IOException {
483
484        this.bitInputStream = bitInputStream;
485        this.bwtBlock = new byte[blockSize];
486
487        final int bwtStartPointer;
488
489        // Read block header
490        this.blockCRC = this.bitInputStream.readInteger();
491        this.blockRandomised = this.bitInputStream.readBoolean();
492        bwtStartPointer = this.bitInputStream.readBits(24);
493
494        // Read block data and decode through to the Inverse Burrows Wheeler Transform stage
495        BZip2HuffmanStageDecoder huffmanDecoder = readHuffmanTables();
496        decodeHuffmanData(huffmanDecoder);
497        initialiseInverseBWT(bwtStartPointer);
498
499    }
