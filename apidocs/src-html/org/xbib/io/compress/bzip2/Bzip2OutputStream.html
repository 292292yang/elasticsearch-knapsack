020package org.xbib.io.compress.bzip2;
021
022import java.io.ByteArrayOutputStream;
023import java.io.IOException;
024import java.io.OutputStream;
025
026/**
027 *  BZip2OutputSream
028 *
029 *  Encapsulates any outputstream to write a bzip2-ed stream to it.
030 *
031 */
032public class Bzip2OutputStream extends OutputStream {
033
034    private ByteArrayOutputStream buffer;
035    private BZip2Deflate deflater;
036    private OutputStream os;
037    private int bufferMax;
038
039    /**
040     * Wraps an OutputStream in a BZip2OutputStream.
041     *
042     * Any bytes written to this stream will be compressed using the bzip2 algorithm on the fly
043     *
044     * @param os OutputStream to encapsulate
045     */
046    public Bzip2OutputStream(OutputStream os) throws IOException {
047        this(os, 8192);
048    }
049
050    /**
051     * Wraps an OutputStream in a BZip2OutputStream.
052     *
053     * Any bytes written to this stream will be compressed using the bzip2 algorithm on the fly
054     *
055     * @param os OutputStream to encapsulate
056     */
057    public Bzip2OutputStream(OutputStream os, int bufferMax) throws IOException {
058        super();
059        this.bufferMax = bufferMax;
060        this.buffer = new ByteArrayOutputStream(bufferMax);
061        this.deflater = new BZip2Deflate(1, bufferMax, true);
062        this.os = os;
063    }
064
065    /**
066     * Write integer to OutputStream
067     *
068     * @param b integer to write to stream
069     */
070    @Override
071    public void write(int b) throws IOException {
072        if (buffer.size() > bufferMax) {
073            flush();
074        }
075        if (deflater.setInput(b) > 0) {
076            deflate();
077        }
078    }
079
080    /**
081     * Write byte array OutputStream
082     *
083     * @param b byte array to write
084     */
085    @Override
086    public void write(byte[] b) throws IOException {
087        write(b, 0, b.length);
088    }
089
090    /**
091     * Write part of a byte array OutputStream
092     *
093     * @param b byte array to write
094     * @param off       index in byte array to start output at
095     * @param len       number of bytes to write
096     */
097    @Override
098    public void write(byte[] b, int off, int len) throws IOException {
099        if (buffer.size() > bufferMax) {
100            flush();
101        }
102        if (deflater.setInput(b, off, len) > 0) {
103            deflate();
104        }
105    }
106
107    /**
108     * Close OutputStream
109     */
110    @Override
111    public void close() throws IOException {
112        deflater.finish();
113        while (deflate() > 0) {
114            flush();
115        }
116        os.close();
117    }
118
119    /**
120     * Flush OutputStream
121     */
122    @Override
123    public void flush() throws IOException {
124        if (buffer.size() > 0) {
125            os.write(buffer.toByteArray());
126            buffer.reset();
127            os.flush();
128        }
129    }
130
131    /**
132     * reads add empties the BZip2Deflate buffer
133     *
134     * @return the number of bytes read into the local output buffer.
135     */
136    private int deflate() {
137        byte[] b = new byte[bufferMax];
138        int c;
139        c = deflater.deflate(b);
140        if (c > 0) {
141            buffer.write(b, 0, c);
142        }
143        return c;
144    }
145}
