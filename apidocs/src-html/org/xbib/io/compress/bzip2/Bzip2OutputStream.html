002    package org.xbib.io.compress.bzip2;
003    
004    import java.io.ByteArrayOutputStream;
005    import java.io.IOException;
006    import java.io.OutputStream;
007    
008    /**
009     * BZip2OutputSream
010     * <p/>
011     * Encapsulates any outputstream to write a bzip2-ed stream to it.
012     */
013    public class Bzip2OutputStream extends OutputStream {
014    
015        private ByteArrayOutputStream buffer;
016        private BZip2Deflate deflater;
017        private OutputStream os;
018        private int bufferMax;
019    
020        /**
021         * Wraps an OutputStream in a BZip2OutputStream.
022         * <p/>
023         * Any bytes written to this stream will be compressed using the bzip2 algorithm on the fly
024         *
025         * @param os OutputStream to encapsulate
026         */
027        public Bzip2OutputStream(OutputStream os) throws IOException {
028            this(os, 8192);
029        }
030    
031        /**
032         * Wraps an OutputStream in a BZip2OutputStream.
033         * <p/>
034         * Any bytes written to this stream will be compressed using the bzip2 algorithm on the fly
035         *
036         * @param os OutputStream to encapsulate
037         */
038        public Bzip2OutputStream(OutputStream os, int bufferMax) throws IOException {
039            super();
040            this.bufferMax = bufferMax;
041            this.buffer = new ByteArrayOutputStream(bufferMax);
042            this.deflater = new BZip2Deflate(1, bufferMax, true);
043            this.os = os;
044        }
045    
046        /**
047         * Write integer to OutputStream
048         *
049         * @param b integer to write to stream
050         */
051        @Override
052        public void write(int b) throws IOException {
053            if (buffer.size() > bufferMax) {
054                flush();
055            }
056            if (deflater.setInput(b) > 0) {
057                deflate();
058            }
059        }
060    
061        /**
062         * Write byte array OutputStream
063         *
064         * @param b byte array to write
065         */
066        @Override
067        public void write(byte[] b) throws IOException {
068            write(b, 0, b.length);
069        }
070    
071        /**
072         * Write part of a byte array OutputStream
073         *
074         * @param b   byte array to write
075         * @param off index in byte array to start output at
076         * @param len number of bytes to write
077         */
078        @Override
079        public void write(byte[] b, int off, int len) throws IOException {
080            if (buffer.size() > bufferMax) {
081                flush();
082            }
083            if (deflater.setInput(b, off, len) > 0) {
084                deflate();
085            }
086        }
087    
088        /**
089         * Close OutputStream
090         */
091        @Override
092        public void close() throws IOException {
093            deflater.finish();
094            while (deflate() > 0) {
095                flush();
096            }
097            os.close();
098        }
099    
100        /**
101         * Flush OutputStream
102         */
103        @Override
104        public void flush() throws IOException {
105            if (buffer.size() > 0) {
106                os.write(buffer.toByteArray());
107                buffer.reset();
108                os.flush();
109            }
110        }
111    
112        /**
113         * reads add empties the BZip2Deflate buffer
114         *
115         * @return the number of bytes read into the local output buffer.
116         */
117        private int deflate() {
118            byte[] b = new byte[bufferMax];
119            int c;
120            c = deflater.deflate(b);
121            if (c > 0) {
122                buffer.write(b, 0, c);
123            }
124            return c;
125        }
126    }
