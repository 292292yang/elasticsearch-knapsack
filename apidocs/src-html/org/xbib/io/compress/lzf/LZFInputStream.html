001    
002    package org.xbib.io.compress.lzf;
003    
004    import org.xbib.io.compress.BufferRecycler;
005    
006    import java.io.IOException;
007    import java.io.InputStream;
008    import java.io.OutputStream;
009    
010    /**
011     * Decorator {@link java.io.InputStream} implementation used for reading compressed data
012     * and uncompressing it on the fly, such that reads return uncompressed data.
013     * Its direct counterpart is {@link LZFOutputStream}; but there is also
014     * {@link LZFCompressingInputStream} which does reverse of this class.
015     */
016    public class LZFInputStream extends InputStream {
017    
018        /**
019         * Underlying decoder in use.
020         */
021        protected final ChunkDecoder _decoder;
022        /**
023         * Object that handles details of buffer recycling
024         */
025        protected final BufferRecycler _recycler;
026        /**
027         * stream to be decompressed
028         */
029        protected final InputStream _inputStream;
030        /**
031         * Flag that indicates if we have already called 'inputStream.close()' (to
032         * avoid calling it multiple times)
033         */
034        protected boolean _inputStreamClosed;
035        /**
036         * Flag that indicates whether we force full reads (reading of as many bytes
037         * as requested), or 'optimal' reads (up to as many as available, but at
038         * least one). Default is false, meaning that 'optimal' read is used.
039         */
040        protected boolean _cfgFullReads = false;
041        /**
042         * the current buffer of compressed bytes (from which to decode)
043         */
044        protected byte[] _inputBuffer;
045        /**
046         * the buffer of uncompressed bytes from which content is read
047         */
048        protected byte[] _decodedBytes;
049        /**
050         * The current position (next char to output) in the uncompressed bytes
051         * buffer.
052         */
053        protected int _bufferPosition = 0;
054        /**
055         * Length of the current uncompressed bytes buffer
056         */
057        protected int _bufferLength = 0;
058        /**
059         * Number of bytes read from the underlying {@link #_inputStream}
060         */
061        protected int _readCount = 0;
062    
063        /*
064         // Construction
065         */
066        public LZFInputStream(final InputStream inputStream) throws IOException {
067            this(inputStream, false);
068        }
069    
070        /**
071         * @param in        Underlying input stream to use
072         * @param fullReads Whether {@link #read(byte[])} should try to read exactly
073         *                  as many bytes as requested (true); or just however many happen to be
074         *                  available (false)
075         */
076        public LZFInputStream(final InputStream in, boolean fullReads) throws IOException {
077            this(in, fullReads, ChunkDecoderFactory.optimalInstance());
078        }
079    
080        public LZFInputStream(final InputStream in, boolean fullReads, ChunkDecoder decoder)
081                throws IOException {
082            super();
083            _decoder = decoder;
084            _recycler = BufferRecycler.instance();
085            _inputStream = in;
086            _inputStreamClosed = false;
087            _cfgFullReads = fullReads;
088    
089            _inputBuffer = _recycler.allocInputBuffer(LZFChunk.MAX_CHUNK_LEN);
090            _decodedBytes = _recycler.allocDecodeBuffer(LZFChunk.MAX_CHUNK_LEN);
091        }
092    
093        /**
094         * Method that can be used define whether reads should be "full" or
095         * "optimal": former means that full compressed blocks are read right away
096         * as needed, optimal that only smaller chunks are read at a time, more
097         * being read as needed.
098         */
099        public void setUseFullReads(boolean b) {
100            _cfgFullReads = b;
101        }
102    
103        /*
104         * InputStream impl
105         */
106    
107        /**
108         * Method is overridden to report number of bytes that can now be read from
109         * decoded data buffer, without reading bytes from the underlying stream.
110         * Never throws an exception; returns number of bytes available without
111         * further reads from underlying source; -1 if stream has been closed, or 0
112         * if an actual read (and possible blocking) is needed to find out.
113         */
114        @Override
115        public int available() {
116            // if closed, return -1;
117            if (_inputStreamClosed) {
118                return -1;
119            }
120            int left = (_bufferLength - _bufferPosition);
121            return (left <= 0) ? 0 : left;
122        }
123    
124        @Override
125        public int read() throws IOException {
126            if (!readyBuffer()) {
127                return -1;
128            }
129            return _decodedBytes[_bufferPosition++] & 255;
130        }
131    
132        @Override
133        public int read(final byte[] buffer) throws IOException {
134            return read(buffer, 0, buffer.length);
135        }
136    
137        @Override
138        public int read(final byte[] buffer, int offset, int length) throws IOException {
139            if (length < 1) {
140                return 0;
141            }
142            if (!readyBuffer()) {
143                return -1;
144            }
145            // First let's read however much data we happen to have...
146            int chunkLength = Math.min(_bufferLength - _bufferPosition, length);
147            System.arraycopy(_decodedBytes, _bufferPosition, buffer, offset, chunkLength);
148            _bufferPosition += chunkLength;
149    
150            if (chunkLength == length || !_cfgFullReads) {
151                return chunkLength;
152            }
153            // Need more data, then
154            int totalRead = chunkLength;
155            do {
156                offset += chunkLength;
157                if (!readyBuffer()) {
158                    break;
159                }
160                chunkLength = Math.min(_bufferLength - _bufferPosition, (length - totalRead));
161                System.arraycopy(_decodedBytes, _bufferPosition, buffer, offset, chunkLength);
162                _bufferPosition += chunkLength;
163                totalRead += chunkLength;
164            } while (totalRead < length);
165    
166            return totalRead;
167        }
168    
169        @Override
170        public void close() throws IOException {
171            _bufferPosition = _bufferLength = 0;
172            byte[] buf = _inputBuffer;
173            if (buf != null) {
174                _inputBuffer = null;
175                _recycler.releaseInputBuffer(buf);
176            }
177            buf = _decodedBytes;
178            if (buf != null) {
179                _decodedBytes = null;
180                _recycler.releaseDecodeBuffer(buf);
181            }
182            if (!_inputStreamClosed) {
183                _inputStreamClosed = true;
184                _inputStream.close();
185            }
186        }
187    
188        /**
189         * Overridden to just skip at most a single chunk at a time
190         */
191        @Override
192        public long skip(long n) throws IOException {
193            if (_inputStreamClosed) {
194                return -1;
195            }
196            int left = (_bufferLength - _bufferPosition);
197            // if none left, must read more:
198            if (left <= 0) {
199                // otherwise must read more to skip...
200                int b = read();
201                if (b < 0) { // EOF
202                    return -1;
203                }
204                // push it back to get accurate skip count
205                --_bufferPosition;
206                left = (_bufferLength - _bufferPosition);
207            }
208            // either way, just skip whatever we have decoded
209            if (left > n) {
210                left = (int) n;
211            }
212            _bufferPosition += left;
213            return left;
214        }
215    
216        /**
217         * Method that can be used to find underlying {@link java.io.InputStream} that we
218         * read from to get LZF encoded data to decode. Will never return null;
219         * although underlying stream may be closed (if this stream has been
220         * closed).
221         */
222        public InputStream getUnderlyingInputStream() {
223            return _inputStream;
224        }
225    
226        /**
227         * Method that can be called to discard any already buffered input, read
228         * from input source. Specialized method that only makes sense if the
229         * underlying {@link java.io.InputStream} can be repositioned reliably.
230         */
231        public void discardBuffered() {
232            _bufferPosition = _bufferLength = 0;
233        }
234    
235        /**
236         * Convenience method that will read and uncompress all data available, and
237         * write it using given {@link java.io.OutputStream}. This avoids having to make an
238         * intermediate copy of uncompressed data which would be needed when doing
239         * the same manually.
240         *
241         * @param out OutputStream to use for writing content
242         * @return Number of bytes written (uncompressed)
243         */
244        public int readAndWrite(OutputStream out) throws IOException {
245            int total = 0;
246    
247            while (readyBuffer()) {
248                int avail = _bufferLength - _bufferPosition;
249                out.write(_decodedBytes, _bufferPosition, avail);
250                _bufferPosition += avail; // to ensure it looks like we consumed it all
251                total += avail;
252            }
253            return total;
254        }
255    
256        /**
257         * Fill the uncompressed bytes buffer by reading the underlying inputStream.
258         *
259         * @throws java.io.IOException
260         */
261        protected boolean readyBuffer() throws IOException {
262            if (_bufferPosition < _bufferLength) {
263                return true;
264            }
265            if (_inputStreamClosed) {
266                return false;
267            }
268            _bufferLength = _decoder.decodeChunk(_inputStream, _inputBuffer, _decodedBytes);
269            if (_bufferLength < 0) {
270                return false;
271            }
272            _bufferPosition = 0;
273            return (_bufferPosition < _bufferLength);
274        }
275    }
