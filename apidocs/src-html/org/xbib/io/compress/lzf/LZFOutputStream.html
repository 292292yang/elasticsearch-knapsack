002    package org.xbib.io.compress.lzf;
003    
004    import org.xbib.io.compress.BufferRecycler;
005    
006    import java.io.IOException;
007    import java.io.OutputStream;
008    
009    /**
010     * Decorator {@link java.io.OutputStream} implementation that will compress output using
011     * LZF compression algorithm, given uncompressed input to write. Its counterpart
012     * is {@link LZFInputStream}; although in some ways
013     * {@link LZFCompressingInputStream} can be seen as the opposite.
014     */
015    public class LZFOutputStream extends OutputStream {
016    
017        private final ChunkEncoder _encoder;
018        private final BufferRecycler _recycler;
019        protected final OutputStream _outputStream;
020        protected byte[] _outputBuffer;
021        protected int _position = 0;
022        /**
023         * Configuration setting that governs whether basic 'flush()' should first
024         * complete a block or not. <p> Default value is 'true'
025         */
026        protected boolean _cfgFinishBlockOnFlush = true;
027        /**
028         * Flag that indicates if we have already called '_outputStream.close()' (to
029         * avoid calling it multiple times)
030         */
031        protected boolean _outputStreamClosed;
032    
033        /*
034         // Construction, configuration
035         */
036        public LZFOutputStream(final OutputStream outputStream) {
037            this(outputStream, LZFChunk.MAX_CHUNK_LEN);
038        }
039    
040        public LZFOutputStream(final OutputStream outputStream, int bufsize) {
041            _encoder = new ChunkEncoder(bufsize);
042            _recycler = BufferRecycler.instance();
043            _outputStream = outputStream;
044            _outputBuffer = _recycler.allocOutputBuffer(bufsize);
045            _outputStreamClosed = false;
046        }
047    
048        /**
049         * Method for defining whether call to {@link #flush} will also complete
050         * current block (similar to calling {@link #finishBlock()}) or not.
051         */
052        public LZFOutputStream setFinishBlockOnFlush(boolean b) {
053            _cfgFinishBlockOnFlush = b;
054            return this;
055        }
056    
057        /*
058         // OutputStream impl
059         */
060        @Override
061        public void write(final int singleByte) throws IOException {
062            checkNotClosed();
063            if (_position >= _outputBuffer.length) {
064                writeCompressedBlock();
065            }
066            _outputBuffer[_position++] = (byte) singleByte;
067        }
068    
069        @Override
070        public void write(final byte[] buffer, int offset, int length) throws IOException {
071            checkNotClosed();
072    
073            final int BUFFER_LEN = _outputBuffer.length;
074    
075            // simple case first: buffering only (for trivially short writes)
076            int free = BUFFER_LEN - _position;
077            if (free >= length) {
078                System.arraycopy(buffer, offset, _outputBuffer, _position, length);
079                _position += length;
080                return;
081            }
082            // otherwise, copy whatever we can, flush
083            System.arraycopy(buffer, offset, _outputBuffer, _position, free);
084            offset += free;
085            length -= free;
086            _position += free;
087            writeCompressedBlock();
088    
089            // then write intermediate full block, if any, without copying:
090            while (length >= BUFFER_LEN) {
091                _encoder.encodeAndWriteChunk(buffer, offset, BUFFER_LEN, _outputStream);
092                offset += BUFFER_LEN;
093                length -= BUFFER_LEN;
094            }
095    
096            // and finally, copy leftovers in buffer, if any
097            if (length > 0) {
098                System.arraycopy(buffer, offset, _outputBuffer, 0, length);
099            }
100            _position = length;
101        }
102    
103        @Override
104        public void flush() throws IOException {
105            checkNotClosed();
106            if (_cfgFinishBlockOnFlush && _position > 0) {
107                writeCompressedBlock();
108            }
109            _outputStream.flush();
110        }
111    
112        @Override
113        public void close() throws IOException {
114            if (!_outputStreamClosed) {
115                if (_position > 0) {
116                    writeCompressedBlock();
117                }
118                _outputStream.flush();
119                _encoder.close();
120                byte[] buf = _outputBuffer;
121                if (buf != null) {
122                    _outputBuffer = null;
123                    _recycler.releaseOutputBuffer(buf);
124                }
125                _outputStreamClosed = true;
126                _outputStream.close();
127            }
128        }
129    
130        /**
131         * Method that can be used to find underlying {@link java.io.OutputStream} that we
132         * write encoded LZF encoded data into, after compressing it. Will never
133         * return null; although underlying stream may be closed (if this stream has
134         * been closed).
135         */
136        public OutputStream getUnderlyingOutputStream() {
137            return _outputStream;
138        }
139    
140        /**
141         * Accessor for checking whether call to "flush()" will first finish the
142         * current block or not
143         */
144        public boolean getFinishBlockOnFlush() {
145            return _cfgFinishBlockOnFlush;
146        }
147    
148        /**
149         * Method that can be used to force completion of the current block, which
150         * means that all buffered data will be compressed into an LZF block. This
151         * typically results in lower compression ratio as larger blocks compress
152         * better; but may be necessary for network connections to ensure timely
153         * sending of data.
154         */
155        public LZFOutputStream finishBlock() throws IOException {
156            checkNotClosed();
157            if (_position > 0) {
158                writeCompressedBlock();
159            }
160            return this;
161        }
162    
163        /**
164         * Compress and write the current block to the OutputStream
165         */
166        protected void writeCompressedBlock() throws IOException {
167            int left = _position;
168            _position = 0;
169            int offset = 0;
170    
171            do {
172                int chunkLen = Math.min(LZFChunk.MAX_CHUNK_LEN, left);
173                _encoder.encodeAndWriteChunk(_outputBuffer, offset, chunkLen, _outputStream);
174                offset += chunkLen;
175                left -= chunkLen;
176            } while (left > 0);
177        }
178    
179        protected void checkNotClosed() throws IOException {
180            if (_outputStreamClosed) {
181                throw new IOException(getClass().getName() + " already closed");
182            }
183        }
184    }
