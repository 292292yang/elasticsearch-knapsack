016package org.xbib.io.compress.lzf;
017
018import java.io.IOException;
019import java.io.OutputStream;
020
021/**
022 * Decorator {@link java.io.OutputStream} implementation that will compress output using
023 * LZF compression algorithm, given uncompressed input to write. Its counterpart
024 * is {@link LZFInputStream}; although in some ways
025 * {@link LZFCompressingInputStream} can be seen as the opposite.
026 */
027public class LZFOutputStream extends OutputStream {
028
029    private final ChunkEncoder _encoder;
030    private final BufferRecycler _recycler;
031    protected final OutputStream _outputStream;
032    protected byte[] _outputBuffer;
033    protected int _position = 0;
034    /**
035     * Configuration setting that governs whether basic 'flush()' should first
036     * complete a block or not. <p> Default value is 'true'
037     */
038    protected boolean _cfgFinishBlockOnFlush = true;
039    /**
040     * Flag that indicates if we have already called '_outputStream.close()' (to
041     * avoid calling it multiple times)
042     */
043    protected boolean _outputStreamClosed;
044
045    /*
046     // Construction, configuration
047     */
048    public LZFOutputStream(final OutputStream outputStream) {
049        this(outputStream, LZFChunk.MAX_CHUNK_LEN);
050    }
051
052    public LZFOutputStream(final OutputStream outputStream, int bufsize) {
053        _encoder = new ChunkEncoder(bufsize);
054        _recycler = BufferRecycler.instance();
055        _outputStream = outputStream;
056        _outputBuffer = _recycler.allocOutputBuffer(bufsize);
057        _outputStreamClosed = false;
058    }
059
060    /**
061     * Method for defining whether call to {@link #flush} will also complete
062     * current block (similar to calling {@link #finishBlock()}) or not.
063     */
064    public LZFOutputStream setFinishBlockOnFlush(boolean b) {
065        _cfgFinishBlockOnFlush = b;
066        return this;
067    }
068
069    /*
070     // OutputStream impl
071     */
072    @Override
073    public void write(final int singleByte) throws IOException {
074        checkNotClosed();
075        if (_position >= _outputBuffer.length) {
076            writeCompressedBlock();
077        }
078        _outputBuffer[_position++] = (byte) singleByte;
079    }
080
081    @Override
082    public void write(final byte[] buffer, int offset, int length) throws IOException {
083        checkNotClosed();
084
085        final int BUFFER_LEN = _outputBuffer.length;
086
087        // simple case first: buffering only (for trivially short writes)
088        int free = BUFFER_LEN - _position;
089        if (free >= length) {
090            System.arraycopy(buffer, offset, _outputBuffer, _position, length);
091            _position += length;
092            return;
093        }
094        // otherwise, copy whatever we can, flush
095        System.arraycopy(buffer, offset, _outputBuffer, _position, free);
096        offset += free;
097        length -= free;
098        _position += free;
099        writeCompressedBlock();
100
101        // then write intermediate full block, if any, without copying:
102        while (length >= BUFFER_LEN) {
103            _encoder.encodeAndWriteChunk(buffer, offset, BUFFER_LEN, _outputStream);
104            offset += BUFFER_LEN;
105            length -= BUFFER_LEN;
106        }
107
108        // and finally, copy leftovers in buffer, if any
109        if (length > 0) {
110            System.arraycopy(buffer, offset, _outputBuffer, 0, length);
111        }
112        _position = length;
113    }
114
115    @Override
116    public void flush() throws IOException {
117        checkNotClosed();
118        if (_cfgFinishBlockOnFlush && _position > 0) {
119            writeCompressedBlock();
120        }
121        _outputStream.flush();
122    }
123
124    @Override
125    public void close() throws IOException {
126        if (!_outputStreamClosed) {
127            if (_position > 0) {
128                writeCompressedBlock();
129            }
130            _outputStream.flush();
131            _encoder.close();
132            byte[] buf = _outputBuffer;
133            if (buf != null) {
134                _outputBuffer = null;
135                _recycler.releaseOutputBuffer(buf);
136            }
137            _outputStreamClosed = true;
138            _outputStream.close();
139        }
140    }
141
142    /**
143     * Method that can be used to find underlying {@link java.io.OutputStream} that we
144     * write encoded LZF encoded data into, after compressing it. Will never
145     * return null; although underlying stream may be closed (if this stream has
146     * been closed).
147     */
148    public OutputStream getUnderlyingOutputStream() {
149        return _outputStream;
150    }
151
152    /**
153     * Accessor for checking whether call to "flush()" will first finish the
154     * current block or not
155     */
156    public boolean getFinishBlockOnFlush() {
157        return _cfgFinishBlockOnFlush;
158    }
159
160    /**
161     * Method that can be used to force completion of the current block, which
162     * means that all buffered data will be compressed into an LZF block. This
163     * typically results in lower compression ratio as larger blocks compress
164     * better; but may be necessary for network connections to ensure timely
165     * sending of data.
166     */
167    public LZFOutputStream finishBlock() throws IOException {
168        checkNotClosed();
169        if (_position > 0) {
170            writeCompressedBlock();
171        }
172        return this;
173    }
174
175    /**
176     * Compress and write the current block to the OutputStream
177     */
178    protected void writeCompressedBlock() throws IOException {
179        int left = _position;
180        _position = 0;
181        int offset = 0;
182
183        do {
184            int chunkLen = Math.min(LZFChunk.MAX_CHUNK_LEN, left);
185            _encoder.encodeAndWriteChunk(_outputBuffer, offset, chunkLen, _outputStream);
186            offset += chunkLen;
187            left -= chunkLen;
188        } while (left > 0);
189    }
190
191    protected void checkNotClosed() throws IOException {
192        if (_outputStreamClosed) {
193            throw new IOException(getClass().getName() + " already closed");
194        }
195    }
196}
