016package org.xbib.io.compress.lzf;
017
018import java.io.IOException;
019import java.io.InputStream;
020
021/**
022 * Decorator {@link java.io.InputStream} implementation used for reading
023 * <b>uncompressed</b> data and <b>compressing</b> it on the fly, such that
024 * reads return compressed data. It is reverse of {@link LZFInputStream} (which
025 * instead uncompresses data).
026 */
027public class LZFCompressingInputStream extends InputStream {
028
029    private final BufferRecycler _recycler;
030    private ChunkEncoder _encoder;
031    /**
032     * stream to be decompressed
033     */
034    protected final InputStream _inputStream;
035    /**
036     * Flag that indicates if we have already called 'inputStream.close()' (to
037     * avoid calling it multiple times)
038     */
039    protected boolean _inputStreamClosed;
040    /**
041     * Flag that indicates whether we force full reads (reading of as many bytes
042     * as requested), or 'optimal' reads (up to as many as available, but at
043     * least one). Default is false, meaning that 'optimal' read is used.
044     */
045    protected boolean _cfgFullReads = false;
046    /**
047     * Buffer in which uncompressed input is first read, before getting encoded
048     * in {@link #_encodedBytes}.
049     */
050    protected byte[] _inputBuffer;
051    /**
052     * Buffer that contains compressed data that is returned to readers.
053     */
054    protected byte[] _encodedBytes;
055    /**
056     * The current position (next char to output) in the uncompressed bytes
057     * buffer.
058     */
059    protected int _bufferPosition = 0;
060    /**
061     * Length of the current uncompressed bytes buffer
062     */
063    protected int _bufferLength = 0;
064    /**
065     * Number of bytes read from the underlying {@link #_inputStream}
066     */
067    protected int _readCount = 0;
068
069    /*
070     ///////////////////////////////////////////////////////////////////////
071     // Construction, configuration
072     ///////////////////////////////////////////////////////////////////////
073     */
074    public LZFCompressingInputStream(InputStream in) {
075        _inputStream = in;
076        _recycler = BufferRecycler.instance();
077        _inputBuffer = _recycler.allocInputBuffer(LZFChunk.MAX_CHUNK_LEN);
078        // let's not yet allocate encoding buffer; don't know optimal size
079    }
080
081    /**
082     * Method that can be used define whether reads should be "full" or
083     * "optimal": former means that full compressed blocks are read right away
084     * as needed, optimal that only smaller chunks are read at a time, more
085     * being read as needed.
086     */
087    public void setUseFullReads(boolean b) {
088        _cfgFullReads = b;
089    }
090
091    @Override
092    public int available() {
093        // if closed, return -1;
094        if (_inputStreamClosed) {
095            return -1;
096        }
097        int left = (_bufferLength - _bufferPosition);
098        return (left <= 0) ? 0 : left;
099    }
100
101    @Override
102    public int read() throws IOException {
103        if (!readyBuffer()) {
104            return -1;
105        }
106        return _encodedBytes[_bufferPosition++] & 255;
107    }
108
109    @Override
110    public int read(final byte[] buffer) throws IOException {
111        return read(buffer, 0, buffer.length);
112    }
113
114    @Override
115    public int read(final byte[] buffer, int offset, int length) throws IOException {
116        if (length < 1) {
117            return 0;
118        }
119        if (!readyBuffer()) {
120            return -1;
121        }
122        // First let's read however much data we happen to have...
123        int chunkLength = Math.min(_bufferLength - _bufferPosition, length);
124        System.arraycopy(_encodedBytes, _bufferPosition, buffer, offset, chunkLength);
125        _bufferPosition += chunkLength;
126
127        if (chunkLength == length || !_cfgFullReads) {
128            return chunkLength;
129        }
130        // Need more data, then
131        int totalRead = chunkLength;
132        do {
133            offset += chunkLength;
134            if (!readyBuffer()) {
135                break;
136            }
137            chunkLength = Math.min(_bufferLength - _bufferPosition, (length - totalRead));
138            System.arraycopy(_encodedBytes, _bufferPosition, buffer, offset, chunkLength);
139            _bufferPosition += chunkLength;
140            totalRead += chunkLength;
141        } while (totalRead < length);
142
143        return totalRead;
144    }
145
146    @Override
147    public void close() throws IOException {
148        _bufferPosition = _bufferLength = 0;
149        byte[] buf = _encodedBytes;
150        if (buf != null) {
151            _encodedBytes = null;
152            _recycler.releaseEncodeBuffer(buf);
153        }
154        if (_encoder != null) {
155            _encoder.close();
156        }
157        _closeInput();
158    }
159
160    private void _closeInput() throws IOException {
161        byte[] buf = _inputBuffer;
162        if (buf != null) {
163            _inputBuffer = null;
164            _recycler.releaseInputBuffer(buf);
165        }
166        if (!_inputStreamClosed) {
167            _inputStreamClosed = true;
168            _inputStream.close();
169        }
170    }
171
172    /**
173     * Overridden to just skip at most a single chunk at a time
174     */
175    @Override
176    public long skip(long n) throws IOException {
177        if (_inputStreamClosed) {
178            return -1;
179        }
180        int left = (_bufferLength - _bufferPosition);
181        // if none left, must read more:
182        if (left <= 0) {
183            // otherwise must read more to skip...
184            int b = read();
185            if (b < 0) { // EOF
186                return -1;
187            }
188            // push it back to get accurate skip count
189            --_bufferPosition;
190            left = (_bufferLength - _bufferPosition);
191        }
192        // either way, just skip whatever we have decoded
193        if (left > n) {
194            left = (int) n;
195        }
196        _bufferPosition += left;
197        return left;
198    }
199    /*
200     ///////////////////////////////////////////////////////////////////////
201     // Internal methods
202     ///////////////////////////////////////////////////////////////////////
203     */
204
205    /**
206     * Fill the uncompressed bytes buffer by reading the underlying inputStream.
207     *
208     * @throws java.io.IOException
209     */
210    protected boolean readyBuffer() throws IOException {
211        if (_bufferPosition < _bufferLength) {
212            return true;
213        }
214        if (_inputStreamClosed) {
215            return false;
216        }
217        // Ok: read as much as we can from input source first
218        int count = _inputStream.read(_inputBuffer, 0, _inputBuffer.length);
219        if (count < 0) { // if no input read, it's EOF
220            _closeInput(); // and we can close input source as well
221            return false;
222        }
223        int chunkLength = count;
224        int left = _inputBuffer.length - count;
225
226        while ((count = _inputStream.read(_inputBuffer, chunkLength, left)) > 0) {
227            chunkLength += count;
228            left -= count;
229            if (left < 1) {
230                break;
231            }
232        }
233
234        _bufferPosition = 0;
235
236        // Ok: if we don't yet have an encoder (and buffer for it), let's get one
237        if (_encoder == null) {
238            // need 7 byte header, plus regular max buffer size:
239            int bufferLen = chunkLength + ((chunkLength + 31) >> 5) + 7;
240            _encoder = ChunkEncoder.nonAllocatingEncoder(bufferLen);
241            _encodedBytes = _recycler.allocEncodingBuffer(bufferLen);
242        }
243        // offset of 7 so we can prepend header as necessary
244        int encodeEnd = _encoder.tryCompress(_inputBuffer, 0, chunkLength, _encodedBytes, 7);
245        // but did it compress?
246        if (encodeEnd < (chunkLength + 5)) { // yes! (compared to 5 byte uncomp prefix, data)
247            // prepend header in situ
248            LZFChunk.appendCompressedHeader(chunkLength, encodeEnd - 7, _encodedBytes, 0);
249            _bufferLength = encodeEnd;
250        } else { // no -- so sad...
251            int ptr = LZFChunk.appendNonCompressedHeader(chunkLength, _encodedBytes, 0);
252            // TODO: figure out a way to avoid this copy; need a header
253            System.arraycopy(_inputBuffer, 0, _encodedBytes, ptr, chunkLength);
254            _bufferLength = ptr + chunkLength;
255        }
256        if (count < 0) { // did we get end-of-input?
257            _closeInput();
258        }
259        return true;
260    }
261}
