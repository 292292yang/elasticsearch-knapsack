002    package org.xbib.io.compress.lzf;
003    
004    import org.xbib.io.compress.BufferRecycler;
005    
006    import java.io.IOException;
007    import java.io.InputStream;
008    
009    /**
010     * Decorator {@link java.io.InputStream} implementation used for reading
011     * <b>uncompressed</b> data and <b>compressing</b> it on the fly, such that
012     * reads return compressed data. It is reverse of {@link LZFInputStream} (which
013     * instead uncompresses data).
014     */
015    public class LZFCompressingInputStream extends InputStream {
016    
017        private final BufferRecycler _recycler;
018        private ChunkEncoder _encoder;
019        /**
020         * stream to be decompressed
021         */
022        protected final InputStream _inputStream;
023        /**
024         * Flag that indicates if we have already called 'inputStream.close()' (to
025         * avoid calling it multiple times)
026         */
027        protected boolean _inputStreamClosed;
028        /**
029         * Flag that indicates whether we force full reads (reading of as many bytes
030         * as requested), or 'optimal' reads (up to as many as available, but at
031         * least one). Default is false, meaning that 'optimal' read is used.
032         */
033        protected boolean _cfgFullReads = false;
034        /**
035         * Buffer in which uncompressed input is first read, before getting encoded
036         * in {@link #_encodedBytes}.
037         */
038        protected byte[] _inputBuffer;
039        /**
040         * Buffer that contains compressed data that is returned to readers.
041         */
042        protected byte[] _encodedBytes;
043        /**
044         * The current position (next char to output) in the uncompressed bytes
045         * buffer.
046         */
047        protected int _bufferPosition = 0;
048        /**
049         * Length of the current uncompressed bytes buffer
050         */
051        protected int _bufferLength = 0;
052        /**
053         * Number of bytes read from the underlying {@link #_inputStream}
054         */
055        protected int _readCount = 0;
056    
057        /*
058         ///////////////////////////////////////////////////////////////////////
059         // Construction, configuration
060         ///////////////////////////////////////////////////////////////////////
061         */
062        public LZFCompressingInputStream(InputStream in) {
063            _inputStream = in;
064            _recycler = BufferRecycler.instance();
065            _inputBuffer = _recycler.allocInputBuffer(LZFChunk.MAX_CHUNK_LEN);
066            // let's not yet allocate encoding buffer; don't know optimal size
067        }
068    
069        /**
070         * Method that can be used define whether reads should be "full" or
071         * "optimal": former means that full compressed blocks are read right away
072         * as needed, optimal that only smaller chunks are read at a time, more
073         * being read as needed.
074         */
075        public void setUseFullReads(boolean b) {
076            _cfgFullReads = b;
077        }
078    
079        @Override
080        public int available() {
081            // if closed, return -1;
082            if (_inputStreamClosed) {
083                return -1;
084            }
085            int left = (_bufferLength - _bufferPosition);
086            return (left <= 0) ? 0 : left;
087        }
088    
089        @Override
090        public int read() throws IOException {
091            if (!readyBuffer()) {
092                return -1;
093            }
094            return _encodedBytes[_bufferPosition++] & 255;
095        }
096    
097        @Override
098        public int read(final byte[] buffer) throws IOException {
099            return read(buffer, 0, buffer.length);
100        }
101    
102        @Override
103        public int read(final byte[] buffer, int offset, int length) throws IOException {
104            if (length < 1) {
105                return 0;
106            }
107            if (!readyBuffer()) {
108                return -1;
109            }
110            // First let's read however much data we happen to have...
111            int chunkLength = Math.min(_bufferLength - _bufferPosition, length);
112            System.arraycopy(_encodedBytes, _bufferPosition, buffer, offset, chunkLength);
113            _bufferPosition += chunkLength;
114    
115            if (chunkLength == length || !_cfgFullReads) {
116                return chunkLength;
117            }
118            // Need more data, then
119            int totalRead = chunkLength;
120            do {
121                offset += chunkLength;
122                if (!readyBuffer()) {
123                    break;
124                }
125                chunkLength = Math.min(_bufferLength - _bufferPosition, (length - totalRead));
126                System.arraycopy(_encodedBytes, _bufferPosition, buffer, offset, chunkLength);
127                _bufferPosition += chunkLength;
128                totalRead += chunkLength;
129            } while (totalRead < length);
130    
131            return totalRead;
132        }
133    
134        @Override
135        public void close() throws IOException {
136            _bufferPosition = _bufferLength = 0;
137            byte[] buf = _encodedBytes;
138            if (buf != null) {
139                _encodedBytes = null;
140                _recycler.releaseEncodeBuffer(buf);
141            }
142            if (_encoder != null) {
143                _encoder.close();
144            }
145            _closeInput();
146        }
147    
148        private void _closeInput() throws IOException {
149            byte[] buf = _inputBuffer;
150            if (buf != null) {
151                _inputBuffer = null;
152                _recycler.releaseInputBuffer(buf);
153            }
154            if (!_inputStreamClosed) {
155                _inputStreamClosed = true;
156                _inputStream.close();
157            }
158        }
159    
160        /**
161         * Overridden to just skip at most a single chunk at a time
162         */
163        @Override
164        public long skip(long n) throws IOException {
165            if (_inputStreamClosed) {
166                return -1;
167            }
168            int left = (_bufferLength - _bufferPosition);
169            // if none left, must read more:
170            if (left <= 0) {
171                // otherwise must read more to skip...
172                int b = read();
173                if (b < 0) { // EOF
174                    return -1;
175                }
176                // push it back to get accurate skip count
177                --_bufferPosition;
178                left = (_bufferLength - _bufferPosition);
179            }
180            // either way, just skip whatever we have decoded
181            if (left > n) {
182                left = (int) n;
183            }
184            _bufferPosition += left;
185            return left;
186        }
187        /*
188         ///////////////////////////////////////////////////////////////////////
189         // Internal methods
190         ///////////////////////////////////////////////////////////////////////
191         */
192    
193        /**
194         * Fill the uncompressed bytes buffer by reading the underlying inputStream.
195         *
196         * @throws java.io.IOException
197         */
198        protected boolean readyBuffer() throws IOException {
199            if (_bufferPosition < _bufferLength) {
200                return true;
201            }
202            if (_inputStreamClosed) {
203                return false;
204            }
205            // Ok: read as much as we can from input source first
206            int count = _inputStream.read(_inputBuffer, 0, _inputBuffer.length);
207            if (count < 0) { // if no input read, it's EOF
208                _closeInput(); // and we can close input source as well
209                return false;
210            }
211            int chunkLength = count;
212            int left = _inputBuffer.length - count;
213    
214            while ((count = _inputStream.read(_inputBuffer, chunkLength, left)) > 0) {
215                chunkLength += count;
216                left -= count;
217                if (left < 1) {
218                    break;
219                }
220            }
221    
222            _bufferPosition = 0;
223    
224            // Ok: if we don't yet have an encoder (and buffer for it), let's get one
225            if (_encoder == null) {
226                // need 7 byte header, plus regular max buffer size:
227                int bufferLen = chunkLength + ((chunkLength + 31) >> 5) + 7;
228                _encoder = ChunkEncoder.nonAllocatingEncoder(bufferLen);
229                _encodedBytes = _recycler.allocEncodingBuffer(bufferLen);
230            }
231            // offset of 7 so we can prepend header as necessary
232            int encodeEnd = _encoder.tryCompress(_inputBuffer, 0, chunkLength, _encodedBytes, 7);
233            // but did it compress?
234            if (encodeEnd < (chunkLength + 5)) { // yes! (compared to 5 byte uncomp prefix, data)
235                // prepend header in situ
236                LZFChunk.appendCompressedHeader(chunkLength, encodeEnd - 7, _encodedBytes, 0);
237                _bufferLength = encodeEnd;
238            } else { // no -- so sad...
239                int ptr = LZFChunk.appendNonCompressedHeader(chunkLength, _encodedBytes, 0);
240                // TODO: figure out a way to avoid this copy; need a header
241                System.arraycopy(_inputBuffer, 0, _encodedBytes, ptr, chunkLength);
242                _bufferLength = ptr + chunkLength;
243            }
244            if (count < 0) { // did we get end-of-input?
245                _closeInput();
246            }
247            return true;
248        }
249    }
