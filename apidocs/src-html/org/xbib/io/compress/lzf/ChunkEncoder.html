009    /**
010     * Class that handles actual encoding of individual chunks. Resulting chunks can
011     * be compressed or non-compressed; compression is only used if it actually
012     * reduces chunk size (including overhead of additional header bytes)
013     */
014    public final class ChunkEncoder {
015        // Beyond certain point we won't be able to compress; let's use 16 bytes as cut-off
016    
017        private static final int MIN_BLOCK_TO_COMPRESS = 16;
018        private static final int MIN_HASH_SIZE = 256;
019        // Not much point in bigger tables, with 8k window
020        private static final int MAX_HASH_SIZE = 16384;
021        private static final int MAX_OFF = 1 << 13; // 8k
022        private static final int MAX_REF = (1 << 8) + (1 << 3); // 264
023        // // Encoding tables etc
024        private final BufferRecycler _recycler;
025        /**
026         * Hash table contains lookup based on 3-byte sequence; key is hash of such
027         * triplet, value is offset in buffer.
028         */
029        private int[] _hashTable;
030        private final int _hashModulo;
031        /**
032         * Buffer in which encoded content is stored during processing
033         */
034        private byte[] _encodeBuffer;
035        /**
036         * Small buffer passed to LZFChunk, needed for writing chunk header
037         */
038        private byte[] _headerBuffer;
039    
040        /**
041         * @param totalLength Total encoded length; used for calculating size of
042         *                    hash table to use
043         */
044        public ChunkEncoder(int totalLength) {
045            int largestChunkLen = Math.max(totalLength, LZFChunk.MAX_CHUNK_LEN);
046    
047            int suggestedHashLen = calcHashLen(largestChunkLen);
048            _recycler = BufferRecycler.instance();
049            _hashTable = _recycler.allocEncodingHash(suggestedHashLen);
050            _hashModulo = _hashTable.length - 1;
051            // Ok, then, what's the worst case output buffer length?
052            // length indicator for each 32 literals, so:
053            int bufferLen = largestChunkLen + ((largestChunkLen + 31) >> 5);
054            _encodeBuffer = _recycler.allocEncodingBuffer(bufferLen);
055        }
056    
057        /**
058         * Alternate constructor used when we want to avoid allocation encoding
059         * buffer, in cases where caller wants full control over allocations.
060         */
061        private ChunkEncoder(int totalLength, boolean bogus) {
062            int largestChunkLen = Math.max(totalLength, LZFChunk.MAX_CHUNK_LEN);
063            int suggestedHashLen = calcHashLen(largestChunkLen);
064            _recycler = BufferRecycler.instance();
065            _hashTable = _recycler.allocEncodingHash(suggestedHashLen);
066            _hashModulo = _hashTable.length - 1;
067            _encodeBuffer = null;
068        }
069    
070        public static ChunkEncoder nonAllocatingEncoder(int totalLength) {
071            return new ChunkEncoder(totalLength, true);
072        }

080        /**
081         * Method to close once encoder is no longer in use. Note: after calling
082         * this method, further calls to {@link #encodeChunk} will fail
083         */
084        public void close() {
085            byte[] buf = _encodeBuffer;
086            if (buf != null) {
087                _encodeBuffer = null;
088                _recycler.releaseEncodeBuffer(buf);
089            }
090            int[] ibuf = _hashTable;
091            if (ibuf != null) {
092                _hashTable = null;
093                _recycler.releaseEncodingHash(ibuf);
094            }
095        }
096    
097        /**
098         * Method for compressing (or not) individual chunks
099         */
100        public LZFChunk encodeChunk(byte[] data, int offset, int len) {
101            if (len >= MIN_BLOCK_TO_COMPRESS) {
102                /* If we have non-trivial block, and can compress it by at least
103                 * 2 bytes (since header is 2 bytes longer), let's compress:
104                 */
105                int compLen = tryCompress(data, offset, offset + len, _encodeBuffer, 0);
106                if (compLen < (len - 2)) { // nah; just return uncompressed
107                    return LZFChunk.createCompressed(len, _encodeBuffer, 0, compLen);
108                }
109            }
110            // Otherwise leave uncompressed:
111            return LZFChunk.createNonCompressed(data, offset, len);
112        }
113    
114        /**
115         * Method for encoding individual chunk, writing it to given output stream.
116         */
117        public void encodeAndWriteChunk(byte[] data, int offset, int len, OutputStream out)
118                throws IOException {
119            byte[] headerBuf = _headerBuffer;
120            if (headerBuf == null) {
121                _headerBuffer = headerBuf = new byte[LZFChunk.MAX_HEADER_LEN];
122            }
123            if (len >= MIN_BLOCK_TO_COMPRESS) {
124                /* If we have non-trivial block, and can compress it by at least
125                 * 2 bytes (since header is 2 bytes longer), let's compress:
126                 */
127                int compLen = tryCompress(data, offset, offset + len, _encodeBuffer, 0);
128                if (compLen < (len - 2)) { // nah; just return uncompressed
129                    LZFChunk.writeCompressedHeader(len, compLen, out, headerBuf);
130                    out.write(_encodeBuffer, 0, compLen);
131                    return;
132                }
133            }
134            // Otherwise leave uncompressed:
135            LZFChunk.writeNonCompressedHeader(len, out, headerBuf);
136            out.write(data, offset, len);
137        }
138    
139        /**
140         * Main workhorse method that will try to compress given chunk, and return
141         * end position (offset to byte after last included byte)
142         */
143        protected int tryCompress(byte[] in, int inPos, int inEnd, byte[] out, int outPos) {
144            final int[] hashTable = _hashTable;
145            ++outPos;
146            int seen = first(in, 0); // past 4 bytes we have seen... (last one is LSB)
147            int literals = 0;
148            inEnd -= 4;
149            final int firstPos = inPos; // so that we won't have back references across block boundary
150    
151            while (inPos < inEnd) {
152                byte p2 = in[inPos + 2];
153                // next
154                seen = (seen << 8) + (p2 & 255);
155                int off = hash(seen);
156                int ref = hashTable[off];
157                hashTable[off] = inPos;
158    
159                // First expected common case: no back-ref (for whatever reason)
160                if (ref >= inPos // can't refer forward (i.e. leftovers)
161                        || ref < firstPos // or to previous block
162                        || (off = inPos - ref) > MAX_OFF
163                        || in[ref + 2] != p2 // must match hash
164                        || in[ref + 1] != (byte) (seen >> 8)
165                        || in[ref] != (byte) (seen >> 16)) {
166                    out[outPos++] = in[inPos++];
167                    literals++;
168                    if (literals == LZFChunk.MAX_LITERAL) {
169                        out[outPos - 33] = (byte) 31; // <= out[outPos - literals - 1] = MAX_LITERAL_MINUS_1;
170                        literals = 0;
171                        outPos++;
172                    }
173                    continue;
174                }
175                // match
176                int maxLen = inEnd - inPos + 2;
177                if (maxLen > MAX_REF) {
178                    maxLen = MAX_REF;
179                }
180                if (literals == 0) {
181                    outPos--;
182                } else {
183                    out[outPos - literals - 1] = (byte) (literals - 1);
184                    literals = 0;
185                }
186                int len = 3;
187                while (len < maxLen && in[ref + len] == in[inPos + len]) {
188                    len++;
189                }
190                len -= 2;
191                --off; // was off by one earlier
192                if (len < 7) {
193                    out[outPos++] = (byte) ((off >> 8) + (len << 5));
194                } else {
195                    out[outPos++] = (byte) ((off >> 8) + (7 << 5));
196                    out[outPos++] = (byte) (len - 7);
197                }
198                out[outPos++] = (byte) off;
199                outPos++;
200                inPos += len;
201                seen = first(in, inPos);
202                seen = (seen << 8) + (in[inPos + 2] & 255);
203                hashTable[hash(seen)] = inPos;
204                ++inPos;
205                seen = (seen << 8) + (in[inPos + 2] & 255); // hash = next(hash, in, inPos);
206                hashTable[hash(seen)] = inPos;
207                ++inPos;
208            }
209            // try offlining the tail
210            return handleTail(in, inPos, inEnd + 4, out, outPos, literals);
211        }
212    
213        private final int handleTail(byte[] in, int inPos, int inEnd, byte[] out, int outPos,
214                                     int literals) {
215            while (inPos < inEnd) {
216                out[outPos++] = in[inPos++];
217                literals++;
218                if (literals == LZFChunk.MAX_LITERAL) {
219                    out[outPos - literals - 1] = (byte) (literals - 1);
220                    literals = 0;
221                    outPos++;
222                }
223            }
224            out[outPos - literals - 1] = (byte) (literals - 1);
225            if (literals == 0) {
226                outPos--;
227            }
228            return outPos;
229        }

236        private static int calcHashLen(int chunkSize) {
237            // in general try get hash table size of 2x input size
238            chunkSize += chunkSize;
239            // but no larger than max size:
240            if (chunkSize >= MAX_HASH_SIZE) {
241                return MAX_HASH_SIZE;
242            }
243            // otherwise just need to round up to nearest 2x
244            int hashLen = MIN_HASH_SIZE;
245            while (hashLen < chunkSize) {
246                hashLen += hashLen;
247            }
248            return hashLen;
249        }
250    
251        private final int first(byte[] in, int inPos) {
252            return (in[inPos] << 8) + (in[inPos + 1] & 0xFF);
253        }
