002    package org.xbib.io.compress.lzf;
003    
004    import java.io.IOException;
005    
006    /**
007     * Encoder that handles splitting of input into chunks to encode, calls
008     * {@link ChunkEncoder} to compress individual chunks and combines resulting
009     * chunks into contiguous output byte array.
010     */
011    public class LZFEncoder {
012        // Static methods only, no point in instantiating
013    
014        private LZFEncoder() {
015        }
016    
017        public static byte[] encode(byte[] data) throws IOException {
018            return encode(data, data.length);
019        }
020    
021        /**
022         * Method for compressing given input data using LZF encoding and block
023         * structure (compatible with lzf command line utility). Result consists of
024         * a sequence of chunks.
025         */
026        public static byte[] encode(byte[] data, int length) throws IOException {
027            return encode(data, 0, length);
028        }
029    
030        /**
031         * Method for compressing given input data using LZF encoding and block
032         * structure (compatible with lzf command line utility). Result consists of
033         * a sequence of chunks.
034         */
035        public static byte[] encode(byte[] data, int offset, int length) throws IOException {
036            ChunkEncoder enc = new ChunkEncoder(length);
037            byte[] result = encode(enc, data, offset, length);
038            // important: may be able to reuse buffers
039            enc.close();
040            return result;
041        }
042    
043        public static byte[] encode(ChunkEncoder enc, byte[] data, int length)
044                throws IOException {
045            return encode(enc, data, 0, length);
046        }
047    
048        public static byte[] encode(ChunkEncoder enc, byte[] data, int offset, int length)
049                throws IOException {
050            int left = length;
051            int chunkLen = Math.min(LZFChunk.MAX_CHUNK_LEN, left);
052            LZFChunk first = enc.encodeChunk(data, offset, chunkLen);
053            left -= chunkLen;
054            // shortcut: if it all fit in, no need to coalesce:
055            if (left < 1) {
056                return first.getData();
057            }
058            // otherwise need to get other chunks:
059            int resultBytes = first.length();
060            offset += chunkLen;
061            LZFChunk last = first;
062    
063            do {
064                chunkLen = Math.min(left, LZFChunk.MAX_CHUNK_LEN);
065                LZFChunk chunk = enc.encodeChunk(data, offset, chunkLen);
066                offset += chunkLen;
067                left -= chunkLen;
068                resultBytes += chunk.length();
069                last.setNext(chunk);
070                last = chunk;
071            } while (left > 0);
072            // and then coalesce returns into single contiguous byte array
073            byte[] result = new byte[resultBytes];
074            int ptr = 0;
075            for (; first != null; first = first.next()) {
076                ptr = first.copyTo(result, ptr);
077            }
078            return result;
079        }
080    }
