003import java.io.IOException;
004import java.io.InputStream;
005import org.xbib.io.compress.lzf.ChunkDecoder;

008/**
009 * Safe {@link ChunkDecoder} implementation that can be used on any platform.
010 */
011public class VanillaChunkDecoder extends ChunkDecoder {
012
013    public VanillaChunkDecoder() {
014    }
015
016    @Override
017    public final int decodeChunk(final InputStream is, final byte[] inputBuffer, final byte[] outputBuffer)
018            throws IOException {
019        int bytesInOutput;
020        /* note: we do NOT read more than 5 bytes because otherwise might need to shuffle bytes
021         * for output buffer (could perhaps optimize in future?)
022         */
023        int bytesRead = readHeader(is, inputBuffer);
024        if ((bytesRead < HEADER_BYTES)
025                || inputBuffer[0] != LZFChunk.BYTE_Z || inputBuffer[1] != LZFChunk.BYTE_V) {
026            if (bytesRead == 0) { // probably fine, clean EOF
027                return -1;
028            }
029            throw new IOException("Corrupt input data, block did not start with 2 byte signature ('ZV') followed by type byte, 2-byte length)");
030        }
031        int type = inputBuffer[2];
032        int compLen = uint16(inputBuffer, 3);
033        if (type == LZFChunk.BLOCK_TYPE_NON_COMPRESSED) { // uncompressed
034            readFully(is, false, outputBuffer, 0, compLen);
035            bytesInOutput = compLen;
036        } else { // compressed
037            readFully(is, true, inputBuffer, 0, 2 + compLen); // first 2 bytes are uncompressed length
038            int uncompLen = uint16(inputBuffer, 0);
039            decodeChunk(inputBuffer, 2, outputBuffer, 0, uncompLen);
040            bytesInOutput = uncompLen;
041        }
042        return bytesInOutput;
043    }
044
045    @Override
046    public final void decodeChunk(byte[] in, int inPos, byte[] out, int outPos, int outEnd)
047            throws IOException {
048        do {
049            int ctrl = in[inPos++] & 255;
050            if (ctrl < LZFChunk.MAX_LITERAL) { // literal run
051                switch (ctrl) {
052                    case 31:
053                        out[outPos++] = in[inPos++];
054                    case 30:
055                        out[outPos++] = in[inPos++];
056                    case 29:
057                        out[outPos++] = in[inPos++];
058                    case 28:
059                        out[outPos++] = in[inPos++];
060                    case 27:
061                        out[outPos++] = in[inPos++];
062                    case 26:
063                        out[outPos++] = in[inPos++];
064                    case 25:
065                        out[outPos++] = in[inPos++];
066                    case 24:
067                        out[outPos++] = in[inPos++];
068                    case 23:
069                        out[outPos++] = in[inPos++];
070                    case 22:
071                        out[outPos++] = in[inPos++];
072                    case 21:
073                        out[outPos++] = in[inPos++];
074                    case 20:
075                        out[outPos++] = in[inPos++];
076                    case 19:
077                        out[outPos++] = in[inPos++];
078                    case 18:
079                        out[outPos++] = in[inPos++];
080                    case 17:
081                        out[outPos++] = in[inPos++];
082                    case 16:
083                        out[outPos++] = in[inPos++];
084                    case 15:
085                        out[outPos++] = in[inPos++];
086                    case 14:
087                        out[outPos++] = in[inPos++];
088                    case 13:
089                        out[outPos++] = in[inPos++];
090                    case 12:
091                        out[outPos++] = in[inPos++];
092                    case 11:
093                        out[outPos++] = in[inPos++];
094                    case 10:
095                        out[outPos++] = in[inPos++];
096                    case 9:
097                        out[outPos++] = in[inPos++];
098                    case 8:
099                        out[outPos++] = in[inPos++];
100                    case 7:
101                        out[outPos++] = in[inPos++];
102                    case 6:
103                        out[outPos++] = in[inPos++];
104                    case 5:
105                        out[outPos++] = in[inPos++];
106                    case 4:
107                        out[outPos++] = in[inPos++];
108                    case 3:
109                        out[outPos++] = in[inPos++];
110                    case 2:
111                        out[outPos++] = in[inPos++];
112                    case 1:
113                        out[outPos++] = in[inPos++];
114                    case 0:
115                        out[outPos++] = in[inPos++];
116                }
117                continue;
118            }
119            // back reference
120            int len = ctrl >> 5;
121            ctrl = -((ctrl & 0x1f) << 8) - 1;
122            if (len < 7) { // 2 bytes; length of 3 - 8 bytes
123                ctrl -= in[inPos++] & 255;
124                out[outPos] = out[outPos++ + ctrl];
125                out[outPos] = out[outPos++ + ctrl];
126                switch (len) {
127                    case 6:
128                        out[outPos] = out[outPos++ + ctrl];
129                    case 5:
130                        out[outPos] = out[outPos++ + ctrl];
131                    case 4:
132                        out[outPos] = out[outPos++ + ctrl];
133                    case 3:
134                        out[outPos] = out[outPos++ + ctrl];
135                    case 2:
136                        out[outPos] = out[outPos++ + ctrl];
137                    case 1:
138                        out[outPos] = out[outPos++ + ctrl];
139                }
140                continue;
141            }
142
143            // long version (3 bytes, length of up to 264 bytes)
144            len = in[inPos++] & 255;
145            ctrl -= in[inPos++] & 255;
146
147            // First: if there is no overlap, can just use arraycopy:
148            if ((ctrl + len) < -9) {
149                len += 9;
150                if (len <= 32) {
151                    copyUpTo32WithSwitch(out, outPos + ctrl, out, outPos, len - 1);
152                } else {
153                    System.arraycopy(out, outPos + ctrl, out, outPos, len);
154                }
155                outPos += len;
156                continue;
157            }
158
159            // otherwise manual copy: so first just copy 9 bytes we know are needed
160            out[outPos] = out[outPos++ + ctrl];
161            out[outPos] = out[outPos++ + ctrl];
162            out[outPos] = out[outPos++ + ctrl];
163            out[outPos] = out[outPos++ + ctrl];
164            out[outPos] = out[outPos++ + ctrl];
165            out[outPos] = out[outPos++ + ctrl];
166            out[outPos] = out[outPos++ + ctrl];
167            out[outPos] = out[outPos++ + ctrl];
168            out[outPos] = out[outPos++ + ctrl];
169
170            // then loop
171            // Odd: after extensive profiling, looks like magic number
172            // for unrolling is 4: with 8 performance is worse (even
173            // bit less than with no unrolling).
174            len += outPos;
175            final int end = len - 3;
176            while (outPos < end) {
177                out[outPos] = out[outPos++ + ctrl];
178                out[outPos] = out[outPos++ + ctrl];
179                out[outPos] = out[outPos++ + ctrl];
180                out[outPos] = out[outPos++ + ctrl];
181            }
182            switch (len - outPos) {
183                case 3:
184                    out[outPos] = out[outPos++ + ctrl];
185                case 2:
186                    out[outPos] = out[outPos++ + ctrl];
187                case 1:
188                    out[outPos] = out[outPos++ + ctrl];
189            }
190        } while (outPos < outEnd);
191
192        // sanity check to guard against corrupt data:
193        if (outPos != outEnd) {
194            throw new IOException("Corrupt data: overrun in decompress, input offset " + inPos + ", output offset " + outPos);
195        }
196    }
