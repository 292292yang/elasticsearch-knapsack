002    package org.xbib.io.compress.lzf;
003    
004    import java.io.IOException;
005    import java.io.OutputStream;
006    
007    /**
008     * Helper class used to store LZF encoded segments (compressed and
009     * non-compressed) that can be sequenced to produce LZF files/streams.
010     */
011    public class LZFChunk {
012    
013        /**
014         * Maximum length of literal run for LZF encoding.
015         */
016        public static final int MAX_LITERAL = 1 << 5; // 32
017        // Chunk length is limited by 2-byte length indicator, to 64k
018        public static final int MAX_CHUNK_LEN = 0xFFFF;
019        /**
020         * Header can be either 7 bytes (compressed) or 5 bytes (uncompressed) long
021         */
022        public static final int MAX_HEADER_LEN = 7;
023        public final static byte BYTE_Z = 'Z';
024        public final static byte BYTE_V = 'V';
025        public final static int BLOCK_TYPE_NON_COMPRESSED = 0;
026        public final static int BLOCK_TYPE_COMPRESSED = 1;
027        protected final byte[] _data;
028        protected LZFChunk _next;
029    
030        private LZFChunk(byte[] data) {
031            _data = data;
032        }
033    
034        /**
035         * Factory method for constructing compressed chunk
036         */
037        public static LZFChunk createCompressed(int origLen, byte[] encData, int encPtr, int encLen) {
038            byte[] result = new byte[encLen + 7];
039            result[0] = BYTE_Z;
040            result[1] = BYTE_V;
041            result[2] = BLOCK_TYPE_COMPRESSED;
042            result[3] = (byte) (encLen >> 8);
043            result[4] = (byte) encLen;
044            result[5] = (byte) (origLen >> 8);
045            result[6] = (byte) origLen;
046            System.arraycopy(encData, encPtr, result, 7, encLen);
047            return new LZFChunk(result);
048        }
049    
050        public static int appendCompressedHeader(int origLen, int encLen, byte[] headerBuffer, int offset)
051                throws IOException {
052            headerBuffer[offset++] = BYTE_Z;
053            headerBuffer[offset++] = BYTE_V;
054            headerBuffer[offset++] = BLOCK_TYPE_COMPRESSED;
055            headerBuffer[offset++] = (byte) (encLen >> 8);
056            headerBuffer[offset++] = (byte) encLen;
057            headerBuffer[offset++] = (byte) (origLen >> 8);
058            headerBuffer[offset++] = (byte) origLen;
059            return offset;
060        }
061    
062        public static void writeCompressedHeader(int origLen, int encLen, OutputStream out, byte[] headerBuffer)
063                throws IOException {
064            headerBuffer[0] = BYTE_Z;
065            headerBuffer[1] = BYTE_V;
066            headerBuffer[2] = BLOCK_TYPE_COMPRESSED;
067            headerBuffer[3] = (byte) (encLen >> 8);
068            headerBuffer[4] = (byte) encLen;
069            headerBuffer[5] = (byte) (origLen >> 8);
070            headerBuffer[6] = (byte) origLen;
071            out.write(headerBuffer, 0, 7);
072        }
073    
074        /**
075         * Factory method for constructing compressed chunk
076         */
077        public static LZFChunk createNonCompressed(byte[] plainData, int ptr, int len) {
078            byte[] result = new byte[len + 5];
079            result[0] = BYTE_Z;
080            result[1] = BYTE_V;
081            result[2] = BLOCK_TYPE_NON_COMPRESSED;
082            result[3] = (byte) (len >> 8);
083            result[4] = (byte) len;
084            System.arraycopy(plainData, ptr, result, 5, len);
085            return new LZFChunk(result);
086        }
087    
088        public static int appendNonCompressedHeader(int len, byte[] headerBuffer, int offset)
089                throws IOException {
090            headerBuffer[offset++] = BYTE_Z;
091            headerBuffer[offset++] = BYTE_V;
092            headerBuffer[offset++] = BLOCK_TYPE_NON_COMPRESSED;
093            headerBuffer[offset++] = (byte) (len >> 8);
094            headerBuffer[offset++] = (byte) len;
095            return offset;
096        }
097    
098        public static void writeNonCompressedHeader(int len, OutputStream out, byte[] headerBuffer)
099                throws IOException {
100            headerBuffer[0] = BYTE_Z;
101            headerBuffer[1] = BYTE_V;
102            headerBuffer[2] = BLOCK_TYPE_NON_COMPRESSED;
103            headerBuffer[3] = (byte) (len >> 8);
104            headerBuffer[4] = (byte) len;
105            out.write(headerBuffer, 0, 5);
106        }
107    
108        public void setNext(LZFChunk next) {
109            _next = next;
110        }
111    
112        public LZFChunk next() {
113            return _next;
114        }
115    
116        public int length() {
117            return _data.length;
118        }
119    
120        public byte[] getData() {
121            return _data;
122        }
123    
124        public int copyTo(byte[] dst, int ptr) {
125            int len = _data.length;
126            System.arraycopy(_data, 0, dst, ptr, len);
127            return ptr + len;
128        }
129    }
