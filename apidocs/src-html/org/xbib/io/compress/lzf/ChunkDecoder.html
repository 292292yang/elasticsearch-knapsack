002    package org.xbib.io.compress.lzf;
003    
004    import java.io.IOException;
005    import java.io.InputStream;
006    
007    /**
008     * Decoder that handles decoding of sequence of encoded LZF chunks, combining
009     * them into a single contiguous result byte array.
010     */
011    public abstract class ChunkDecoder {
012    
013        protected final static byte BYTE_NULL = 0;
014        protected final static int HEADER_BYTES = 5;
015    
016        public ChunkDecoder() {
017        }
018    
019        /*
020         * Public API
021         */
022    
023        /**
024         * Method for decompressing a block of input data encoded in LZF block
025         * structure (compatible with lzf command line utility), and can consist of
026         * any number of blocks. Note that input MUST consists of a sequence of one
027         * or more complete chunks; partial chunks can not be handled.
028         */
029        public final byte[] decode(final byte[] inputBuffer) throws IOException {
030            byte[] result = new byte[calculateUncompressedSize(inputBuffer, 0, inputBuffer.length)];
031            decode(inputBuffer, 0, inputBuffer.length, result);
032            return result;
033        }
034    
035        /**
036         * Method for decompressing a block of input data encoded in LZF block
037         * structure (compatible with lzf command line utility), and can consist of
038         * any number of blocks. Note that input MUST consists of a sequence of one
039         * or more complete chunks; partial chunks can not be handled.
040         */
041        public final byte[] decode(final byte[] inputBuffer, int inputPtr, int inputLen) throws IOException {
042            byte[] result = new byte[calculateUncompressedSize(inputBuffer, inputPtr, inputLen)];
043            decode(inputBuffer, inputPtr, inputLen, result);
044            return result;
045        }
046    
047        /**
048         * Method for decompressing a block of input data encoded in LZF block
049         * structure (compatible with lzf command line utility), and can consist of
050         * any number of blocks. Note that input MUST consists of a sequence of one
051         * or more complete chunks; partial chunks can not be handled.
052         */
053        public final int decode(final byte[] inputBuffer, final byte[] targetBuffer) throws IOException {
054            return decode(inputBuffer, 0, inputBuffer.length, targetBuffer);
055        }
056    
057        /**
058         * Method for decompressing a block of input data encoded in LZF block
059         * structure (compatible with lzf command line utility), and can consist of
060         * any number of blocks. Note that input MUST consists of a sequence of one
061         * or more complete chunks; partial chunks can not be handled.
062         */
063        public int decode(final byte[] sourceBuffer, int inPtr, int inLength,
064                          final byte[] targetBuffer) throws IOException {
065            int outPtr = 0;
066            int blockNr = 0;
067    
068            final int end = inPtr + inLength - 1; // -1 to offset possible end marker
069    
070            while (inPtr < end) {
071                // let's do basic sanity checks; no point in skimping with these checks
072                if (sourceBuffer[inPtr] != LZFChunk.BYTE_Z || sourceBuffer[inPtr + 1] != LZFChunk.BYTE_V) {
073                    throw new IOException("Corrupt input data, block #" + blockNr + " (at offset " + inPtr + "): did not start with 'ZV' signature bytes");
074                }
075                inPtr += 2;
076                int type = sourceBuffer[inPtr++];
077                int len = uint16(sourceBuffer, inPtr);
078                inPtr += 2;
079                if (type == LZFChunk.BLOCK_TYPE_NON_COMPRESSED) { // uncompressed
080                    if ((outPtr + len) > targetBuffer.length) {
081                        _reportArrayOverflow(targetBuffer, outPtr, len);
082                    }
083                    System.arraycopy(sourceBuffer, inPtr, targetBuffer, outPtr, len);
084                    outPtr += len;
085                } else { // compressed
086                    int uncompLen = uint16(sourceBuffer, inPtr);
087                    if ((outPtr + uncompLen) > targetBuffer.length) {
088                        _reportArrayOverflow(targetBuffer, outPtr, uncompLen);
089                    }
090                    inPtr += 2;
091                    decodeChunk(sourceBuffer, inPtr, targetBuffer, outPtr, outPtr + uncompLen);
092                    outPtr += uncompLen;
093                }
094                inPtr += len;
095                ++blockNr;
096            }
097            return outPtr;
098        }
099    
100        /**
101         * Main decode from a stream. Decompressed bytes are placed in the
102         * outputBuffer, inputBuffer is a "scratch-area".
103         *
104         * @param is           An input stream of LZF compressed bytes
105         * @param inputBuffer  A byte array used as a scratch area.
106         * @param outputBuffer A byte array in which the result is returned
107         * @return The number of bytes placed in the outputBuffer.
108         */
109        public abstract int decodeChunk(final InputStream is, final byte[] inputBuffer, final byte[] outputBuffer)
110                throws IOException;
111    
112        /**
113         * Main decode method for individual chunks.
114         */
115        public abstract void decodeChunk(byte[] in, int inPos, byte[] out, int outPos, int outEnd)
116                throws IOException;
117    
118        /*
119         ///////////////////////////////////////////////////////////////////////
120         // Public static methods
121         ///////////////////////////////////////////////////////////////////////
122         */
123    
124        /**
125         * Helper method that will calculate total uncompressed size, for sequence
126         * of one or more LZF blocks stored in given byte array. Will do basic
127         * sanity checking, so that this method can be called to verify against some
128         * types of corruption.
129         */
130        public static int calculateUncompressedSize(byte[] data, int ptr, int length) throws IOException {
131            int uncompressedSize = 0;
132            int blockNr = 0;
133            final int end = ptr + length;
134    
135            while (ptr < end) {
136                // can use optional end marker
137                if (ptr == (data.length + 1) && data[ptr] == BYTE_NULL) {
138                    ++ptr; // so that we'll be at end
139                    break;
140                }
141                // simpler to handle bounds checks by catching exception here...
142                try {
143                    if (data[ptr] != LZFChunk.BYTE_Z || data[ptr + 1] != LZFChunk.BYTE_V) {
144                        throw new IOException("Corrupt input data, block #" + blockNr + " (at offset " + ptr + "): did not start with 'ZV' signature bytes");
145                    }
146                    int type = (int) data[ptr + 2];
147                    int blockLen = uint16(data, ptr + 3);
148                    if (type == LZFChunk.BLOCK_TYPE_NON_COMPRESSED) { // uncompressed
149                        ptr += 5;
150                        uncompressedSize += blockLen;
151                    } else if (type == LZFChunk.BLOCK_TYPE_COMPRESSED) { // compressed
152                        uncompressedSize += uint16(data, ptr + 5);
153                        ptr += 7;
154                    } else { // unknown... CRC-32 would be 2, but that's not implemented by cli tool
155                        throw new IOException("Corrupt input data, block #" + blockNr + " (at offset " + ptr + "): unrecognized block type " + (type & 0xFF));
156                    }
157                    ptr += blockLen;
158                } catch (ArrayIndexOutOfBoundsException e) {
159                    throw new IOException("Corrupt input data, block #" + blockNr + " (at offset " + ptr + "): truncated block header");
160                }
161                ++blockNr;
162            }
163            // one more sanity check:
164            if (ptr != end) {
165                throw new IOException("Corrupt input data: block #" + blockNr + " extends " + (data.length - ptr) + " beyond end of input");
166            }
167            return uncompressedSize;
168        }
169    
170        /*
171         ///////////////////////////////////////////////////////////////////////
172         // Internal methods
173         ///////////////////////////////////////////////////////////////////////
174         */
175        protected final static int uint16(byte[] data, int ptr) {
176            return ((data[ptr] & 0xFF) << 8) + (data[ptr + 1] & 0xFF);
177        }
178    
179        /**
180         * Helper method to forcibly load header bytes that must be read before
181         * chunk can be handled.
182         */
183        protected final static int readHeader(final InputStream is, final byte[] inputBuffer)
184                throws IOException {
185            // Ok: simple case first, where we just get all data we need
186            int needed = HEADER_BYTES;
187            int count = is.read(inputBuffer, 0, needed);
188    
189            if (count == needed) {
190                return count;
191            }
192            if (count <= 0) {
193                return 0;
194            }
195    
196            // if not, a source that trickles data (network etc); must loop
197            int offset = count;
198            needed -= count;
199    
200            do {
201                count = is.read(inputBuffer, offset, needed);
202                if (count <= 0) {
203                    break;
204                }
205                offset += count;
206                needed -= count;
207            } while (needed > 0);
208            return offset;
209        }
210    
211        protected final static void readFully(InputStream is, boolean compressed,
212                                              byte[] outputBuffer, int offset, int len) throws IOException {
213            int left = len;
214            while (left > 0) {
215                int count = is.read(outputBuffer, offset, left);
216                if (count < 0) { // EOF not allowed here
217                    throw new IOException("EOF in " + len + " byte ("
218                            + (compressed ? "" : "un") + "compressed) block: could only read "
219                            + (len - left) + " bytes");
220                }
221                offset += count;
222                left -= count;
223            }
224        }
225    
226        /**
227         * Helper method called when it is determined that the target buffer can not
228         * hold all data to copy or uncompress
229         */
230        protected void _reportArrayOverflow(byte[] targetBuffer, int outPtr, int dataLen)
231                throws IOException {
232            throw new IOException("Target buffer too small (" + targetBuffer.length + "): can not copy/uncompress "
233                    + dataLen + " bytes to offset " + outPtr);
234        }
235    }
