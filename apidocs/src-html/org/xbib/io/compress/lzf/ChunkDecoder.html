016package org.xbib.io.compress.lzf;
017
018import java.io.IOException;
019import java.io.InputStream;
020
021/**
022 * Decoder that handles decoding of sequence of encoded LZF chunks, combining
023 * them into a single contiguous result byte array.
024 */
025public abstract class ChunkDecoder {
026
027    protected final static byte BYTE_NULL = 0;
028    protected final static int HEADER_BYTES = 5;
029
030    public ChunkDecoder() {
031    }
032
033    /**
034     * Method for decompressing a block of input data encoded in LZF block
035     * structure (compatible with lzf command line utility), and can consist of
036     * any number of blocks. Note that input MUST consists of a sequence of one
037     * or more complete chunks; partial chunks can not be handled.
038     */
039    public final byte[] decode(final byte[] inputBuffer) throws IOException {
040        byte[] result = new byte[calculateUncompressedSize(inputBuffer, 0, inputBuffer.length)];
041        decode(inputBuffer, 0, inputBuffer.length, result);
042        return result;
043    }
044
045    /**
046     * Method for decompressing a block of input data encoded in LZF block
047     * structure (compatible with lzf command line utility), and can consist of
048     * any number of blocks. Note that input MUST consists of a sequence of one
049     * or more complete chunks; partial chunks can not be handled.
050     */
051    public final byte[] decode(final byte[] inputBuffer, int inputPtr, int inputLen) throws IOException {
052        byte[] result = new byte[calculateUncompressedSize(inputBuffer, inputPtr, inputLen)];
053        decode(inputBuffer, inputPtr, inputLen, result);
054        return result;
055    }
056
057    /**
058     * Method for decompressing a block of input data encoded in LZF block
059     * structure (compatible with lzf command line utility), and can consist of
060     * any number of blocks. Note that input MUST consists of a sequence of one
061     * or more complete chunks; partial chunks can not be handled.
062     */
063    public final int decode(final byte[] inputBuffer, final byte[] targetBuffer) throws IOException {
064        return decode(inputBuffer, 0, inputBuffer.length, targetBuffer);
065    }
066
067    /**
068     * Method for decompressing a block of input data encoded in LZF block
069     * structure (compatible with lzf command line utility), and can consist of
070     * any number of blocks. Note that input MUST consists of a sequence of one
071     * or more complete chunks; partial chunks can not be handled.
072     */
073    public int decode(final byte[] sourceBuffer, int inPtr, int inLength,
074                      final byte[] targetBuffer) throws IOException {
075        int outPtr = 0;
076        int blockNr = 0;
077
078        final int end = inPtr + inLength - 1; // -1 to offset possible end marker
079
080        while (inPtr < end) {
081            // let's do basic sanity checks; no point in skimping with these checks
082            if (sourceBuffer[inPtr] != LZFChunk.BYTE_Z || sourceBuffer[inPtr + 1] != LZFChunk.BYTE_V) {
083                throw new IOException("Corrupt input data, block #" + blockNr + " (at offset " + inPtr + "): did not start with 'ZV' signature bytes");
084            }
085            inPtr += 2;
086            int type = sourceBuffer[inPtr++];
087            int len = uint16(sourceBuffer, inPtr);
088            inPtr += 2;
089            if (type == LZFChunk.BLOCK_TYPE_NON_COMPRESSED) { // uncompressed
090                if ((outPtr + len) > targetBuffer.length) {
091                    _reportArrayOverflow(targetBuffer, outPtr, len);
092                }
093                System.arraycopy(sourceBuffer, inPtr, targetBuffer, outPtr, len);
094                outPtr += len;
095            } else { // compressed
096                int uncompLen = uint16(sourceBuffer, inPtr);
097                if ((outPtr + uncompLen) > targetBuffer.length) {
098                    _reportArrayOverflow(targetBuffer, outPtr, uncompLen);
099                }
100                inPtr += 2;
101                decodeChunk(sourceBuffer, inPtr, targetBuffer, outPtr, outPtr + uncompLen);
102                outPtr += uncompLen;
103            }
104            inPtr += len;
105            ++blockNr;
106        }
107        return outPtr;
108    }
109
110    /**
111     * Main decode from a stream. Decompressed bytes are placed in the
112     * outputBuffer, inputBuffer is a "scratch-area".
113     *
114     * @param is           An input stream of LZF compressed bytes
115     * @param inputBuffer  A byte array used as a scratch area.
116     * @param outputBuffer A byte array in which the result is returned
117     * @return The number of bytes placed in the outputBuffer.
118     */
119    public abstract int decodeChunk(final InputStream is, final byte[] inputBuffer, final byte[] outputBuffer)
120            throws IOException;
121
122    /**
123     * Main decode method for individual chunks.
124     */
125    public abstract void decodeChunk(byte[] in, int inPos, byte[] out, int outPos, int outEnd)
126            throws IOException;
127
128    /**
129     * Helper method that will calculate total uncompressed size, for sequence
130     * of one or more LZF blocks stored in given byte array. Will do basic
131     * sanity checking, so that this method can be called to verify against some
132     * types of corruption.
133     */
134    public static int calculateUncompressedSize(byte[] data, int ptr, int length) throws IOException {
135        int uncompressedSize = 0;
136        int blockNr = 0;
137        final int end = ptr + length;
138
139        while (ptr < end) {
140            // can use optional end marker
141            if (ptr == (data.length + 1) && data[ptr] == BYTE_NULL) {
142                ++ptr; // so that we'll be at end
143                break;
144            }
145            // simpler to handle bounds checks by catching exception here...
146            try {
147                if (data[ptr] != LZFChunk.BYTE_Z || data[ptr + 1] != LZFChunk.BYTE_V) {
148                    throw new IOException("Corrupt input data, block #" + blockNr + " (at offset " + ptr + "): did not start with 'ZV' signature bytes");
149                }
150                int type = (int) data[ptr + 2];
151                int blockLen = uint16(data, ptr + 3);
152                if (type == LZFChunk.BLOCK_TYPE_NON_COMPRESSED) { // uncompressed
153                    ptr += 5;
154                    uncompressedSize += blockLen;
155                } else if (type == LZFChunk.BLOCK_TYPE_COMPRESSED) { // compressed
156                    uncompressedSize += uint16(data, ptr + 5);
157                    ptr += 7;
158                } else { // unknown... CRC-32 would be 2, but that's not implemented by cli tool
159                    throw new IOException("Corrupt input data, block #" + blockNr + " (at offset " + ptr + "): unrecognized block type " + (type & 0xFF));
160                }
161                ptr += blockLen;
162            } catch (ArrayIndexOutOfBoundsException e) {
163                throw new IOException("Corrupt input data, block #" + blockNr + " (at offset " + ptr + "): truncated block header");
164            }
165            ++blockNr;
166        }
167        // one more sanity check:
168        if (ptr != end) {
169            throw new IOException("Corrupt input data: block #" + blockNr + " extends " + (data.length - ptr) + " beyond end of input");
170        }
171        return uncompressedSize;
172    }
173
174    protected static int uint16(byte[] data, int ptr) {
175        return ((data[ptr] & 0xFF) << 8) + (data[ptr + 1] & 0xFF);
176    }
177
178    /**
179     * Helper method to forcibly load header bytes that must be read before
180     * chunk can be handled.
181     */
182    protected static int readHeader(final InputStream is, final byte[] inputBuffer)
183            throws IOException {
184        // Ok: simple case first, where we just get all data we need
185        int needed = HEADER_BYTES;
186        int count = is.read(inputBuffer, 0, needed);
187
188        if (count == needed) {
189            return count;
190        }
191        if (count <= 0) {
192            return 0;
193        }
194
195        // if not, a source that trickles data (network etc); must loop
196        int offset = count;
197        needed -= count;
198
199        do {
200            count = is.read(inputBuffer, offset, needed);
201            if (count <= 0) {
202                break;
203            }
204            offset += count;
205            needed -= count;
206        } while (needed > 0);
207        return offset;
208    }
209
210    protected static void readFully(InputStream is, boolean compressed,
211                                    byte[] outputBuffer, int offset, int len) throws IOException {
212        int left = len;
213        while (left > 0) {
214            int count = is.read(outputBuffer, offset, left);
215            if (count < 0) { // EOF not allowed here
216                throw new IOException("EOF in " + len + " byte ("
217                        + (compressed ? "" : "un") + "compressed) block: could only read "
218                        + (len - left) + " bytes");
219            }
220            offset += count;
221            left -= count;
222        }
223    }
224
225    /**
226     * Helper method called when it is determined that the target buffer can not
227     * hold all data to copy or uncompress
228     */
229    protected void _reportArrayOverflow(byte[] targetBuffer, int outPtr, int dataLen)
230            throws IOException {
231        throw new IOException("Target buffer too small (" + targetBuffer.length + "): can not copy/uncompress "
232                + dataLen + " bytes to offset " + outPtr);
233    }
234}
