002package org.xbib.io.compress.lzf;
003
004import java.io.IOException;
005
006/**
007 * Decoder that handles decoding of sequence of encoded LZF chunks, combining
008 * them into a single contiguous result byte array. As of version 0.9, this
009 * class has been mostly replaced by {@link ChunkDecoder}, although static
010 * methods are left here and may still be used for convenience. All static
011 * methods use {@link ChunkDecoderFactory#optimalInstance} to find actual
012 * {@link ChunkDecoder} instance to use.
013 */
014public class LZFDecoder {
015
016    public static byte[] decode(final byte[] inputBuffer) throws IOException {
017        return decode(inputBuffer, 0, inputBuffer.length);
018    }
019
020    public static byte[] decode(final byte[] inputBuffer, int offset, int length) throws IOException {
021        return ChunkDecoderFactory.optimalInstance().decode(inputBuffer, offset, length);
022    }
023
024    public static int decode(final byte[] inputBuffer, final byte[] targetBuffer) throws IOException {
025        return decode(inputBuffer, 0, inputBuffer.length, targetBuffer);
026    }
027
028    public static int decode(final byte[] sourceBuffer, int offset, int length, final byte[] targetBuffer) throws IOException {
029        return ChunkDecoderFactory.optimalInstance().decode(sourceBuffer, offset, length, targetBuffer);
030    }
031
032    public static int calculateUncompressedSize(byte[] data, int offset, int length) throws IOException {
033        return ChunkDecoder.calculateUncompressedSize(data, length, length);
034    }
035}
