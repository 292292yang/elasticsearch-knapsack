001package org.xbib.io.compress.lzf.util;
002
003import java.io.File;
004import java.io.FileDescriptor;
005import java.io.FileNotFoundException;
006import java.io.FileOutputStream;
007import java.io.IOException;
008import java.io.OutputStream;
009import org.xbib.io.compress.BufferRecycler;
010import org.xbib.io.compress.lzf.ChunkEncoder;
011import org.xbib.io.compress.lzf.LZFChunk;
012
013/**
014 * Helper class that allows use of LZF compression even if a library requires
015 * use of {@link FileOutputStream}. <p> Note that use of this class is not
016 * recommended unless you absolutely must use a {@link FileOutputStream}
017 * instance; otherwise basic {@link LZFOutputStream} (which uses aggregation for
018 * underlying streams) is more appropriate <p> Implementation note: much of the
019 * code is just copied from {@link LZFOutputStream}, so care must be taken to
020 * keep implementations in sync if there are fixes.
021 *
022 */
023public class LZFFileOutputStream extends FileOutputStream {
024
025    private static final int OUTPUT_BUFFER_SIZE = LZFChunk.MAX_CHUNK_LEN;
026    private final ChunkEncoder _encoder;
027    private final BufferRecycler _recycler;
028    protected byte[] _outputBuffer;
029    protected int _position = 0;
030    /**
031     * Configuration setting that governs whether basic 'flush()' should first
032     * complete a block or not. <p> Default value is 'true'
033     *
034     * @since 0.8
035     */
036    protected boolean _cfgFinishBlockOnFlush = true;
037    /**
038     * Flag that indicates if we have already called '_outputStream.close()' (to
039     * avoid calling it multiple times)
040     */
041    protected boolean _outputStreamClosed;
042    /**
043     * Wrapper object we use to allow decoder to write directly to the stream,
044     * without ending in infinite loop...
045     */
046    private final Wrapper _wrapper;
047
048    /*
049     // Construction, configuration
050     */
051    public LZFFileOutputStream(File file) throws FileNotFoundException {
052        super(file);
053        _encoder = new ChunkEncoder(OUTPUT_BUFFER_SIZE);
054        _recycler = BufferRecycler.instance();
055        _outputBuffer = _recycler.allocOutputBuffer(OUTPUT_BUFFER_SIZE);
056        _wrapper = new Wrapper();
057    }
058
059    public LZFFileOutputStream(File file, boolean append) throws FileNotFoundException {
060        super(file, append);
061        _encoder = new ChunkEncoder(OUTPUT_BUFFER_SIZE);
062        _recycler = BufferRecycler.instance();
063        _outputBuffer = _recycler.allocOutputBuffer(OUTPUT_BUFFER_SIZE);
064        _wrapper = new Wrapper();
065    }
066
067    public LZFFileOutputStream(FileDescriptor fdObj) {
068        super(fdObj);
069        _encoder = new ChunkEncoder(OUTPUT_BUFFER_SIZE);
070        _recycler = BufferRecycler.instance();
071        _outputBuffer = _recycler.allocOutputBuffer(OUTPUT_BUFFER_SIZE);
072        _wrapper = new Wrapper();
073    }
074
075    public LZFFileOutputStream(String name) throws FileNotFoundException {
076        super(name);
077        _encoder = new ChunkEncoder(OUTPUT_BUFFER_SIZE);
078        _recycler = BufferRecycler.instance();
079        _outputBuffer = _recycler.allocOutputBuffer(OUTPUT_BUFFER_SIZE);
080        _wrapper = new Wrapper();
081    }
082
083    public LZFFileOutputStream(String name, boolean append) throws FileNotFoundException {
084        super(name, append);
085        _encoder = new ChunkEncoder(OUTPUT_BUFFER_SIZE);
086        _recycler = BufferRecycler.instance();
087        _outputBuffer = _recycler.allocOutputBuffer(OUTPUT_BUFFER_SIZE);
088        _wrapper = new Wrapper();
089    }
090
091    /**
092     * Method for defining whether call to {@link #flush} will also complete
093     * current block (similar to calling {@link #finishBlock()}) or not.
094     */
095    public LZFFileOutputStream setFinishBlockOnFlush(boolean b) {
096        _cfgFinishBlockOnFlush = b;
097        return this;
098    }
099
100    /*
101     // FileOutputStream overrides
102     */
103    @Override
104    public void close() throws IOException {
105        if (!_outputStreamClosed) {
106            if (_position > 0) {
107                writeCompressedBlock();
108            }
109            super.flush();
110            _encoder.close();
111            byte[] buf = _outputBuffer;
112            if (buf != null) {
113                _outputBuffer = null;
114                _recycler.releaseOutputBuffer(buf);
115            }
116            _outputStreamClosed = true;
117            super.close();
118        }
119    }
120
121    @Override
122    public void flush() throws IOException {
123        checkNotClosed();
124        if (_cfgFinishBlockOnFlush && _position > 0) {
125            writeCompressedBlock();
126        }
127        super.flush();
128    }
129
130    // fine as is: don't override
131    // public FileChannel getChannel();
132    // final, can't override:
133    // public FileDescriptor getFD();
134    @Override
135    public void write(byte[] b) throws IOException {
136        write(b, 0, b.length);
137    }
138
139    @Override
140    public void write(byte[] buffer, int offset, int length) throws IOException {
141        checkNotClosed();
142
143        final int BUFFER_LEN = _outputBuffer.length;
144
145        // simple case first: buffering only (for trivially short writes)
146        int free = BUFFER_LEN - _position;
147        if (free >= length) {
148            System.arraycopy(buffer, offset, _outputBuffer, _position, length);
149            _position += length;
150            return;
151        }
152        // otherwise, copy whatever we can, flush
153        System.arraycopy(buffer, offset, _outputBuffer, _position, free);
154        offset += free;
155        length -= free;
156        _position += free;
157        writeCompressedBlock();
158
159        // then write intermediate full block, if any, without copying:
160        while (length >= BUFFER_LEN) {
161            _encoder.encodeAndWriteChunk(buffer, offset, BUFFER_LEN, _wrapper);
162            offset += BUFFER_LEN;
163            length -= BUFFER_LEN;
164        }
165
166        // and finally, copy leftovers in buffer, if any
167        if (length > 0) {
168            System.arraycopy(buffer, offset, _outputBuffer, 0, length);
169        }
170        _position = length;
171    }
172
173    @Override
174    public void write(int b) throws IOException {
175        checkNotClosed();
176        if (_position >= _outputBuffer.length) {
177            writeCompressedBlock();
178        }
179        _outputBuffer[_position++] = (byte) b;
180    }
181
182    /*
183     // Additional public methods
184     */
185    /**
186     * Accessor for checking whether call to "flush()" will first finish the
187     * current block or not
188     */
189    public boolean getFinishBlockOnFlush() {
190        return _cfgFinishBlockOnFlush;
191    }
192
193    /**
194     * Method that can be used to force completion of the current block, which
195     * means that all buffered data will be compressed into an LZF block. This
196     * typically results in lower compression ratio as larger blocks compress
197     * better; but may be necessary for network connections to ensure timely
198     * sending of data.
199     */
200    public LZFFileOutputStream finishBlock() throws IOException {
201        checkNotClosed();
202        if (_position > 0) {
203            writeCompressedBlock();
204        }
205        return this;
206    }
207
208    /*
209     ///////////////////////////////////////////////////////////////////////
210     // Internal methods
211     ///////////////////////////////////////////////////////////////////////
212     */
213    /**
214     * Compress and write the current block to the OutputStream
215     */
216    protected void writeCompressedBlock() throws IOException {
217        int left = _position;
218        _position = 0;
219        int offset = 0;
220
221        do {
222            int chunkLen = Math.min(LZFChunk.MAX_CHUNK_LEN, left);
223            _encoder.encodeAndWriteChunk(_outputBuffer, offset, chunkLen, _wrapper);
224            offset += chunkLen;
225            left -= chunkLen;
226        } while (left > 0);
227    }
228
229    protected void rawWrite(byte[] buffer, int offset, int length) throws IOException {
230        super.write(buffer, offset, length);
231    }
232
233    protected void checkNotClosed() throws IOException {
234        if (_outputStreamClosed) {
235            throw new IOException(getClass().getName() + " already closed");
236        }
237    }
238
239    /*
240     // Helper class(es)
241     */
242    /**
243     * This simple wrapper is needed to re-route read calls so that they will
244     * use "raw" writes
245     */
246    private final class Wrapper extends OutputStream {
247
248        @Override
249        public void write(int arg0) throws IOException {
250            throw new UnsupportedOperationException();
251        }
252
253        @Override
254        public void write(byte[] buffer, int offset, int length) throws IOException {
255            rawWrite(buffer, offset, length);
256        }
257    }
258}
