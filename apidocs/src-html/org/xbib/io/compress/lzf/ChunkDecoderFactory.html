001package org.xbib.io.compress.lzf;
002
003/**
004 * Simple helper class used for loading {@link ChunkDecoder} implementations,
005 * based on criteria such as "fastest available". <p> Yes, it looks butt-ugly,
006 * but does the job. Nonetheless, if anyone has lipstick for this pig, let me
007 * know.
008 */
009public class ChunkDecoderFactory {
010
011    private final static ChunkDecoderFactory _instance;
012
013    static {
014        Class<?> impl = VanillaChunkDecoder.class;
015        _instance = new ChunkDecoderFactory(impl);
016    }
017
018    private final Class<? extends ChunkDecoder> _implClass;
019
020    @SuppressWarnings("unchecked")
021    private ChunkDecoderFactory(Class<?> imp) {
022        _implClass = (Class<? extends ChunkDecoder>) imp;
023    }
024
025    /*
026     // Public API
027     */
028
029    /**
030     * Method to use for getting decompressor instance that uses the most
031     * optimal available methods for underlying data access. It should be safe
032     * to call this method as implementations are dynamically loaded; however,
033     * on some non-standard platforms it may be necessary to either directly
034     * load instances, or use {@link #safeInstance()}.
035     */
036    public static ChunkDecoder optimalInstance() {
037        try {
038            return _instance._implClass.newInstance();
039        } catch (Exception e) {
040            throw new IllegalStateException("Failed to load a ChunkDecoder instance (" + e.getClass().getName() + "): "
041                    + e.getMessage(), e);
042        }
043    }
044
045    /**
046     * Method that can be used to ensure that a "safe" decompressor instance is
047     * loaded. Safe here means that it should work on any and all Java
048     * platforms.
049     */
050    public static ChunkDecoder safeInstance() {
051        // this will always succeed loading; no need to use dynamic class loading or instantiation
052        return new VanillaChunkDecoder();
053    }
054}
