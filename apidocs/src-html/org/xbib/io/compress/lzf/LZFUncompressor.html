001package org.xbib.io.compress.lzf;
002
003import java.io.IOException;
004import org.xbib.io.compress.BufferRecycler;
005import org.xbib.io.compress.DataHandler;
006import org.xbib.io.compress.Uncompressor;
007import org.xbib.io.compress.lzf.util.ChunkDecoderFactory;
008
009/**
010 * {@link com.ning.compress.Uncompressor} implementation for uncompressing LZF
011 * encoded data in "push" mode, in which input is not read using
012 * {@link java.io.InputStream} but rather pushed to uncompressor in variable
013 * length chunks.
014 */
015public class LZFUncompressor extends Uncompressor {
016    /*
017     // State constants
018     */
019
020    /**
021     * State in which a new block or end-of-stream is expected.
022     */
023    protected final static int STATE_INITIAL = 0;
024    protected final static int STATE_HEADER_Z_GOTTEN = 1;
025    protected final static int STATE_HEADER_ZV_GOTTEN = 2;
026    protected final static int STATE_HEADER_COMPRESSED_0 = 3;
027    protected final static int STATE_HEADER_COMPRESSED_1 = 4;
028    protected final static int STATE_HEADER_COMPRESSED_2 = 5;
029    protected final static int STATE_HEADER_COMPRESSED_3 = 6;
030    protected final static int STATE_HEADER_COMPRESSED_BUFFERING = 7;
031    protected final static int STATE_HEADER_UNCOMPRESSED_0 = 8;
032    protected final static int STATE_HEADER_UNCOMPRESSED_1 = 9;
033    protected final static int STATE_HEADER_UNCOMPRESSED_STREAMING = 10;
034    /*
035     ///////////////////////////////////////////////////////////////////////
036     // Configuration, helper objects
037     ///////////////////////////////////////////////////////////////////////
038     */
039    /**
040     * Handler that will receive uncompressed data.
041     */
042    protected final DataHandler _handler;
043    /**
044     * Underlying decompressor we use for chunk decompression.
045     */
046    protected final ChunkDecoder _decoder;
047    protected final BufferRecycler _recycler;
048    /*
049     ///////////////////////////////////////////////////////////////////////
050     // Decoder state
051     ///////////////////////////////////////////////////////////////////////
052     */
053    /**
054     * Current decoding state, which determines meaning of following byte(s).
055     */
056    protected int _state = STATE_INITIAL;
057    /**
058     * Number of bytes in current, compressed block
059     */
060    protected int _compressedLength;
061    /**
062     * Number of bytes from current block, either after uncompressing data (for
063     * compressed blocks), or included in stream (for uncompressed).
064     */
065    protected int _uncompressedLength;
066    /**
067     * Buffer in which compressed input is buffered if necessary, to get full
068     * chunks for decoding.
069     */
070    protected byte[] _inputBuffer;
071    /**
072     * Buffer used for data uncompressed from
073     * <code>_inputBuffer</code>.
074     */
075    protected byte[] _decodeBuffer;
076    /**
077     * Number of bytes that have been buffered in {@link #_inputBuffer} to be
078     * uncompressed; or copied directly from uncompressed block.
079     */
080    protected int _bytesReadFromBlock;
081
082    /*
083     ///////////////////////////////////////////////////////////////////////
084     // Instance creation
085     ///////////////////////////////////////////////////////////////////////
086     */
087    public LZFUncompressor(DataHandler handler) {
088        this(handler, ChunkDecoderFactory.optimalInstance());
089    }
090
091    public LZFUncompressor(DataHandler handler, ChunkDecoder dec) {
092        _handler = handler;
093        _decoder = dec;
094        _recycler = BufferRecycler.instance();
095    }
096
097    /*
098     ///////////////////////////////////////////////////////////////////////
099     // Uncompressor API implementation
100     ///////////////////////////////////////////////////////////////////////
101     */
102    @Override
103    public void feedCompressedData(byte[] comp, int offset, int len) throws IOException {
104        final int end = offset + len;
105
106        while (offset < end) {
107            byte b = comp[offset++];
108
109            switch (_state) {
110                case STATE_INITIAL:
111                    if (b != LZFChunk.BYTE_Z) {
112                        _reportBadHeader(comp, offset, len, 0);
113                    }
114                    if (offset >= end) {
115                        _state = STATE_HEADER_Z_GOTTEN;
116                        break;
117                    }
118                    b = comp[offset++];
119                // fall through
120                case STATE_HEADER_Z_GOTTEN:
121                    if (b != LZFChunk.BYTE_V) {
122                        _reportBadHeader(comp, offset, len, 1);
123                    }
124                    if (offset >= end) {
125                        _state = STATE_HEADER_ZV_GOTTEN;
126                        break;
127                    }
128                    b = comp[offset++];
129                // fall through
130                case STATE_HEADER_ZV_GOTTEN:
131                    _bytesReadFromBlock = 0; {
132                    int type = b & 0xFF;
133                    if (type != LZFChunk.BLOCK_TYPE_COMPRESSED) {
134                        if (type == LZFChunk.BLOCK_TYPE_NON_COMPRESSED) {
135                            _state = STATE_HEADER_UNCOMPRESSED_0;
136                            continue;
137                        }
138                        _reportBadBlockType(comp, offset, len, type);
139                    }
140                }
141                _state = STATE_HEADER_COMPRESSED_0;
142                if (offset >= end) {
143                    break;
144                }
145                b = comp[offset++];
146                // fall through for compressed blocks
147                case STATE_HEADER_COMPRESSED_0: // first byte of compressed-length
148                    _compressedLength = b & 0xFF;
149                    if (offset >= end) {
150                        _state = STATE_HEADER_COMPRESSED_1;
151                        break;
152                    }
153                    b = comp[offset++];
154                // fall through
155                case STATE_HEADER_COMPRESSED_1:
156                    _compressedLength = (_compressedLength << 8) + (b & 0xFF);
157                    if (offset >= end) {
158                        _state = STATE_HEADER_COMPRESSED_2;
159                        break;
160                    }
161                    b = comp[offset++];
162                // fall through
163                case STATE_HEADER_COMPRESSED_2:
164                    _uncompressedLength = b & 0xFF;
165                    if (offset >= end) {
166                        _state = STATE_HEADER_COMPRESSED_3;
167                        break;
168                    }
169                    b = comp[offset++];
170                // fall through
171                case STATE_HEADER_COMPRESSED_3:
172                    _uncompressedLength = (_uncompressedLength << 8) + (b & 0xFF);
173                    _state = STATE_HEADER_COMPRESSED_BUFFERING;
174                    if (offset >= end) {
175                        break;
176                    }
177                    b = comp[offset++];
178                // fall through
179                case STATE_HEADER_COMPRESSED_BUFFERING:
180                    offset = _handleCompressed(comp, --offset, end);
181                    // either state changes, or we run out of input...
182                    break;
183
184                case STATE_HEADER_UNCOMPRESSED_0:
185                    _uncompressedLength = b & 0xFF;
186                    if (offset >= end) {
187                        _state = STATE_HEADER_UNCOMPRESSED_1;
188                        break;
189                    }
190                    b = comp[offset++];
191                // fall through
192                case STATE_HEADER_UNCOMPRESSED_1:
193                    _uncompressedLength = (_uncompressedLength << 8) + (b & 0xFF);
194                    _state = STATE_HEADER_UNCOMPRESSED_STREAMING;
195                    if (offset >= end) {
196                        break;
197                    }
198                    b = comp[offset++];
199                // fall through
200                case STATE_HEADER_UNCOMPRESSED_STREAMING:
201                    offset = _handleUncompressed(comp, --offset, end);
202                    // All done?
203                    if (_bytesReadFromBlock == _uncompressedLength) {
204                        _state = STATE_INITIAL;
205                    }
206                    break;
207            }
208        }
209    }
210
211    @Override
212    public void complete() throws IOException {
213        byte[] b = _inputBuffer;
214        if (b != null) {
215            _inputBuffer = null;
216            _recycler.releaseInputBuffer(b);
217        }
218        b = _decodeBuffer;
219        if (b != null) {
220            _decodeBuffer = null;
221            _recycler.releaseDecodeBuffer(b);
222        }
223        // 24-May-2012, tatu: Should we call this here; or fail with exception?
224        _handler.allDataHandled();
225        if (_state != STATE_INITIAL) {
226            if (_state == STATE_HEADER_COMPRESSED_BUFFERING) {
227                throw new IOException("Incomplete compressed LZF block; only got " + _bytesReadFromBlock
228                        + " bytes, needed " + _compressedLength);
229            }
230            if (_state == STATE_HEADER_UNCOMPRESSED_STREAMING) {
231                throw new IOException("Incomplete uncompressed LZF block; only got " + _bytesReadFromBlock
232                        + " bytes, needed " + _uncompressedLength);
233            }
234            throw new IOException("Incomplete LZF block; decoding state = " + _state);
235        }
236    }
237
238    /*
239     ///////////////////////////////////////////////////////////////////////
240     // Helper methods, decompression
241     ///////////////////////////////////////////////////////////////////////
242     */
243    private final int _handleUncompressed(byte[] comp, int offset, int end) throws IOException {
244        // Simple, we just do pass through...
245        int amount = Math.min(end - offset, _uncompressedLength - _bytesReadFromBlock);
246        _handler.handleData(comp, offset, amount);
247        _bytesReadFromBlock += amount;
248        return offset + amount;
249    }
250
251    private final int _handleCompressed(byte[] comp, int offset, int end) throws IOException {
252        // One special case: if we get the whole block, can avoid buffering:
253        int available = end - offset;
254        if (_bytesReadFromBlock == 0 && available >= _compressedLength) {
255            _uncompress(comp, offset, _compressedLength);
256            offset += _compressedLength;
257            _state = STATE_INITIAL;
258            return offset;
259        }
260        // otherwise need to buffer
261        if (_inputBuffer == null) {
262            _inputBuffer = _recycler.allocInputBuffer(LZFChunk.MAX_CHUNK_LEN);
263        }
264        int amount = Math.min(available, _compressedLength - _bytesReadFromBlock);
265        System.arraycopy(comp, offset, _inputBuffer, _bytesReadFromBlock, amount);
266        offset += amount;
267        _bytesReadFromBlock += amount;
268        // Got it all?
269        if (_bytesReadFromBlock == _compressedLength) {
270            _uncompress(_inputBuffer, 0, _compressedLength);
271            _state = STATE_INITIAL;
272        }
273        return offset;
274    }
275
276    private final void _uncompress(byte[] src, int srcOffset, int len) throws IOException {
277        if (_decodeBuffer == null) {
278            _decodeBuffer = _recycler.allocDecodeBuffer(LZFChunk.MAX_CHUNK_LEN);
279        }
280        _decoder.decodeChunk(src, srcOffset, _decodeBuffer, 0, _uncompressedLength);
281        _handler.handleData(_decodeBuffer, 0, _uncompressedLength);
282    }
283
284    /*
285     ///////////////////////////////////////////////////////////////////////
286     // Helper methods, error reporting
287     ///////////////////////////////////////////////////////////////////////
288     */
289    protected void _reportBadHeader(byte[] comp, int nextOffset, int len, int relative)
290            throws IOException {
291        char exp = (relative == 0) ? 'Z' : 'V';
292        --nextOffset;
293        throw new IOException("Bad block: byte #" + relative + " of block header not '"
294                + exp + "' (0x" + Integer.toHexString(exp)
295                + ") but 0x" + Integer.toHexString(comp[nextOffset] & 0xFF)
296                + " (at " + (nextOffset - 1) + "/" + (len) + ")");
297    }
298
299    protected void _reportBadBlockType(byte[] comp, int nextOffset, int len, int type)
300            throws IOException {
301        throw new IOException("Bad block: unrecognized type 0x" + Integer.toHexString(type & 0xFF)
302                + " (at " + (nextOffset - 1) + "/" + len + ")");
303    }
