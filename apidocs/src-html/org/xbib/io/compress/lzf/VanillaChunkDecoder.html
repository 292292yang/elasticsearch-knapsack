016package org.xbib.io.compress.lzf;
017
018import java.io.IOException;
019import java.io.InputStream;
020
021/**
022 * Safe {@link ChunkDecoder} implementation that can be used on any platform.
023 */
024public class VanillaChunkDecoder extends ChunkDecoder {
025
026    public VanillaChunkDecoder() {
027    }
028
029    @Override
030    public final int decodeChunk(final InputStream is, final byte[] inputBuffer, final byte[] outputBuffer)
031            throws IOException {
032        int bytesInOutput;
033        /* note: we do NOT read more than 5 bytes because otherwise might need to shuffle bytes
034         * for output buffer (could perhaps optimize in future?)
035         */
036        int bytesRead = readHeader(is, inputBuffer);
037        if ((bytesRead < HEADER_BYTES)
038                || inputBuffer[0] != LZFChunk.BYTE_Z || inputBuffer[1] != LZFChunk.BYTE_V) {
039            if (bytesRead == 0) { // probably fine, clean EOF
040                return -1;
041            }
042            throw new IOException("Corrupt input data, block did not start with 2 byte signature ('ZV') followed by type byte, 2-byte length)");
043        }
044        int type = inputBuffer[2];
045        int compLen = uint16(inputBuffer, 3);
046        if (type == LZFChunk.BLOCK_TYPE_NON_COMPRESSED) { // uncompressed
047            readFully(is, false, outputBuffer, 0, compLen);
048            bytesInOutput = compLen;
049        } else { // compressed
050            readFully(is, true, inputBuffer, 0, 2 + compLen); // first 2 bytes are uncompressed length
051            int uncompLen = uint16(inputBuffer, 0);
052            decodeChunk(inputBuffer, 2, outputBuffer, 0, uncompLen);
053            bytesInOutput = uncompLen;
054        }
055        return bytesInOutput;
056    }
057
058    @Override
059    public final void decodeChunk(byte[] in, int inPos, byte[] out, int outPos, int outEnd)
060            throws IOException {
061        do {
062            int ctrl = in[inPos++] & 255;
063            if (ctrl < LZFChunk.MAX_LITERAL) { // literal run
064                switch (ctrl) {
065                    case 31:
066                        out[outPos++] = in[inPos++];
067                    case 30:
068                        out[outPos++] = in[inPos++];
069                    case 29:
070                        out[outPos++] = in[inPos++];
071                    case 28:
072                        out[outPos++] = in[inPos++];
073                    case 27:
074                        out[outPos++] = in[inPos++];
075                    case 26:
076                        out[outPos++] = in[inPos++];
077                    case 25:
078                        out[outPos++] = in[inPos++];
079                    case 24:
080                        out[outPos++] = in[inPos++];
081                    case 23:
082                        out[outPos++] = in[inPos++];
083                    case 22:
084                        out[outPos++] = in[inPos++];
085                    case 21:
086                        out[outPos++] = in[inPos++];
087                    case 20:
088                        out[outPos++] = in[inPos++];
089                    case 19:
090                        out[outPos++] = in[inPos++];
091                    case 18:
092                        out[outPos++] = in[inPos++];
093                    case 17:
094                        out[outPos++] = in[inPos++];
095                    case 16:
096                        out[outPos++] = in[inPos++];
097                    case 15:
098                        out[outPos++] = in[inPos++];
099                    case 14:
100                        out[outPos++] = in[inPos++];
101                    case 13:
102                        out[outPos++] = in[inPos++];
103                    case 12:
104                        out[outPos++] = in[inPos++];
105                    case 11:
106                        out[outPos++] = in[inPos++];
107                    case 10:
108                        out[outPos++] = in[inPos++];
109                    case 9:
110                        out[outPos++] = in[inPos++];
111                    case 8:
112                        out[outPos++] = in[inPos++];
113                    case 7:
114                        out[outPos++] = in[inPos++];
115                    case 6:
116                        out[outPos++] = in[inPos++];
117                    case 5:
118                        out[outPos++] = in[inPos++];
119                    case 4:
120                        out[outPos++] = in[inPos++];
121                    case 3:
122                        out[outPos++] = in[inPos++];
123                    case 2:
124                        out[outPos++] = in[inPos++];
125                    case 1:
126                        out[outPos++] = in[inPos++];
127                    case 0:
128                        out[outPos++] = in[inPos++];
129                }
130                continue;
131            }
132            // back reference
133            int len = ctrl >> 5;
134            ctrl = -((ctrl & 0x1f) << 8) - 1;
135            if (len < 7) { // 2 bytes; length of 3 - 8 bytes
136                ctrl -= in[inPos++] & 255;
137                out[outPos] = out[outPos++ + ctrl];
138                out[outPos] = out[outPos++ + ctrl];
139                switch (len) {
140                    case 6:
141                        out[outPos] = out[outPos++ + ctrl];
142                    case 5:
143                        out[outPos] = out[outPos++ + ctrl];
144                    case 4:
145                        out[outPos] = out[outPos++ + ctrl];
146                    case 3:
147                        out[outPos] = out[outPos++ + ctrl];
148                    case 2:
149                        out[outPos] = out[outPos++ + ctrl];
150                    case 1:
151                        out[outPos] = out[outPos++ + ctrl];
152                }
153                continue;
154            }
155
156            // long version (3 bytes, length of up to 264 bytes)
157            len = in[inPos++] & 255;
158            ctrl -= in[inPos++] & 255;
159
160            // First: if there is no overlap, can just use arraycopy:
161            if ((ctrl + len) < -9) {
162                len += 9;
163                if (len <= 32) {
164                    copyUpTo32WithSwitch(out, outPos + ctrl, out, outPos, len - 1);
165                } else {
166                    System.arraycopy(out, outPos + ctrl, out, outPos, len);
167                }
168                outPos += len;
169                continue;
170            }
171
172            // otherwise manual copy: so first just copy 9 bytes we know are needed
173            out[outPos] = out[outPos++ + ctrl];
174            out[outPos] = out[outPos++ + ctrl];
175            out[outPos] = out[outPos++ + ctrl];
176            out[outPos] = out[outPos++ + ctrl];
177            out[outPos] = out[outPos++ + ctrl];
178            out[outPos] = out[outPos++ + ctrl];
179            out[outPos] = out[outPos++ + ctrl];
180            out[outPos] = out[outPos++ + ctrl];
181            out[outPos] = out[outPos++ + ctrl];
182
183            // then loop
184            // Odd: after extensive profiling, looks like magic number
185            // for unrolling is 4: with 8 performance is worse (even
186            // bit less than with no unrolling).
187            len += outPos;
188            final int end = len - 3;
189            while (outPos < end) {
190                out[outPos] = out[outPos++ + ctrl];
191                out[outPos] = out[outPos++ + ctrl];
192                out[outPos] = out[outPos++ + ctrl];
193                out[outPos] = out[outPos++ + ctrl];
194            }
195            switch (len - outPos) {
196                case 3:
197                    out[outPos] = out[outPos++ + ctrl];
198                case 2:
199                    out[outPos] = out[outPos++ + ctrl];
200                case 1:
201                    out[outPos] = out[outPos++ + ctrl];
202            }
203        } while (outPos < outEnd);
204
205        // sanity check to guard against corrupt data:
206        if (outPos != outEnd) {
207            throw new IOException("Corrupt data: overrun in decompress, input offset " + inPos + ", output offset " + outPos);
208        }
209    }
