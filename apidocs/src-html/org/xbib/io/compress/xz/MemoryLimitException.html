010package org.xbib.io.compress.xz;
011
012/**
013 * Thrown when the memory usage limit given to the XZ decompressor
014 * would be exceeded.
015 * <p>
016 * The amount of memory required and the memory usage limit are
017 * included in the error detail message in human readable format.
018 */
019public class MemoryLimitException extends XZIOException {
020    private static final long serialVersionUID = 3L;
021
022    private final int memoryNeeded;
023    private final int memoryLimit;
024
025    /**
026     * Creates a new MemoryLimitException.
027     * <p>
028     * The amount of memory needed and the memory usage limit are
029     * included in the error detail message.
030     *
031     * @param   memoryNeeded    amount of memory needed as kibibytes (KiB)
032     * @param   memoryLimit     specified memory usage limit as kibibytes (KiB)
033     */
034    public MemoryLimitException(int memoryNeeded, int memoryLimit) {
035        super("" + memoryNeeded + " KiB of memory would be needed; limit was "
036              + memoryLimit + " KiB");
037
038        this.memoryNeeded = memoryNeeded;
039        this.memoryLimit = memoryLimit;
040    }
041
042    /**
043     * Gets how much memory is required to decompress the data.
044     *
045     * @return      amount of memory needed as kibibytes (KiB)
046     */
047    public int getMemoryNeeded() {
048        return memoryNeeded;
049    }
050
051    /**
052     * Gets what the memory usage limit was at the time the exception
053     * was created.
054     *
055     * @return      memory usage limit as kibibytes (KiB)
056     */
057    public int getMemoryLimit() {
058        return memoryLimit;
059    }
060}
