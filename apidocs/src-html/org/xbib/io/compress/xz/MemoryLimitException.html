002    package org.xbib.io.compress.xz;
003    
004    /**
005     * Thrown when the memory usage limit given to the XZ decompressor
006     * would be exceeded.
007     * <p/>
008     * The amount of memory required and the memory usage limit are
009     * included in the error detail message in human readable format.
010     */
011    public class MemoryLimitException extends XZIOException {
012        private static final long serialVersionUID = 3L;
013    
014        private final int memoryNeeded;
015        private final int memoryLimit;
016    
017        /**
018         * Creates a new MemoryLimitException.
019         * <p/>
020         * The amount of memory needed and the memory usage limit are
021         * included in the error detail message.
022         *
023         * @param memoryNeeded amount of memory needed as kibibytes (KiB)
024         * @param memoryLimit  specified memory usage limit as kibibytes (KiB)
025         */
026        public MemoryLimitException(int memoryNeeded, int memoryLimit) {
027            super("" + memoryNeeded + " KiB of memory would be needed; limit was "
028                    + memoryLimit + " KiB");
029    
030            this.memoryNeeded = memoryNeeded;
031            this.memoryLimit = memoryLimit;
032        }
033    
034        /**
035         * Gets how much memory is required to decompress the data.
036         *
037         * @return amount of memory needed as kibibytes (KiB)
038         */
039        public int getMemoryNeeded() {
040            return memoryNeeded;
041        }
042    
043        /**
044         * Gets what the memory usage limit was at the time the exception
045         * was created.
046         *
047         * @return memory usage limit as kibibytes (KiB)
048         */
049        public int getMemoryLimit() {
050            return memoryLimit;
051        }
052    }
