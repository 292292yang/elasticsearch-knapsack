001package org.xbib.io.compress.xz.common;
002
003import java.io.IOException;
004import java.io.OutputStream;
005import java.util.zip.CRC32;
006
007public class EncoderUtil extends Util {
008    public static void writeCRC32(OutputStream out, byte[] buf)
009            throws IOException {
010        CRC32 crc32 = new CRC32();
011        crc32.update(buf);
012        long value = crc32.getValue();
013
014        for (int i = 0; i < 4; ++i) {
015            out.write((byte) (value >>> (i * 8)));
016        }
017    }
018
019    public static void encodeVLI(OutputStream out, long num)
020            throws IOException {
021        while (num >= 0x80) {
022            out.write((byte) (num | 0x80));
023            num >>>= 7;
024        }
025
026        out.write((byte) num);
027    }
028}
