001package org.xbib.io.compress.xz.simple;
002
003public final class ARM implements SimpleFilter {
004    private final boolean isEncoder;
005    private int pos;
006
007    public ARM(boolean isEncoder, int startPos) {
008        this.isEncoder = isEncoder;
009        pos = startPos + 8;
010    }
011
012    public int code(byte[] buf, int off, int len) {
013        int end = off + len - 4;
014        int i;
015
016        for (i = off; i <= end; i += 4) {
017            if ((buf[i + 3] & 0xFF) == 0xEB) {
018                int src = ((buf[i + 2] & 0xFF) << 16)
019                        | ((buf[i + 1] & 0xFF) << 8)
020                        | (buf[i] & 0xFF);
021                src <<= 2;
022
023                int dest;
024                if (isEncoder) {
025                    dest = src + (pos + i - off);
026                } else {
027                    dest = src - (pos + i - off);
028                }
029
030                dest >>>= 2;
031                buf[i + 2] = (byte) (dest >>> 16);
032                buf[i + 1] = (byte) (dest >>> 8);
033                buf[i] = (byte) dest;
034            }
035        }
036
037        i -= off;
038        pos += i;
039        return i;
040    }
041}
