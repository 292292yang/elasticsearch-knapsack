002    package org.xbib.io.compress.xz.simple;
003    
004    public final class ARMThumb implements SimpleFilter {
005        private final boolean isEncoder;
006        private int pos;
007    
008        public ARMThumb(boolean isEncoder, int startPos) {
009            this.isEncoder = isEncoder;
010            pos = startPos + 4;
011        }
012    
013        public int code(byte[] buf, int off, int len) {
014            int end = off + len - 4;
015            int i;
016    
017            for (i = off; i <= end; i += 2) {
018                if ((buf[i + 1] & 0xF8) == 0xF0 && (buf[i + 3] & 0xF8) == 0xF8) {
019                    int src = ((buf[i + 1] & 0x07) << 19)
020                            | ((buf[i] & 0xFF) << 11)
021                            | ((buf[i + 3] & 0x07) << 8)
022                            | (buf[i + 2] & 0xFF);
023                    src <<= 1;
024    
025                    int dest;
026                    if (isEncoder) {
027                        dest = src + (pos + i - off);
028                    } else {
029                        dest = src - (pos + i - off);
030                    }
031    
032                    dest >>>= 1;
033                    buf[i + 1] = (byte) (0xF0 | ((dest >>> 19) & 0x07));
034                    buf[i] = (byte) (dest >>> 11);
035                    buf[i + 3] = (byte) (0xF8 | ((dest >>> 8) & 0x07));
036                    buf[i + 2] = (byte) dest;
037                    i += 2;
038                }
039            }
040    
041            i -= off;
042            pos += i;
043            return i;
044        }
045    }
