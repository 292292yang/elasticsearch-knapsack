002package org.xbib.io.compress.xz.simple;
003
004public final class PowerPC implements SimpleFilter {
005    private final boolean isEncoder;
006    private int pos;
007
008    public PowerPC(boolean isEncoder, int startPos) {
009        this.isEncoder = isEncoder;
010        pos = startPos;
011    }
012
013    public int code(byte[] buf, int off, int len) {
014        int end = off + len - 4;
015        int i;
016
017        for (i = off; i <= end; i += 4) {
018            if ((buf[i] & 0xFC) == 0x48 && (buf[i + 3] & 0x03) == 0x01) {
019                int src = ((buf[i] & 0x03) << 24)
020                        | ((buf[i + 1] & 0xFF) << 16)
021                        | ((buf[i + 2] & 0xFF) << 8)
022                        | (buf[i + 3] & 0xFC);
023
024                int dest;
025                if (isEncoder) {
026                    dest = src + (pos + i - off);
027                } else {
028                    dest = src - (pos + i - off);
029                }
030
031                buf[i] = (byte) (0x48 | ((dest >>> 24) & 0x03));
032                buf[i + 1] = (byte) (dest >>> 16);
033                buf[i + 2] = (byte) (dest >>> 8);
034                buf[i + 3] = (byte) ((buf[i + 3] & 0x03) | dest);
035            }
036        }
037
038        i -= off;
039        pos += i;
040        return i;
041    }
042}
