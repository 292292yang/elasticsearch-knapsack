002    package org.xbib.io.compress.xz.simple;
003    
004    public final class SPARC implements SimpleFilter {
005        private final boolean isEncoder;
006        private int pos;
007    
008        public SPARC(boolean isEncoder, int startPos) {
009            this.isEncoder = isEncoder;
010            pos = startPos;
011        }
012    
013        public int code(byte[] buf, int off, int len) {
014            int end = off + len - 4;
015            int i;
016    
017            for (i = off; i <= end; i += 4) {
018                if ((buf[i] == 0x40 && (buf[i + 1] & 0xC0) == 0x00)
019                        || (buf[i] == 0x7F && (buf[i + 1] & 0xC0) == 0xC0)) {
020                    int src = ((buf[i] & 0xFF) << 24)
021                            | ((buf[i + 1] & 0xFF) << 16)
022                            | ((buf[i + 2] & 0xFF) << 8)
023                            | (buf[i + 3] & 0xFF);
024                    src <<= 2;
025    
026                    int dest;
027                    if (isEncoder) {
028                        dest = src + (pos + i - off);
029                    } else {
030                        dest = src - (pos + i - off);
031                    }
032    
033                    dest >>>= 2;
034                    dest = (((0 - ((dest >>> 22) & 1)) << 22) & 0x3FFFFFFF)
035                            | (dest & 0x3FFFFF) | 0x40000000;
036    
037                    buf[i] = (byte) (dest >>> 24);
038                    buf[i + 1] = (byte) (dest >>> 16);
039                    buf[i + 2] = (byte) (dest >>> 8);
040                    buf[i + 3] = (byte) dest;
041                }
042            }
043    
044            i -= off;
045            pos += i;
046            return i;
047        }
048    }
