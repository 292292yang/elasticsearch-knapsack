011package org.xbib.io.compress.xz.simple;
012
013public final class X86 implements SimpleFilter {
014    private static final boolean[] MASK_TO_ALLOWED_STATUS
015            = {true, true, true, false, true, false, false, false};
016
017    private static final int[] MASK_TO_BIT_NUMBER = {0, 1, 2, 2, 3, 3, 3, 3};
018
019    private final boolean isEncoder;
020    private int pos;
021    private int prevMask = 0;
022
023    private static boolean test86MSByte(byte b) {
024        int i = b & 0xFF;
025        return i == 0x00 || i == 0xFF;
026    }
027
028    public X86(boolean isEncoder, int startPos) {
029        this.isEncoder = isEncoder;
030        pos = startPos + 5;
031    }
032
033    public int code(byte[] buf, int off, int len) {
034        int prevPos = off - 1;
035        int end = off + len - 5;
036        int i;
037
038        for (i = off; i <= end; ++i) {
039            if ((buf[i] & 0xFE) != 0xE8)
040                continue;
041
042            prevPos = i - prevPos;
043            if ((prevPos & ~3) != 0) { // (unsigned)prevPos > 3
044                prevMask = 0;
045            } else {
046                prevMask = (prevMask << (prevPos - 1)) & 7;
047                if (prevMask != 0) {
048                    if (!MASK_TO_ALLOWED_STATUS[prevMask] || test86MSByte(
049                            buf[i + 4 - MASK_TO_BIT_NUMBER[prevMask]])) {
050                        prevPos = i;
051                        prevMask = (prevMask << 1) | 1;
052                        continue;
053                    }
054                }
055            }
056
057            prevPos = i;
058
059            if (test86MSByte(buf[i + 4])) {
060                int src = (buf[i + 1] & 0xFF)
061                          | ((buf[i + 2] & 0xFF) << 8)
062                          | ((buf[i + 3] & 0xFF) << 16)
063                          | ((buf[i + 4] & 0xFF) << 24);
064                int dest;
065                while (true) {
066                    if (isEncoder)
067                        dest = src + (pos + i - off);
068                    else
069                        dest = src - (pos + i - off);
070
071                    if (prevMask == 0)
072                        break;
073
074                    int index = MASK_TO_BIT_NUMBER[prevMask] * 8;
075                    if (!test86MSByte((byte)(dest >>> (24 - index))))
076                        break;
077
078                    src = dest ^ ((1 << (32 - index)) - 1);
079                }
080
081                buf[i + 1] = (byte)dest;
082                buf[i + 2] = (byte)(dest >>> 8);
083                buf[i + 3] = (byte)(dest >>> 16);
084                buf[i + 4] = (byte)(~(((dest >>> 24) & 1) - 1));
085                i += 4;
086            } else {
087                prevMask = (prevMask << 1) | 1;
088            }
089        }
090
091        prevPos = i - prevPos;
092        prevMask = ((prevPos & ~3) != 0) ? 0 : prevMask << (prevPos - 1);
093
094        i -= off;
095        pos += i;
096        return i;
097    }
098}
