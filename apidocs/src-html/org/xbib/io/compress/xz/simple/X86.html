001package org.xbib.io.compress.xz.simple;
002
003public final class X86 implements SimpleFilter {
004    private static final boolean[] MASK_TO_ALLOWED_STATUS
005            = {true, true, true, false, true, false, false, false};
006
007    private static final int[] MASK_TO_BIT_NUMBER = {0, 1, 2, 2, 3, 3, 3, 3};
008
009    private final boolean isEncoder;
010    private int pos;
011    private int prevMask = 0;
012
013    private static boolean test86MSByte(byte b) {
014        int i = b & 0xFF;
015        return i == 0x00 || i == 0xFF;
016    }
017
018    public X86(boolean isEncoder, int startPos) {
019        this.isEncoder = isEncoder;
020        pos = startPos + 5;
021    }
022
023    public int code(byte[] buf, int off, int len) {
024        int prevPos = off - 1;
025        int end = off + len - 5;
026        int i;
027
028        for (i = off; i <= end; ++i) {
029            if ((buf[i] & 0xFE) != 0xE8) {
030                continue;
031            }
032
033            prevPos = i - prevPos;
034            if ((prevPos & ~3) != 0) { // (unsigned)prevPos > 3
035                prevMask = 0;
036            } else {
037                prevMask = (prevMask << (prevPos - 1)) & 7;
038                if (prevMask != 0) {
039                    if (!MASK_TO_ALLOWED_STATUS[prevMask] || test86MSByte(
040                            buf[i + 4 - MASK_TO_BIT_NUMBER[prevMask]])) {
041                        prevPos = i;
042                        prevMask = (prevMask << 1) | 1;
043                        continue;
044                    }
045                }
046            }
047
048            prevPos = i;
049
050            if (test86MSByte(buf[i + 4])) {
051                int src = (buf[i + 1] & 0xFF)
052                        | ((buf[i + 2] & 0xFF) << 8)
053                        | ((buf[i + 3] & 0xFF) << 16)
054                        | ((buf[i + 4] & 0xFF) << 24);
055                int dest;
056                while (true) {
057                    if (isEncoder) {
058                        dest = src + (pos + i - off);
059                    } else {
060                        dest = src - (pos + i - off);
061                    }
062
063                    if (prevMask == 0) {
064                        break;
065                    }
066
067                    int index = MASK_TO_BIT_NUMBER[prevMask] * 8;
068                    if (!test86MSByte((byte) (dest >>> (24 - index)))) {
069                        break;
070                    }
071
072                    src = dest ^ ((1 << (32 - index)) - 1);
073                }
074
075                buf[i + 1] = (byte) dest;
076                buf[i + 2] = (byte) (dest >>> 8);
077                buf[i + 3] = (byte) (dest >>> 16);
078                buf[i + 4] = (byte) (~(((dest >>> 24) & 1) - 1));
079                i += 4;
080            } else {
081                prevMask = (prevMask << 1) | 1;
082            }
083        }
084
085        prevPos = i - prevPos;
086        prevMask = ((prevPos & ~3) != 0) ? 0 : prevMask << (prevPos - 1);
087
088        i -= off;
089        pos += i;
090        return i;
091    }
092}
