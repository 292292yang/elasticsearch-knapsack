001package org.xbib.io.compress.xz.simple;
002
003public final class IA64 implements SimpleFilter {
004    private static final int[] BRANCH_TABLE = {
005            0, 0, 0, 0, 0, 0, 0, 0,
006            0, 0, 0, 0, 0, 0, 0, 0,
007            4, 4, 6, 6, 0, 0, 7, 7,
008            4, 4, 0, 0, 4, 4, 0, 0};
009
010    private final boolean isEncoder;
011    private int pos;
012
013    public IA64(boolean isEncoder, int startPos) {
014        this.isEncoder = isEncoder;
015        pos = startPos;
016    }
017
018    public int code(byte[] buf, int off, int len) {
019        int end = off + len - 16;
020        int i;
021
022        for (i = off; i <= end; i += 16) {
023            int instrTemplate = buf[i] & 0x1F;
024            int mask = BRANCH_TABLE[instrTemplate];
025
026            for (int slot = 0, bitPos = 5; slot < 3; ++slot, bitPos += 41) {
027                if (((mask >>> slot) & 1) == 0) {
028                    continue;
029                }
030
031                int bytePos = bitPos >>> 3;
032                int bitRes = bitPos & 7;
033
034                long instr = 0;
035                for (int j = 0; j < 6; ++j) {
036                    instr |= (buf[i + bytePos + j] & 0xFFL) << (8 * j);
037                }
038
039                long instrNorm = instr >>> bitRes;
040
041                if (((instrNorm >>> 37) & 0x0F) != 0x05
042                        || ((instrNorm >>> 9) & 0x07) != 0x00) {
043                    continue;
044                }
045
046                int src = (int) ((instrNorm >>> 13) & 0x0FFFFF);
047                src |= ((int) (instrNorm >>> 36) & 1) << 20;
048                src <<= 4;
049
050                int dest;
051                if (isEncoder) {
052                    dest = src + (pos + i - off);
053                } else {
054                    dest = src - (pos + i - off);
055                }
056
057                dest >>>= 4;
058
059                instrNorm &= ~(0x8FFFFFL << 13);
060                instrNorm |= (dest & 0x0FFFFFL) << 13;
061                instrNorm |= (dest & 0x100000L) << (36 - 20);
062
063                instr &= (1 << bitRes) - 1;
064                instr |= instrNorm << bitRes;
065
066                for (int j = 0; j < 6; ++j) {
067                    buf[i + bytePos + j] = (byte) (instr >>> (8 * j));
068                }
069            }
070        }
071
072        i -= off;
073        pos += i;
074        return i;
075    }
076}
