002    package org.xbib.io.compress.xz.delta;
003    
004    public class DeltaEncoder extends DeltaCoder {
005        public DeltaEncoder(int distance) {
006            super(distance);
007        }
008    
009        public void encode(byte[] in, int in_off, int len, byte[] out) {
010            for (int i = 0; i < len; ++i) {
011                byte tmp = history[(distance + pos) & DISTANCE_MASK];
012                history[pos-- & DISTANCE_MASK] = in[in_off + i];
013                out[i] = (byte) (in[in_off + i] - tmp);
014            }
015        }
016    }
