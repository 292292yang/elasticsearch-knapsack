010package org.xbib.io.compress.xz.delta;
011
012public class DeltaEncoder extends DeltaCoder {
013    public DeltaEncoder(int distance) {
014        super(distance);
015    }
016
017    public void encode(byte[] in, int in_off, int len, byte[] out) {
018        for (int i = 0; i < len; ++i) {
019            byte tmp = history[(distance + pos) & DISTANCE_MASK];
020            history[pos-- & DISTANCE_MASK] = in[in_off + i];
021            out[i] = (byte)(in[in_off + i] - tmp);
022        }
023    }
024}
