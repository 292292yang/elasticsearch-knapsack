002    package org.xbib.io.compress.xz.delta;
003    
004    public class DeltaDecoder extends DeltaCoder {
005        public DeltaDecoder(int distance) {
006            super(distance);
007        }
008    
009        public void decode(byte[] buf, int off, int len) {
010            int end = off + len;
011            for (int i = off; i < end; ++i) {
012                buf[i] += history[(distance + pos) & DISTANCE_MASK];
013                history[pos-- & DISTANCE_MASK] = buf[i];
014            }
015        }
016    }
