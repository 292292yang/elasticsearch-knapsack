002package org.xbib.io.compress.xz;
003
004import org.xbib.io.compress.xz.common.DecoderUtil;
005
006import java.io.DataInputStream;
007import java.io.IOException;
008import java.io.InputStream;
009
010/**
011 * Decompresses a .xz file in streamed mode (no seeking).
012 * <p/>
013 * Use this to decompress regular standalone .xz files. This reads from
014 * its input stream until the end of the input or until an error occurs.
015 * This supports decompressing concatenated .xz files.
016 * <p/>
017 * <h4>Typical use cases</h4>
018 * <p/>
019 * Getting an input stream to decompress a .xz file:
020 * <p><blockquote><pre>
021 * InputStream infile = new FileInputStream("foo.xz");
022 * XZInputStream inxz = new XZInputStream(infile);
023 * </pre></blockquote>
024 * <p/>
025 * It's important to keep in mind that decompressor memory usage depends
026 * on the settings used to compress the file. The worst-case memory usage
027 * of XZInputStream is currently 1.5&nbsp;GiB. Still, very few files will
028 * require more than about 65&nbsp;MiB because that's how much decompressing
029 * a file created with the highest preset level will need, and only a few
030 * people use settings other than the predefined presets.
031 * <p/>
032 * It is possible to specify a memory usage limit for
033 * <code>XZInputStream</code>. If decompression requires more memory than
034 * the specified limit, MemoryLimitException will be thrown when reading
035 * from the stream. For example, the following sets the memory usage limit
036 * to 100&nbsp;MiB:
037 * <p><blockquote><pre>
038 * InputStream infile = new FileInputStream("foo.xz");
039 * XZInputStream inxz = new XZInputStream(infile, 100 * 1024);
040 * </pre></blockquote>
041 * <p/>
042 * <h4>When uncompressed size is known beforehand</h4>
043 * <p/>
044 * If you are decompressing complete files and your application knows
045 * exactly how much uncompressed data there should be, it is good to try
046 * reading one more byte by calling <code>read()</code> and checking
047 * that it returns <code>-1</code>. This way the decompressor will parse the
048 * file footers and verify the integrity checks, giving the caller more
049 * confidence that the uncompressed data is valid. (This advice seems to
050 * apply to <code>java.util.zip.GZIPInputStream</code> too.)
051 *
052 * @see SingleXZInputStream
053 */
054public class XZInputStream extends InputStream {
055    private final int memoryLimit;
056    private InputStream in;
057    private SingleXZInputStream xzIn;
058    private boolean endReached = false;
059    private IOException exception = null;
060
061    /**
062     * Creates a new XZ decompressor without a memory usage limit.
063     * <p/>
064     * This constructor reads and parses the XZ Stream Header (12 bytes)
065     * from <code>in</code>. The header of the first Block is not read
066     * until <code>read</code> is called.
067     *
068     * @param in input stream from which XZ-compressed
069     *           data is read
070     * @throws XZFormatException           input is not in the XZ format
071     * @throws CorruptedInputException     XZ header CRC32 doesn't match
072     * @throws UnsupportedOptionsException XZ header is valid but specifies options
073     *                                     not supported by this implementation
074     * @throws java.io.EOFException        less than 12 bytes of input was available
075     *                                     from <code>in</code>
076     * @throws java.io.IOException         may be thrown by <code>in</code>
077     */
078    public XZInputStream(InputStream in) throws IOException {
079        this(in, -1);
080    }
081
082    /**
083     * Creates a new XZ decompressor with an optional memory usage limit.
084     * <p/>
085     * This is identical to <code>XZInputStream(InputStream)</code> except
086     * that this takes also the <code>memoryLimit</code> argument.
087     *
088     * @param in          input stream from which XZ-compressed
089     *                    data is read
090     * @param memoryLimit memory usage limit in kibibytes (KiB)
091     *                    or <code>-1</code> to impose no
092     *                    memory usage limit
093     * @throws XZFormatException           input is not in the XZ format
094     * @throws CorruptedInputException     XZ header CRC32 doesn't match
095     * @throws UnsupportedOptionsException XZ header is valid but specifies options
096     *                                     not supported by this implementation
097     * @throws java.io.EOFException        less than 12 bytes of input was available
098     *                                     from <code>in</code>
099     * @throws java.io.IOException         may be thrown by <code>in</code>
100     */
101    public XZInputStream(InputStream in, int memoryLimit) throws IOException {
102        this.in = in;
103        this.memoryLimit = memoryLimit;
104        this.xzIn = new SingleXZInputStream(in, memoryLimit);
105    }
106
107    /**
108     * Decompresses the next byte from this input stream.
109     * <p/>
110     * Reading lots of data with <code>read()</code> from this input stream
111     * may be inefficient. Wrap it in {@link java.io.BufferedInputStream}
112     * if you need to read lots of data one byte at a time.
113     *
114     * @return the next decompressed byte, or <code>-1</code>
115     * to indicate the end of the compressed stream
116     * @throws CorruptedInputException
117     * @throws UnsupportedOptionsException
118     * @throws MemoryLimitException
119     * @throws XZIOException               if the stream has been closed
120     * @throws java.io.EOFException        compressed input is truncated or corrupt
121     * @throws java.io.IOException         may be thrown by <code>in</code>
122     */
123    public int read() throws IOException {
124        byte[] buf = new byte[1];
125        return read(buf, 0, 1) == -1 ? -1 : (buf[0] & 0xFF);
126    }
127
128    /**
129     * Decompresses into an array of bytes.
130     * <p/>
131     * If <code>len</code> is zero, no bytes are read and <code>0</code>
132     * is returned. Otherwise this will try to decompress <code>len</code>
133     * bytes of uncompressed data. Less than <code>len</code> bytes may
134     * be read only in the following situations:
135     * <ul>
136     * <li>The end of the compressed data was reached successfully.</li>
137     * <li>An error is detected after at least one but less <code>len</code>
138     * bytes have already been successfully decompressed.
139     * The next call with non-zero <code>len</code> will immediately
140     * throw the pending exception.</li>
141     * <li>An exception is thrown.</li>
142     * </ul>
143     *
144     * @param buf target buffer for uncompressed data
145     * @param off start offset in <code>buf</code>
146     * @param len maximum number of uncompressed bytes to read
147     * @return number of bytes read, or <code>-1</code> to indicate
148     * the end of the compressed stream
149     * @throws CorruptedInputException
150     * @throws UnsupportedOptionsException
151     * @throws MemoryLimitException
152     * @throws XZIOException               if the stream has been closed
153     * @throws java.io.EOFException        compressed input is truncated or corrupt
154     * @throws java.io.IOException         may be thrown by <code>in</code>
155     */
156    public int read(byte[] buf, int off, int len) throws IOException {
157        if (off < 0 || len < 0 || off + len < 0 || off + len > buf.length) {
158            throw new IndexOutOfBoundsException();
159        }
160
161        if (len == 0) {
162            return 0;
163        }
164
165        if (in == null) {
166            throw new XZIOException("Stream closed");
167        }
168
169        if (exception != null) {
170            throw exception;
171        }
172
173        if (endReached) {
174            return -1;
175        }
176
177        int size = 0;
178
179        try {
180            while (len > 0) {
181                if (xzIn == null) {
182                    prepareNextStream();
183                    if (endReached) {
184                        return size == 0 ? -1 : size;
185                    }
186                }
187
188                int ret = xzIn.read(buf, off, len);
189
190                if (ret > 0) {
191                    size += ret;
192                    off += ret;
193                    len -= ret;
194                } else if (ret == -1) {
195                    xzIn = null;
196                }
197            }
198        } catch (IOException e) {
199            exception = e;
200            if (size == 0) {
201                throw e;
202            }
203        }
204
205        return size;
206    }
207
208    private void prepareNextStream() throws IOException {
209        DataInputStream inData = new DataInputStream(in);
210        byte[] buf = new byte[DecoderUtil.STREAM_HEADER_SIZE];
211
212        // The size of Stream Padding must be a multiple of four bytes,
213        // all bytes zero.
214        do {
215            // First try to read one byte to see if we have reached the end
216            // of the file.
217            int ret = inData.read(buf, 0, 1);
218            if (ret == -1) {
219                endReached = true;
220                return;
221            }
222
223            // Since we got one byte of input, there must be at least
224            // three more available in a valid file.
225            inData.readFully(buf, 1, 3);
226
227        } while (buf[0] == 0 && buf[1] == 0 && buf[2] == 0 && buf[3] == 0);
228
229        // Not all bytes are zero. In a valid Stream it indicates the
230        // beginning of the next Stream. Read the rest of the Stream Header
231        // and initialize the XZ decoder.
232        inData.readFully(buf, 4, DecoderUtil.STREAM_HEADER_SIZE - 4);
233
234        try {
235            xzIn = new SingleXZInputStream(in, memoryLimit, buf);
236        } catch (XZFormatException e) {
237            // Since this isn't the first .xz Stream, it is more
238            // logical to tell that the data is corrupt.
239            throw new CorruptedInputException(
240                    "Garbage after a valid XZ Stream");
241        }
242    }
243
244    /**
245     * Returns the number of uncompressed bytes that can be read
246     * without blocking. The value is returned with an assumption
247     * that the compressed input data will be valid. If the compressed
248     * data is corrupt, <code>CorruptedInputException</code> may get
249     * thrown before the number of bytes claimed to be available have
250     * been read from this input stream.
251     *
252     * @return the number of uncompressed bytes that can be read
253     * without blocking
254     */
255    public int available() throws IOException {
256        if (in == null) {
257            throw new XZIOException("Stream closed");
258        }
259
260        if (exception != null) {
261            throw exception;
262        }
263
264        return xzIn == null ? 0 : xzIn.available();
265    }
266
267    /**
268     * Closes the stream and calls <code>in.close()</code>.
269     * If the stream was already closed, this does nothing.
270     *
271     * @throws java.io.IOException if thrown by <code>in.close()</code>
272     */
273    public void close() throws IOException {
274        if (in != null) {
275            try {
276                in.close();
277            } finally {
278                in = null;
279            }
280        }
281    }
282}
