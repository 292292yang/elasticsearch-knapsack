010package org.xbib.io.compress.xz;
011
012import java.io.InputStream;
013import java.io.DataInputStream;
014import java.io.IOException;
015import java.io.EOFException;
016import org.xbib.io.compress.xz.common.DecoderUtil;
017
018/**
019 * Decompresses a .xz file in streamed mode (no seeking).
020 * <p>
021 * Use this to decompress regular standalone .xz files. This reads from
022 * its input stream until the end of the input or until an error occurs.
023 * This supports decompressing concatenated .xz files.
024 *
025 * <h4>Typical use cases</h4>
026 * <p>
027 * Getting an input stream to decompress a .xz file:
028 * <p><blockquote><pre>
029 * InputStream infile = new FileInputStream("foo.xz");
030 * XZInputStream inxz = new XZInputStream(infile);
031 * </pre></blockquote>
032 * <p>
033 * It's important to keep in mind that decompressor memory usage depends
034 * on the settings used to compress the file. The worst-case memory usage
035 * of XZInputStream is currently 1.5&nbsp;GiB. Still, very few files will
036 * require more than about 65&nbsp;MiB because that's how much decompressing
037 * a file created with the highest preset level will need, and only a few
038 * people use settings other than the predefined presets.
039 * <p>
040 * It is possible to specify a memory usage limit for
041 * <code>XZInputStream</code>. If decompression requires more memory than
042 * the specified limit, MemoryLimitException will be thrown when reading
043 * from the stream. For example, the following sets the memory usage limit
044 * to 100&nbsp;MiB:
045 * <p><blockquote><pre>
046 * InputStream infile = new FileInputStream("foo.xz");
047 * XZInputStream inxz = new XZInputStream(infile, 100 * 1024);
048 * </pre></blockquote>
049 *
050 * <h4>When uncompressed size is known beforehand</h4>
051 * <p>
052 * If you are decompressing complete files and your application knows
053 * exactly how much uncompressed data there should be, it is good to try
054 * reading one more byte by calling <code>read()</code> and checking
055 * that it returns <code>-1</code>. This way the decompressor will parse the
056 * file footers and verify the integrity checks, giving the caller more
057 * confidence that the uncompressed data is valid. (This advice seems to
058 * apply to <code>java.util.zip.GZIPInputStream</code> too.)
059 *
060 * @see SingleXZInputStream
061 */
062public class XZInputStream extends InputStream {
063    private final int memoryLimit;
064    private InputStream in;
065    private SingleXZInputStream xzIn;
066    private boolean endReached = false;
067    private IOException exception = null;
068
069    /**
070     * Creates a new XZ decompressor without a memory usage limit.
071     * <p>
072     * This constructor reads and parses the XZ Stream Header (12 bytes)
073     * from <code>in</code>. The header of the first Block is not read
074     * until <code>read</code> is called.
075     *
076     * @param       in          input stream from which XZ-compressed
077     *                          data is read
078     *
079     * @throws      XZFormatException
080     *                          input is not in the XZ format
081     *
082     * @throws      CorruptedInputException
083     *                          XZ header CRC32 doesn't match
084     *
085     * @throws      UnsupportedOptionsException
086     *                          XZ header is valid but specifies options
087     *                          not supported by this implementation
088     *
089     * @throws      EOFException
090     *                          less than 12 bytes of input was available
091     *                          from <code>in</code>
092     *
093     * @throws      IOException may be thrown by <code>in</code>
094     */
095    public XZInputStream(InputStream in) throws IOException {
096        this(in, -1);
097    }
098
099    /**
100     * Creates a new XZ decompressor with an optional memory usage limit.
101     * <p>
102     * This is identical to <code>XZInputStream(InputStream)</code> except
103     * that this takes also the <code>memoryLimit</code> argument.
104     *
105     * @param       in          input stream from which XZ-compressed
106     *                          data is read
107     *
108     * @param       memoryLimit memory usage limit in kibibytes (KiB)
109     *                          or <code>-1</code> to impose no
110     *                          memory usage limit
111     *
112     * @throws      XZFormatException
113     *                          input is not in the XZ format
114     *
115     * @throws      CorruptedInputException
116     *                          XZ header CRC32 doesn't match
117     *
118     * @throws      UnsupportedOptionsException
119     *                          XZ header is valid but specifies options
120     *                          not supported by this implementation
121     *
122     * @throws      EOFException
123     *                          less than 12 bytes of input was available
124     *                          from <code>in</code>
125     *
126     * @throws      IOException may be thrown by <code>in</code>
127     */
128    public XZInputStream(InputStream in, int memoryLimit) throws IOException {
129        this.in = in;
130        this.memoryLimit = memoryLimit;
131        this.xzIn = new SingleXZInputStream(in, memoryLimit);
132    }
133
134    /**
135     * Decompresses the next byte from this input stream.
136     * <p>
137     * Reading lots of data with <code>read()</code> from this input stream
138     * may be inefficient. Wrap it in {@link java.io.BufferedInputStream}
139     * if you need to read lots of data one byte at a time.
140     *
141     * @return      the next decompressed byte, or <code>-1</code>
142     *              to indicate the end of the compressed stream
143     *
144     * @throws      CorruptedInputException
145     * @throws      UnsupportedOptionsException
146     * @throws      MemoryLimitException
147     *
148     * @throws      XZIOException if the stream has been closed
149     *
150     * @throws      EOFException
151     *                          compressed input is truncated or corrupt
152     *
153     * @throws      IOException may be thrown by <code>in</code>
154     */
155    public int read() throws IOException {
156        byte[] buf = new byte[1];
157        return read(buf, 0, 1) == -1 ? -1 : (buf[0] & 0xFF);
158    }
159
160    /**
161     * Decompresses into an array of bytes.
162     * <p>
163     * If <code>len</code> is zero, no bytes are read and <code>0</code>
164     * is returned. Otherwise this will try to decompress <code>len</code>
165     * bytes of uncompressed data. Less than <code>len</code> bytes may
166     * be read only in the following situations:
167     * <ul>
168     *   <li>The end of the compressed data was reached successfully.</li>
169     *   <li>An error is detected after at least one but less <code>len</code>
170     *       bytes have already been successfully decompressed.
171     *       The next call with non-zero <code>len</code> will immediately
172     *       throw the pending exception.</li>
173     *   <li>An exception is thrown.</li>
174     * </ul>
175     *
176     * @param       buf         target buffer for uncompressed data
177     * @param       off         start offset in <code>buf</code>
178     * @param       len         maximum number of uncompressed bytes to read
179     *
180     * @return      number of bytes read, or <code>-1</code> to indicate
181     *              the end of the compressed stream
182     *
183     * @throws      CorruptedInputException
184     * @throws      UnsupportedOptionsException
185     * @throws      MemoryLimitException
186     *
187     * @throws      XZIOException if the stream has been closed
188     *
189     * @throws      EOFException
190     *                          compressed input is truncated or corrupt
191     *
192     * @throws      IOException may be thrown by <code>in</code>
193     */
194    public int read(byte[] buf, int off, int len) throws IOException {
195        if (off < 0 || len < 0 || off + len < 0 || off + len > buf.length)
196            throw new IndexOutOfBoundsException();
197
198        if (len == 0)
199            return 0;
200
201        if (in == null)
202            throw new XZIOException("Stream closed");
203
204        if (exception != null)
205            throw exception;
206
207        if (endReached)
208            return -1;
209
210        int size = 0;
211
212        try {
213            while (len > 0) {
214                if (xzIn == null) {
215                    prepareNextStream();
216                    if (endReached)
217                        return size == 0 ? -1 : size;
218                }
219
220                int ret = xzIn.read(buf, off, len);
221
222                if (ret > 0) {
223                    size += ret;
224                    off += ret;
225                    len -= ret;
226                } else if (ret == -1) {
227                    xzIn = null;
228                }
229            }
230        } catch (IOException e) {
231            exception = e;
232            if (size == 0)
233                throw e;
234        }
235
236        return size;
237    }
238
239    private void prepareNextStream() throws IOException {
240        DataInputStream inData = new DataInputStream(in);
241        byte[] buf = new byte[DecoderUtil.STREAM_HEADER_SIZE];
242
243        // The size of Stream Padding must be a multiple of four bytes,
244        // all bytes zero.
245        do {
246            // First try to read one byte to see if we have reached the end
247            // of the file.
248            int ret = inData.read(buf, 0, 1);
249            if (ret == -1) {
250                endReached = true;
251                return;
252            }
253
254            // Since we got one byte of input, there must be at least
255            // three more available in a valid file.
256            inData.readFully(buf, 1, 3);
257
258        } while (buf[0] == 0 && buf[1] == 0 && buf[2] == 0 && buf[3] == 0);
259
260        // Not all bytes are zero. In a valid Stream it indicates the
261        // beginning of the next Stream. Read the rest of the Stream Header
262        // and initialize the XZ decoder.
263        inData.readFully(buf, 4, DecoderUtil.STREAM_HEADER_SIZE - 4);
264
265        try {
266            xzIn = new SingleXZInputStream(in, memoryLimit, buf);
267        } catch (XZFormatException e) {
268            // Since this isn't the first .xz Stream, it is more
269            // logical to tell that the data is corrupt.
270            throw new CorruptedInputException(
271                    "Garbage after a valid XZ Stream");
272        }
273    }
274
275    /**
276     * Returns the number of uncompressed bytes that can be read
277     * without blocking. The value is returned with an assumption
278     * that the compressed input data will be valid. If the compressed
279     * data is corrupt, <code>CorruptedInputException</code> may get
280     * thrown before the number of bytes claimed to be available have
281     * been read from this input stream.
282     *
283     * @return      the number of uncompressed bytes that can be read
284     *              without blocking
285     */
286    public int available() throws IOException {
287        if (in == null)
288            throw new XZIOException("Stream closed");
289
290        if (exception != null)
291            throw exception;
292
293        return xzIn == null ? 0 : xzIn.available();
294    }
295
296    /**
297     * Closes the stream and calls <code>in.close()</code>.
298     * If the stream was already closed, this does nothing.
299     *
300     * @throws  IOException if thrown by <code>in.close()</code>
301     */
302    public void close() throws IOException {
303        if (in != null) {
304            try {
305                in.close();
306            } finally {
307                in = null;
308            }
309        }
310    }
