001package org.xbib.io.compress.xz;
002
003import org.xbib.io.compress.xz.check.Check;
004import org.xbib.io.compress.xz.common.DecoderUtil;
005import org.xbib.io.compress.xz.common.StreamFlags;
006import org.xbib.io.compress.xz.index.IndexHash;
007
008import java.io.DataInputStream;
009import java.io.IOException;
010import java.io.InputStream;
011
012/**
013 * Decompresses exactly one XZ Stream in streamed mode (no seeking).
014 * The decompression stops after the first XZ Stream has been decompressed,
015 * and the read position in the input stream is left at the first byte
016 * after the end of the XZ Stream. This can be useful when XZ data has
017 * been stored inside some other file format or protocol.
018 * <p>
019 * Unless you know what you are doing, don't use this class to decompress
020 * standalone .xz files. For that purpose, use <code>XZInputStream</code>.
021 * <p>
022 * When uncompressed size is known beforehand
023 * <p>
024 * If you are decompressing complete XZ streams and your application knows
025 * exactly how much uncompressed data there should be, it is good to try
026 * reading one more byte by calling <code>read()</code> and checking
027 * that it returns <code>-1</code>. This way the decompressor will parse the
028 * file footers and verify the integrity checks, giving the caller more
029 * confidence that the uncompressed data is valid.
030 *
031 * @see XZInputStream
032 */
033public class SingleXZInputStream extends InputStream {
034    private InputStream in;
035    private int memoryLimit;
036    private StreamFlags streamHeaderFlags;
037    private Check check;
038    private BlockInputStream blockDecoder = null;
039    private final IndexHash indexHash = new IndexHash();
040    private boolean endReached = false;
041    private IOException exception = null;
042
043    /**
044     * Creates a new XZ decompressor that decompresses exactly one
045     * XZ Stream from <code>in</code> without a memory usage limit.
046     * <p>
047     * This constructor reads and parses the XZ Stream Header (12 bytes)
048     * from <code>in</code>. The header of the first Block is not read
049     * until <code>read</code> is called.
050     *
051     * @param in input stream from which XZ-compressed
052     *           data is read
053     * @throws XZFormatException           input is not in the XZ format
054     * @throws CorruptedInputException     XZ header CRC32 doesn't match
055     * @throws UnsupportedOptionsException XZ header is valid but specifies options
056     *                                     not supported by this implementation
057     * @throws java.io.EOFException        less than 12 bytes of input was available
058     *                                     from <code>in</code>
059     * @throws java.io.IOException         may be thrown by <code>in</code>
060     */
061    public SingleXZInputStream(InputStream in) throws IOException {
062        initialize(in, -1);
063    }
064
065    /**
066     * Creates a new XZ decompressor that decompresses exactly one
067     * XZ Stream from <code>in</code> with an optional memory usage limit.
068     * <p>
069     * This is identical to <code>SingleXZInputStream(InputStream)</code>
070     * except that this takes also the <code>memoryLimit</code> argument.
071     *
072     * @param in          input stream from which XZ-compressed
073     *                    data is read
074     * @param memoryLimit memory usage limit in kibibytes (KiB)
075     *                    or <code>-1</code> to impose no
076     *                    memory usage limit
077     * @throws XZFormatException           input is not in the XZ format
078     * @throws CorruptedInputException     XZ header CRC32 doesn't match
079     * @throws UnsupportedOptionsException XZ header is valid but specifies options
080     *                                     not supported by this implementation
081     * @throws java.io.EOFException        less than 12 bytes of input was available
082     *                                     from <code>in</code>
083     * @throws java.io.IOException         may be thrown by <code>in</code>
084     */
085    public SingleXZInputStream(InputStream in, int memoryLimit)
086            throws IOException {
087        initialize(in, memoryLimit);
088    }
089
090    SingleXZInputStream(InputStream in, int memoryLimit,
091                        byte[] streamHeader) throws IOException {
092        initialize(in, memoryLimit, streamHeader);
093    }
094
095    private void initialize(InputStream in, int memoryLimit)
096            throws IOException {
097        byte[] streamHeader = new byte[DecoderUtil.STREAM_HEADER_SIZE];
098        new DataInputStream(in).readFully(streamHeader);
099        initialize(in, memoryLimit, streamHeader);
100    }
101
102    private void initialize(InputStream in, int memoryLimit,
103                            byte[] streamHeader) throws IOException {
104        this.in = in;
105        this.memoryLimit = memoryLimit;
106        streamHeaderFlags = DecoderUtil.decodeStreamHeader(streamHeader);
107        check = Check.getInstance(streamHeaderFlags.checkType);
108    }
109
110    /**
111     * Gets the ID of the integrity check used in this XZ Stream.
112     *
113     * @return the Check ID specified in the XZ Stream Header
114     */
115    public int getCheckType() {
116        return streamHeaderFlags.checkType;
117    }
118
119    /**
120     * Gets the name of the integrity check used in this XZ Stream.
121     *
122     * @return the name of the check specified in the XZ Stream Header
123     */
124    public String getCheckName() {
125        return check.getName();
126    }
127
128    /**
129     * Decompresses the next byte from this input stream.
130     * <p>
131     * Reading lots of data with <code>read()</code> from this input stream
132     * may be inefficient. Wrap it in {@link java.io.BufferedInputStream}
133     * if you need to read lots of data one byte at a time.
134     *
135     * @return the next decompressed byte, or <code>-1</code>
136     * to indicate the end of the compressed stream
137     * @throws CorruptedInputException
138     * @throws UnsupportedOptionsException
139     * @throws MemoryLimitException
140     * @throws XZIOException               if the stream has been closed
141     * @throws java.io.EOFException        compressed input is truncated or corrupt
142     * @throws java.io.IOException         may be thrown by <code>in</code>
143     */
144    public int read() throws IOException {
145        byte[] buf = new byte[1];
146        return read(buf, 0, 1) == -1 ? -1 : (buf[0] & 0xFF);
147    }
148
149    /**
150     * Decompresses into an array of bytes.
151     * <p>
152     * If <code>len</code> is zero, no bytes are read and <code>0</code>
153     * is returned. Otherwise this will try to decompress <code>len</code>
154     * bytes of uncompressed data. Less than <code>len</code> bytes may
155     * be read only in the following situations:
156     * <ul>
157     * <li>The end of the compressed data was reached successfully.</li>
158     * <li>An error is detected after at least one but less <code>len</code>
159     * bytes have already been successfully decompressed.
160     * The next call with non-zero <code>len</code> will immediately
161     * throw the pending exception.</li>
162     * <li>An exception is thrown.</li>
163     * </ul>
164     *
165     * @param buf target buffer for uncompressed data
166     * @param off start offset in <code>buf</code>
167     * @param len maximum number of uncompressed bytes to read
168     * @return number of bytes read, or <code>-1</code> to indicate
169     * the end of the compressed stream
170     * @throws CorruptedInputException
171     * @throws UnsupportedOptionsException
172     * @throws MemoryLimitException
173     * @throws XZIOException               if the stream has been closed
174     * @throws java.io.EOFException        compressed input is truncated or corrupt
175     * @throws java.io.IOException         may be thrown by <code>in</code>
176     */
177    public int read(byte[] buf, int off, int len) throws IOException {
178        if (off < 0 || len < 0 || off + len < 0 || off + len > buf.length) {
179            throw new IndexOutOfBoundsException();
180        }
181
182        if (len == 0) {
183            return 0;
184        }
185
186        if (in == null) {
187            throw new XZIOException("Stream closed");
188        }
189
190        if (exception != null) {
191            throw exception;
192        }
193
194        if (endReached) {
195            return -1;
196        }
197
198        int size = 0;
199
200        try {
201            while (len > 0) {
202                if (blockDecoder == null) {
203                    try {
204                        blockDecoder = new BlockInputStream(
205                                in, check, memoryLimit, -1, -1);
206                    } catch (IndexIndicatorException e) {
207                        indexHash.validate(in);
208                        validateStreamFooter();
209                        endReached = true;
210                        return size > 0 ? size : -1;
211                    }
212                }
213
214                int ret = blockDecoder.read(buf, off, len);
215
216                if (ret > 0) {
217                    size += ret;
218                    off += ret;
219                    len -= ret;
220                } else if (ret == -1) {
221                    indexHash.add(blockDecoder.getUnpaddedSize(),
222                            blockDecoder.getUncompressedSize());
223                    blockDecoder = null;
224                }
225            }
226        } catch (IOException e) {
227            exception = e;
228            if (size == 0) {
229                throw e;
230            }
231        }
232
233        return size;
234    }
235
236    private void validateStreamFooter() throws IOException {
237        byte[] buf = new byte[DecoderUtil.STREAM_HEADER_SIZE];
238        new DataInputStream(in).readFully(buf);
239        StreamFlags streamFooterFlags = DecoderUtil.decodeStreamFooter(buf);
240
241        if (!DecoderUtil.areStreamFlagsEqual(streamHeaderFlags,
242                streamFooterFlags)
243                || indexHash.getIndexSize() != streamFooterFlags.backwardSize) {
244            throw new CorruptedInputException(
245                    "XZ Stream Footer does not match Stream Header");
246        }
247    }
248
249    /**
250     * Returns the number of uncompressed bytes that can be read
251     * without blocking. The value is returned with an assumption
252     * that the compressed input data will be valid. If the compressed
253     * data is corrupt, <code>CorruptedInputException</code> may get
254     * thrown before the number of bytes claimed to be available have
255     * been read from this input stream.
256     *
257     * @return the number of uncompressed bytes that can be read
258     * without blocking
259     */
260    public int available() throws IOException {
261        if (in == null) {
262            throw new XZIOException("Stream closed");
263        }
264
265        if (exception != null) {
266            throw exception;
267        }
268
269        return blockDecoder == null ? 0 : blockDecoder.available();
270    }
271
272    /**
273     * Closes the stream and calls <code>in.close()</code>.
274     * If the stream was already closed, this does nothing.
275     *
276     * @throws java.io.IOException if thrown by <code>in.close()</code>
277     */
278    public void close() throws IOException {
279        if (in != null) {
280            try {
281                in.close();
282            } finally {
283                in = null;
284            }
285        }
286    }
