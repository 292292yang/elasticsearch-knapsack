002    package org.xbib.io.compress.xz;
003    
004    import org.xbib.io.compress.xz.check.Check;
005    import org.xbib.io.compress.xz.common.DecoderUtil;
006    import org.xbib.io.compress.xz.common.StreamFlags;
007    import org.xbib.io.compress.xz.index.IndexHash;
008    
009    import java.io.DataInputStream;
010    import java.io.IOException;
011    import java.io.InputStream;
012    
013    /**
014     * Decompresses exactly one XZ Stream in streamed mode (no seeking).
015     * The decompression stops after the first XZ Stream has been decompressed,
016     * and the read position in the input stream is left at the first byte
017     * after the end of the XZ Stream. This can be useful when XZ data has
018     * been stored inside some other file format or protocol.
019     * <p/>
020     * Unless you know what you are doing, don't use this class to decompress
021     * standalone .xz files. For that purpose, use <code>XZInputStream</code>.
022     * <p/>
023     * <h4>When uncompressed size is known beforehand</h4>
024     * <p/>
025     * If you are decompressing complete XZ streams and your application knows
026     * exactly how much uncompressed data there should be, it is good to try
027     * reading one more byte by calling <code>read()</code> and checking
028     * that it returns <code>-1</code>. This way the decompressor will parse the
029     * file footers and verify the integrity checks, giving the caller more
030     * confidence that the uncompressed data is valid.
031     *
032     * @see XZInputStream
033     */
034    public class SingleXZInputStream extends InputStream {
035        private InputStream in;
036        private int memoryLimit;
037        private StreamFlags streamHeaderFlags;
038        private Check check;
039        private BlockInputStream blockDecoder = null;
040        private final IndexHash indexHash = new IndexHash();
041        private boolean endReached = false;
042        private IOException exception = null;
043    
044        /**
045         * Creates a new XZ decompressor that decompresses exactly one
046         * XZ Stream from <code>in</code> without a memory usage limit.
047         * <p/>
048         * This constructor reads and parses the XZ Stream Header (12 bytes)
049         * from <code>in</code>. The header of the first Block is not read
050         * until <code>read</code> is called.
051         *
052         * @param in input stream from which XZ-compressed
053         *           data is read
054         * @throws XZFormatException           input is not in the XZ format
055         * @throws CorruptedInputException     XZ header CRC32 doesn't match
056         * @throws UnsupportedOptionsException XZ header is valid but specifies options
057         *                                     not supported by this implementation
058         * @throws java.io.EOFException        less than 12 bytes of input was available
059         *                                     from <code>in</code>
060         * @throws java.io.IOException         may be thrown by <code>in</code>
061         */
062        public SingleXZInputStream(InputStream in) throws IOException {
063            initialize(in, -1);
064        }
065    
066        /**
067         * Creates a new XZ decompressor that decompresses exactly one
068         * XZ Stream from <code>in</code> with an optional memory usage limit.
069         * <p/>
070         * This is identical to <code>SingleXZInputStream(InputStream)</code>
071         * except that this takes also the <code>memoryLimit</code> argument.
072         *
073         * @param in          input stream from which XZ-compressed
074         *                    data is read
075         * @param memoryLimit memory usage limit in kibibytes (KiB)
076         *                    or <code>-1</code> to impose no
077         *                    memory usage limit
078         * @throws XZFormatException           input is not in the XZ format
079         * @throws CorruptedInputException     XZ header CRC32 doesn't match
080         * @throws UnsupportedOptionsException XZ header is valid but specifies options
081         *                                     not supported by this implementation
082         * @throws java.io.EOFException        less than 12 bytes of input was available
083         *                                     from <code>in</code>
084         * @throws java.io.IOException         may be thrown by <code>in</code>
085         */
086        public SingleXZInputStream(InputStream in, int memoryLimit)
087                throws IOException {
088            initialize(in, memoryLimit);
089        }
090    
091        SingleXZInputStream(InputStream in, int memoryLimit,
092                            byte[] streamHeader) throws IOException {
093            initialize(in, memoryLimit, streamHeader);
094        }
095    
096        private void initialize(InputStream in, int memoryLimit)
097                throws IOException {
098            byte[] streamHeader = new byte[DecoderUtil.STREAM_HEADER_SIZE];
099            new DataInputStream(in).readFully(streamHeader);
100            initialize(in, memoryLimit, streamHeader);
101        }
102    
103        private void initialize(InputStream in, int memoryLimit,
104                                byte[] streamHeader) throws IOException {
105            this.in = in;
106            this.memoryLimit = memoryLimit;
107            streamHeaderFlags = DecoderUtil.decodeStreamHeader(streamHeader);
108            check = Check.getInstance(streamHeaderFlags.checkType);
109        }
110    
111        /**
112         * Gets the ID of the integrity check used in this XZ Stream.
113         *
114         * @return the Check ID specified in the XZ Stream Header
115         */
116        public int getCheckType() {
117            return streamHeaderFlags.checkType;
118        }
119    
120        /**
121         * Gets the name of the integrity check used in this XZ Stream.
122         *
123         * @return the name of the check specified in the XZ Stream Header
124         */
125        public String getCheckName() {
126            return check.getName();
127        }
128    
129        /**
130         * Decompresses the next byte from this input stream.
131         * <p/>
132         * Reading lots of data with <code>read()</code> from this input stream
133         * may be inefficient. Wrap it in {@link java.io.BufferedInputStream}
134         * if you need to read lots of data one byte at a time.
135         *
136         * @return the next decompressed byte, or <code>-1</code>
137         * to indicate the end of the compressed stream
138         * @throws CorruptedInputException
139         * @throws UnsupportedOptionsException
140         * @throws MemoryLimitException
141         * @throws XZIOException               if the stream has been closed
142         * @throws java.io.EOFException        compressed input is truncated or corrupt
143         * @throws java.io.IOException         may be thrown by <code>in</code>
144         */
145        public int read() throws IOException {
146            byte[] buf = new byte[1];
147            return read(buf, 0, 1) == -1 ? -1 : (buf[0] & 0xFF);
148        }
149    
150        /**
151         * Decompresses into an array of bytes.
152         * <p/>
153         * If <code>len</code> is zero, no bytes are read and <code>0</code>
154         * is returned. Otherwise this will try to decompress <code>len</code>
155         * bytes of uncompressed data. Less than <code>len</code> bytes may
156         * be read only in the following situations:
157         * <ul>
158         * <li>The end of the compressed data was reached successfully.</li>
159         * <li>An error is detected after at least one but less <code>len</code>
160         * bytes have already been successfully decompressed.
161         * The next call with non-zero <code>len</code> will immediately
162         * throw the pending exception.</li>
163         * <li>An exception is thrown.</li>
164         * </ul>
165         *
166         * @param buf target buffer for uncompressed data
167         * @param off start offset in <code>buf</code>
168         * @param len maximum number of uncompressed bytes to read
169         * @return number of bytes read, or <code>-1</code> to indicate
170         * the end of the compressed stream
171         * @throws CorruptedInputException
172         * @throws UnsupportedOptionsException
173         * @throws MemoryLimitException
174         * @throws XZIOException               if the stream has been closed
175         * @throws java.io.EOFException        compressed input is truncated or corrupt
176         * @throws java.io.IOException         may be thrown by <code>in</code>
177         */
178        public int read(byte[] buf, int off, int len) throws IOException {
179            if (off < 0 || len < 0 || off + len < 0 || off + len > buf.length) {
180                throw new IndexOutOfBoundsException();
181            }
182    
183            if (len == 0) {
184                return 0;
185            }
186    
187            if (in == null) {
188                throw new XZIOException("Stream closed");
189            }
190    
191            if (exception != null) {
192                throw exception;
193            }
194    
195            if (endReached) {
196                return -1;
197            }
198    
199            int size = 0;
200    
201            try {
202                while (len > 0) {
203                    if (blockDecoder == null) {
204                        try {
205                            blockDecoder = new BlockInputStream(
206                                    in, check, memoryLimit, -1, -1);
207                        } catch (IndexIndicatorException e) {
208                            indexHash.validate(in);
209                            validateStreamFooter();
210                            endReached = true;
211                            return size > 0 ? size : -1;
212                        }
213                    }
214    
215                    int ret = blockDecoder.read(buf, off, len);
216    
217                    if (ret > 0) {
218                        size += ret;
219                        off += ret;
220                        len -= ret;
221                    } else if (ret == -1) {
222                        indexHash.add(blockDecoder.getUnpaddedSize(),
223                                blockDecoder.getUncompressedSize());
224                        blockDecoder = null;
225                    }
226                }
227            } catch (IOException e) {
228                exception = e;
229                if (size == 0) {
230                    throw e;
231                }
232            }
233    
234            return size;
235        }
236    
237        private void validateStreamFooter() throws IOException {
238            byte[] buf = new byte[DecoderUtil.STREAM_HEADER_SIZE];
239            new DataInputStream(in).readFully(buf);
240            StreamFlags streamFooterFlags = DecoderUtil.decodeStreamFooter(buf);
241    
242            if (!DecoderUtil.areStreamFlagsEqual(streamHeaderFlags,
243                    streamFooterFlags)
244                    || indexHash.getIndexSize() != streamFooterFlags.backwardSize) {
245                throw new CorruptedInputException(
246                        "XZ Stream Footer does not match Stream Header");
247            }
248        }
249    
250        /**
251         * Returns the number of uncompressed bytes that can be read
252         * without blocking. The value is returned with an assumption
253         * that the compressed input data will be valid. If the compressed
254         * data is corrupt, <code>CorruptedInputException</code> may get
255         * thrown before the number of bytes claimed to be available have
256         * been read from this input stream.
257         *
258         * @return the number of uncompressed bytes that can be read
259         * without blocking
260         */
261        public int available() throws IOException {
262            if (in == null) {
263                throw new XZIOException("Stream closed");
264            }
265    
266            if (exception != null) {
267                throw exception;
268            }
269    
270            return blockDecoder == null ? 0 : blockDecoder.available();
271        }
272    
273        /**
274         * Closes the stream and calls <code>in.close()</code>.
275         * If the stream was already closed, this does nothing.
276         *
277         * @throws java.io.IOException if thrown by <code>in.close()</code>
278         */
279        public void close() throws IOException {
280            if (in != null) {
281                try {
282                    in.close();
283                } finally {
284                    in = null;
285                }
286            }
287        }
