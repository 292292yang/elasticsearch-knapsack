010package org.xbib.io.compress.xz.index;
011
012import java.io.IOException;
013import java.io.EOFException;
014import java.util.zip.CheckedInputStream;
015import org.xbib.io.compress.xz.common.DecoderUtil;
016import org.xbib.io.compress.xz.common.StreamFlags;
017import org.xbib.io.compress.xz.SeekableInputStream;
018import org.xbib.io.compress.xz.CorruptedInputException;
019import org.xbib.io.compress.xz.MemoryLimitException;
020import org.xbib.io.compress.xz.UnsupportedOptionsException;
021
022public class IndexDecoder extends IndexBase {
023    private final BlockInfo info = new BlockInfo();
024    private final long streamPadding;
025    private final int memoryUsage;
026    private final long[] unpadded;
027    private final long[] uncompressed;
028    private long largestBlockSize = 0;
029
030    /**
031     * Current position in the arrays. This is initialized to <code>-1</code>
032     * because then it is possible to use <code>hasNext()</code> and
033     * <code>getNext()</code> to get BlockInfo of the first Block.
034     */
035    private int pos = -1;
036
037    public IndexDecoder(SeekableInputStream in, StreamFlags streamFooterFlags,
038                        long streamPadding, int memoryLimit)
039            throws IOException {
040        super(new CorruptedInputException("XZ Index is corrupt"));
041        info.streamFlags = streamFooterFlags;
042        this.streamPadding = streamPadding;
043
044        // If endPos is exceeded before the CRC32 field has been decoded,
045        // the Index is corrupt.
046        long endPos = in.position() + streamFooterFlags.backwardSize - 4;
047
048        java.util.zip.CRC32 crc32 = new java.util.zip.CRC32();
049        CheckedInputStream inChecked = new CheckedInputStream(in, crc32);
050
051        // Index Indicator
052        if (inChecked.read() != 0x00)
053            throw new CorruptedInputException("XZ Index is corrupt");
054
055        try {
056            // Number of Records
057            long count = DecoderUtil.decodeVLI(inChecked);
058
059            // Catch Record counts that obviously too high to be valid.
060            // This test isn't exact because it ignores Index Indicator,
061            // Number of Records, and CRC32 fields, but this is good enough
062            // to catch the most obvious problems.
063            if (count >= streamFooterFlags.backwardSize / 2)
064                throw new CorruptedInputException("XZ Index is corrupt");
065
066            // If the Record count doesn't fit into an int, we cannot
067            // allocate the arrays to hold the Records.
068            if (count > Integer.MAX_VALUE)
069                throw new UnsupportedOptionsException("XZ Index has over "
070                        + Integer.MAX_VALUE + " Records");
071
072            // Calculate approximate memory requirements and check the
073            // memory usage limit.
074            memoryUsage = 1 + (int)((16L * count + 1023) / 1024);
075            if (memoryLimit >= 0 && memoryUsage > memoryLimit)
076                throw new MemoryLimitException(memoryUsage, memoryLimit);
077
078            // Allocate the arrays for the Records.
079            unpadded = new long[(int)count];
080            uncompressed = new long[(int)count];
081            int record = 0;
082
083            // Decode the Records.
084            for (int i = (int)count; i > 0; --i) {
085                // Get the next Record.
086                long unpaddedSize = DecoderUtil.decodeVLI(inChecked);
087                long uncompressedSize = DecoderUtil.decodeVLI(inChecked);
088
089                // Check that the input position stays sane. Since this is
090                // checked only once per loop iteration instead of for
091                // every input byte read, it's still possible that
092                // EOFException gets thrown with corrupt input.
093                if (in.position() > endPos)
094                    throw new CorruptedInputException("XZ Index is corrupt");
095
096                // Add the new Record.
097                unpadded[record] = blocksSum + unpaddedSize;
098                uncompressed[record] = uncompressedSum + uncompressedSize;
099                ++record;
100                super.add(unpaddedSize, uncompressedSize);
101                assert record == recordCount;
102
103                // Remember the uncompressed size of the largest Block.
104                if (largestBlockSize < uncompressedSize)
105                    largestBlockSize = uncompressedSize;
106            }
107        } catch (EOFException e) {
108            // EOFException is caught just in case a corrupt input causes
109            // DecoderUtil.decodeVLI to read too much at once.
110            throw new CorruptedInputException("XZ Index is corrupt");
111        }
112
113        // Validate that the size of the Index field matches
114        // Backward Size.
115        int indexPaddingSize = getIndexPaddingSize();
116        if (in.position() + indexPaddingSize != endPos)
117            throw new CorruptedInputException("XZ Index is corrupt");
118
119        // Index Padding
120        while (indexPaddingSize-- > 0)
121            if (inChecked.read() != 0x00)
122                throw new CorruptedInputException("XZ Index is corrupt");
123
124        // CRC32
125        long value = crc32.getValue();
126        for (int i = 0; i < 4; ++i)
127            if (((value >>> (i * 8)) & 0xFF) != in.read())
128                throw new CorruptedInputException("XZ Index is corrupt");
129    }
130
131    public BlockInfo locate(long target) {
132        assert target < uncompressedSum;
133
134        int left = 0;
135        int right = unpadded.length - 1;
136
137        while (left < right) {
138            int i = left + (right - left) / 2;
139
140            if (uncompressed[i] <= target)
141                left = i + 1;
142            else
143                right = i;
144        }
145
146        pos = left;
147        return getInfo();
148    }
149
150    public int getMemoryUsage() {
151        return memoryUsage;
152    }
153
154    public long getStreamAndPaddingSize() {
155        return getStreamSize() + streamPadding;
156    }
157
158    public long getUncompressedSize() {
159        return uncompressedSum;
160    }
161
162    public long getLargestBlockSize() {
163        return largestBlockSize;
164    }
165
166    public boolean hasNext() {
167        return pos + 1 < recordCount;
168    }
169
170    public BlockInfo getNext() {
171        ++pos;
172        return getInfo();
173    }
174
175    private BlockInfo getInfo() {
176        if (pos == 0) {
177            info.compressedOffset = 0;
178            info.uncompressedOffset = 0;
179        } else {
180            info.compressedOffset = (unpadded[pos - 1] + 3) & ~3;
181            info.uncompressedOffset = uncompressed[pos - 1];
182        }
183
184        info.unpaddedSize = unpadded[pos] - info.compressedOffset;
185        info.uncompressedSize = uncompressed[pos] - info.uncompressedOffset;
186
187        info.compressedOffset += DecoderUtil.STREAM_HEADER_SIZE;
188        return info;
189    }
190}
