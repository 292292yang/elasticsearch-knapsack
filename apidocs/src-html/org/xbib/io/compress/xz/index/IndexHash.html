002    package org.xbib.io.compress.xz.index;
003    
004    import org.xbib.io.compress.xz.CorruptedInputException;
005    import org.xbib.io.compress.xz.XZIOException;
006    import org.xbib.io.compress.xz.common.DecoderUtil;
007    
008    import java.io.DataInputStream;
009    import java.io.IOException;
010    import java.io.InputStream;
011    import java.nio.ByteBuffer;
012    import java.util.Arrays;
013    import java.util.zip.CheckedInputStream;
014    
015    public class IndexHash extends IndexBase {
016        private org.xbib.io.compress.xz.check.Check hash;
017    
018        public IndexHash() {
019            super(new CorruptedInputException());
020    
021            try {
022                hash = new org.xbib.io.compress.xz.check.SHA256();
023            } catch (java.security.NoSuchAlgorithmException e) {
024                hash = new org.xbib.io.compress.xz.check.CRC32();
025            }
026        }
027    
028        public void add(long unpaddedSize, long uncompressedSize)
029                throws XZIOException {
030            super.add(unpaddedSize, uncompressedSize);
031    
032            ByteBuffer buf = ByteBuffer.allocate(2 * 8);
033            buf.putLong(unpaddedSize);
034            buf.putLong(uncompressedSize);
035            hash.update(buf.array());
036        }
037    
038        public void validate(InputStream in) throws IOException {
039            // Index Indicator (0x00) has already been read by BlockInputStream
040            // so add 0x00 to the CRC32 here.
041            java.util.zip.CRC32 crc32 = new java.util.zip.CRC32();
042            crc32.update('\0');
043            CheckedInputStream inChecked = new CheckedInputStream(in, crc32);
044    
045            // Get and validate the Number of Records field.
046            long storedRecordCount = DecoderUtil.decodeVLI(inChecked);
047            if (storedRecordCount != recordCount) {
048                throw new CorruptedInputException("XZ Index is corrupt");
049            }
050    
051            // Decode and hash the Index field and compare it to
052            // the hash value calculated from the decoded Blocks.
053            IndexHash stored = new IndexHash();
054            for (long i = 0; i < recordCount; ++i) {
055                long unpaddedSize = DecoderUtil.decodeVLI(inChecked);
056                long uncompressedSize = DecoderUtil.decodeVLI(inChecked);
057    
058                try {
059                    stored.add(unpaddedSize, uncompressedSize);
060                } catch (XZIOException e) {
061                    throw new CorruptedInputException("XZ Index is corrupt");
062                }
063    
064                if (stored.blocksSum > blocksSum
065                        || stored.uncompressedSum > uncompressedSum
066                        || stored.indexListSize > indexListSize) {
067                    throw new CorruptedInputException("XZ Index is corrupt");
068                }
069            }
070    
071            if (stored.blocksSum != blocksSum
072                    || stored.uncompressedSum != uncompressedSum
073                    || stored.indexListSize != indexListSize
074                    || !Arrays.equals(stored.hash.finish(), hash.finish())) {
075                throw new CorruptedInputException("XZ Index is corrupt");
076            }
077    
078            // Index Padding
079            DataInputStream inData = new DataInputStream(inChecked);
080            for (int i = getIndexPaddingSize(); i > 0; --i) {
081                if (inData.readUnsignedByte() != 0x00) {
082                    throw new CorruptedInputException("XZ Index is corrupt");
083                }
084            }
085    
086            // CRC32
087            long value = crc32.getValue();
088            for (int i = 0; i < 4; ++i) {
089                if (((value >>> (i * 8)) & 0xFF) != inData.readUnsignedByte()) {
090                    throw new CorruptedInputException("XZ Index is corrupt");
091                }
092            }
093        }
094    }
