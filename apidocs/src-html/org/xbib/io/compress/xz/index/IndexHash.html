010package org.xbib.io.compress.xz.index;
011
012import java.io.InputStream;
013import java.io.DataInputStream;
014import java.io.IOException;
015import java.nio.ByteBuffer;
016import java.util.Arrays;
017import java.util.zip.CheckedInputStream;
018import org.xbib.io.compress.xz.common.DecoderUtil;
019import org.xbib.io.compress.xz.XZIOException;
020import org.xbib.io.compress.xz.CorruptedInputException;
021
022public class IndexHash extends IndexBase {
023    private org.xbib.io.compress.xz.check.Check hash;
024
025    public IndexHash() {
026        super(new CorruptedInputException());
027
028        try {
029            hash = new org.xbib.io.compress.xz.check.SHA256();
030        } catch (java.security.NoSuchAlgorithmException e) {
031            hash = new org.xbib.io.compress.xz.check.CRC32();
032        }
033    }
034
035    public void add(long unpaddedSize, long uncompressedSize)
036            throws XZIOException {
037        super.add(unpaddedSize, uncompressedSize);
038
039        ByteBuffer buf = ByteBuffer.allocate(2 * 8);
040        buf.putLong(unpaddedSize);
041        buf.putLong(uncompressedSize);
042        hash.update(buf.array());
043    }
044
045    public void validate(InputStream in) throws IOException {
046        // Index Indicator (0x00) has already been read by BlockInputStream
047        // so add 0x00 to the CRC32 here.
048        java.util.zip.CRC32 crc32 = new java.util.zip.CRC32();
049        crc32.update('\0');
050        CheckedInputStream inChecked = new CheckedInputStream(in, crc32);
051
052        // Get and validate the Number of Records field.
053        long storedRecordCount = DecoderUtil.decodeVLI(inChecked);
054        if (storedRecordCount != recordCount)
055            throw new CorruptedInputException("XZ Index is corrupt");
056
057        // Decode and hash the Index field and compare it to
058        // the hash value calculated from the decoded Blocks.
059        IndexHash stored = new IndexHash();
060        for (long i = 0; i < recordCount; ++i) {
061            long unpaddedSize = DecoderUtil.decodeVLI(inChecked);
062            long uncompressedSize = DecoderUtil.decodeVLI(inChecked);
063
064            try {
065                stored.add(unpaddedSize, uncompressedSize);
066            } catch (XZIOException e) {
067                throw new CorruptedInputException("XZ Index is corrupt");
068            }
069
070            if (stored.blocksSum > blocksSum
071                    || stored.uncompressedSum > uncompressedSum
072                    || stored.indexListSize > indexListSize)
073                throw new CorruptedInputException("XZ Index is corrupt");
074        }
075
076        if (stored.blocksSum != blocksSum
077                || stored.uncompressedSum != uncompressedSum
078                || stored.indexListSize != indexListSize
079                || !Arrays.equals(stored.hash.finish(), hash.finish()))
080            throw new CorruptedInputException("XZ Index is corrupt");
081
082        // Index Padding
083        DataInputStream inData = new DataInputStream(inChecked);
084        for (int i = getIndexPaddingSize(); i > 0; --i)
085            if (inData.readUnsignedByte() != 0x00)
086                throw new CorruptedInputException("XZ Index is corrupt");
087
088        // CRC32
089        long value = crc32.getValue();
090        for (int i = 0; i < 4; ++i)
091            if (((value >>> (i * 8)) & 0xFF) != inData.readUnsignedByte())
092                throw new CorruptedInputException("XZ Index is corrupt");
093    }
094}
