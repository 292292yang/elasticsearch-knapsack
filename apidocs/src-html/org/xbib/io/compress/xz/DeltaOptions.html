010package org.xbib.io.compress.xz;
011
012import java.io.InputStream;
013
014/**
015 * Delta filter options. The Delta filter can be used only as a non-last
016 * filter in the chain, for example Delta + LZMA2.
017 * <p>
018 * Currently only simple byte-wise delta is supported. The only option
019 * is the delta distance, which you should set to match your data.
020 * It's not possible to provide a generic default value for it.
021 * <p>
022 * For example, with distance = 2 and eight-byte input
023 * A1 B1 A2 B3 A3 B5 A4 B7, the output will be A1 B1 01 02 01 02 01 02.
024 * <p>
025 * The Delta filter can be good with uncompressed bitmap images. It can
026 * also help with PCM audio, although special-purpose compressors like
027 * FLAC will give much smaller result at much better compression speed.
028 */
029public class DeltaOptions extends FilterOptions {
030    /**
031     * Smallest supported delta calculation distance.
032     */
033    public static final int DISTANCE_MIN = 1;
034
035    /**
036     * Largest supported delta calculation distance.
037     */
038    public static final int DISTANCE_MAX = 256;
039
040    private int distance = DISTANCE_MIN;
041
042    /**
043     * Creates new Delta options and sets the delta distance to 1 byte.
044     */
045    public DeltaOptions() {}
046
047    /**
048     * Creates new Delta options and sets the distance to the given value.
049     */
050    public DeltaOptions(int distance) throws UnsupportedOptionsException {
051        setDistance(distance);
052    }
053
054    /**
055     * Sets the delta distance in bytes. The new distance must be in
056     * the range [DISTANCE_MIN, DISTANCE_MAX].
057     */
058    public void setDistance(int distance) throws UnsupportedOptionsException {
059        if (distance < DISTANCE_MIN || distance > DISTANCE_MAX)
060            throw new UnsupportedOptionsException(
061                    "Delta distance must be in the range [" + DISTANCE_MIN
062                    + ", " + DISTANCE_MAX + "]: " + distance);
063
064        this.distance = distance;
065    }
066
067    /**
068     * Gets the delta distance.
069     */
070    public int getDistance() {
071        return distance;
072    }
073
074    public int getEncoderMemoryUsage() {
075        return DeltaOutputStream.getMemoryUsage();
076    }
077
078    public FinishableOutputStream getOutputStream(FinishableOutputStream out) {
079        return new DeltaOutputStream(out, this);
080    }
081
082    public int getDecoderMemoryUsage() {
083        return 1;
084    }
085
086    public InputStream getInputStream(InputStream in) {
087        return new DeltaInputStream(in, distance);
088    }
089
090    FilterEncoder getFilterEncoder() {
091        return new DeltaEncoder(this);
092    }
093
094    public Object clone() {
095        try {
096            return super.clone();
097        } catch (CloneNotSupportedException e) {
098            assert false;
099            throw new RuntimeException();
100        }
101    }
102}
