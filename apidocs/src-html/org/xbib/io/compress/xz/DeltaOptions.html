002    package org.xbib.io.compress.xz;
003    
004    import java.io.InputStream;
005    
006    /**
007     * Delta filter options. The Delta filter can be used only as a non-last
008     * filter in the chain, for example Delta + LZMA2.
009     * <p/>
010     * Currently only simple byte-wise delta is supported. The only option
011     * is the delta distance, which you should set to match your data.
012     * It's not possible to provide a generic default value for it.
013     * <p/>
014     * For example, with distance = 2 and eight-byte input
015     * A1 B1 A2 B3 A3 B5 A4 B7, the output will be A1 B1 01 02 01 02 01 02.
016     * <p/>
017     * The Delta filter can be good with uncompressed bitmap images. It can
018     * also help with PCM audio, although special-purpose compressors like
019     * FLAC will give much smaller result at much better compression speed.
020     */
021    public class DeltaOptions extends FilterOptions {
022        /**
023         * Smallest supported delta calculation distance.
024         */
025        public static final int DISTANCE_MIN = 1;
026    
027        /**
028         * Largest supported delta calculation distance.
029         */
030        public static final int DISTANCE_MAX = 256;
031    
032        private int distance = DISTANCE_MIN;
033    
034        /**
035         * Creates new Delta options and sets the delta distance to 1 byte.
036         */
037        public DeltaOptions() {
038        }
039    
040        /**
041         * Creates new Delta options and sets the distance to the given value.
042         */
043        public DeltaOptions(int distance) throws UnsupportedOptionsException {
044            setDistance(distance);
045        }
046    
047        /**
048         * Sets the delta distance in bytes. The new distance must be in
049         * the range [DISTANCE_MIN, DISTANCE_MAX].
050         */
051        public void setDistance(int distance) throws UnsupportedOptionsException {
052            if (distance < DISTANCE_MIN || distance > DISTANCE_MAX) {
053                throw new UnsupportedOptionsException(
054                        "Delta distance must be in the range [" + DISTANCE_MIN
055                                + ", " + DISTANCE_MAX + "]: " + distance);
056            }
057    
058            this.distance = distance;
059        }
060    
061        /**
062         * Gets the delta distance.
063         */
064        public int getDistance() {
065            return distance;
066        }
067    
068        public int getEncoderMemoryUsage() {
069            return DeltaOutputStream.getMemoryUsage();
070        }
071    
072        public FinishableOutputStream getOutputStream(FinishableOutputStream out) {
073            return new DeltaOutputStream(out, this);
074        }
075    
076        public int getDecoderMemoryUsage() {
077            return 1;
078        }
079    
080        public InputStream getInputStream(InputStream in) {
081            return new DeltaInputStream(in, distance);
082        }
083    
084        FilterEncoder getFilterEncoder() {
085            return new DeltaEncoder(this);
086        }
087    
088        public Object clone() {
089            try {
090                return super.clone();
091            } catch (CloneNotSupportedException e) {
092                assert false;
093                throw new RuntimeException();
094            }
095        }
096    }
