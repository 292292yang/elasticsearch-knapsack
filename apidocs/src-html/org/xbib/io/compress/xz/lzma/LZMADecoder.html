011package org.xbib.io.compress.xz.lzma;
012
013import java.io.IOException;
014import org.xbib.io.compress.xz.lz.LZDecoder;
015import org.xbib.io.compress.xz.rangecoder.RangeDecoder;
016import org.xbib.io.compress.xz.CorruptedInputException;
017
018public final class LZMADecoder extends LZMACoder {
019    private final LZDecoder lz;
020    private final RangeDecoder rc;
021    private final LiteralDecoder literalDecoder;
022    private final LengthDecoder matchLenDecoder = new LengthDecoder();
023    private final LengthDecoder repLenDecoder = new LengthDecoder();
024
025    public LZMADecoder(LZDecoder lz, RangeDecoder rc, int lc, int lp, int pb) {
026        super(pb);
027        this.lz = lz;
028        this.rc = rc;
029        this.literalDecoder = new LiteralDecoder(lc, lp);
030        reset();
031    }
032
033    public void reset() {
034        super.reset();
035        literalDecoder.reset();
036        matchLenDecoder.reset();
037        repLenDecoder.reset();
038    }
039
040    public void decode() throws IOException {
041        lz.repeatPending();
042
043        while (lz.hasSpace()) {
044            int posState = lz.getPos() & posMask;
045
046            if (rc.decodeBit(isMatch[state.get()], posState) == 0) {
047                literalDecoder.decode();
048            } else {
049                int len = rc.decodeBit(isRep, state.get()) == 0
050                          ? decodeMatch(posState)
051                          : decodeRepMatch(posState);
052                lz.repeat(reps[0], len);
053            }
054        }
055
056        rc.normalize();
057
058        if (!rc.isInBufferOK())
059            throw new CorruptedInputException();
060    }
061
062    private int decodeMatch(int posState) throws IOException {
063        state.updateMatch();
064
065        reps[3] = reps[2];
066        reps[2] = reps[1];
067        reps[1] = reps[0];
068
069        int len = matchLenDecoder.decode(posState);
070        int distSlot = rc.decodeBitTree(distSlots[getDistState(len)]);
071
072        if (distSlot < DIST_MODEL_START) {
073            reps[0] = distSlot;
074        } else {
075            int limit = (distSlot >> 1) - 1;
076            reps[0] = (2 | (distSlot & 1)) << limit;
077
078            if (distSlot < DIST_MODEL_END) {
079                reps[0] |= rc.decodeReverseBitTree(
080                        distSpecial[distSlot - DIST_MODEL_START]);
081            } else {
082                reps[0] |= rc.decodeDirectBits(limit - ALIGN_BITS)
083                           << ALIGN_BITS;
084                reps[0] |= rc.decodeReverseBitTree(distAlign);
085            }
086        }
087
088        return len;
089    }
090
091    private int decodeRepMatch(int posState) throws IOException {
092        if (rc.decodeBit(isRep0, state.get()) == 0) {
093            if (rc.decodeBit(isRep0Long[state.get()], posState) == 0) {
094                state.updateShortRep();
095                return 1;
096            }
097        } else {
098            int tmp;
099
100            if (rc.decodeBit(isRep1, state.get()) == 0) {
101                tmp = reps[1];
102            } else {
103                if (rc.decodeBit(isRep2, state.get()) == 0) {
104                    tmp = reps[2];
105                } else {
106                    tmp = reps[3];
107                    reps[3] = reps[2];
108                }
109
110                reps[2] = reps[1];
111            }
112
113            reps[1] = reps[0];
114            reps[0] = tmp;
115        }
116
117        state.updateLongRep();
118
119        return repLenDecoder.decode(posState);
120    }
121
122
123    private class LiteralDecoder extends LiteralCoder {
124        LiteralSubdecoder[] subdecoders;
125
126        LiteralDecoder(int lc, int lp) {
127            super(lc, lp);
128
129            subdecoders = new LiteralSubdecoder[1 << (lc + lp)];
130            for (int i = 0; i < subdecoders.length; ++i)
131                subdecoders[i] = new LiteralSubdecoder();
132        }
133
134        void reset() {
135            for (int i = 0; i < subdecoders.length; ++i)
136                subdecoders[i].reset();
137        }
138
139        void decode() throws IOException {
140            int i = getSubcoderIndex(lz.getByte(0), lz.getPos());
141            subdecoders[i].decode();
142        }
143
144
145        private class LiteralSubdecoder extends LiteralSubcoder {
146            void decode() throws IOException {
147                int symbol = 1;
148
149                if (state.isLiteral()) {
150                    do {
151                        symbol = (symbol << 1) | rc.decodeBit(probs, symbol);
152                    } while (symbol < 0x100);
153
154                } else {
155                    int matchByte = lz.getByte(reps[0]);
156                    int offset = 0x100;
157                    int matchBit;
158                    int bit;
159
160                    do {
161                        matchByte <<= 1;
162                        matchBit = matchByte & offset;
163                        bit = rc.decodeBit(probs, offset + matchBit + symbol);
164                        symbol = (symbol << 1) | bit;
165                        offset &= (0 - bit) ^ ~matchBit;
166                    } while (symbol < 0x100);
167                }
168
169                lz.putByte((byte)symbol);
170                state.updateLiteral();
171            }
172        }
173    }
174
175
176    private class LengthDecoder extends LengthCoder {
177        int decode(int posState) throws IOException {
178            if (rc.decodeBit(choice, 0) == 0)
179                return rc.decodeBitTree(low[posState]) + MATCH_LEN_MIN;
180
181            if (rc.decodeBit(choice, 1) == 0)
182                return rc.decodeBitTree(mid[posState])
183                       + MATCH_LEN_MIN + LOW_SYMBOLS;
184
185            return rc.decodeBitTree(high)
186                   + MATCH_LEN_MIN + LOW_SYMBOLS + MID_SYMBOLS;
187        }
188    }
189}
