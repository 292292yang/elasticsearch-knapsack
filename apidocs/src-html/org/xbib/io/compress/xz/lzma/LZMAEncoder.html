009    public static final int MODE_NORMAL = 2;

011    /**
012     * LZMA2 chunk is considered full when its uncompressed size exceeds
013     * <code>LZMA2_UNCOMPRESSED_LIMIT</code>.
014     * <p>
015     * A compressed LZMA2 chunk can hold 2 MiB of uncompressed data.
016     * A single LZMA symbol may indicate up to MATCH_LEN_MAX bytes
017     * of data, so the LZMA2 chunk is considered full when there is
018     * less space than MATCH_LEN_MAX bytes.
019     */
020    private static final int LZMA2_UNCOMPRESSED_LIMIT
021            = (2 << 20) - MATCH_LEN_MAX;
022
023    /**
024     * LZMA2 chunk is considered full when its compressed size exceeds
025     * <code>LZMA2_COMPRESSED_LIMIT</code>.
026     * <p>
027     * The maximum compressed size of a LZMA2 chunk is 64 KiB.
028     * A single LZMA symbol might use 20 bytes of space even though
029     * it usually takes just one byte or so. Two more bytes are needed
030     * for LZMA2 uncompressed chunks (see LZMA2OutputStream.writeChunk).
031     * Leave a little safety margin and use 26 bytes.
032     */
033    private static final int LZMA2_COMPRESSED_LIMIT = (64 << 10) - 26;
034
035    private static final int DIST_PRICE_UPDATE_INTERVAL = FULL_DISTANCES;
036    private static final int ALIGN_PRICE_UPDATE_INTERVAL = ALIGN_SIZE;
037
038    private final RangeEncoder rc;
039    final LZEncoder lz;
040    final LiteralEncoder literalEncoder;
041    final LengthEncoder matchLenEncoder;
042    final LengthEncoder repLenEncoder;
043    final int niceLen;

046    private int alignPriceCount = 0;

050    private final int[][] fullDistPrices
051            = new int[DIST_STATES][FULL_DISTANCES];

058    public static int getMemoryUsage(int mode, int dictSize,
059                                     int extraSizeBefore, int mf) {
060        int m = 80;
061
062        switch (mode) {
063            case MODE_FAST:
064                m += LZMAEncoderFast.getMemoryUsage(
065                        dictSize, extraSizeBefore, mf);
066                break;
067
068            case MODE_NORMAL:
069                m += LZMAEncoderNormal.getMemoryUsage(
070                        dictSize, extraSizeBefore, mf);
071                break;
072
073            default:
074                throw new IllegalArgumentException();
075        }
076
077        return m;
078    }

099    /**
100     * Gets an integer [0, 63] matching the highest two bits of an integer.
101     * This is like bit scan reverse (BSR) on x86 except that this also
102     * cares about the second highest bit.
103     */
104    public static int getDistSlot(int dist) {
105        if (dist <= DIST_MODEL_START) {
106            return dist;
107        }
108
109        int n = dist;
110        int i = 31;
111
112        if ((n & 0xFFFF0000) == 0) {
113            n <<= 16;
114            i = 15;
115        }
116
117        if ((n & 0xFF000000) == 0) {
118            n <<= 8;
119            i -= 8;
120        }
121
122        if ((n & 0xF0000000) == 0) {
123            n <<= 4;
124            i -= 4;
125        }
126
127        if ((n & 0xC0000000) == 0) {
128            n <<= 2;
129            i -= 2;
130        }
131
132        if ((n & 0x80000000) == 0) {
133            --i;
134        }
135
136        return (i << 1) + ((dist >>> (i - 1)) & 1);
137    }

139    /**
140     * Gets the next LZMA symbol.
141     * <p>
142     * There are three types of symbols: literal (a single byte),
143     * repeated match, and normal match. The symbol is indicated
144     * by the return value and by the variable <code>back</code>.
145     * <p>
146     * Literal: <code>back == -1</code> and return value is <code>1</code>.
147     * The literal itself needs to be read from <code>lz</code> separately.
148     * <p>
149     * Repeated match: <code>back</code> is in the range [0, 3] and
150     * the return value is the length of the repeated match.
151     * <p>
152     * Normal match: <code>back - REPS<code> (<code>back - 4</code>)
153     * is the distance of the match and the return value is the length
154     * of the match.
155     */
156    abstract int getNextSymbol();
157
158    LZMAEncoder(RangeEncoder rc, LZEncoder lz,
159                int lc, int lp, int pb, int dictSize, int niceLen) {
160        super(pb);
161        this.rc = rc;
162        this.lz = lz;
163        this.niceLen = niceLen;
164
165        literalEncoder = new LiteralEncoder(lc, lp);
166        matchLenEncoder = new LengthEncoder(pb, niceLen);
167        repLenEncoder = new LengthEncoder(pb, niceLen);
168
169        distSlotPricesSize = getDistSlot(dictSize - 1) + 1;
170        distSlotPrices = new int[DIST_STATES][distSlotPricesSize];
171
172        reset();
173    }

184        distPriceCount = 0;

199    /**
200     * Compresses for LZMA2.
201     *
202     * @return true if the LZMA2 chunk became full, false otherwise
203     */
204    public boolean encodeForLZMA2() {
205        if (!lz.isStarted() && !encodeInit()) {
206            return false;
207        }
208
209        while (uncompressedSize <= LZMA2_UNCOMPRESSED_LIMIT
210                && rc.getPendingSize() <= LZMA2_COMPRESSED_LIMIT) {
211            if (!encodeSymbol()) {
212                return false;
213            }
214        }
215
216        return true;
217    }
218
219    private boolean encodeInit() {
220        assert readAhead == -1;
221        if (!lz.hasEnoughData(0)) {
222            return false;
223        }
224
225        // The first symbol must be a literal unless using
226        // a preset dictionary. This code isn't run if using
227        // a preset dictionary.
228        skip(1);
229        rc.encodeBit(isMatch[state.get()], 0, 0);
230        literalEncoder.encodeInit();
231
232        --readAhead;
233        assert readAhead == -1;
234
235        ++uncompressedSize;
236        assert uncompressedSize == 1;
237
238        return true;
239    }
240
241    private boolean encodeSymbol() {
242        if (!lz.hasEnoughData(readAhead + 1)) {
243            return false;
244        }
245
246        int len = getNextSymbol();
247
248        assert readAhead >= 0;
249        int posState = (lz.getPos() - readAhead) & posMask;
250
251        if (back == -1) {
252            // Literal i.e. eight-bit byte
253            assert len == 1;
254            rc.encodeBit(isMatch[state.get()], posState, 0);
255            literalEncoder.encode();
256        } else {
257            // Some type of match
258            rc.encodeBit(isMatch[state.get()], posState, 1);
259            if (back < REPS) {
260                // Repeated match i.e. the same distance
261                // has been used earlier.
262                assert lz.getMatchLen(-readAhead, reps[back], len) == len;
263                rc.encodeBit(isRep, state.get(), 1);
264                encodeRepMatch(back, len, posState);
265            } else {
266                // Normal match
267                assert lz.getMatchLen(-readAhead, back - REPS, len) == len;
268                rc.encodeBit(isRep, state.get(), 0);
269                encodeMatch(back - REPS, len, posState);
270            }
271        }
272
273        readAhead -= len;
274        uncompressedSize += len;
275
276        return true;
277    }
278
279    private void encodeMatch(int dist, int len, int posState) {
280        state.updateMatch();
281        matchLenEncoder.encode(len, posState);
282
283        int distSlot = getDistSlot(dist);
284        rc.encodeBitTree(distSlots[getDistState(len)], distSlot);
285
286        if (distSlot >= DIST_MODEL_START) {
287            int footerBits = (distSlot >>> 1) - 1;
288            int base = (2 | (distSlot & 1)) << footerBits;
289            int distReduced = dist - base;
290
291            if (distSlot < DIST_MODEL_END) {
292                rc.encodeReverseBitTree(
293                        distSpecial[distSlot - DIST_MODEL_START],
294                        distReduced);
295            } else {
296                rc.encodeDirectBits(distReduced >>> ALIGN_BITS,
297                        footerBits - ALIGN_BITS);
298                rc.encodeReverseBitTree(distAlign, distReduced & ALIGN_MASK);
299                --alignPriceCount;
300            }
301        }
302
303        reps[3] = reps[2];
304        reps[2] = reps[1];
305        reps[1] = reps[0];
306        reps[0] = dist;
307
308        --distPriceCount;
309    }

311    private void encodeRepMatch(int rep, int len, int posState) {
312        if (rep == 0) {
313            rc.encodeBit(isRep0, state.get(), 0);
314            rc.encodeBit(isRep0Long[state.get()], posState, len == 1 ? 0 : 1);
315        } else {
316            int dist = reps[rep];
317            rc.encodeBit(isRep0, state.get(), 1);
318
319            if (rep == 1) {
320                rc.encodeBit(isRep1, state.get(), 0);
321            } else {
322                rc.encodeBit(isRep1, state.get(), 1);
323                rc.encodeBit(isRep2, state.get(), rep - 2);
324
325                if (rep == 3) {
326                    reps[3] = reps[2];
327                }
328
329                reps[2] = reps[1];
330            }
331
332            reps[1] = reps[0];
333            reps[0] = dist;
334        }
335
336        if (len == 1) {
337            state.updateShortRep();
338        } else {
339            repLenEncoder.encode(len, posState);
340            state.updateLongRep();
341        }
342    }

370    int getShortRepPrice(int anyRepPrice, State state, int posState) {
371        return anyRepPrice
372                + RangeEncoder.getBitPrice(isRep0[state.get()], 0)
373                + RangeEncoder.getBitPrice(isRep0Long[state.get()][posState],
374                0);
375    }

377    int getLongRepPrice(int anyRepPrice, int rep, State state, int posState) {
378        int price = anyRepPrice;
379
380        if (rep == 0) {
381            price += RangeEncoder.getBitPrice(isRep0[state.get()], 0)
382                    + RangeEncoder.getBitPrice(
383                    isRep0Long[state.get()][posState], 1);
384        } else {
385            price += RangeEncoder.getBitPrice(isRep0[state.get()], 1);
386
387            if (rep == 1) {
388                price += RangeEncoder.getBitPrice(isRep1[state.get()], 0);
389            } else {
390                price += RangeEncoder.getBitPrice(isRep1[state.get()], 1)
391                        + RangeEncoder.getBitPrice(isRep2[state.get()],
392                        rep - 2);
393            }
394        }
395
396        return price;
397    }

399    int getLongRepAndLenPrice(int rep, int len, State state, int posState) {
400        int anyMatchPrice = getAnyMatchPrice(state, posState);
401        int anyRepPrice = getAnyRepPrice(anyMatchPrice, state);
402        int longRepPrice = getLongRepPrice(anyRepPrice, rep, state, posState);
403        return longRepPrice + repLenEncoder.getPrice(len, posState);
404    }

406    int getMatchAndLenPrice(int normalMatchPrice,
407                            int dist, int len, int posState) {
408        int price = normalMatchPrice
409                + matchLenEncoder.getPrice(len, posState);
410        int distState = getDistState(len);
411
412        if (dist < FULL_DISTANCES) {
413            price += fullDistPrices[distState][dist];
414        } else {
415            // Note that distSlotPrices includes also
416            // the price of direct bits.
417            int distSlot = getDistSlot(dist);
418            price += distSlotPrices[distState][distSlot]
419                    + alignPrices[dist & ALIGN_MASK];
420        }
421
422        return price;
423    }

425    private void updateDistPrices() {
426        distPriceCount = DIST_PRICE_UPDATE_INTERVAL;
427
428        for (int distState = 0; distState < DIST_STATES; ++distState) {
429            for (int distSlot = 0; distSlot < distSlotPricesSize; ++distSlot) {
430                distSlotPrices[distState][distSlot]
431                        = RangeEncoder.getBitTreePrice(
432                        distSlots[distState], distSlot);
433            }
434
435            for (int distSlot = DIST_MODEL_END; distSlot < distSlotPricesSize;
436                 ++distSlot) {
437                int count = (distSlot >>> 1) - 1 - ALIGN_BITS;
438                distSlotPrices[distState][distSlot]
439                        += RangeEncoder.getDirectBitsPrice(count);
440            }
441
442            for (int dist = 0; dist < DIST_MODEL_START; ++dist) {
443                fullDistPrices[distState][dist]
444                        = distSlotPrices[distState][dist];
445            }
446        }
447
448        int dist = DIST_MODEL_START;
449        for (int distSlot = DIST_MODEL_START; distSlot < DIST_MODEL_END;
450             ++distSlot) {
451            int footerBits = (distSlot >>> 1) - 1;
452            int base = (2 | (distSlot & 1)) << footerBits;
453
454            int limit = distSpecial[distSlot - DIST_MODEL_START].length;
455            for (int i = 0; i < limit; ++i) {
456                int distReduced = dist - base;
457                int price = RangeEncoder.getReverseBitTreePrice(
458                        distSpecial[distSlot - DIST_MODEL_START],
459                        distReduced);
460
461                for (int distState = 0; distState < DIST_STATES; ++distState) {
462                    fullDistPrices[distState][dist]
463                            = distSlotPrices[distState][distSlot] + price;
464                }
465
466                ++dist;
467            }
468        }
469
470        assert dist == FULL_DISTANCES;
471    }
472
473    private void updateAlignPrices() {
474        alignPriceCount = ALIGN_PRICE_UPDATE_INTERVAL;
475
476        for (int i = 0; i < ALIGN_SIZE; ++i) {
477            alignPrices[i] = RangeEncoder.getReverseBitTreePrice(distAlign,
478                    i);
479        }
480    }

482    /**
483     * Updates the lookup tables used for calculating match distance
484     * and length prices. The updating is skipped for performance reasons
485     * if the tables haven't changed much since the previous update.
486     */
487    void updatePrices() {
488        if (distPriceCount <= 0) {
489            updateDistPrices();
490        }
491
492        if (alignPriceCount <= 0) {
493            updateAlignPrices();
494        }
495
496        matchLenEncoder.updatePrices();
497        repLenEncoder.updatePrices();
498    }

501    class LiteralEncoder extends LiteralCoder {
502        LiteralSubencoder[] subencoders;
503
504        LiteralEncoder(int lc, int lp) {
505            super(lc, lp);
506
507            subencoders = new LiteralSubencoder[1 << (lc + lp)];
508            for (int i = 0; i < subencoders.length; ++i) {
509                subencoders[i] = new LiteralSubencoder();
510            }
511        }
512
513        void reset() {
514            for (int i = 0; i < subencoders.length; ++i) {
515                subencoders[i].reset();
516            }
517        }
518
519        void encodeInit() {
520            // When encoding the first byte of the stream, there is
521            // no previous byte in the dictionary so the encode function
522            // wouldn't work.
523            assert readAhead >= 0;
524            subencoders[0].encode();
525        }
526
527        void encode() {
528            assert readAhead >= 0;
529            int i = getSubcoderIndex(lz.getByte(1 + readAhead),
530                    lz.getPos() - readAhead);
531            subencoders[i].encode();
532        }
533
534        int getPrice(int curByte, int matchByte,
535                     int prevByte, int pos, State state) {
536            int price = RangeEncoder.getBitPrice(
537                    isMatch[state.get()][pos & posMask], 0);
538
539            int i = getSubcoderIndex(prevByte, pos);
540            price += state.isLiteral()
541                    ? subencoders[i].getNormalPrice(curByte)
542                    : subencoders[i].getMatchedPrice(curByte, matchByte);
543
544            return price;
545        }
546
547        private class LiteralSubencoder extends LiteralSubcoder {
548            void encode() {
549                int symbol = lz.getByte(readAhead) | 0x100;
550
551                if (state.isLiteral()) {
552                    int subencoderIndex;
553                    int bit;
554
555                    do {
556                        subencoderIndex = symbol >>> 8;
557                        bit = (symbol >>> 7) & 1;
558                        rc.encodeBit(probs, subencoderIndex, bit);
559                        symbol <<= 1;
560                    } while (symbol < 0x10000);
561
562                } else {
563                    int matchByte = lz.getByte(reps[0] + 1 + readAhead);
564                    int offset = 0x100;
565                    int subencoderIndex;
566                    int matchBit;
567                    int bit;
568
569                    do {
570                        matchByte <<= 1;
571                        matchBit = matchByte & offset;
572                        subencoderIndex = offset + matchBit + (symbol >>> 8);
573                        bit = (symbol >>> 7) & 1;
574                        rc.encodeBit(probs, subencoderIndex, bit);
575                        symbol <<= 1;
576                        offset &= ~(matchByte ^ symbol);
577                    } while (symbol < 0x10000);
578                }
579
580                state.updateLiteral();
581            }
582
583            int getNormalPrice(int symbol) {
584                int price = 0;
585                int subencoderIndex;
586                int bit;
587
588                symbol |= 0x100;
589
590                do {
591                    subencoderIndex = symbol >>> 8;
592                    bit = (symbol >>> 7) & 1;
593                    price += RangeEncoder.getBitPrice(probs[subencoderIndex],
594                            bit);
595                    symbol <<= 1;
596                } while (symbol < (0x100 << 8));
597
598                return price;
599            }
600
601            int getMatchedPrice(int symbol, int matchByte) {
602                int price = 0;
603                int offset = 0x100;
604                int subencoderIndex;
605                int matchBit;
606                int bit;
607
608                symbol |= 0x100;
609
610                do {
611                    matchByte <<= 1;
612                    matchBit = matchByte & offset;
613                    subencoderIndex = offset + matchBit + (symbol >>> 8);
614                    bit = (symbol >>> 7) & 1;
615                    price += RangeEncoder.getBitPrice(probs[subencoderIndex],
616                            bit);
617                    symbol <<= 1;
618                    offset &= ~(matchByte ^ symbol);
619                } while (symbol < (0x100 << 8));
620
621                return price;
622            }
623        }
624    }

627    class LengthEncoder extends LengthCoder {
628        /**
629         * The prices are updated after at least
630         * <code>PRICE_UPDATE_INTERVAL</code> many lengths
631         * have been encoded with the same posState.
632         */
633        private static final int PRICE_UPDATE_INTERVAL = 32; // FIXME?
634
635        private final int[] counters;
636        private final int[][] prices;
637
638        LengthEncoder(int pb, int niceLen) {
639            int posStates = 1 << pb;
640            counters = new int[posStates];
641
642            // Always allocate at least LOW_SYMBOLS + MID_SYMBOLS because
643            // it makes updatePrices slightly simpler. The prices aren't
644            // usually needed anyway if niceLen < 18.
645            int lenSymbols = Math.max(niceLen - MATCH_LEN_MIN + 1,
646                    LOW_SYMBOLS + MID_SYMBOLS);
647            prices = new int[posStates][lenSymbols];
648        }
649
650        void reset() {
651            super.reset();
652
653            // Reset counters to zero to force price update before
654            // the prices are needed.
655            for (int i = 0; i < counters.length; ++i) {
656                counters[i] = 0;
657            }
658        }
659
660        void encode(int len, int posState) {
661            len -= MATCH_LEN_MIN;
662
663            if (len < LOW_SYMBOLS) {
664                rc.encodeBit(choice, 0, 0);
665                rc.encodeBitTree(low[posState], len);
666            } else {
667                rc.encodeBit(choice, 0, 1);
668                len -= LOW_SYMBOLS;
669
670                if (len < MID_SYMBOLS) {
671                    rc.encodeBit(choice, 1, 0);
672                    rc.encodeBitTree(mid[posState], len);
673                } else {
674                    rc.encodeBit(choice, 1, 1);
675                    rc.encodeBitTree(high, len - MID_SYMBOLS);
676                }
677            }
678
679            --counters[posState];
680        }
681
682        int getPrice(int len, int posState) {
683            return prices[posState][len - MATCH_LEN_MIN];
684        }
685
686        void updatePrices() {
687            for (int posState = 0; posState < counters.length; ++posState) {
688                if (counters[posState] <= 0) {
689                    counters[posState] = PRICE_UPDATE_INTERVAL;
690                    updatePrices(posState);
691                }
692            }
693        }
694
695        private void updatePrices(int posState) {
696            int choice0Price = RangeEncoder.getBitPrice(choice[0], 0);
697
698            int i = 0;
699            for (; i < LOW_SYMBOLS; ++i) {
700                prices[posState][i] = choice0Price
701                        + RangeEncoder.getBitTreePrice(low[posState], i);
702            }
703
704            choice0Price = RangeEncoder.getBitPrice(choice[0], 1);
705            int choice1Price = RangeEncoder.getBitPrice(choice[1], 0);
706
707            for (; i < LOW_SYMBOLS + MID_SYMBOLS; ++i) {
708                prices[posState][i] = choice0Price + choice1Price
709                        + RangeEncoder.getBitTreePrice(mid[posState],
710                        i - LOW_SYMBOLS);
711            }
712
713            choice1Price = RangeEncoder.getBitPrice(choice[1], 1);
714
715            for (; i < prices[posState].length; ++i) {
716                prices[posState][i] = choice0Price + choice1Price
717                        + RangeEncoder.getBitTreePrice(high, i - LOW_SYMBOLS
718                        - MID_SYMBOLS);
719            }
720        }
721    }
