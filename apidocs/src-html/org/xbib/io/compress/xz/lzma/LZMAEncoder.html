008    public abstract class LZMAEncoder extends LZMACoder {
009        public static final int MODE_FAST = 1;
010        public static final int MODE_NORMAL = 2;
011    
012        /**
013         * LZMA2 chunk is considered full when its uncompressed size exceeds
014         * <code>LZMA2_UNCOMPRESSED_LIMIT</code>.
015         * <p/>
016         * A compressed LZMA2 chunk can hold 2 MiB of uncompressed data.
017         * A single LZMA symbol may indicate up to MATCH_LEN_MAX bytes
018         * of data, so the LZMA2 chunk is considered full when there is
019         * less space than MATCH_LEN_MAX bytes.
020         */
021        private static final int LZMA2_UNCOMPRESSED_LIMIT
022                = (2 << 20) - MATCH_LEN_MAX;
023    
024        /**
025         * LZMA2 chunk is considered full when its compressed size exceeds
026         * <code>LZMA2_COMPRESSED_LIMIT</code>.
027         * <p/>
028         * The maximum compressed size of a LZMA2 chunk is 64 KiB.
029         * A single LZMA symbol might use 20 bytes of space even though
030         * it usually takes just one byte or so. Two more bytes are needed
031         * for LZMA2 uncompressed chunks (see LZMA2OutputStream.writeChunk).
032         * Leave a little safety margin and use 26 bytes.
033         */
034        private static final int LZMA2_COMPRESSED_LIMIT = (64 << 10) - 26;
035    
036        private static final int DIST_PRICE_UPDATE_INTERVAL = FULL_DISTANCES;
037        private static final int ALIGN_PRICE_UPDATE_INTERVAL = ALIGN_SIZE;
038    
039        private final RangeEncoder rc;
040        final LZEncoder lz;
041        final LiteralEncoder literalEncoder;
042        final LengthEncoder matchLenEncoder;
043        final LengthEncoder repLenEncoder;
044        final int niceLen;
045    
046        private int distPriceCount = 0;
047        private int alignPriceCount = 0;
048    
049        private final int distSlotPricesSize;
050        private final int[][] distSlotPrices;
051        private final int[][] fullDistPrices
052                = new int[DIST_STATES][FULL_DISTANCES];
053        private final int[] alignPrices = new int[ALIGN_SIZE];
054    
055        int back = 0;
056        int readAhead = -1;
057        private int uncompressedSize = 0;

059        public static int getMemoryUsage(int mode, int dictSize,
060                                         int extraSizeBefore, int mf) {
061            int m = 80;
062    
063            switch (mode) {
064                case MODE_FAST:
065                    m += LZMAEncoderFast.getMemoryUsage(
066                            dictSize, extraSizeBefore, mf);
067                    break;
068    
069                case MODE_NORMAL:
070                    m += LZMAEncoderNormal.getMemoryUsage(
071                            dictSize, extraSizeBefore, mf);
072                    break;
073    
074                default:
075                    throw new IllegalArgumentException();
076            }
077    
078            return m;
079        }

081        public static LZMAEncoder getInstance(
082                RangeEncoder rc, int lc, int lp, int pb, int mode,
083                int dictSize, int extraSizeBefore,
084                int niceLen, int mf, int depthLimit) {
085            switch (mode) {
086                case MODE_FAST:

089                            niceLen, mf, depthLimit);

091                case MODE_NORMAL:

093                            dictSize, extraSizeBefore,
094                            niceLen, mf, depthLimit);

097            throw new IllegalArgumentException();
098        }

100        /**
101         * Gets an integer [0, 63] matching the highest two bits of an integer.
102         * This is like bit scan reverse (BSR) on x86 except that this also
103         * cares about the second highest bit.
104         */
105        public static int getDistSlot(int dist) {
106            if (dist <= DIST_MODEL_START) {
107                return dist;
108            }
109    
110            int n = dist;
111            int i = 31;
112    
113            if ((n & 0xFFFF0000) == 0) {
114                n <<= 16;
115                i = 15;
116            }
117    
118            if ((n & 0xFF000000) == 0) {
119                n <<= 8;
120                i -= 8;
121            }
122    
123            if ((n & 0xF0000000) == 0) {
124                n <<= 4;
125                i -= 4;
126            }
127    
128            if ((n & 0xC0000000) == 0) {
129                n <<= 2;
130                i -= 2;
131            }
132    
133            if ((n & 0x80000000) == 0) {
134                --i;
135            }
136    
137            return (i << 1) + ((dist >>> (i - 1)) & 1);
138        }

140        /**
141         * Gets the next LZMA symbol.
142         * <p/>
143         * There are three types of symbols: literal (a single byte),
144         * repeated match, and normal match. The symbol is indicated
145         * by the return value and by the variable <code>back</code>.
146         * <p/>
147         * Literal: <code>back == -1</code> and return value is <code>1</code>.
148         * The literal itself needs to be read from <code>lz</code> separately.
149         * <p/>
150         * Repeated match: <code>back</code> is in the range [0, 3] and
151         * the return value is the length of the repeated match.
152         * <p/>
153         * Normal match: <code>back - REPS<code> (<code>back - 4</code>)
154         * is the distance of the match and the return value is the length
155         * of the match.
156         */
157        abstract int getNextSymbol();

159        LZMAEncoder(RangeEncoder rc, LZEncoder lz,
160                    int lc, int lp, int pb, int dictSize, int niceLen) {
161            super(pb);
162            this.rc = rc;
163            this.lz = lz;
164            this.niceLen = niceLen;
165    
166            literalEncoder = new LiteralEncoder(lc, lp);
167            matchLenEncoder = new LengthEncoder(pb, niceLen);
168            repLenEncoder = new LengthEncoder(pb, niceLen);
169    
170            distSlotPricesSize = getDistSlot(dictSize - 1) + 1;
171            distSlotPrices = new int[DIST_STATES][distSlotPricesSize];
172    
173            reset();
174        }

181            super.reset();
182            literalEncoder.reset();
183            matchLenEncoder.reset();
184            repLenEncoder.reset();
185            distPriceCount = 0;
186            alignPriceCount = 0;
187    
188            uncompressedSize += readAhead + 1;
189            readAhead = -1;

200        /**
201         * Compresses for LZMA2.
202         *
203         * @return true if the LZMA2 chunk became full, false otherwise
204         */
205        public boolean encodeForLZMA2() {
206            if (!lz.isStarted() && !encodeInit()) {
207                return false;
208            }
209    
210            while (uncompressedSize <= LZMA2_UNCOMPRESSED_LIMIT
211                    && rc.getPendingSize() <= LZMA2_COMPRESSED_LIMIT) {
212                if (!encodeSymbol()) {
213                    return false;
214                }
215            }
216    
217            return true;
218        }
219    
220        private boolean encodeInit() {
221            assert readAhead == -1;
222            if (!lz.hasEnoughData(0)) {
223                return false;
224            }
225    
226            // The first symbol must be a literal unless using
227            // a preset dictionary. This code isn't run if using
228            // a preset dictionary.
229            skip(1);
230            rc.encodeBit(isMatch[state.get()], 0, 0);
231            literalEncoder.encodeInit();
232    
233            --readAhead;
234            assert readAhead == -1;
235    
236            ++uncompressedSize;
237            assert uncompressedSize == 1;
238    
239            return true;
240        }
241    
242        private boolean encodeSymbol() {
243            if (!lz.hasEnoughData(readAhead + 1)) {
244                return false;
245            }
246    
247            int len = getNextSymbol();
248    
249            assert readAhead >= 0;
250            int posState = (lz.getPos() - readAhead) & posMask;
251    
252            if (back == -1) {
253                // Literal i.e. eight-bit byte
254                assert len == 1;
255                rc.encodeBit(isMatch[state.get()], posState, 0);
256                literalEncoder.encode();
257            } else {
258                // Some type of match
259                rc.encodeBit(isMatch[state.get()], posState, 1);
260                if (back < REPS) {
261                    // Repeated match i.e. the same distance
262                    // has been used earlier.
263                    assert lz.getMatchLen(-readAhead, reps[back], len) == len;
264                    rc.encodeBit(isRep, state.get(), 1);
265                    encodeRepMatch(back, len, posState);
266                } else {
267                    // Normal match
268                    assert lz.getMatchLen(-readAhead, back - REPS, len) == len;
269                    rc.encodeBit(isRep, state.get(), 0);
270                    encodeMatch(back - REPS, len, posState);
271                }
272            }
273    
274            readAhead -= len;
275            uncompressedSize += len;
276    
277            return true;
278        }
279    
280        private void encodeMatch(int dist, int len, int posState) {
281            state.updateMatch();
282            matchLenEncoder.encode(len, posState);
283    
284            int distSlot = getDistSlot(dist);
285            rc.encodeBitTree(distSlots[getDistState(len)], distSlot);
286    
287            if (distSlot >= DIST_MODEL_START) {
288                int footerBits = (distSlot >>> 1) - 1;
289                int base = (2 | (distSlot & 1)) << footerBits;
290                int distReduced = dist - base;
291    
292                if (distSlot < DIST_MODEL_END) {
293                    rc.encodeReverseBitTree(
294                            distSpecial[distSlot - DIST_MODEL_START],
295                            distReduced);
296                } else {
297                    rc.encodeDirectBits(distReduced >>> ALIGN_BITS,
298                            footerBits - ALIGN_BITS);
299                    rc.encodeReverseBitTree(distAlign, distReduced & ALIGN_MASK);
300                    --alignPriceCount;
301                }
302            }
303    
304            reps[3] = reps[2];
305            reps[2] = reps[1];
306            reps[1] = reps[0];
307            reps[0] = dist;
308    
309            --distPriceCount;
310        }

312        private void encodeRepMatch(int rep, int len, int posState) {
313            if (rep == 0) {
314                rc.encodeBit(isRep0, state.get(), 0);
315                rc.encodeBit(isRep0Long[state.get()], posState, len == 1 ? 0 : 1);
316            } else {
317                int dist = reps[rep];
318                rc.encodeBit(isRep0, state.get(), 1);
319    
320                if (rep == 1) {
321                    rc.encodeBit(isRep1, state.get(), 0);
322                } else {
323                    rc.encodeBit(isRep1, state.get(), 1);
324                    rc.encodeBit(isRep2, state.get(), rep - 2);
325    
326                    if (rep == 3) {
327                        reps[3] = reps[2];
328                    }
329    
330                    reps[2] = reps[1];
331                }
332    
333                reps[1] = reps[0];
334                reps[0] = dist;
335            }
336    
337            if (len == 1) {
338                state.updateShortRep();
339            } else {
340                repLenEncoder.encode(len, posState);
341                state.updateLongRep();
342            }
343        }

345        Matches getMatches() {
346            ++readAhead;
347            Matches matches = lz.getMatches();
348            assert lz.verifyMatches(matches);
349            return matches;

361        int getNormalMatchPrice(int anyMatchPrice, State state) {
362            return anyMatchPrice
363                    + RangeEncoder.getBitPrice(isRep[state.get()], 0);
364        }

366        int getAnyRepPrice(int anyMatchPrice, State state) {
367            return anyMatchPrice
368                    + RangeEncoder.getBitPrice(isRep[state.get()], 1);
369        }
370    
371        int getShortRepPrice(int anyRepPrice, State state, int posState) {
372            return anyRepPrice
373                    + RangeEncoder.getBitPrice(isRep0[state.get()], 0)
374                    + RangeEncoder.getBitPrice(isRep0Long[state.get()][posState],
375                    0);
376        }
377    
378        int getLongRepPrice(int anyRepPrice, int rep, State state, int posState) {
379            int price = anyRepPrice;
380    
381            if (rep == 0) {
382                price += RangeEncoder.getBitPrice(isRep0[state.get()], 0)
383                        + RangeEncoder.getBitPrice(
384                        isRep0Long[state.get()][posState], 1);
385            } else {
386                price += RangeEncoder.getBitPrice(isRep0[state.get()], 1);
387    
388                if (rep == 1) {
389                    price += RangeEncoder.getBitPrice(isRep1[state.get()], 0);
390                } else {
391                    price += RangeEncoder.getBitPrice(isRep1[state.get()], 1)
392                            + RangeEncoder.getBitPrice(isRep2[state.get()],
393                            rep - 2);
394                }
395            }
396    
397            return price;
398        }

400        int getLongRepAndLenPrice(int rep, int len, State state, int posState) {
401            int anyMatchPrice = getAnyMatchPrice(state, posState);
402            int anyRepPrice = getAnyRepPrice(anyMatchPrice, state);
403            int longRepPrice = getLongRepPrice(anyRepPrice, rep, state, posState);
404            return longRepPrice + repLenEncoder.getPrice(len, posState);
405        }
406    
407        int getMatchAndLenPrice(int normalMatchPrice,
408                                int dist, int len, int posState) {
409            int price = normalMatchPrice
410                    + matchLenEncoder.getPrice(len, posState);
411            int distState = getDistState(len);
412    
413            if (dist < FULL_DISTANCES) {
414                price += fullDistPrices[distState][dist];
415            } else {
416                // Note that distSlotPrices includes also
417                // the price of direct bits.
418                int distSlot = getDistSlot(dist);
419                price += distSlotPrices[distState][distSlot]
420                        + alignPrices[dist & ALIGN_MASK];
421            }
422    
423            return price;
424        }

426        private void updateDistPrices() {
427            distPriceCount = DIST_PRICE_UPDATE_INTERVAL;
428    
429            for (int distState = 0; distState < DIST_STATES; ++distState) {
430                for (int distSlot = 0; distSlot < distSlotPricesSize; ++distSlot) {
431                    distSlotPrices[distState][distSlot]
432                            = RangeEncoder.getBitTreePrice(
433                            distSlots[distState], distSlot);
434                }
435    
436                for (int distSlot = DIST_MODEL_END; distSlot < distSlotPricesSize;
437                     ++distSlot) {
438                    int count = (distSlot >>> 1) - 1 - ALIGN_BITS;
439                    distSlotPrices[distState][distSlot]
440                            += RangeEncoder.getDirectBitsPrice(count);
441                }
442    
443                for (int dist = 0; dist < DIST_MODEL_START; ++dist) {
444                    fullDistPrices[distState][dist]
445                            = distSlotPrices[distState][dist];
446                }
447            }
448    
449            int dist = DIST_MODEL_START;
450            for (int distSlot = DIST_MODEL_START; distSlot < DIST_MODEL_END;
451                 ++distSlot) {
452                int footerBits = (distSlot >>> 1) - 1;
453                int base = (2 | (distSlot & 1)) << footerBits;
454    
455                int limit = distSpecial[distSlot - DIST_MODEL_START].length;
456                for (int i = 0; i < limit; ++i) {
457                    int distReduced = dist - base;
458                    int price = RangeEncoder.getReverseBitTreePrice(
459                            distSpecial[distSlot - DIST_MODEL_START],
460                            distReduced);
461    
462                    for (int distState = 0; distState < DIST_STATES; ++distState) {
463                        fullDistPrices[distState][dist]
464                                = distSlotPrices[distState][distSlot] + price;
465                    }
466    
467                    ++dist;
468                }
469            }
470    
471            assert dist == FULL_DISTANCES;
472        }
473    
474        private void updateAlignPrices() {
475            alignPriceCount = ALIGN_PRICE_UPDATE_INTERVAL;
476    
477            for (int i = 0; i < ALIGN_SIZE; ++i) {
478                alignPrices[i] = RangeEncoder.getReverseBitTreePrice(distAlign,
479                        i);
480            }
481        }

483        /**
484         * Updates the lookup tables used for calculating match distance
485         * and length prices. The updating is skipped for performance reasons
486         * if the tables haven't changed much since the previous update.
487         */
488        void updatePrices() {
489            if (distPriceCount <= 0) {
490                updateDistPrices();
491            }
492    
493            if (alignPriceCount <= 0) {
494                updateAlignPrices();
495            }
496    
497            matchLenEncoder.updatePrices();
498            repLenEncoder.updatePrices();
499        }

502        class LiteralEncoder extends LiteralCoder {
503            LiteralSubencoder[] subencoders;
504    
505            LiteralEncoder(int lc, int lp) {
506                super(lc, lp);
507    
508                subencoders = new LiteralSubencoder[1 << (lc + lp)];
509                for (int i = 0; i < subencoders.length; ++i) {
510                    subencoders[i] = new LiteralSubencoder();
511                }
512            }
513    
514            void reset() {
515                for (int i = 0; i < subencoders.length; ++i) {
516                    subencoders[i].reset();
517                }
518            }
519    
520            void encodeInit() {
521                // When encoding the first byte of the stream, there is
522                // no previous byte in the dictionary so the encode function
523                // wouldn't work.
524                assert readAhead >= 0;
525                subencoders[0].encode();
526            }
527    
528            void encode() {
529                assert readAhead >= 0;
530                int i = getSubcoderIndex(lz.getByte(1 + readAhead),
531                        lz.getPos() - readAhead);
532                subencoders[i].encode();
533            }
534    
535            int getPrice(int curByte, int matchByte,
536                         int prevByte, int pos, State state) {
537                int price = RangeEncoder.getBitPrice(
538                        isMatch[state.get()][pos & posMask], 0);
539    
540                int i = getSubcoderIndex(prevByte, pos);
541                price += state.isLiteral()
542                        ? subencoders[i].getNormalPrice(curByte)
543                        : subencoders[i].getMatchedPrice(curByte, matchByte);
544    
545                return price;
546            }
547    
548            private class LiteralSubencoder extends LiteralSubcoder {
549                void encode() {
550                    int symbol = lz.getByte(readAhead) | 0x100;
551    
552                    if (state.isLiteral()) {
553                        int subencoderIndex;
554                        int bit;
555    
556                        do {
557                            subencoderIndex = symbol >>> 8;
558                            bit = (symbol >>> 7) & 1;
559                            rc.encodeBit(probs, subencoderIndex, bit);
560                            symbol <<= 1;
561                        } while (symbol < 0x10000);
562    
563                    } else {
564                        int matchByte = lz.getByte(reps[0] + 1 + readAhead);
565                        int offset = 0x100;
566                        int subencoderIndex;
567                        int matchBit;
568                        int bit;
569    
570                        do {
571                            matchByte <<= 1;
572                            matchBit = matchByte & offset;
573                            subencoderIndex = offset + matchBit + (symbol >>> 8);
574                            bit = (symbol >>> 7) & 1;
575                            rc.encodeBit(probs, subencoderIndex, bit);
576                            symbol <<= 1;
577                            offset &= ~(matchByte ^ symbol);
578                        } while (symbol < 0x10000);
579                    }
580    
581                    state.updateLiteral();
582                }
583    
584                int getNormalPrice(int symbol) {
585                    int price = 0;
586                    int subencoderIndex;
587                    int bit;
588    
589                    symbol |= 0x100;
590    
591                    do {
592                        subencoderIndex = symbol >>> 8;
593                        bit = (symbol >>> 7) & 1;
594                        price += RangeEncoder.getBitPrice(probs[subencoderIndex],
595                                bit);
596                        symbol <<= 1;
597                    } while (symbol < (0x100 << 8));
598    
599                    return price;
600                }
601    
602                int getMatchedPrice(int symbol, int matchByte) {
603                    int price = 0;
604                    int offset = 0x100;
605                    int subencoderIndex;
606                    int matchBit;
607                    int bit;
608    
609                    symbol |= 0x100;
610    
611                    do {
612                        matchByte <<= 1;
613                        matchBit = matchByte & offset;
614                        subencoderIndex = offset + matchBit + (symbol >>> 8);
615                        bit = (symbol >>> 7) & 1;
616                        price += RangeEncoder.getBitPrice(probs[subencoderIndex],
617                                bit);
618                        symbol <<= 1;
619                        offset &= ~(matchByte ^ symbol);
620                    } while (symbol < (0x100 << 8));
621    
622                    return price;
623                }
624            }
625        }

628        class LengthEncoder extends LengthCoder {
629            /**
630             * The prices are updated after at least
631             * <code>PRICE_UPDATE_INTERVAL</code> many lengths
632             * have been encoded with the same posState.
633             */
634            private static final int PRICE_UPDATE_INTERVAL = 32; // FIXME?
635    
636            private final int[] counters;
637            private final int[][] prices;
638    
639            LengthEncoder(int pb, int niceLen) {
640                int posStates = 1 << pb;
641                counters = new int[posStates];
642    
643                // Always allocate at least LOW_SYMBOLS + MID_SYMBOLS because
644                // it makes updatePrices slightly simpler. The prices aren't
645                // usually needed anyway if niceLen < 18.
646                int lenSymbols = Math.max(niceLen - MATCH_LEN_MIN + 1,
647                        LOW_SYMBOLS + MID_SYMBOLS);
648                prices = new int[posStates][lenSymbols];
649            }
650    
651            void reset() {
652                super.reset();
653    
654                // Reset counters to zero to force price update before
655                // the prices are needed.
656                for (int i = 0; i < counters.length; ++i) {
657                    counters[i] = 0;
658                }
659            }
660    
661            void encode(int len, int posState) {
662                len -= MATCH_LEN_MIN;
663    
664                if (len < LOW_SYMBOLS) {
665                    rc.encodeBit(choice, 0, 0);
666                    rc.encodeBitTree(low[posState], len);
667                } else {
668                    rc.encodeBit(choice, 0, 1);
669                    len -= LOW_SYMBOLS;
670    
671                    if (len < MID_SYMBOLS) {
672                        rc.encodeBit(choice, 1, 0);
673                        rc.encodeBitTree(mid[posState], len);
674                    } else {
675                        rc.encodeBit(choice, 1, 1);
676                        rc.encodeBitTree(high, len - MID_SYMBOLS);
677                    }
678                }
679    
680                --counters[posState];
681            }
682    
683            int getPrice(int len, int posState) {
684                return prices[posState][len - MATCH_LEN_MIN];
685            }
686    
687            void updatePrices() {
688                for (int posState = 0; posState < counters.length; ++posState) {
689                    if (counters[posState] <= 0) {
690                        counters[posState] = PRICE_UPDATE_INTERVAL;
691                        updatePrices(posState);
692                    }
693                }
694            }
695    
696            private void updatePrices(int posState) {
697                int choice0Price = RangeEncoder.getBitPrice(choice[0], 0);
698    
699                int i = 0;
700                for (; i < LOW_SYMBOLS; ++i) {
701                    prices[posState][i] = choice0Price
702                            + RangeEncoder.getBitTreePrice(low[posState], i);
703                }
704    
705                choice0Price = RangeEncoder.getBitPrice(choice[0], 1);
706                int choice1Price = RangeEncoder.getBitPrice(choice[1], 0);
707    
708                for (; i < LOW_SYMBOLS + MID_SYMBOLS; ++i) {
709                    prices[posState][i] = choice0Price + choice1Price
710                            + RangeEncoder.getBitTreePrice(mid[posState],
711                            i - LOW_SYMBOLS);
712                }
713    
714                choice1Price = RangeEncoder.getBitPrice(choice[1], 1);
715    
716                for (; i < prices[posState].length; ++i) {
717                    prices[posState][i] = choice0Price + choice1Price
718                            + RangeEncoder.getBitTreePrice(high, i - LOW_SYMBOLS
719                            - MID_SYMBOLS);
720                }
721            }
722        }
