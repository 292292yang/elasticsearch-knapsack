012/**
013 * Generic {@link java.io.IOException IOException} specific to this package.

015 * from <code>XZIOException</code>.
016 */
017public class XZIOException extends java.io.IOException {
018    private static final long serialVersionUID = 3L;
019
020    public XZIOException() {
021        super();
022    }
023
024    public XZIOException(String s) {
025        super(s);
026    }
027}
