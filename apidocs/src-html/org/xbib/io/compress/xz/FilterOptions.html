002    package org.xbib.io.compress.xz;
003    
004    import java.io.IOException;
005    import java.io.InputStream;
006    
007    /**
008     * Base class for filter-specific options classes.
009     */
010    public abstract class FilterOptions implements Cloneable {
011        /**
012         * Gets how much memory the encoder will need with
013         * the given filter chain. This function simply calls
014         * <code>getEncoderMemoryUsage()</code> for every filter
015         * in the array and returns the sum of the returned values.
016         */
017        public static int getEncoderMemoryUsage(FilterOptions[] options) {
018            int m = 0;
019    
020            for (int i = 0; i < options.length; ++i) {
021                m += options[i].getEncoderMemoryUsage();
022            }
023    
024            return m;
025        }
026    
027        /**
028         * Gets how much memory the decoder will need with
029         * the given filter chain. This function simply calls
030         * <code>getDecoderMemoryUsage()</code> for every filter
031         * in the array and returns the sum of the returned values.
032         */
033        public static int getDecoderMemoryUsage(FilterOptions[] options) {
034            int m = 0;
035    
036            for (int i = 0; i < options.length; ++i) {
037                m += options[i].getDecoderMemoryUsage();
038            }
039    
040            return m;
041        }
042    
043        /**
044         * Gets how much memory the encoder will need with these options.
045         */
046        public abstract int getEncoderMemoryUsage();
047    
048        /**
049         * Gets a raw (no XZ headers) encoder output stream using these options.
050         * Raw streams are an advanced feature. In most cases you want to store
051         * the compressed data in the .xz container format instead of using
052         * a raw stream. To use this filter in a .xz file, pass this object
053         * to XZOutputStream.
054         */
055        public abstract FinishableOutputStream getOutputStream(
056                FinishableOutputStream out);
057    
058        /**
059         * Gets how much memory the decoder will need to decompress the data
060         * that was encoded with these options.
061         */
062        public abstract int getDecoderMemoryUsage();
063    
064        /**
065         * Gets a raw (no XZ headers) decoder input stream using these options.
066         */
067        public abstract InputStream getInputStream(InputStream in)
068                throws IOException;
069    
070        abstract FilterEncoder getFilterEncoder();
071    
072        FilterOptions() {
073        }
074    }
