002package org.xbib.io.compress.xz;
003
004import java.io.IOException;
005import java.io.OutputStream;
006
007/**
008 * Output stream that supports finishing without closing
009 * the underlying stream.
010 */
011public abstract class FinishableOutputStream extends OutputStream {
012    /**
013     * Finish the stream without closing the underlying stream.
014     * No more data may be written to the stream after finishing.
015     * <p/>
016     * The <code>finish</code> method of <code>FinishableOutputStream</code>
017     * does nothing. Subclasses should override it if they need finishing
018     * support, which is the case, for example, with compressors.
019     *
020     * @throws java.io.IOException
021     */
022    public void finish() throws IOException {
023    }
024
025    ;
026}
