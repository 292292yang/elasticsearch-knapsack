010package org.xbib.io.compress.xz;
011
012import java.io.OutputStream;
013import java.io.IOException;
014
015/**
016 * Output stream that supports finishing without closing
017 * the underlying stream.
018 */
019public abstract class FinishableOutputStream extends OutputStream {
020    /**
021     * Finish the stream without closing the underlying stream.
022     * No more data may be written to the stream after finishing.
023     * <p>
024     * The <code>finish</code> method of <code>FinishableOutputStream</code>
025     * does nothing. Subclasses should override it if they need finishing
026     * support, which is the case, for example, with compressors.
027     *
028     * @throws      IOException
029     */
030    public void finish() throws IOException {};
031}
