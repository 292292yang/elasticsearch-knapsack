001package org.xbib.io.compress.xz.check;
002
003import org.xbib.io.compress.xz.UnsupportedOptionsException;
004import org.xbib.io.compress.xz.XZ;
005
006public abstract class Check {
007    int size;
008    String name;
009
010    public abstract void update(byte[] buf, int off, int len);
011
012    public abstract byte[] finish();
013
014    public void update(byte[] buf) {
015        update(buf, 0, buf.length);
016    }
017
018    public int getSize() {
019        return size;
020    }
021
022    public String getName() {
023        return name;
024    }
025
026    public static Check getInstance(int checkType)
027            throws UnsupportedOptionsException {
028        switch (checkType) {
029            case XZ.CHECK_NONE:
030                return new None();
031
032            case XZ.CHECK_CRC32:
033                return new CRC32();
034
035            case XZ.CHECK_CRC64:
036                return new CRC64();
037
038            case XZ.CHECK_SHA256:
039                try {
040                    return new SHA256();
041                } catch (java.security.NoSuchAlgorithmException e) {
042                }
043
044                break;
045        }
046
047        throw new UnsupportedOptionsException(
048                "Unsupported Check ID " + checkType);
049    }
050}
