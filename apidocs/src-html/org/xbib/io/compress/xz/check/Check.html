010package org.xbib.io.compress.xz.check;
011
012import org.xbib.io.compress.xz.XZ;
013import org.xbib.io.compress.xz.UnsupportedOptionsException;
014
015public abstract class Check {
016    int size;
017    String name;
018
019    public abstract void update(byte[] buf, int off, int len);
020    public abstract byte[] finish();
021
022    public void update(byte[] buf) {
023        update(buf, 0, buf.length);
024    }
025
026    public int getSize() {
027        return size;
028    }
029
030    public String getName() {
031        return name;
032    }
033
034    public static Check getInstance(int checkType)
035            throws UnsupportedOptionsException {
036        switch (checkType) {
037            case XZ.CHECK_NONE:
038                return new None();
039
040            case XZ.CHECK_CRC32:
041                return new CRC32();
042
043            case XZ.CHECK_CRC64:
044                return new CRC64();
045
046            case XZ.CHECK_SHA256:
047                try {
048                    return new SHA256();
049                } catch (java.security.NoSuchAlgorithmException e) {}
050
051                break;
052        }
053
054        throw new UnsupportedOptionsException(
055                "Unsupported Check ID " + checkType);
056    }
057}
