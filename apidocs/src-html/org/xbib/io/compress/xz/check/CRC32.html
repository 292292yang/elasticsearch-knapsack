010package org.xbib.io.compress.xz.check;
011
012public class CRC32 extends Check {
013    private final java.util.zip.CRC32 state = new java.util.zip.CRC32();
014
015    public CRC32() {
016        size = 4;
017        name = "CRC32";
018    }
019
020    public void update(byte[] buf, int off, int len) {
021        state.update(buf, off, len);
022    }
023
024    public byte[] finish() {
025        long value = state.getValue();
026        byte[] buf = new byte[] { (byte)(value),
027                                  (byte)(value >>> 8),
028                                  (byte)(value >>> 16),
029                                  (byte)(value >>> 24) };
030        state.reset();
031        return buf;
032    }
033}
