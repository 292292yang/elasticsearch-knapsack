001package org.xbib.io.compress.xz.check;
002
003public class CRC32 extends Check {
004    private final java.util.zip.CRC32 state = new java.util.zip.CRC32();
005
006    public CRC32() {
007        size = 4;
008        name = "CRC32";
009    }
010
011    public void update(byte[] buf, int off, int len) {
012        state.update(buf, off, len);
013    }
014
015    public byte[] finish() {
016        long value = state.getValue();
017        byte[] buf = new byte[]{(byte) (value),
018                (byte) (value >>> 8),
019                (byte) (value >>> 16),
020                (byte) (value >>> 24)};
021        state.reset();
022        return buf;
023    }
024}
