002package org.xbib.io.compress.xz.check;
003
004public class CRC32 extends Check {
005    private final java.util.zip.CRC32 state = new java.util.zip.CRC32();
006
007    public CRC32() {
008        size = 4;
009        name = "CRC32";
010    }
011
012    public void update(byte[] buf, int off, int len) {
013        state.update(buf, off, len);
014    }
015
016    public byte[] finish() {
017        long value = state.getValue();
018        byte[] buf = new byte[]{(byte) (value),
019                (byte) (value >>> 8),
020                (byte) (value >>> 16),
021                (byte) (value >>> 24)};
022        state.reset();
023        return buf;
024    }
025}
