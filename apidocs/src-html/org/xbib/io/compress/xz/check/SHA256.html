002    package org.xbib.io.compress.xz.check;
003    
004    public class SHA256 extends Check {
005        private final java.security.MessageDigest sha256;
006    
007        public SHA256() throws java.security.NoSuchAlgorithmException {
008            size = 32;
009            name = "SHA-256";
010            sha256 = java.security.MessageDigest.getInstance("SHA-256");
011        }
012    
013        public void update(byte[] buf, int off, int len) {
014            sha256.update(buf, off, len);
015        }
016    
017        public byte[] finish() {
018            byte[] buf = sha256.digest();
019            sha256.reset();
020            return buf;
021        }
022    }
