002    package org.xbib.io.compress.xz.check;
003    
004    public class CRC64 extends Check {
005        private static final long poly = 0xC96C5795D7870F42L;
006        private static final long crcTable[] = new long[256];
007    
008        private long crc = -1;
009    
010        static {
011            for (int b = 0; b < crcTable.length; ++b) {
012                long r = b;
013                for (int i = 0; i < 8; ++i) {
014                    if ((r & 1) == 1) {
015                        r = (r >>> 1) ^ poly;
016                    } else {
017                        r >>>= 1;
018                    }
019                }
020    
021                crcTable[b] = r;
022            }
023        }
024    
025        public CRC64() {
026            size = 8;
027            name = "CRC64";
028        }
029    
030        public void update(byte[] buf, int off, int len) {
031            int end = off + len;
032    
033            while (off < end) {
034                crc = crcTable[(buf[off++] ^ (int) crc) & 0xFF] ^ (crc >>> 8);
035            }
036        }
037    
038        public byte[] finish() {
039            long value = ~crc;
040            crc = -1;
041    
042            byte[] buf = new byte[8];
043            for (int i = 0; i < buf.length; ++i) {
044                buf[i] = (byte) (value >> (i * 8));
045            }
046    
047            return buf;
048        }
049    }
