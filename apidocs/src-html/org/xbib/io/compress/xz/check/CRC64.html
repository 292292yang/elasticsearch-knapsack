001package org.xbib.io.compress.xz.check;
002
003public class CRC64 extends Check {
004    private static final long poly = 0xC96C5795D7870F42L;
005    private static final long crcTable[] = new long[256];
006
007    private long crc = -1;
008
009    static {
010        for (int b = 0; b < crcTable.length; ++b) {
011            long r = b;
012            for (int i = 0; i < 8; ++i) {
013                if ((r & 1) == 1) {
014                    r = (r >>> 1) ^ poly;
015                } else {
016                    r >>>= 1;
017                }
018            }
019
020            crcTable[b] = r;
021        }
022    }
023
024    public CRC64() {
025        size = 8;
026        name = "CRC64";
027    }
028
029    public void update(byte[] buf, int off, int len) {
030        int end = off + len;
031
032        while (off < end) {
033            crc = crcTable[(buf[off++] ^ (int) crc) & 0xFF] ^ (crc >>> 8);
034        }
035    }
036
037    public byte[] finish() {
038        long value = ~crc;
039        crc = -1;
040
041        byte[] buf = new byte[8];
042        for (int i = 0; i < buf.length; ++i) {
043            buf[i] = (byte) (value >> (i * 8));
044        }
045
046        return buf;
047    }
048}
