011package org.xbib.io.compress.xz.lz;
012
013import java.io.OutputStream;
014import java.io.IOException;
015
016public abstract class LZEncoder {
017    public static final int MF_HC4 = 0x04;
018    public static final int MF_BT4 = 0x14;
019
020    /**
021     * Number of bytes to keep available before the current byte
022     * when moving the LZ window.
023     */
024    private final int keepSizeBefore;
025
026    /**
027     * Number of bytes that must be available, the current byte included,
028     * to make hasEnoughInput return true. Flushing and finishing are
029     * naturally exceptions to this since there cannot be any data after
030     * the end of the uncompressed input.
031     */
032    private final int keepSizeAfter;
033
034    final int matchLenMax;
035    final int niceLen;
036
037    final byte[] buf;
038
039    int readPos = -1;
040    private int readLimit = -1;
041    private boolean finishing = false;
042    private int writePos = 0;
043    private int pendingSize = 0;
044
045    static void normalize(int[] positions, int normalizationOffset) {
046        for (int i = 0; i < positions.length; ++i) {
047            if (positions[i] <= normalizationOffset)
048                positions[i] = 0;
049            else
050                positions[i] -= normalizationOffset;
051        }
052    }
053
054    /**
055     * Gets the size of the LZ window buffer that needs to be allocated.
056     */
057    private static int getBufSize(
058            int dictSize, int extraSizeBefore, int extraSizeAfter,
059            int matchLenMax) {
060        int keepSizeBefore = extraSizeBefore + dictSize;
061        int keepSizeAfter = extraSizeAfter + matchLenMax;
062        int reserveSize = Math.min(dictSize / 2 + (256 << 10), 512 << 20);
063        return keepSizeBefore + keepSizeAfter + reserveSize;
064    }
065
066    /**
067     * Gets approximate memory usage of the LZEncoder base structure and
068     * the match finder as kibibytes.
069     */
070    public static int getMemoryUsage(
071            int dictSize, int extraSizeBefore, int extraSizeAfter,
072            int matchLenMax, int mf) {
073        // Buffer size + a little extra
074        int m = getBufSize(dictSize, extraSizeBefore, extraSizeAfter,
075                           matchLenMax) / 1024 + 10;
076
077        switch (mf) {
078            case MF_HC4:
079                m += HC4.getMemoryUsage(dictSize);
080                break;
081
082            case MF_BT4:
083                m += BT4.getMemoryUsage(dictSize);
084                break;
085
086            default:
087                throw new IllegalArgumentException();
088        }
089
090        return m;
091    }
092
093    /**
094     * Creates a new LZEncoder.
095     * <p>
096     * @param       dictSize    dictionary size
097     *
098     * @param       extraSizeBefore
099     *                          number of bytes to keep available in the
100     *                          history in addition to dictSize
101     *
102     * @param       extraSizeAfter
103     *                          number of bytes that must be available
104     *                          after current position + matchLenMax
105     *
106     * @param       niceLen     if a match of at least <code>niceLen</code>
107     *                          bytes is found, be happy with it and don't
108     *                          stop looking for longer matches
109     *
110     * @param       matchLenMax don't test for matches longer than
111     *                          <code>matchLenMax</code> bytes
112     *
113     * @param       mf          match finder ID
114     *
115     * @param       depthLimit  match finder search depth limit
116     */
117    public static LZEncoder getInstance(
118            int dictSize, int extraSizeBefore, int extraSizeAfter,
119            int niceLen, int matchLenMax, int mf, int depthLimit) {
120        switch (mf) {
121            case MF_HC4:
122                return new HC4(dictSize, extraSizeBefore, extraSizeAfter,
123                               niceLen, matchLenMax, depthLimit);
124
125            case MF_BT4:
126                return new BT4(dictSize, extraSizeBefore, extraSizeAfter,
127                               niceLen, matchLenMax, depthLimit);
128        }
129
130        throw new IllegalArgumentException();
131    }
132
133    /**
134     * Creates a new LZEncoder. See <code>getInstance</code>.
135     */
136    LZEncoder(int dictSize, int extraSizeBefore, int extraSizeAfter,
137              int niceLen, int matchLenMax) {
138        buf = new byte[getBufSize(dictSize, extraSizeBefore, extraSizeAfter,
139                                  matchLenMax)];
140
141        keepSizeBefore = extraSizeBefore + dictSize;
142        keepSizeAfter = extraSizeAfter + matchLenMax;
143
144        this.matchLenMax = matchLenMax;
145        this.niceLen = niceLen;
146    }
147
148    /**
149     * Sets a preset dictionary. If a preset dictionary is wanted, this
150     * function must be called immediately after creating the LZEncoder
151     * before any data has been encoded.
152     */
153    public void setPresetDict(int dictSize, byte[] presetDict) {
154        assert !isStarted();
155        assert writePos == 0;
156
157        if (presetDict != null) {
158            // If the preset dictionary buffer is bigger than the dictionary
159            // size, copy only the tail of the preset dictionary.
160            int copySize = Math.min(presetDict.length, dictSize);
161            int offset = presetDict.length - copySize;
162            System.arraycopy(presetDict, offset, buf, 0, copySize);
163            writePos += copySize;
164            skip(copySize);
165        }
166    }
167
168    /**
169     * Moves data from the end of the buffer to the beginning, discarding
170     * old data and making space for new input.
171     */
172    private void moveWindow() {
173        // Align the move to a multiple of 16 bytes. LZMA2 needs this
174        // because it uses the lowest bits from readPos to get the
175        // alignment of the uncompressed data.
176        int moveOffset = (readPos + 1 - keepSizeBefore) & ~15;
177        int moveSize = writePos - moveOffset;
178        System.arraycopy(buf, moveOffset, buf, 0, moveSize);
179
180        readPos -= moveOffset;
181        readLimit -= moveOffset;
182        writePos -= moveOffset;
183    }
184
185    /**
186     * Copies new data into the LZEncoder's buffer.
187     */
188    public int fillWindow(byte[] in, int off, int len) {
189        assert !finishing;
190
191        // Move the sliding window if needed.
192        if (readPos >= buf.length - keepSizeAfter)
193            moveWindow();
194
195        // Try to fill the dictionary buffer. If it becomes full,
196        // some of the input bytes may be left unused.
197        if (len > buf.length - writePos)
198            len = buf.length - writePos;
199
200        System.arraycopy(in, off, buf, writePos, len);
201        writePos += len;
202
203        // Set the new readLimit but only if there's enough data to allow
204        // encoding of at least one more byte.
205        if (writePos >= keepSizeAfter)
206            readLimit = writePos - keepSizeAfter;
207
208        // After flushing or setting a preset dictionary there may be pending
209        // data that hasn't been ran through the match finder yet.
210        //
211        // NOTE: The test for readLimit is only to avoid wasting time
212        // if we get very little new input and thus readLimit wasn't
213        // increased above.
214        if (pendingSize > 0 && readPos < readLimit) {
215            readPos -= pendingSize;
216            int oldPendingSize = pendingSize;
217            pendingSize = 0;
218            skip(oldPendingSize);
219            assert pendingSize < oldPendingSize;
220        }
221
222        // Tell the caller how much input we actually copied into
223        // the dictionary.
224        return len;
225    }
226
227    /**
228     * Returns true if at least one byte has already been run through
229     * the match finder.
230     */
231    public boolean isStarted() {
232        return readPos != -1;
233    }
234
235    /**
236     * Marks that all the input needs to be made available in
237     * the encoded output.
238     */
239    public void setFlushing() {
240        readLimit = writePos - 1;
241    }
242
243    /**
244     * Marks that there is no more input remaining. The read position
245     * can be advanced until the end of the data.
246     */
247    public void setFinishing() {
248        readLimit = writePos - 1;
249        finishing = true;
250    }
251
252    /**
253     * Tests if there is enough input available to let the caller encode
254     * at least one more byte.
255     */
256    public boolean hasEnoughData(int alreadyReadLen) {
257        return readPos - alreadyReadLen < readLimit;
258    }
259
260    public void copyUncompressed(OutputStream out, int backward, int len)
261            throws IOException {
262        out.write(buf, readPos + 1 - backward, len);
263    }
264
265    /**
266     * Get the number of bytes available, including the current byte.
267     * <p>
268     * Note that the result is undefined if <code>getMatches</code> or
269     * <code>skip</code> hasn't been called yet and no preset dictionary
270     * is being used.
271     */
272    public int getAvail() {
273        assert isStarted();
274        return writePos - readPos;
275    }
276
277    /**
278     * Gets the lowest four bits of the absolute offset of the current byte.
279     * Bits other than the lowest four are undefined.
280     */
281    public int getPos() {
282        return readPos;
283    }
284
285    /**
286     * Gets the byte from the given backward offset.
287     * <p>
288     * The current byte is at <code>0</code>, the previous byte
289     * at <code>1</code> etc. To get a byte at zero-based distance,
290     * use <code>getByte(dist + 1)<code>.
291     * <p>
292     * This function is equivalent to <code>getByte(0, backward)</code>.
293     */
294    public int getByte(int backward) {
295        return buf[readPos - backward] & 0xFF;
296    }
297
298    /**
299     * Gets the byte from the given forward minus backward offset.
300     * The forward offset is added to the current position. This lets
301     * one read bytes ahead of the current byte.
302     */
303    public int getByte(int forward, int backward) {
304        return buf[readPos + forward - backward] & 0xFF;
305    }
306
307    /**
308     * Get the length of a match at the given distance.
309     *
310     * @param       dist        zero-based distance of the match to test
311     * @param       lenLimit    don't test for a match longer than this
312     *
313     * @return      length of the match; it is in the range [0, lenLimit]
314     */
315    public int getMatchLen(int dist, int lenLimit) {
316        int backPos = readPos - dist - 1;
317        int len = 0;
318
319        while (len < lenLimit && buf[readPos + len] == buf[backPos + len])
320            ++len;
321
322        return len;
323    }
324
325    /**
326     * Get the length of a match at the given distance and forward offset.
327     *
328     * @param       forward     forward offset
329     * @param       dist        zero-based distance of the match to test
330     * @param       lenLimit    don't test for a match longer than this
331     *
332     * @return      length of the match; it is in the range [0, lenLimit]
333     */
334    public int getMatchLen(int forward, int dist, int lenLimit) {
335        int curPos = readPos + forward;
336        int backPos = curPos - dist - 1;
337        int len = 0;
338
339        while (len < lenLimit && buf[curPos + len] == buf[backPos + len])
340            ++len;
341
342        return len;
343    }
344
345    /**
346     * Verifies that the matches returned by the match finder are valid.
347     * This is meant to be used in an assert statement. This is totally
348     * useless for actual encoding since match finder's results should
349     * naturally always be valid if it isn't broken.
350     *
351     * @param       matches     return value from <code>getMatches</code>
352     *
353     * @return      true if matches are valid, false if match finder is broken
354     */
355    public boolean verifyMatches(Matches matches) {
356        int lenLimit = Math.min(getAvail(), matchLenMax);
357
358        for (int i = 0; i < matches.count; ++i)
359            if (getMatchLen(matches.dist[i], lenLimit) != matches.len[i])
360                return false;
361
362        return true;
363    }
364
365    /**
366     * Moves to the next byte, checks if there is enough input available,
367     * and returns the amount of input available.
368     *
369     * @param       requiredForFlushing
370     *                          minimum number of available bytes when
371     *                          flushing; encoding may be continued with
372     *                          new input after flushing
373     * @param       requiredForFinishing
374     *                          minimum number of available bytes when
375     *                          finishing; encoding must not be continued
376     *                          after finishing or the match finder state
377     *                          may be corrupt
378     *
379     * @return      the number of bytes available or zero if there
380     *              is not enough input available
381     */
382    int movePos(int requiredForFlushing, int requiredForFinishing) {
383        assert requiredForFlushing >= requiredForFinishing;
384
385        ++readPos;
386        int avail = writePos - readPos;
387
388        if (avail < requiredForFlushing) {
389            if (avail < requiredForFinishing || !finishing) {
390                ++pendingSize;
391                avail = 0;
392            }
393        }
394
395        return avail;
396    }
397
398    /**
399     * Runs match finder for the next byte and returns the matches found.
400     */
401    public abstract Matches getMatches();
402
403    /**
404     * Skips the given number of bytes in the match finder.
405     */
406    public abstract void skip(int len);
407}
