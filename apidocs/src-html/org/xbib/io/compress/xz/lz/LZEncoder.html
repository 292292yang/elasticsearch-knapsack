002    package org.xbib.io.compress.xz.lz;
003    
004    import java.io.IOException;
005    import java.io.OutputStream;
006    
007    public abstract class LZEncoder {
008        public static final int MF_HC4 = 0x04;
009        public static final int MF_BT4 = 0x14;
010    
011        /**
012         * Number of bytes to keep available before the current byte
013         * when moving the LZ window.
014         */
015        private final int keepSizeBefore;
016    
017        /**
018         * Number of bytes that must be available, the current byte included,
019         * to make hasEnoughInput return true. Flushing and finishing are
020         * naturally exceptions to this since there cannot be any data after
021         * the end of the uncompressed input.
022         */
023        private final int keepSizeAfter;
024    
025        final int matchLenMax;
026        final int niceLen;
027    
028        final byte[] buf;
029    
030        int readPos = -1;
031        private int readLimit = -1;
032        private boolean finishing = false;
033        private int writePos = 0;
034        private int pendingSize = 0;
035    
036        static void normalize(int[] positions, int normalizationOffset) {
037            for (int i = 0; i < positions.length; ++i) {
038                if (positions[i] <= normalizationOffset) {
039                    positions[i] = 0;
040                } else {
041                    positions[i] -= normalizationOffset;
042                }
043            }
044        }
045    
046        /**
047         * Gets the size of the LZ window buffer that needs to be allocated.
048         */
049        private static int getBufSize(
050                int dictSize, int extraSizeBefore, int extraSizeAfter,
051                int matchLenMax) {
052            int keepSizeBefore = extraSizeBefore + dictSize;
053            int keepSizeAfter = extraSizeAfter + matchLenMax;
054            int reserveSize = Math.min(dictSize / 2 + (256 << 10), 512 << 20);
055            return keepSizeBefore + keepSizeAfter + reserveSize;
056        }
057    
058        /**
059         * Gets approximate memory usage of the LZEncoder base structure and
060         * the match finder as kibibytes.
061         */
062        public static int getMemoryUsage(
063                int dictSize, int extraSizeBefore, int extraSizeAfter,
064                int matchLenMax, int mf) {
065            // Buffer size + a little extra
066            int m = getBufSize(dictSize, extraSizeBefore, extraSizeAfter,
067                    matchLenMax) / 1024 + 10;
068    
069            switch (mf) {
070                case MF_HC4:
071                    m += HC4.getMemoryUsage(dictSize);
072                    break;
073    
074                case MF_BT4:
075                    m += BT4.getMemoryUsage(dictSize);
076                    break;
077    
078                default:
079                    throw new IllegalArgumentException();
080            }
081    
082            return m;
083        }
084    
085        /**
086         * Creates a new LZEncoder.
087         * <p/>
088         *
089         * @param dictSize        dictionary size
090         * @param extraSizeBefore number of bytes to keep available in the
091         *                        history in addition to dictSize
092         * @param extraSizeAfter  number of bytes that must be available
093         *                        after current position + matchLenMax
094         * @param niceLen         if a match of at least <code>niceLen</code>
095         *                        bytes is found, be happy with it and don't
096         *                        stop looking for longer matches
097         * @param matchLenMax     don't test for matches longer than
098         *                        <code>matchLenMax</code> bytes
099         * @param mf              match finder ID
100         * @param depthLimit      match finder search depth limit
101         */
102        public static LZEncoder getInstance(
103                int dictSize, int extraSizeBefore, int extraSizeAfter,
104                int niceLen, int matchLenMax, int mf, int depthLimit) {
105            switch (mf) {
106                case MF_HC4:
107                    return new HC4(dictSize, extraSizeBefore, extraSizeAfter,
108                            niceLen, matchLenMax, depthLimit);
109    
110                case MF_BT4:
111                    return new BT4(dictSize, extraSizeBefore, extraSizeAfter,
112                            niceLen, matchLenMax, depthLimit);
113            }
114    
115            throw new IllegalArgumentException();
116        }
117    
118        /**
119         * Creates a new LZEncoder. See <code>getInstance</code>.
120         */
121        LZEncoder(int dictSize, int extraSizeBefore, int extraSizeAfter,
122                  int niceLen, int matchLenMax) {
123            buf = new byte[getBufSize(dictSize, extraSizeBefore, extraSizeAfter,
124                    matchLenMax)];
125    
126            keepSizeBefore = extraSizeBefore + dictSize;
127            keepSizeAfter = extraSizeAfter + matchLenMax;
128    
129            this.matchLenMax = matchLenMax;
130            this.niceLen = niceLen;
131        }
132    
133        /**
134         * Sets a preset dictionary. If a preset dictionary is wanted, this
135         * function must be called immediately after creating the LZEncoder
136         * before any data has been encoded.
137         */
138        public void setPresetDict(int dictSize, byte[] presetDict) {
139            assert !isStarted();
140            assert writePos == 0;
141    
142            if (presetDict != null) {
143                // If the preset dictionary buffer is bigger than the dictionary
144                // size, copy only the tail of the preset dictionary.
145                int copySize = Math.min(presetDict.length, dictSize);
146                int offset = presetDict.length - copySize;
147                System.arraycopy(presetDict, offset, buf, 0, copySize);
148                writePos += copySize;
149                skip(copySize);
150            }
151        }
152    
153        /**
154         * Moves data from the end of the buffer to the beginning, discarding
155         * old data and making space for new input.
156         */
157        private void moveWindow() {
158            // Align the move to a multiple of 16 bytes. LZMA2 needs this
159            // because it uses the lowest bits from readPos to get the
160            // alignment of the uncompressed data.
161            int moveOffset = (readPos + 1 - keepSizeBefore) & ~15;
162            int moveSize = writePos - moveOffset;
163            System.arraycopy(buf, moveOffset, buf, 0, moveSize);
164    
165            readPos -= moveOffset;
166            readLimit -= moveOffset;
167            writePos -= moveOffset;
168        }
169    
170        /**
171         * Copies new data into the LZEncoder's buffer.
172         */
173        public int fillWindow(byte[] in, int off, int len) {
174            assert !finishing;
175    
176            // Move the sliding window if needed.
177            if (readPos >= buf.length - keepSizeAfter) {
178                moveWindow();
179            }
180    
181            // Try to fill the dictionary buffer. If it becomes full,
182            // some of the input bytes may be left unused.
183            if (len > buf.length - writePos) {
184                len = buf.length - writePos;
185            }
186    
187            System.arraycopy(in, off, buf, writePos, len);
188            writePos += len;
189    
190            // Set the new readLimit but only if there's enough data to allow
191            // encoding of at least one more byte.
192            if (writePos >= keepSizeAfter) {
193                readLimit = writePos - keepSizeAfter;
194            }
195    
196            // After flushing or setting a preset dictionary there may be pending
197            // data that hasn't been ran through the match finder yet.
198            //
199            // NOTE: The test for readLimit is only to avoid wasting time
200            // if we get very little new input and thus readLimit wasn't
201            // increased above.
202            if (pendingSize > 0 && readPos < readLimit) {
203                readPos -= pendingSize;
204                int oldPendingSize = pendingSize;
205                pendingSize = 0;
206                skip(oldPendingSize);
207                assert pendingSize < oldPendingSize;
208            }
209    
210            // Tell the caller how much input we actually copied into
211            // the dictionary.
212            return len;
213        }
214    
215        /**
216         * Returns true if at least one byte has already been run through
217         * the match finder.
218         */
219        public boolean isStarted() {
220            return readPos != -1;
221        }
222    
223        /**
224         * Marks that all the input needs to be made available in
225         * the encoded output.
226         */
227        public void setFlushing() {
228            readLimit = writePos - 1;
229        }
230    
231        /**
232         * Marks that there is no more input remaining. The read position
233         * can be advanced until the end of the data.
234         */
235        public void setFinishing() {
236            readLimit = writePos - 1;
237            finishing = true;
238        }
239    
240        /**
241         * Tests if there is enough input available to let the caller encode
242         * at least one more byte.
243         */
244        public boolean hasEnoughData(int alreadyReadLen) {
245            return readPos - alreadyReadLen < readLimit;
246        }
247    
248        public void copyUncompressed(OutputStream out, int backward, int len)
249                throws IOException {
250            out.write(buf, readPos + 1 - backward, len);
251        }
252    
253        /**
254         * Get the number of bytes available, including the current byte.
255         * <p/>
256         * Note that the result is undefined if <code>getMatches</code> or
257         * <code>skip</code> hasn't been called yet and no preset dictionary
258         * is being used.
259         */
260        public int getAvail() {
261            assert isStarted();
262            return writePos - readPos;
263        }
264    
265        /**
266         * Gets the lowest four bits of the absolute offset of the current byte.
267         * Bits other than the lowest four are undefined.
268         */
269        public int getPos() {
270            return readPos;
271        }
272    
273        /**
274         * Gets the byte from the given backward offset.
275         * <p/>
276         * The current byte is at <code>0</code>, the previous byte
277         * at <code>1</code> etc. To get a byte at zero-based distance,
278         * use <code>getByte(dist + 1)<code>.
279         * <p/>
280         * This function is equivalent to <code>getByte(0, backward)</code>.
281         */
282        public int getByte(int backward) {
283            return buf[readPos - backward] & 0xFF;
284        }
285    
286        /**
287         * Gets the byte from the given forward minus backward offset.
288         * The forward offset is added to the current position. This lets
289         * one read bytes ahead of the current byte.
290         */
291        public int getByte(int forward, int backward) {
292            return buf[readPos + forward - backward] & 0xFF;
293        }
294    
295        /**
296         * Get the length of a match at the given distance.
297         *
298         * @param dist     zero-based distance of the match to test
299         * @param lenLimit don't test for a match longer than this
300         * @return length of the match; it is in the range [0, lenLimit]
301         */
302        public int getMatchLen(int dist, int lenLimit) {
303            int backPos = readPos - dist - 1;
304            int len = 0;
305    
306            while (len < lenLimit && buf[readPos + len] == buf[backPos + len]) {
307                ++len;
308            }
309    
310            return len;
311        }
312    
313        /**
314         * Get the length of a match at the given distance and forward offset.
315         *
316         * @param forward  forward offset
317         * @param dist     zero-based distance of the match to test
318         * @param lenLimit don't test for a match longer than this
319         * @return length of the match; it is in the range [0, lenLimit]
320         */
321        public int getMatchLen(int forward, int dist, int lenLimit) {
322            int curPos = readPos + forward;
323            int backPos = curPos - dist - 1;
324            int len = 0;
325    
326            while (len < lenLimit && buf[curPos + len] == buf[backPos + len]) {
327                ++len;
328            }
329    
330            return len;
331        }
332    
333        /**
334         * Verifies that the matches returned by the match finder are valid.
335         * This is meant to be used in an assert statement. This is totally
336         * useless for actual encoding since match finder's results should
337         * naturally always be valid if it isn't broken.
338         *
339         * @param matches return value from <code>getMatches</code>
340         * @return true if matches are valid, false if match finder is broken
341         */
342        public boolean verifyMatches(Matches matches) {
343            int lenLimit = Math.min(getAvail(), matchLenMax);
344    
345            for (int i = 0; i < matches.count; ++i) {
346                if (getMatchLen(matches.dist[i], lenLimit) != matches.len[i]) {
347                    return false;
348                }
349            }
350    
351            return true;
352        }
353    
354        /**
355         * Moves to the next byte, checks if there is enough input available,
356         * and returns the amount of input available.
357         *
358         * @param requiredForFlushing  minimum number of available bytes when
359         *                             flushing; encoding may be continued with
360         *                             new input after flushing
361         * @param requiredForFinishing minimum number of available bytes when
362         *                             finishing; encoding must not be continued
363         *                             after finishing or the match finder state
364         *                             may be corrupt
365         * @return the number of bytes available or zero if there
366         * is not enough input available
367         */
368        int movePos(int requiredForFlushing, int requiredForFinishing) {
369            assert requiredForFlushing >= requiredForFinishing;
370    
371            ++readPos;
372            int avail = writePos - readPos;
373    
374            if (avail < requiredForFlushing) {
375                if (avail < requiredForFinishing || !finishing) {
376                    ++pendingSize;
377                    avail = 0;
378                }
379            }
380    
381            return avail;
382        }
383    
384        /**
385         * Runs match finder for the next byte and returns the matches found.
386         */
387        public abstract Matches getMatches();
388    
389        /**
390         * Skips the given number of bytes in the match finder.
391         */
392        public abstract void skip(int len);
393    }
