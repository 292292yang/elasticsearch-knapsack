002    package org.xbib.io.compress.xz;
003    
004    import org.xbib.io.compress.xz.lz.LZDecoder;
005    import org.xbib.io.compress.xz.lzma.LZMADecoder;
006    import org.xbib.io.compress.xz.rangecoder.RangeDecoder;
007    
008    import java.io.DataInputStream;
009    import java.io.IOException;
010    import java.io.InputStream;
011    
012    /**
013     * Decompresses a raw LZMA2 stream (no XZ headers).
014     */
015    public class LZMA2InputStream extends InputStream {
016        /**
017         * Smallest valid LZMA2 dictionary size.
018         * <p/>
019         * Very tiny dictionaries would be a performance problem, so
020         * the minimum is 4 KiB.
021         */
022        public static final int DICT_SIZE_MIN = 4096;
023    
024        /**
025         * Largest dictionary size supported by this implementation.
026         * <p/>
027         * The LZMA2 algorithm allows dictionaries up to one byte less than 4 GiB.
028         * This implementation supports only 16 bytes less than 2 GiB for raw
029         * LZMA2 streams, and for .xz files the maximum is 1.5 GiB. This
030         * limitation is due to Java using signed 32-bit integers for array
031         * indexing. The limitation shouldn't matter much in practice since so
032         * huge dictionaries are not normally used.
033         */
034        public static final int DICT_SIZE_MAX = Integer.MAX_VALUE & ~15;
035    
036        private static final int COMPRESSED_SIZE_MAX = 1 << 16;
037    
038        private DataInputStream in;
039    
040        private final LZDecoder lz;
041        private final RangeDecoder rc = new RangeDecoder(COMPRESSED_SIZE_MAX);
042        private LZMADecoder lzma;
043    
044        private int uncompressedSize = 0;
045        private boolean isLZMAChunk;
046    
047        private boolean needDictReset = true;
048        private boolean needProps = true;
049        private boolean endReached = false;
050    
051        private IOException exception = null;
052    
053        /**
054         * Gets approximate decompressor memory requirements as kibibytes for
055         * the given dictionary size.
056         *
057         * @param dictSize LZMA2 dictionary size as bytes, must be
058         *                 in the range [<code>DICT_SIZE_MIN</code>,
059         *                 <code>DICT_SIZE_MAX</code>]
060         * @return approximate memory requirements as kibibytes (KiB)
061         */
062        public static int getMemoryUsage(int dictSize) {
063            // The base state is aroudn 30-40 KiB (probabilities etc.),
064            // range decoder needs COMPRESSED_SIZE_MAX bytes for buffering,
065            // and LZ decoder needs a dictionary buffer.
066            return 40 + COMPRESSED_SIZE_MAX / 1024 + getDictSize(dictSize) / 1024;
067        }
068    
069        private static int getDictSize(int dictSize) {
070            if (dictSize < DICT_SIZE_MIN || dictSize > DICT_SIZE_MAX) {
071                throw new IllegalArgumentException(
072                        "Unsupported dictionary size " + dictSize);
073            }
074    
075            // Round dictionary size upward to a multiple of 16. This way LZMA
076            // can use LZDecoder.getPos() for calculating LZMA's posMask.
077            // Note that this check is needed only for raw LZMA2 streams; it is
078            // redundant with .xz.
079            return (dictSize + 15) & ~15;
080        }
081    
082        /**
083         * Creates a new input stream that decompresses raw LZMA2 data
084         * from <code>in</code>.
085         * <p/>
086         * The caller needs to know the dictionary size used when compressing;
087         * the dictionary size isn't stored as part of a raw LZMA2 stream.
088         * <p/>
089         * Specifying a too small dictionary size will prevent decompressing
090         * the stream. Specifying a too big dictionary is waste of memory but
091         * decompression will work.
092         * <p/>
093         * There is no need to specify a dictionary bigger than
094         * the uncompressed size of the data even if a bigger dictionary
095         * was used when compressing. If you know the uncompressed size
096         * of the data, this might allow saving some memory.
097         *
098         * @param in       input stream from which LZMA2-compressed
099         *                 data is read
100         * @param dictSize LZMA2 dictionary size as bytes, must be
101         *                 in the range [<code>DICT_SIZE_MIN</code>,
102         *                 <code>DICT_SIZE_MAX</code>]
103         */
104        public LZMA2InputStream(InputStream in, int dictSize) {
105            this(in, dictSize, null);
106        }
107    
108        /**
109         * Creates a new LZMA2 decompressor using a preset dictionary.
110         * <p/>
111         * This is like <code>LZMAInputStream(InputStream, int)</code> except
112         * that the dictionary may be initialized using a preset dictionary.
113         * If a preset dictionary was used when compressing the data, the
114         * same preset dictionary must be provided when decompressing.
115         *
116         * @param in         input stream from which LZMA2-compressed
117         *                   data is read
118         * @param dictSize   LZMA2 dictionary size as bytes, must be
119         *                   in the range [<code>DICT_SIZE_MIN</code>,
120         *                   <code>DICT_SIZE_MAX</code>]
121         * @param presetDict preset dictionary or <code>null</code>
122         *                   to use no preset dictionary
123         */
124        public LZMA2InputStream(InputStream in, int dictSize, byte[] presetDict) {
125            // Check for null because otherwise null isn't detect
126            // in this constructor.
127            if (in == null) {
128                throw new NullPointerException();
129            }
130    
131            this.in = new DataInputStream(in);
132            this.lz = new LZDecoder(getDictSize(dictSize), presetDict);
133    
134            if (presetDict != null && presetDict.length > 0) {
135                needDictReset = false;
136            }
137        }
138    
139        /**
140         * Decompresses the next byte from this input stream.
141         * <p/>
142         * Reading lots of data with <code>read()</code> from this input stream
143         * may be inefficient. Wrap it in <code>java.io.BufferedInputStream</code>
144         * if you need to read lots of data one byte at a time.
145         *
146         * @return the next decompressed byte, or <code>-1</code>
147         * to indicate the end of the compressed stream
148         * @throws CorruptedInputException
149         * @throws XZIOException           if the stream has been closed
150         * @throws EOFException            compressed input is truncated or corrupt
151         * @throws java.io.IOException     may be thrown by <code>in</code>
152         */
153        public int read() throws IOException {
154            byte[] buf = new byte[1];
155            return read(buf, 0, 1) == -1 ? -1 : (buf[0] & 0xFF);
156        }
157    
158        /**
159         * Decompresses into an array of bytes.
160         * <p/>
161         * If <code>len</code> is zero, no bytes are read and <code>0</code>
162         * is returned. Otherwise this will block until <code>len</code>
163         * bytes have been decompressed, the end of LZMA2 stream is reached,
164         * or an exception is thrown.
165         *
166         * @param buf target buffer for uncompressed data
167         * @param off start offset in <code>buf</code>
168         * @param len maximum number of uncompressed bytes to read
169         * @return number of bytes read, or <code>-1</code> to indicate
170         * the end of the compressed stream
171         * @throws CorruptedInputException
172         * @throws XZIOException           if the stream has been closed
173         * @throws EOFException            compressed input is truncated or corrupt
174         * @throws java.io.IOException     may be thrown by <code>in</code>
175         */
176        public int read(byte[] buf, int off, int len) throws IOException {
177            if (off < 0 || len < 0 || off + len < 0 || off + len > buf.length) {
178                throw new IndexOutOfBoundsException();
179            }
180    
181            if (len == 0) {
182                return 0;
183            }
184    
185            if (in == null) {
186                throw new XZIOException("Stream closed");
187            }
188    
189            if (exception != null) {
190                throw exception;
191            }
192    
193            if (endReached) {
194                return -1;
195            }
196    
197            try {
198                int size = 0;
199    
200                while (len > 0) {
201                    if (uncompressedSize == 0) {
202                        decodeChunkHeader();
203                        if (endReached) {
204                            return size == 0 ? -1 : size;
205                        }
206                    }
207    
208                    int copySizeMax = Math.min(uncompressedSize, len);
209    
210                    if (!isLZMAChunk) {
211                        lz.copyUncompressed(in, copySizeMax);
212                    } else {
213                        lz.setLimit(copySizeMax);
214                        lzma.decode();
215                    }
216    
217                    int copiedSize = lz.flush(buf, off);
218                    off += copiedSize;
219                    len -= copiedSize;
220                    size += copiedSize;
221                    uncompressedSize -= copiedSize;
222    
223                    if (uncompressedSize == 0) {
224                        if (!rc.isFinished() || lz.hasPending()) {
225                            throw new CorruptedInputException();
226                        }
227                    }
228                }
229    
230                return size;
231    
232            } catch (IOException e) {
233                exception = e;
234                throw e;
235            }
236        }
237    
238        private void decodeChunkHeader() throws IOException {
239            int control = in.readUnsignedByte();
240    
241            if (control == 0x00) {
242                endReached = true;
243                return;
244            }
245    
246            if (control >= 0xE0 || control == 0x01) {
247                needProps = true;
248                needDictReset = false;
249                lz.reset();
250            } else if (needDictReset) {
251                throw new CorruptedInputException();
252            }
253    
254            if (control >= 0x80) {
255                isLZMAChunk = true;
256    
257                uncompressedSize = (control & 0x1F) << 16;
258                uncompressedSize += in.readUnsignedShort() + 1;
259    
260                int compressedSize = in.readUnsignedShort() + 1;
261    
262                if (control >= 0xC0) {
263                    needProps = false;
264                    decodeProps();
265    
266                } else if (needProps) {
267                    throw new CorruptedInputException();
268    
269                } else if (control >= 0xA0) {
270                    lzma.reset();
271                }
272    
273                rc.prepareInputBuffer(in, compressedSize);
274    
275            } else if (control > 0x02) {
276                throw new CorruptedInputException();
277    
278            } else {
279                isLZMAChunk = false;
280                uncompressedSize = in.readUnsignedShort() + 1;
281            }
282        }
283    
284        private void decodeProps() throws IOException {
285            int props = in.readUnsignedByte();
286    
287            if (props > (4 * 5 + 4) * 9 + 8) {
288                throw new CorruptedInputException();
289            }
290    
291            int pb = props / (9 * 5);
292            props -= pb * 9 * 5;
293            int lp = props / 9;
294            int lc = props - lp * 9;
295    
296            if (lc + lp > 4) {
297                throw new CorruptedInputException();
298            }
299    
300            lzma = new LZMADecoder(lz, rc, lc, lp, pb);
301        }
302    
303        /**
304         * Returns the number of uncompressed bytes that can be read
305         * without blocking. The value is returned with an assumption
306         * that the compressed input data will be valid. If the compressed
307         * data is corrupt, <code>CorruptedInputException</code> may get
308         * thrown before the number of bytes claimed to be available have
309         * been read from this input stream.
310         * <p/>
311         * In LZMAInputStream, the return value will be non-zero when the
312         * decompressor is in the middle of an LZMA2 chunk. The return value
313         * will then be the number of uncompressed bytes remaining from that
314         * chunk.
315         *
316         * @return the number of uncompressed bytes that can be read
317         * without blocking
318         */
319        public int available() throws IOException {
320            if (in == null) {
321                throw new XZIOException("Stream closed");
322            }
323    
324            if (exception != null) {
325                throw exception;
326            }
327    
328            return uncompressedSize;
329        }
330    
331        /**
332         * Closes the stream and calls <code>in.close()</code>.
333         * If the stream was already closed, this does nothing.
334         *
335         * @throws java.io.IOException if thrown by <code>in.close()</code>
336         */
337        public void close() throws IOException {
338            if (in != null) {
339                try {
340                    in.close();
341                } finally {
342                    in = null;
343                }
344            }
345        }
