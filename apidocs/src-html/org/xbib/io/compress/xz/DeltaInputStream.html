010package org.xbib.io.compress.xz;
011
012import java.io.InputStream;
013import java.io.IOException;
014import org.xbib.io.compress.xz.delta.DeltaDecoder;
015
016/**
017 * Decodes raw Delta-filtered data (no XZ headers).
018 * <p>
019 * The delta filter doesn't change the size of the data and thus it
020 * cannot have an end-of-payload marker. It will simply decode until
021 * its input stream indicates end of input.
022 */
023public class DeltaInputStream extends InputStream {
024    /**
025     * Smallest supported delta calculation distance.
026     */
027    public static final int DISTANCE_MIN = 1;
028
029    /**
030     * Largest supported delta calculation distance.
031     */
032    public static final int DISTANCE_MAX = 256;
033
034    private InputStream in;
035    private final DeltaDecoder delta;
036
037    private IOException exception = null;
038
039    /**
040     * Creates a new Delta decoder with the given delta calculation distance.
041     *
042     * @param       in          input stream from which Delta filtered data
043     *                          is read
044     *
045     * @param       distance    delta calculation distance, must be in the
046     *                          range [<code>DISTANCE_MIN</code>,
047     *                          <code>DISTANCE_MAX</code>]
048     */
049    public DeltaInputStream(InputStream in, int distance) {
050        // Check for null because otherwise null isn't detect
051        // in this constructor.
052        if (in == null)
053            throw new NullPointerException();
054
055        this.in = in;
056        this.delta = new DeltaDecoder(distance);
057    }
058
059    /**
060     * Decode the next byte from this input stream.
061     *
062     * @return      the next decoded byte, or <code>-1</code> to indicate
063     *              the end of input on the input stream <code>in</code>
064     *
065     * @throws      IOException may be thrown by <code>in</code>
066     */
067    public int read() throws IOException {
068        byte[] buf = new byte[1];
069        return read(buf, 0, 1) == -1 ? -1 : (buf[0] & 0xFF);
070    }
071
072    /**
073     * Decode into an array of bytes.
074     * <p>
075     * This calls <code>in.read(buf, off, len)</code> and defilters the
076     * returned data.
077     *
078     * @param       buf         target buffer for decoded data
079     * @param       off         start offset in <code>buf</code>
080     * @param       len         maximum number of bytes to read
081     *
082     * @return      number of bytes read, or <code>-1</code> to indicate
083     *              the end of the input stream <code>in</code>
084     *
085     * @throws      XZIOException if the stream has been closed
086     *
087     * @throws      IOException may be thrown by underlaying input
088     *                          stream <code>in</code>
089     */
090    public int read(byte[] buf, int off, int len) throws IOException {
091        if (len == 0)
092            return 0;
093
094        if (in == null)
095            throw new XZIOException("Stream closed");
096
097        if (exception != null)
098            throw exception;
099
100        int size;
101        try {
102            size = in.read(buf, off, len);
103        } catch (IOException e) {
104            exception = e;
105            throw e;
106        }
107
108        if (size == -1)
109            return -1;
110
111        delta.decode(buf, off, size);
112        return size;
113    }
114
115    /**
116     * Calls <code>in.available()</code>.
117     *
118     * @return      the value returned by <code>in.available()</code>
119     */
120    public int available() throws IOException {
121        if (in == null)
122            throw new XZIOException("Stream closed");
123
124        if (exception != null)
125            throw exception;
126
127        return in.available();
128    }
129
130    /**
131     * Closes the stream and calls <code>in.close()</code>.
132     * If the stream was already closed, this does nothing.
133     *
134     * @throws  IOException if thrown by <code>in.close()</code>
135     */
136    public void close() throws IOException {
137        if (in != null) {
138            try {
139                in.close();
140            } finally {
141                in = null;
142            }
143        }
144    }
145}
