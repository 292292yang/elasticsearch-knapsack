010    /**
011     * LZMA2 compression options.
012     * <p/>
013     * While this allows setting the LZMA2 compression options in detail,
014     * often you only need <code>LZMA2Options()</code> or
015     * <code>LZMA2Options(int)</code>.
016     */
017    public class LZMA2Options extends FilterOptions {
018        /**
019         * Minimum valid compression preset level is 0.
020         */
021        public static final int PRESET_MIN = 0;
022    
023        /**
024         * Maximum valid compression preset level is 9.
025         */
026        public static final int PRESET_MAX = 9;
027    
028        /**
029         * Default compression preset level is 6.
030         */
031        public static final int PRESET_DEFAULT = 6;
032    
033        /**
034         * Minimum dictionary size is 4 KiB.
035         */
036        public static final int DICT_SIZE_MIN = 4096;
037    
038        /**
039         * Maximum dictionary size for compression is 768 MiB.
040         * <p/>
041         * The decompressor supports bigger dictionaries, up to almost 2 GiB.
042         * With HC4 the encoder would support dictionaries bigger than 768 MiB.
043         * The 768 MiB limit comes from the current implementation of BT4 where
044         * we would otherwise hit the limits of signed ints in array indexing.
045         * <p/>
046         * If you really need bigger dictionary for decompression,
047         * use {@link LZMA2InputStream} directly.
048         */
049        public static final int DICT_SIZE_MAX = 768 << 20;
050    
051        /**
052         * The default dictionary size is 8 MiB.
053         */
054        public static final int DICT_SIZE_DEFAULT = 8 << 20;
055    
056        /**
057         * Maximum value for lc + lp is 4.
058         */
059        public static final int LC_LP_MAX = 4;
060    
061        /**
062         * The default number of literal context bits is 3.
063         */
064        public static final int LC_DEFAULT = 3;
065    
066        /**
067         * The default number of literal position bits is 0.
068         */
069        public static final int LP_DEFAULT = 0;
070    
071        /**
072         * Maximum value for pb is 4.
073         */
074        public static final int PB_MAX = 4;
075    
076        /**
077         * The default number of position bits is 2.
078         */
079        public static final int PB_DEFAULT = 2;
080    
081        /**
082         * Compression mode: uncompressed.
083         * The data is wrapped into a LZMA2 stream without compression.
084         */
085        public static final int MODE_UNCOMPRESSED = 0;
086    
087        /**
088         * Compression mode: fast.
089         * This is usually combined with a hash chain match finder.
090         */
091        public static final int MODE_FAST = LZMAEncoder.MODE_FAST;
092    
093        /**
094         * Compression mode: normal.
095         * This is usually combined with a binary tree match finder.
096         */
097        public static final int MODE_NORMAL = LZMAEncoder.MODE_NORMAL;
098    
099        /**
100         * Minimum value for <code>niceLen</code> is 8.
101         */
102        public static final int NICE_LEN_MIN = 8;
103    
104        /**
105         * Maximum value for <code>niceLen</code> is 273.
106         */
107        public static final int NICE_LEN_MAX = 273;
108    
109        /**
110         * Match finder: Hash Chain 2-3-4
111         */
112        public static final int MF_HC4 = LZEncoder.MF_HC4;
113    
114        /**
115         * Match finder: Binary tree 2-3-4
116         */
117        public static final int MF_BT4 = LZEncoder.MF_BT4;
118    
119        private static final int[] presetToDictSize = {
120                1 << 18, 1 << 20, 1 << 21, 1 << 22, 1 << 22,
121                1 << 23, 1 << 23, 1 << 24, 1 << 25, 1 << 26};
122    
123        private static final int[] presetToDepthLimit = {4, 8, 24, 48};
124    
125        private int dictSize;
126        private byte[] presetDict = null;
127        private int lc;
128        private int lp;
129        private int pb;
130        private int mode;
131        private int niceLen;
132        private int mf;
133        private int depthLimit;
134    
135        /**
136         * Creates new LZMA2 options and sets them to the default values.
137         * This is equivalent to <code>LZMA2Options(PRESET_DEFAULT)</code>.
138         */
139        public LZMA2Options() {
140            try {
141                setPreset(PRESET_DEFAULT);
142            } catch (UnsupportedOptionsException e) {
143                assert false;
144                throw new RuntimeException();
145            }
146        }
147    
148        /**
149         * Creates new LZMA2 options and sets them to the given preset.
150         *
151         * @throws UnsupportedOptionsException <code>preset</code> is not supported
152         */
153        public LZMA2Options(int preset) throws UnsupportedOptionsException {
154            setPreset(preset);
155        }
156    
157        /**
158         * Creates new LZMA2 options and sets them to the given custom values.
159         *
160         * @throws UnsupportedOptionsException unsupported options were specified
161         */
162        public LZMA2Options(int dictSize, int lc, int lp, int pb, int mode,
163                            int niceLen, int mf, int depthLimit)
164                throws UnsupportedOptionsException {
165            setDictSize(dictSize);
166            setLcLp(lc, lp);
167            setPb(pb);
168            setMode(mode);
169            setNiceLen(niceLen);
170            setMatchFinder(mf);
171            setDepthLimit(depthLimit);
172        }
173    
174        /**
175         * Sets the compression options to the given preset.
176         * <p/>
177         * The presets 0-3 are fast presets with medium compression.
178         * The presets 4-6 are fairly slow presets with high compression.
179         * The default preset (<code>PRESET_DEFAULT</code>) is 6.
180         * <p/>
181         * The presets 7-9 are like the preset 6 but use bigger dictionaries
182         * and have higher compressor and decompressor memory requirements.
183         * Unless the uncompressed size of the file exceeds 8&nbsp;MiB,
184         * 16&nbsp;MiB, or 32&nbsp;MiB, it is waste of memory to use the
185         * presets 7, 8, or 9, respectively.
186         *
187         * @throws UnsupportedOptionsException <code>preset</code> is not supported
188         */
189        public void setPreset(int preset) throws UnsupportedOptionsException {
190            if (preset < 0 || preset > 9) {
191                throw new UnsupportedOptionsException(
192                        "Unsupported preset: " + preset);
193            }
194    
195            lc = LC_DEFAULT;
196            lp = LP_DEFAULT;
197            pb = PB_DEFAULT;
198            dictSize = presetToDictSize[preset];
199    
200            if (preset <= 3) {
201                mode = MODE_FAST;
202                mf = MF_HC4;
203                niceLen = preset <= 1 ? 128 : NICE_LEN_MAX;
204                depthLimit = presetToDepthLimit[preset];
205            } else {
206                mode = MODE_NORMAL;
207                mf = MF_BT4;
208                niceLen = (preset == 4) ? 16 : (preset == 5) ? 32 : 64;
209                depthLimit = 0;
210            }
211        }
212    
213        /**
214         * Sets the dictionary size in bytes.
215         * <p/>
216         * The dictionary (or history buffer) holds the most recently seen
217         * uncompressed data. Bigger dictionary usually means better compression.
218         * However, using a dictioanary bigger than the size of the uncompressed
219         * data is waste of memory.
220         * <p/>
221         * Any value in the range [DICT_SIZE_MIN, DICT_SIZE_MAX] is valid,
222         * but sizes of 2^n and 2^n&nbsp;+&nbsp;2^(n-1) bytes are somewhat
223         * recommended.
224         *
225         * @throws UnsupportedOptionsException <code>dictSize</code> is not supported
226         */
227        public void setDictSize(int dictSize) throws UnsupportedOptionsException {
228            if (dictSize < DICT_SIZE_MIN) {
229                throw new UnsupportedOptionsException(
230                        "LZMA2 dictionary size must be at least 4 KiB: "
231                                + dictSize + " B");
232            }
233    
234            if (dictSize > DICT_SIZE_MAX) {
235                throw new UnsupportedOptionsException(
236                        "LZMA2 dictionary size must not exceed "
237                                + (DICT_SIZE_MAX >> 20) + " MiB: " + dictSize + " B");
238            }
239    
240            this.dictSize = dictSize;
241        }
242    
243        /**
244         * Gets the dictionary size in bytes.
245         */
246        public int getDictSize() {
247            return dictSize;
248        }

250        /**
251         * Sets a preset dictionary. Use null to disable the use of
252         * a preset dictionary. By default there is no preset dictionary.
253         * <p/>
254         * <b>The .xz format doesn't support a preset dictionary for now.
255         * Do not set a preset dictionary unless you use raw LZMA2.</b>
256         * <p/>
257         * Preset dictionary can be useful when compressing many similar,
258         * relatively small chunks of data independently from each other.
259         * A preset dictionary should contain typical strings that occur in
260         * the files being compressed. The most probable strings should be
261         * near the end of the preset dictionary. The preset dictionary used
262         * for compression is also needed for decompression.
263         */
264        public void setPresetDict(byte[] presetDict) {
265            this.presetDict = presetDict;
266        }
267    
268        /**
269         * Gets the preset dictionary.
270         */
271        public byte[] getPresetDict() {
272            return presetDict;
273        }

275        /**
276         * Sets the number of literal context bits and literal position bits.
277         * <p/>
278         * The sum of <code>lc</code> and <code>lp</code> is limited to 4.
279         * Trying to exceed it will throw an exception. This function lets
280         * you change both at the same time.
281         *
282         * @throws UnsupportedOptionsException <code>lc</code> and <code>lp</code>
283         *                                     are invalid
284         */
285        public void setLcLp(int lc, int lp) throws UnsupportedOptionsException {
286            if (lc < 0 || lp < 0 || lc > LC_LP_MAX || lp > LC_LP_MAX
287                    || lc + lp > LC_LP_MAX) {
288                throw new UnsupportedOptionsException(
289                        "lc + lp must not exceed " + LC_LP_MAX + ": "
290                                + lc + " + " + lp);
291            }
292    
293            this.lc = lc;
294            this.lp = lp;
295        }
296    
297        /**
298         * Sets the number of literal context bits.
299         * <p/>
300         * All bytes that cannot be encoded as matches are encoded as literals.
301         * That is, literals are simply 8-bit bytes that are encoded one at
302         * a time.
303         * <p/>
304         * The literal coding makes an assumption that the highest <code>lc</code>
305         * bits of the previous uncompressed byte correlate with the next byte.
306         * For example, in typical English text, an upper-case letter is often
307         * followed by a lower-case letter, and a lower-case letter is usually
308         * followed by another lower-case letter. In the US-ASCII character set,
309         * the highest three bits are 010 for upper-case letters and 011 for
310         * lower-case letters. When <code>lc</code> is at least 3, the literal
311         * coding can take advantage of this property in the  uncompressed data.
312         * <p/>
313         * The default value (3) is usually good. If you want maximum compression,
314         * try <code>setLc(4)</code>. Sometimes it helps a little, and sometimes it
315         * makes compression worse. If it makes it worse, test for example
316         * <code>setLc(2)</code> too.
317         *
318         * @throws UnsupportedOptionsException <code>lc</code> is invalid, or the sum
319         *                                     of <code>lc</code> and <code>lp</code>
320         *                                     exceed LC_LP_MAX
321         */
322        public void setLc(int lc) throws UnsupportedOptionsException {
323            setLcLp(lc, lp);
324        }
325    
326        /**
327         * Sets the number of literal position bits.
328         * <p/>
329         * This affets what kind of alignment in the uncompressed data is
330         * assumed when encoding literals. See {@link #setPb(int) setPb} for
331         * more information about alignment.
332         *
333         * @throws UnsupportedOptionsException <code>lp</code> is invalid, or the sum
334         *                                     of <code>lc</code> and <code>lp</code>
335         *                                     exceed LC_LP_MAX
336         */
337        public void setLp(int lp) throws UnsupportedOptionsException {
338            setLcLp(lc, lp);
339        }
340    
341        /**
342         * Gets the number of literal context bits.
343         */
344        public int getLc() {
345            return lc;
346        }
347    
348        /**
349         * Gets the number of literal position bits.
350         */
351        public int getLp() {
352            return lp;
353        }
354    
355        /**
356         * Sets the number of position bits.
357         * <p/>
358         * This affects what kind of alignment in the uncompressed data is
359         * assumed in general. The default (2) means four-byte alignment
360         * (2^<code>pb</code> = 2^2 = 4), which is often a good choice when
361         * there's no better guess.
362         * <p/>
363         * When the alignment is known, setting the number of position bits
364         * accordingly may reduce the file size a little. For example with text
365         * files having one-byte alignment (US-ASCII, ISO-8859-*, UTF-8), using
366         * <code>setPb(0)</code> can improve compression slightly. For UTF-16
367         * text, <code>setPb(1)</code> is a good choice. If the alignment is
368         * an odd number like 3 bytes, <code>setPb(0)</code> might be the best
369         * choice.
370         * <p/>
371         * Even though the assumed alignment can be adjusted with
372         * <code>setPb</code> and <code>setLp</code>, LZMA2 still slightly favors
373         * 16-byte alignment. It might be worth taking into account when designing
374         * file formats that are likely to be often compressed with LZMA2.
375         *
376         * @throws UnsupportedOptionsException <code>pb</code> is invalid
377         */
378        public void setPb(int pb) throws UnsupportedOptionsException {
379            if (pb < 0 || pb > PB_MAX) {
380                throw new UnsupportedOptionsException(
381                        "pb must not exceed " + PB_MAX + ": " + pb);
382            }
383    
384            this.pb = pb;
385        }
386    
387        /**
388         * Gets the number of position bits.
389         */
390        public int getPb() {
391            return pb;
392        }

394        /**
395         * Sets the compression mode.
396         * <p/>
397         * This specifies the method to analyze the data produced by
398         * a match finder. The default is <code>MODE_FAST</code> for presets
399         * 0-3 and <code>MODE_NORMAL</code> for presets 4-9.
400         * <p/>
401         * Usually <code>MODE_FAST</code> is used with Hash Chain match finders
402         * and <code>MODE_NORMAL</code> with Binary Tree match finders. This is
403         * also what the presets do.
404         * <p/>
405         * The special mode <code>MODE_UNCOMPRESSED</code> doesn't try to
406         * compress the data at all (and doesn't use a match finder) and will
407         * simply wrap it in uncompressed LZMA2 chunks.
408         *
409         * @throws UnsupportedOptionsException <code>mode</code> is not supported
410         */
411        public void setMode(int mode) throws UnsupportedOptionsException {
412            if (mode < MODE_UNCOMPRESSED || mode > MODE_NORMAL) {
413                throw new UnsupportedOptionsException(
414                        "Unsupported compression mode: " + mode);
415            }
416    
417            this.mode = mode;
418        }
419    
420        /**
421         * Gets the compression mode.
422         */
423        public int getMode() {
424            return mode;
425        }

427        /**
428         * Sets the nice length of matches.
429         * Once a match of at least <code>niceLen</code> bytes is found,
430         * the algorithm stops looking for better matches. Higher values tend
431         * to give better compression at the expense of speed. The default
432         * depends on the preset.
433         *
434         * @throws UnsupportedOptionsException <code>niceLen</code> is invalid
435         */
436        public void setNiceLen(int niceLen) throws UnsupportedOptionsException {
437            if (niceLen < NICE_LEN_MIN) {
438                throw new UnsupportedOptionsException(
439                        "Minimum nice length of matches is "
440                                + NICE_LEN_MIN + " bytes: " + niceLen);
441            }
442    
443            if (niceLen > NICE_LEN_MAX) {
444                throw new UnsupportedOptionsException(
445                        "Maximum nice length of matches is " + NICE_LEN_MAX
446                                + ": " + niceLen);
447            }
448    
449            this.niceLen = niceLen;
450        }
451    
452        /**
453         * Gets the nice length of matches.
454         */
455        public int getNiceLen() {
456            return niceLen;
457        }

459        /**
460         * Sets the match finder type.
461         * <p/>
462         * Match finder has a major effect on compression speed, memory usage,
463         * and compression ratio. Usually Hash Chain match finders are faster
464         * than Binary Tree match finders. The default depends on the preset:
465         * 0-3 use <code>MF_HC4</code> and 4-9 use <code>MF_BT4</code>.
466         *
467         * @throws UnsupportedOptionsException <code>mf</code> is not supported
468         */
469        public void setMatchFinder(int mf) throws UnsupportedOptionsException {
470            if (mf != MF_HC4 && mf != MF_BT4) {
471                throw new UnsupportedOptionsException(
472                        "Unsupported match finder: " + mf);
473            }
474    
475            this.mf = mf;
476        }
477    
478        /**
479         * Gets the match finder type.
480         */
481        public int getMatchFinder() {
482            return mf;
483        }
484    
485        /**
486         * Sets the match finder search depth limit.
487         * <p/>
488         * The default is a special value of <code>0</code> which indicates that
489         * the depth limit should be automatically calculated by the selected
490         * match finder from the nice length of matches.
491         * <p/>
492         * Reasonable depth limit for Hash Chain match finders is 4-100 and
493         * 16-1000 for Binary Tree match finders. Using very high values can
494         * make the compressor extremely slow with some files. Avoid settings
495         * higher than 1000 unless you are prepared to interrupt the compression
496         * in case it is taking far too long.
497         *
498         * @throws UnsupportedOptionsException <code>depthLimit</code> is invalid
499         */
500        public void setDepthLimit(int depthLimit)
501                throws UnsupportedOptionsException {
502            if (depthLimit < 0) {
503                throw new UnsupportedOptionsException(
504                        "Depth limit cannot be negative: " + depthLimit);
505            }
506    
507            this.depthLimit = depthLimit;
508        }
509    
510        /**
511         * Gets the match finder search depth limit.
512         */
513        public int getDepthLimit() {
514            return depthLimit;
515        }

517        public int getEncoderMemoryUsage() {
518            return (mode == MODE_UNCOMPRESSED)
519                    ? UncompressedLZMA2OutputStream.getMemoryUsage()
520                    : LZMA2OutputStream.getMemoryUsage(this);
521        }
522    
523        public FinishableOutputStream getOutputStream(FinishableOutputStream out) {
524            if (mode == MODE_UNCOMPRESSED) {
525                return new UncompressedLZMA2OutputStream(out);
526            }
527    
528            return new LZMA2OutputStream(out, this);
529        }

531        /**
532         * Gets how much memory the LZMA2 decoder will need to decompress the data
533         * that was encoded with these options and stored in a .xz file.
534         * <p/>
535         * The returned value may bigger than the value returned by a direct call
536         * to {@link LZMA2InputStream#getMemoryUsage(int)} if the dictionary size
537         * is not 2^n or 2^n&nbsp;+&nbsp;2^(n-1) bytes. This is because the .xz
538         * headers store the dictionary size in such a format and other values
539         * are rounded up to the next such value. Such rounding is harmess except
540         * it might waste some memory if an unsual dictionary size is used.
541         * <p/>
542         * If you use raw LZMA2 streams and unusual dictioanary size, call
543         * {@link LZMA2InputStream#getMemoryUsage} directly to get raw decoder
544         * memory requirements.
545         */
546        public int getDecoderMemoryUsage() {
547            // Round the dictionary size up to the next 2^n or 2^n + 2^(n-1).
548            int d = dictSize - 1;
549            d |= d >>> 2;
550            d |= d >>> 3;
551            d |= d >>> 4;
552            d |= d >>> 8;
553            d |= d >>> 16;
554            return LZMA2InputStream.getMemoryUsage(d + 1);
555        }
556    
557        public InputStream getInputStream(InputStream in) throws IOException {
558            return new LZMA2InputStream(in, dictSize);
559        }

561        FilterEncoder getFilterEncoder() {
562            return new LZMA2Encoder(this);
563        }
