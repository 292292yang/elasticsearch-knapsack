009/**
010 * LZMA2 compression options.
011 * <p>
012 * While this allows setting the LZMA2 compression options in detail,
013 * often you only need <code>LZMA2Options()</code> or
014 * <code>LZMA2Options(int)</code>.
015 */
016public class LZMA2Options extends FilterOptions {
017    /**
018     * Minimum valid compression preset level is 0.
019     */
020    public static final int PRESET_MIN = 0;
021
022    /**
023     * Maximum valid compression preset level is 9.
024     */
025    public static final int PRESET_MAX = 9;
026
027    /**
028     * Default compression preset level is 6.
029     */
030    public static final int PRESET_DEFAULT = 6;
031
032    /**
033     * Minimum dictionary size is 4 KiB.
034     */
035    public static final int DICT_SIZE_MIN = 4096;
036
037    /**
038     * Maximum dictionary size for compression is 768 MiB.
039     * <p>
040     * The decompressor supports bigger dictionaries, up to almost 2 GiB.
041     * With HC4 the encoder would support dictionaries bigger than 768 MiB.
042     * The 768 MiB limit comes from the current implementation of BT4 where
043     * we would otherwise hit the limits of signed ints in array indexing.
044     * <p>
045     * If you really need bigger dictionary for decompression,
046     * use {@link LZMA2InputStream} directly.
047     */
048    public static final int DICT_SIZE_MAX = 768 << 20;
049
050    /**
051     * The default dictionary size is 8 MiB.
052     */
053    public static final int DICT_SIZE_DEFAULT = 8 << 20;
054
055    /**
056     * Maximum value for lc + lp is 4.
057     */
058    public static final int LC_LP_MAX = 4;
059
060    /**
061     * The default number of literal context bits is 3.
062     */
063    public static final int LC_DEFAULT = 3;
064
065    /**
066     * The default number of literal position bits is 0.
067     */
068    public static final int LP_DEFAULT = 0;
069
070    /**
071     * Maximum value for pb is 4.
072     */
073    public static final int PB_MAX = 4;
074
075    /**
076     * The default number of position bits is 2.
077     */
078    public static final int PB_DEFAULT = 2;
079
080    /**
081     * Compression mode: uncompressed.
082     * The data is wrapped into a LZMA2 stream without compression.
083     */
084    public static final int MODE_UNCOMPRESSED = 0;
085
086    /**
087     * Compression mode: fast.
088     * This is usually combined with a hash chain match finder.
089     */
090    public static final int MODE_FAST = LZMAEncoder.MODE_FAST;
091
092    /**
093     * Compression mode: normal.
094     * This is usually combined with a binary tree match finder.
095     */
096    public static final int MODE_NORMAL = LZMAEncoder.MODE_NORMAL;
097
098    /**
099     * Minimum value for <code>niceLen</code> is 8.
100     */
101    public static final int NICE_LEN_MIN = 8;
102
103    /**
104     * Maximum value for <code>niceLen</code> is 273.
105     */
106    public static final int NICE_LEN_MAX = 273;
107
108    /**
109     * Match finder: Hash Chain 2-3-4
110     */
111    public static final int MF_HC4 = LZEncoder.MF_HC4;
112
113    /**
114     * Match finder: Binary tree 2-3-4
115     */
116    public static final int MF_BT4 = LZEncoder.MF_BT4;

118    private static final int[] presetToDictSize = {
119            1 << 18, 1 << 20, 1 << 21, 1 << 22, 1 << 22,
120            1 << 23, 1 << 23, 1 << 24, 1 << 25, 1 << 26};
121
122    private static final int[] presetToDepthLimit = {4, 8, 24, 48};
123
124    private int dictSize;
125    private byte[] presetDict = null;
126    private int lc;
127    private int lp;
128    private int pb;
129    private int mode;
130    private int niceLen;
131    private int mf;
132    private int depthLimit;

134    /**
135     * Creates new LZMA2 options and sets them to the default values.
136     * This is equivalent to <code>LZMA2Options(PRESET_DEFAULT)</code>.
137     */
138    public LZMA2Options() {
139        try {
140            setPreset(PRESET_DEFAULT);
141        } catch (UnsupportedOptionsException e) {
142            assert false;
143            throw new RuntimeException();
144        }
145    }

148     * Creates new LZMA2 options and sets them to the given preset.
149     *
150     * @throws UnsupportedOptionsException <code>preset</code> is not supported
151     */
152    public LZMA2Options(int preset) throws UnsupportedOptionsException {
153        setPreset(preset);
154    }

156    /**
157     * Creates new LZMA2 options and sets them to the given custom values.
158     *
159     * @throws UnsupportedOptionsException unsupported options were specified
160     */
161    public LZMA2Options(int dictSize, int lc, int lp, int pb, int mode,
162                        int niceLen, int mf, int depthLimit)
163            throws UnsupportedOptionsException {
164        setDictSize(dictSize);
165        setLcLp(lc, lp);
166        setPb(pb);
167        setMode(mode);
168        setNiceLen(niceLen);
169        setMatchFinder(mf);
170        setDepthLimit(depthLimit);
171    }

173    /**
174     * Sets the compression options to the given preset.
175     * <p>
176     * The presets 0-3 are fast presets with medium compression.
177     * The presets 4-6 are fairly slow presets with high compression.
178     * The default preset (<code>PRESET_DEFAULT</code>) is 6.
179     * <p>
180     * The presets 7-9 are like the preset 6 but use bigger dictionaries
181     * and have higher compressor and decompressor memory requirements.
182     * Unless the uncompressed size of the file exceeds 8&nbsp;MiB,
183     * 16&nbsp;MiB, or 32&nbsp;MiB, it is waste of memory to use the
184     * presets 7, 8, or 9, respectively.
185     *
186     * @throws UnsupportedOptionsException <code>preset</code> is not supported
187     */
188    public void setPreset(int preset) throws UnsupportedOptionsException {
189        if (preset < 0 || preset > 9) {
190            throw new UnsupportedOptionsException(
191                    "Unsupported preset: " + preset);
192        }
193
194        lc = LC_DEFAULT;
195        lp = LP_DEFAULT;
196        pb = PB_DEFAULT;
197        dictSize = presetToDictSize[preset];
198
199        if (preset <= 3) {
200            mode = MODE_FAST;
201            mf = MF_HC4;
202            niceLen = preset <= 1 ? 128 : NICE_LEN_MAX;
203            depthLimit = presetToDepthLimit[preset];
204        } else {
205            mode = MODE_NORMAL;
206            mf = MF_BT4;
207            niceLen = (preset == 4) ? 16 : (preset == 5) ? 32 : 64;
208            depthLimit = 0;
209        }
210    }
211
212    /**
213     * Sets the dictionary size in bytes.
214     * <p>
215     * The dictionary (or history buffer) holds the most recently seen
216     * uncompressed data. Bigger dictionary usually means better compression.
217     * However, using a dictioanary bigger than the size of the uncompressed
218     * data is waste of memory.
219     * <p>
220     * Any value in the range [DICT_SIZE_MIN, DICT_SIZE_MAX] is valid,
221     * but sizes of 2^n and 2^n&nbsp;+&nbsp;2^(n-1) bytes are somewhat
222     * recommended.
223     *
224     * @throws UnsupportedOptionsException <code>dictSize</code> is not supported
225     */
226    public void setDictSize(int dictSize) throws UnsupportedOptionsException {
227        if (dictSize < DICT_SIZE_MIN) {
228            throw new UnsupportedOptionsException(
229                    "LZMA2 dictionary size must be at least 4 KiB: "
230                            + dictSize + " B");
231        }
232
233        if (dictSize > DICT_SIZE_MAX) {
234            throw new UnsupportedOptionsException(
235                    "LZMA2 dictionary size must not exceed "
236                            + (DICT_SIZE_MAX >> 20) + " MiB: " + dictSize + " B");
237        }
238
239        this.dictSize = dictSize;
240    }
241
242    /**
243     * Gets the dictionary size in bytes.
244     */
245    public int getDictSize() {
246        return dictSize;
247    }

249    /**
250     * Sets a preset dictionary. Use null to disable the use of
251     * a preset dictionary. By default there is no preset dictionary.
252     * <p>
253     * <b>The .xz format doesn't support a preset dictionary for now.
254     * Do not set a preset dictionary unless you use raw LZMA2.</b>
255     * <p>
256     * Preset dictionary can be useful when compressing many similar,
257     * relatively small chunks of data independently from each other.
258     * A preset dictionary should contain typical strings that occur in
259     * the files being compressed. The most probable strings should be
260     * near the end of the preset dictionary. The preset dictionary used
261     * for compression is also needed for decompression.
262     */
263    public void setPresetDict(byte[] presetDict) {
264        this.presetDict = presetDict;
265    }

267    /**
268     * Gets the preset dictionary.
269     */
270    public byte[] getPresetDict() {
271        return presetDict;
272    }

274    /**
275     * Sets the number of literal context bits and literal position bits.
276     * <p>
277     * The sum of <code>lc</code> and <code>lp</code> is limited to 4.
278     * Trying to exceed it will throw an exception. This function lets
279     * you change both at the same time.
280     *
281     * @throws UnsupportedOptionsException <code>lc</code> and <code>lp</code>
282     *                                     are invalid
283     */
284    public void setLcLp(int lc, int lp) throws UnsupportedOptionsException {
285        if (lc < 0 || lp < 0 || lc > LC_LP_MAX || lp > LC_LP_MAX
286                || lc + lp > LC_LP_MAX) {
287            throw new UnsupportedOptionsException(
288                    "lc + lp must not exceed " + LC_LP_MAX + ": "
289                            + lc + " + " + lp);
290        }
291
292        this.lc = lc;
293        this.lp = lp;
294    }
295
296    /**
297     * Sets the number of literal context bits.
298     * <p>
299     * All bytes that cannot be encoded as matches are encoded as literals.
300     * That is, literals are simply 8-bit bytes that are encoded one at
301     * a time.
302     * <p>
303     * The literal coding makes an assumption that the highest <code>lc</code>
304     * bits of the previous uncompressed byte correlate with the next byte.
305     * For example, in typical English text, an upper-case letter is often
306     * followed by a lower-case letter, and a lower-case letter is usually
307     * followed by another lower-case letter. In the US-ASCII character set,
308     * the highest three bits are 010 for upper-case letters and 011 for
309     * lower-case letters. When <code>lc</code> is at least 3, the literal
310     * coding can take advantage of this property in the  uncompressed data.
311     * <p>
312     * The default value (3) is usually good. If you want maximum compression,
313     * try <code>setLc(4)</code>. Sometimes it helps a little, and sometimes it
314     * makes compression worse. If it makes it worse, test for example
315     * <code>setLc(2)</code> too.
316     *
317     * @throws UnsupportedOptionsException <code>lc</code> is invalid, or the sum
318     *                                     of <code>lc</code> and <code>lp</code>
319     *                                     exceed LC_LP_MAX
320     */
321    public void setLc(int lc) throws UnsupportedOptionsException {
322        setLcLp(lc, lp);
323    }
324
325    /**
326     * Sets the number of literal position bits.
327     * <p>
328     * This affets what kind of alignment in the uncompressed data is
329     * assumed when encoding literals. See {@link #setPb(int) setPb} for
330     * more information about alignment.
331     *
332     * @throws UnsupportedOptionsException <code>lp</code> is invalid, or the sum
333     *                                     of <code>lc</code> and <code>lp</code>
334     *                                     exceed LC_LP_MAX
335     */
336    public void setLp(int lp) throws UnsupportedOptionsException {
337        setLcLp(lc, lp);
338    }

341     * Gets the number of literal context bits.

344        return lc;
345    }

348     * Gets the number of literal position bits.

351        return lp;

354    /**
355     * Sets the number of position bits.
356     * <p>
357     * This affects what kind of alignment in the uncompressed data is
358     * assumed in general. The default (2) means four-byte alignment
359     * (2^<code>pb</code> = 2^2 = 4), which is often a good choice when
360     * there's no better guess.
361     * <p>
362     * When the alignment is known, setting the number of position bits
363     * accordingly may reduce the file size a little. For example with text
364     * files having one-byte alignment (US-ASCII, ISO-8859-*, UTF-8), using
365     * <code>setPb(0)</code> can improve compression slightly. For UTF-16
366     * text, <code>setPb(1)</code> is a good choice. If the alignment is
367     * an odd number like 3 bytes, <code>setPb(0)</code> might be the best
368     * choice.
369     * <p>
370     * Even though the assumed alignment can be adjusted with
371     * <code>setPb</code> and <code>setLp</code>, LZMA2 still slightly favors
372     * 16-byte alignment. It might be worth taking into account when designing
373     * file formats that are likely to be often compressed with LZMA2.
374     *
375     * @throws UnsupportedOptionsException <code>pb</code> is invalid
376     */
377    public void setPb(int pb) throws UnsupportedOptionsException {
378        if (pb < 0 || pb > PB_MAX) {
379            throw new UnsupportedOptionsException(
380                    "pb must not exceed " + PB_MAX + ": " + pb);
381        }
382
383        this.pb = pb;
384    }
385
386    /**
387     * Gets the number of position bits.
388     */
389    public int getPb() {
390        return pb;
391    }

393    /**
394     * Sets the compression mode.
395     * <p>
396     * This specifies the method to analyze the data produced by
397     * a match finder. The default is <code>MODE_FAST</code> for presets
398     * 0-3 and <code>MODE_NORMAL</code> for presets 4-9.
399     * <p>
400     * Usually <code>MODE_FAST</code> is used with Hash Chain match finders
401     * and <code>MODE_NORMAL</code> with Binary Tree match finders. This is
402     * also what the presets do.
403     * <p>
404     * The special mode <code>MODE_UNCOMPRESSED</code> doesn't try to
405     * compress the data at all (and doesn't use a match finder) and will
406     * simply wrap it in uncompressed LZMA2 chunks.
407     *
408     * @throws UnsupportedOptionsException <code>mode</code> is not supported
409     */
410    public void setMode(int mode) throws UnsupportedOptionsException {
411        if (mode < MODE_UNCOMPRESSED || mode > MODE_NORMAL) {
412            throw new UnsupportedOptionsException(
413                    "Unsupported compression mode: " + mode);
414        }
415
416        this.mode = mode;
417    }

420     * Gets the compression mode.

422    public int getMode() {
423        return mode;
424    }

426    /**
427     * Sets the nice length of matches.
428     * Once a match of at least <code>niceLen</code> bytes is found,
429     * the algorithm stops looking for better matches. Higher values tend
430     * to give better compression at the expense of speed. The default
431     * depends on the preset.
432     *
433     * @throws UnsupportedOptionsException <code>niceLen</code> is invalid
434     */
435    public void setNiceLen(int niceLen) throws UnsupportedOptionsException {
436        if (niceLen < NICE_LEN_MIN) {
437            throw new UnsupportedOptionsException(
438                    "Minimum nice length of matches is "
439                            + NICE_LEN_MIN + " bytes: " + niceLen);
440        }
441
442        if (niceLen > NICE_LEN_MAX) {
443            throw new UnsupportedOptionsException(
444                    "Maximum nice length of matches is " + NICE_LEN_MAX
445                            + ": " + niceLen);
446        }
447
448        this.niceLen = niceLen;
449    }

452     * Gets the nice length of matches.
453     */
454    public int getNiceLen() {
455        return niceLen;
456    }

458    /**
459     * Sets the match finder type.
460     * <p>
461     * Match finder has a major effect on compression speed, memory usage,
462     * and compression ratio. Usually Hash Chain match finders are faster
463     * than Binary Tree match finders. The default depends on the preset:
464     * 0-3 use <code>MF_HC4</code> and 4-9 use <code>MF_BT4</code>.
465     *
466     * @throws UnsupportedOptionsException <code>mf</code> is not supported
467     */
468    public void setMatchFinder(int mf) throws UnsupportedOptionsException {
469        if (mf != MF_HC4 && mf != MF_BT4) {
470            throw new UnsupportedOptionsException(
471                    "Unsupported match finder: " + mf);
472        }
473
474        this.mf = mf;
475    }
476
477    /**
478     * Gets the match finder type.
479     */
480    public int getMatchFinder() {
481        return mf;
482    }

484    /**
485     * Sets the match finder search depth limit.
486     * <p>
487     * The default is a special value of <code>0</code> which indicates that
488     * the depth limit should be automatically calculated by the selected
489     * match finder from the nice length of matches.
490     * <p>
491     * Reasonable depth limit for Hash Chain match finders is 4-100 and
492     * 16-1000 for Binary Tree match finders. Using very high values can
493     * make the compressor extremely slow with some files. Avoid settings
494     * higher than 1000 unless you are prepared to interrupt the compression
495     * in case it is taking far too long.
496     *
497     * @throws UnsupportedOptionsException <code>depthLimit</code> is invalid
498     */
499    public void setDepthLimit(int depthLimit)
500            throws UnsupportedOptionsException {
501        if (depthLimit < 0) {
502            throw new UnsupportedOptionsException(
503                    "Depth limit cannot be negative: " + depthLimit);
504        }
505
506        this.depthLimit = depthLimit;
507    }

510     * Gets the match finder search depth limit.
511     */

513        return depthLimit;

516    public int getEncoderMemoryUsage() {
517        return (mode == MODE_UNCOMPRESSED)
518                ? UncompressedLZMA2OutputStream.getMemoryUsage()
519                : LZMA2OutputStream.getMemoryUsage(this);
520    }
521
522    public FinishableOutputStream getOutputStream(FinishableOutputStream out) {
523        if (mode == MODE_UNCOMPRESSED) {
524            return new UncompressedLZMA2OutputStream(out);
525        }
526
527        return new LZMA2OutputStream(out, this);
528    }

530    /**
531     * Gets how much memory the LZMA2 decoder will need to decompress the data
532     * that was encoded with these options and stored in a .xz file.
533     * <p>
534     * The returned value may bigger than the value returned by a direct call
535     * to {@link LZMA2InputStream#getMemoryUsage(int)} if the dictionary size
536     * is not 2^n or 2^n&nbsp;+&nbsp;2^(n-1) bytes. This is because the .xz
537     * headers store the dictionary size in such a format and other values
538     * are rounded up to the next such value. Such rounding is harmess except
539     * it might waste some memory if an unsual dictionary size is used.
540     * <p>
541     * If you use raw LZMA2 streams and unusual dictioanary size, call
542     * {@link LZMA2InputStream#getMemoryUsage} directly to get raw decoder
543     * memory requirements.
544     */
545    public int getDecoderMemoryUsage() {
546        // Round the dictionary size up to the next 2^n or 2^n + 2^(n-1).
547        int d = dictSize - 1;
548        d |= d >>> 2;
549        d |= d >>> 3;
550        d |= d >>> 4;
551        d |= d >>> 8;
552        d |= d >>> 16;
553        return LZMA2InputStream.getMemoryUsage(d + 1);
554    }
555
556    public InputStream getInputStream(InputStream in) throws IOException {
557        return new LZMA2InputStream(in, dictSize);
558    }

560    FilterEncoder getFilterEncoder() {
561        return new LZMA2Encoder(this);
562    }
