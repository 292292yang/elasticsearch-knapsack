002    package org.xbib.io.compress.xz;
003    
004    import org.xbib.io.compress.xz.check.Check;
005    import org.xbib.io.compress.xz.common.EncoderUtil;
006    import org.xbib.io.compress.xz.common.StreamFlags;
007    import org.xbib.io.compress.xz.index.IndexEncoder;
008    
009    import java.io.IOException;
010    import java.io.OutputStream;
011    
012    /**
013     * Compresses into the .xz file format.
014     * <p/>
015     * <h4>Examples</h4>
016     * <p/>
017     * Getting an output stream to compress with LZMA2 using the default
018     * settings and the default integrity check type (CRC64):
019     * <p><blockquote><pre>
020     * FileOutputStream outfile = new FileOutputStream("foo.xz");
021     * XZOutputStream outxz = new XZOutputStream(outfile, new LZMA2Options());
022     * </pre></blockquote>
023     * <p/>
024     * Using the preset level <code>8</code> for LZMA2 (the default
025     * is <code>6</code>) and SHA-256 instead of CRC64 for integrity checking:
026     * <p><blockquote><pre>
027     * XZOutputStream outxz = new XZOutputStream(outfile, new LZMA2Options(8),
028     *                                           XZ.CHECK_SHA256);
029     * </pre></blockquote>
030     * <p/>
031     * Using the x86 BCJ filter together with LZMA2 to compress x86 executables
032     * and printing the memory usage information before creating the
033     * XZOutputStream:
034     * <p><blockquote><pre>
035     * X86Options x86 = new X86Options();
036     * LZMA2Options lzma2 = new LZMA2Options();
037     * FilterOptions[] options = { x86, lzma2 };
038     * String msg = "Encoder memory usage: "
039     *                    + FilterOptions.getEncoderMemoryUsage(options)
040     *                    + " KiB";
041     * trign msg2 = "Decoder memory usage: "
042     *                    + FilterOptions.getDecoderMemoryUsage(options)
043     *                    + " KiB";
044     * XZOutputStream outxz = new XZOutputStream(outfile, options);
045     * </pre></blockquote>
046     */
047    public class XZOutputStream extends FinishableOutputStream {
048        private OutputStream out;
049        private final StreamFlags streamFlags = new StreamFlags();
050        private final Check check;
051        private final IndexEncoder index = new IndexEncoder();
052    
053        private BlockOutputStream blockEncoder = null;
054        private FilterEncoder[] filters;
055    
056        /**
057         * True if the current filter chain supports flushing.
058         * If it doesn't support flushing, <code>flush()</code>
059         * will use <code>endBlock()</code> as a fallback.
060         */
061        private boolean filtersSupportFlushing;
062    
063        private IOException exception = null;
064        private boolean finished = false;
065    
066        /**
067         * Creates a new XZ compressor using one filter and CRC64 as
068         * the integrity check. This constructor is equivalent to passing
069         * a single-member FilterOptions array to
070         * <code>XZOutputStream(OutputStream, FilterOptions[])</code>.
071         *
072         * @param out           output stream to which the compressed data
073         *                      will be written
074         * @param filterOptions filter options to use
075         * @throws UnsupportedOptionsException invalid filter chain
076         * @throws java.io.IOException         may be thrown from <code>out</code>
077         */
078        public XZOutputStream(OutputStream out, FilterOptions filterOptions)
079                throws IOException {
080            this(out, filterOptions, XZ.CHECK_CRC64);
081        }
082    
083        /**
084         * Creates a new XZ compressor using one filter and the specified
085         * integrity check type. This constructor is equivalent to
086         * passing a single-member FilterOptions array to
087         * <code>XZOutputStream(OutputStream, FilterOptions[], int)</code>.
088         *
089         * @param out           output stream to which the compressed data
090         *                      will be written
091         * @param filterOptions filter options to use
092         * @param checkType     type of the integrity check,
093         *                      for example XZ.CHECK_CRC32
094         * @throws UnsupportedOptionsException invalid filter chain
095         * @throws java.io.IOException         may be thrown from <code>out</code>
096         */
097        public XZOutputStream(OutputStream out, FilterOptions filterOptions,
098                              int checkType) throws IOException {
099            this(out, new FilterOptions[]{filterOptions}, checkType);
100        }
101    
102        /**
103         * Creates a new XZ compressor using 1-4 filters and CRC64 as
104         * the integrity check. This constructor is equivalent
105         * <code>XZOutputStream(out, filterOptions, XZ.CHECK_CRC64)</code>.
106         *
107         * @param out           output stream to which the compressed data
108         *                      will be written
109         * @param filterOptions array of filter options to use
110         * @throws UnsupportedOptionsException invalid filter chain
111         * @throws java.io.IOException         may be thrown from <code>out</code>
112         */
113        public XZOutputStream(OutputStream out, FilterOptions[] filterOptions)
114                throws IOException {
115            this(out, filterOptions, XZ.CHECK_CRC64);
116        }
117    
118        /**
119         * Creates a new XZ compressor using 1-4 filters and the specified
120         * integrity check type.
121         *
122         * @param out           output stream to which the compressed data
123         *                      will be written
124         * @param filterOptions array of filter options to use
125         * @param checkType     type of the integrity check,
126         *                      for example XZ.CHECK_CRC32
127         * @throws UnsupportedOptionsException invalid filter chain
128         * @throws java.io.IOException         may be thrown from <code>out</code>
129         */
130        public XZOutputStream(OutputStream out, FilterOptions[] filterOptions,
131                              int checkType) throws IOException {
132            this.out = out;
133            updateFilters(filterOptions);
134    
135            streamFlags.checkType = checkType;
136            check = Check.getInstance(checkType);
137    
138            encodeStreamHeader();
139        }
140    
141        /**
142         * Updates the filter chain with a single filter.
143         * This is equivalent to passing a single-member FilterOptions array
144         * to <code>updateFilters(FilterOptions[])</code>.
145         *
146         * @param filterOptions new filter to use
147         * @throws UnsupportedOptionsException unsupported filter chain, or trying to change
148         *                                     the filter chain in the middle of a Block
149         */
150        public void updateFilters(FilterOptions filterOptions)
151                throws XZIOException {
152            FilterOptions[] opts = new FilterOptions[1];
153            opts[0] = filterOptions;
154            updateFilters(opts);
155        }
156    
157        /**
158         * Updates the filter chain with 1-4 filters.
159         * <p/>
160         * Currently this cannot be used to update e.g. LZMA2 options in the
161         * middle of a XZ Block. Use <code>endBlock()</code> to finish the
162         * current XZ Block before calling this function. The new filter chain
163         * will then be used for the next XZ Block.
164         *
165         * @param filterOptions new filter chain to use
166         * @throws UnsupportedOptionsException unsupported filter chain, or trying to change
167         *                                     the filter chain in the middle of a Block
168         */
169        public void updateFilters(FilterOptions[] filterOptions)
170                throws XZIOException {
171            if (blockEncoder != null) {
172                throw new UnsupportedOptionsException("Changing filter options "
173                        + "in the middle of a XZ Block not implemented");
174            }
175    
176            if (filterOptions.length < 1 || filterOptions.length > 4) {
177                throw new UnsupportedOptionsException(
178                        "XZ filter chain must be 1-4 filters");
179            }
180    
181            filtersSupportFlushing = true;
182            FilterEncoder[] newFilters = new FilterEncoder[filterOptions.length];
183            for (int i = 0; i < filterOptions.length; ++i) {
184                newFilters[i] = filterOptions[i].getFilterEncoder();
185                filtersSupportFlushing &= newFilters[i].supportsFlushing();
186            }
187    
188            RawCoder.validate(newFilters);
189            filters = newFilters;
190        }
191    
192        /**
193         * Writes one byte to be compressed.
194         *
195         * @throws XZIOException       XZ Stream has grown too big
196         * @throws XZIOException       <code>finish()</code> or <code>close()</code>
197         *                             was already called
198         * @throws java.io.IOException may be thrown by the underlying output stream
199         */
200        public void write(int b) throws IOException {
201            byte[] buf = new byte[]{(byte) b};
202            write(buf, 0, 1);
203        }
204    
205        /**
206         * Writes an array of bytes to be compressed.
207         * The compressors tend to do internal buffering and thus the written
208         * data won't be readable from the compressed output immediately.
209         * Use <code>flush()</code> to force everything written so far to
210         * be written to the underlaying output stream, but be aware that
211         * flushing reduces compression ratio.
212         *
213         * @param buf buffer of bytes to be written
214         * @param off start offset in <code>buf</code>
215         * @param len number of bytes to write
216         * @throws XZIOException       XZ Stream has grown too big: total file size
217         *                             about 8&nbsp;EiB or the Index field exceeds
218         *                             16&nbsp;GiB; you shouldn't reach these sizes
219         *                             in practice
220         * @throws XZIOException       <code>finish()</code> or <code>close()</code>
221         *                             was already called and len &gt; 0
222         * @throws java.io.IOException may be thrown by the underlying output stream
223         */
224        public void write(byte[] buf, int off, int len) throws IOException {
225            if (off < 0 || len < 0 || off + len < 0 || off + len > buf.length) {
226                throw new IndexOutOfBoundsException();
227            }
228    
229            if (exception != null) {
230                throw exception;
231            }
232    
233            if (finished) {
234                throw new XZIOException("Stream finished or closed");
235            }
236    
237            try {
238                if (blockEncoder == null) {
239                    blockEncoder = new BlockOutputStream(out, filters, check);
240                }
241    
242                blockEncoder.write(buf, off, len);
243            } catch (IOException e) {
244                exception = e;
245                throw e;
246            }
247        }
248    
249        /**
250         * Finishes the current XZ Block (but not the whole XZ Stream).
251         * This doesn't flush the stream so it's possible that not all data will
252         * be decompressible from the output stream when this function returns.
253         * Call also <code>flush()</code> if flushing is wanted in addition to
254         * finishing the current XZ Block.
255         * <p/>
256         * If there is no unfinished Block open, this function will do nothing.
257         * (No empty XZ Block will be created.)
258         * <p/>
259         * This function can be useful, for example, to create
260         * random-accessible .xz files.
261         * <p/>
262         * Starting a new XZ Block means that the encoder state is reset.
263         * Doing this very often will increase the size of the compressed
264         * file a lot (more than plain <code>flush()</code> would do).
265         *
266         * @throws XZIOException       XZ Stream has grown too big
267         * @throws XZIOException       stream finished or closed
268         * @throws java.io.IOException may be thrown by the underlying output stream
269         */
270        public void endBlock() throws IOException {
271            if (exception != null) {
272                throw exception;
273            }
274    
275            if (finished) {
276                throw new XZIOException("Stream finished or closed");
277            }
278    
279            // NOTE: Once there is threading with multiple Blocks, it's possible
280            // that this function will be more like a barrier that returns
281            // before the last Block has been finished.
282            if (blockEncoder != null) {
283                try {
284                    blockEncoder.finish();
285                    index.add(blockEncoder.getUnpaddedSize(),
286                            blockEncoder.getUncompressedSize());
287                    blockEncoder = null;
288                } catch (IOException e) {
289                    exception = e;
290                    throw e;
291                }
292            }
293        }
294    
295        /**
296         * Flushes the encoder and calls <code>out.flush()</code>.
297         * All buffered pending data will then be decompressible from
298         * the output stream.
299         * <p/>
300         * Calling this function very often may increase the compressed
301         * file size a lot. The filter chain options may affect the size
302         * increase too. For example, with LZMA2 the HC4 match finder has
303         * smaller penalty with flushing than BT4.
304         * <p/>
305         * Some filters don't support flushing. If the filter chain has
306         * such a filter, <code>flush()</code> will call <code>endBlock()</code>
307         * before flushing.
308         *
309         * @throws XZIOException       XZ Stream has grown too big
310         * @throws XZIOException       stream finished or closed
311         * @throws java.io.IOException may be thrown by the underlying output stream
312         */
313        public void flush() throws IOException {
314            if (exception != null) {
315                throw exception;
316            }
317    
318            if (finished) {
319                throw new XZIOException("Stream finished or closed");
320            }
321    
322            try {
323                if (blockEncoder != null) {
324                    if (filtersSupportFlushing) {
325                        // This will eventually call out.flush() so
326                        // no need to do it here again.
327                        blockEncoder.flush();
328                    } else {
329                        endBlock();
330                        out.flush();
331                    }
332                } else {
333                    out.flush();
334                }
335            } catch (IOException e) {
336                exception = e;
337                throw e;
338            }
339        }
340    
341        /**
342         * Finishes compression without closing the underlying stream.
343         * No more data can be written to this stream after finishing
344         * (calling <code>write</code> with an empty buffer is OK).
345         * <p/>
346         * Repeated calls to <code>finish()</code> do nothing unless
347         * an exception was thrown by this stream earlier. In that case
348         * the same exception is thrown again.
349         * <p/>
350         * After finishing, the stream may be closed normally with
351         * <code>close()</code>. If the stream will be closed anyway, there
352         * usually is no need to call <code>finish()</code> separately.
353         *
354         * @throws XZIOException       XZ Stream has grown too big
355         * @throws java.io.IOException may be thrown by the underlying output stream
356         */
357        public void finish() throws IOException {
358            if (!finished) {
359                // This checks for pending exceptions so we don't need to
360                // worry about it here.
361                endBlock();
362    
363                try {
364                    index.encode(out);
365                    encodeStreamFooter();
366                } catch (IOException e) {
367                    exception = e;
368                    throw e;
369                }
370    
371                // Set it to true only if everything goes fine. Setting it earlier
372                // would cause repeated calls to finish() do nothing instead of
373                // throwing an exception to indicate an earlier error.
374                finished = true;
375            }
376        }
377    
378        /**
379         * Finishes compression and closes the underlying stream.
380         * The underlying stream <code>out</code> is closed even if finishing
381         * fails. If both finishing and closing fail, the exception thrown
382         * by <code>finish()</code> is thrown and the exception from the failed
383         * <code>out.close()</code> is lost.
384         *
385         * @throws XZIOException       XZ Stream has grown too big
386         * @throws java.io.IOException may be thrown by the underlying output stream
387         */
388        public void close() throws IOException {
389            if (out != null) {
390                // If finish() throws an exception, it stores the exception to
391                // the variable "exception". So we can ignore the possible
392                // exception here.
393                try {
394                    finish();
395                } catch (IOException e) {
396                }
397    
398                try {
399                    out.close();
400                } catch (IOException e) {
401                    // Remember the exception but only if there is no previous
402                    // pending exception.
403                    if (exception == null) {
404                        exception = e;
405                    }
406                }
407    
408                out = null;
409            }
410    
411            if (exception != null) {
412                throw exception;
413            }
414        }
415    
416        private void encodeStreamFlags(byte[] buf, int off) {
417            buf[off] = 0x00;
418            buf[off + 1] = (byte) streamFlags.checkType;
419        }
420    
421        private void encodeStreamHeader() throws IOException {
422            out.write(XZ.HEADER_MAGIC);
423    
424            byte[] buf = new byte[2];
425            encodeStreamFlags(buf, 0);
426            out.write(buf);
427    
428            EncoderUtil.writeCRC32(out, buf);
429        }
430    
431        private void encodeStreamFooter() throws IOException {
432            byte[] buf = new byte[6];
433            long backwardSize = index.getIndexSize() / 4 - 1;
434            for (int i = 0; i < 4; ++i) {
435                buf[i] = (byte) (backwardSize >>> (i * 8));
436            }
437    
438            encodeStreamFlags(buf, 4);
439    
440            EncoderUtil.writeCRC32(out, buf);
441            out.write(buf);
442            out.write(XZ.FOOTER_MAGIC);
443        }
