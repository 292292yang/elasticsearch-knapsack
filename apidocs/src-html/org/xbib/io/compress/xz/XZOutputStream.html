001package org.xbib.io.compress.xz;
002
003import org.xbib.io.compress.xz.check.Check;
004import org.xbib.io.compress.xz.common.EncoderUtil;
005import org.xbib.io.compress.xz.common.StreamFlags;
006import org.xbib.io.compress.xz.index.IndexEncoder;
007
008import java.io.IOException;
009import java.io.OutputStream;
010
011/**
012 * Compresses into the .xz file format.
013 * <p>
014 * Examples
015 * <p>
016 * Getting an output stream to compress with LZMA2 using the default
017 * settings and the default integrity check type (CRC64):
018 * <p><blockquote><pre>
019 * FileOutputStream outfile = new FileOutputStream("foo.xz");
020 * XZOutputStream outxz = new XZOutputStream(outfile, new LZMA2Options());
021 * </pre></blockquote>
022 * <p>
023 * Using the preset level <code>8</code> for LZMA2 (the default
024 * is <code>6</code>) and SHA-256 instead of CRC64 for integrity checking:
025 * <p><blockquote><pre>
026 * XZOutputStream outxz = new XZOutputStream(outfile, new LZMA2Options(8),
027 *                                           XZ.CHECK_SHA256);
028 * </pre></blockquote>
029 * <p>
030 * Using the x86 BCJ filter together with LZMA2 to compress x86 executables
031 * and printing the memory usage information before creating the
032 * XZOutputStream:
033 * <p><blockquote><pre>
034 * X86Options x86 = new X86Options();
035 * LZMA2Options lzma2 = new LZMA2Options();
036 * FilterOptions[] options = { x86, lzma2 };
037 * String msg = "Encoder memory usage: "
038 *                    + FilterOptions.getEncoderMemoryUsage(options)
039 *                    + " KiB";
040 * trign msg2 = "Decoder memory usage: "
041 *                    + FilterOptions.getDecoderMemoryUsage(options)
042 *                    + " KiB";
043 * XZOutputStream outxz = new XZOutputStream(outfile, options);
044 * </pre></blockquote>
045 */
046public class XZOutputStream extends FinishableOutputStream {
047    private OutputStream out;
048    private final StreamFlags streamFlags = new StreamFlags();
049    private final Check check;
050    private final IndexEncoder index = new IndexEncoder();
051
052    private BlockOutputStream blockEncoder = null;
053    private FilterEncoder[] filters;
054
055    /**
056     * True if the current filter chain supports flushing.
057     * If it doesn't support flushing, <code>flush()</code>
058     * will use <code>endBlock()</code> as a fallback.
059     */
060    private boolean filtersSupportFlushing;
061
062    private IOException exception = null;
063    private boolean finished = false;
064
065    /**
066     * Creates a new XZ compressor using one filter and CRC64 as
067     * the integrity check. This constructor is equivalent to passing
068     * a single-member FilterOptions array to
069     * <code>XZOutputStream(OutputStream, FilterOptions[])</code>.
070     *
071     * @param out           output stream to which the compressed data
072     *                      will be written
073     * @param filterOptions filter options to use
074     * @throws UnsupportedOptionsException invalid filter chain
075     * @throws java.io.IOException         may be thrown from <code>out</code>
076     */
077    public XZOutputStream(OutputStream out, FilterOptions filterOptions)
078            throws IOException {
079        this(out, filterOptions, XZ.CHECK_CRC64);
080    }
081
082    /**
083     * Creates a new XZ compressor using one filter and the specified
084     * integrity check type. This constructor is equivalent to
085     * passing a single-member FilterOptions array to
086     * <code>XZOutputStream(OutputStream, FilterOptions[], int)</code>.
087     *
088     * @param out           output stream to which the compressed data
089     *                      will be written
090     * @param filterOptions filter options to use
091     * @param checkType     type of the integrity check,
092     *                      for example XZ.CHECK_CRC32
093     * @throws UnsupportedOptionsException invalid filter chain
094     * @throws java.io.IOException         may be thrown from <code>out</code>
095     */
096    public XZOutputStream(OutputStream out, FilterOptions filterOptions,
097                          int checkType) throws IOException {
098        this(out, new FilterOptions[]{filterOptions}, checkType);
099    }
100
101    /**
102     * Creates a new XZ compressor using 1-4 filters and CRC64 as
103     * the integrity check. This constructor is equivalent
104     * <code>XZOutputStream(out, filterOptions, XZ.CHECK_CRC64)</code>.
105     *
106     * @param out           output stream to which the compressed data
107     *                      will be written
108     * @param filterOptions array of filter options to use
109     * @throws UnsupportedOptionsException invalid filter chain
110     * @throws java.io.IOException         may be thrown from <code>out</code>
111     */
112    public XZOutputStream(OutputStream out, FilterOptions[] filterOptions)
113            throws IOException {
114        this(out, filterOptions, XZ.CHECK_CRC64);
115    }
116
117    /**
118     * Creates a new XZ compressor using 1-4 filters and the specified
119     * integrity check type.
120     *
121     * @param out           output stream to which the compressed data
122     *                      will be written
123     * @param filterOptions array of filter options to use
124     * @param checkType     type of the integrity check,
125     *                      for example XZ.CHECK_CRC32
126     * @throws UnsupportedOptionsException invalid filter chain
127     * @throws java.io.IOException         may be thrown from <code>out</code>
128     */
129    public XZOutputStream(OutputStream out, FilterOptions[] filterOptions,
130                          int checkType) throws IOException {
131        this.out = out;
132        updateFilters(filterOptions);
133
134        streamFlags.checkType = checkType;
135        check = Check.getInstance(checkType);
136
137        encodeStreamHeader();
138    }
139
140    /**
141     * Updates the filter chain with a single filter.
142     * This is equivalent to passing a single-member FilterOptions array
143     * to <code>updateFilters(FilterOptions[])</code>.
144     *
145     * @param filterOptions new filter to use
146     * @throws UnsupportedOptionsException unsupported filter chain, or trying to change
147     *                                     the filter chain in the middle of a Block
148     */
149    public void updateFilters(FilterOptions filterOptions)
150            throws XZIOException {
151        FilterOptions[] opts = new FilterOptions[1];
152        opts[0] = filterOptions;
153        updateFilters(opts);
154    }
155
156    /**
157     * Updates the filter chain with 1-4 filters.
158     * <p>
159     * Currently this cannot be used to update e.g. LZMA2 options in the
160     * middle of a XZ Block. Use <code>endBlock()</code> to finish the
161     * current XZ Block before calling this function. The new filter chain
162     * will then be used for the next XZ Block.
163     *
164     * @param filterOptions new filter chain to use
165     * @throws UnsupportedOptionsException unsupported filter chain, or trying to change
166     *                                     the filter chain in the middle of a Block
167     */
168    public void updateFilters(FilterOptions[] filterOptions)
169            throws XZIOException {
170        if (blockEncoder != null) {
171            throw new UnsupportedOptionsException("Changing filter options "
172                    + "in the middle of a XZ Block not implemented");
173        }
174
175        if (filterOptions.length < 1 || filterOptions.length > 4) {
176            throw new UnsupportedOptionsException(
177                    "XZ filter chain must be 1-4 filters");
178        }
179
180        filtersSupportFlushing = true;
181        FilterEncoder[] newFilters = new FilterEncoder[filterOptions.length];
182        for (int i = 0; i < filterOptions.length; ++i) {
183            newFilters[i] = filterOptions[i].getFilterEncoder();
184            filtersSupportFlushing &= newFilters[i].supportsFlushing();
185        }
186
187        RawCoder.validate(newFilters);
188        filters = newFilters;
189    }
190
191    /**
192     * Writes one byte to be compressed.
193     *
194     * @throws XZIOException       XZ Stream has grown too big
195     * @throws XZIOException       <code>finish()</code> or <code>close()</code>
196     *                             was already called
197     * @throws java.io.IOException may be thrown by the underlying output stream
198     */
199    public void write(int b) throws IOException {
200        byte[] buf = new byte[]{(byte) b};
201        write(buf, 0, 1);
202    }
203
204    /**
205     * Writes an array of bytes to be compressed.
206     * The compressors tend to do internal buffering and thus the written
207     * data won't be readable from the compressed output immediately.
208     * Use <code>flush()</code> to force everything written so far to
209     * be written to the underlaying output stream, but be aware that
210     * flushing reduces compression ratio.
211     *
212     * @param buf buffer of bytes to be written
213     * @param off start offset in <code>buf</code>
214     * @param len number of bytes to write
215     * @throws XZIOException       XZ Stream has grown too big: total file size
216     *                             about 8&nbsp;EiB or the Index field exceeds
217     *                             16&nbsp;GiB; you shouldn't reach these sizes
218     *                             in practice
219     * @throws XZIOException       <code>finish()</code> or <code>close()</code>
220     *                             was already called and len &gt; 0
221     * @throws java.io.IOException may be thrown by the underlying output stream
222     */
223    public void write(byte[] buf, int off, int len) throws IOException {
224        if (off < 0 || len < 0 || off + len < 0 || off + len > buf.length) {
225            throw new IndexOutOfBoundsException();
226        }
227
228        if (exception != null) {
229            throw exception;
230        }
231
232        if (finished) {
233            throw new XZIOException("Stream finished or closed");
234        }
235
236        try {
237            if (blockEncoder == null) {
238                blockEncoder = new BlockOutputStream(out, filters, check);
239            }
240
241            blockEncoder.write(buf, off, len);
242        } catch (IOException e) {
243            exception = e;
244            throw e;
245        }
246    }
247
248    /**
249     * Finishes the current XZ Block (but not the whole XZ Stream).
250     * This doesn't flush the stream so it's possible that not all data will
251     * be decompressible from the output stream when this function returns.
252     * Call also <code>flush()</code> if flushing is wanted in addition to
253     * finishing the current XZ Block.
254     * <p>
255     * If there is no unfinished Block open, this function will do nothing.
256     * (No empty XZ Block will be created.)
257     * <p>
258     * This function can be useful, for example, to create
259     * random-accessible .xz files.
260     * <p>
261     * Starting a new XZ Block means that the encoder state is reset.
262     * Doing this very often will increase the size of the compressed
263     * file a lot (more than plain <code>flush()</code> would do).
264     *
265     * @throws XZIOException       XZ Stream has grown too big
266     * @throws XZIOException       stream finished or closed
267     * @throws java.io.IOException may be thrown by the underlying output stream
268     */
269    public void endBlock() throws IOException {
270        if (exception != null) {
271            throw exception;
272        }
273
274        if (finished) {
275            throw new XZIOException("Stream finished or closed");
276        }
277
278        // NOTE: Once there is threading with multiple Blocks, it's possible
279        // that this function will be more like a barrier that returns
280        // before the last Block has been finished.
281        if (blockEncoder != null) {
282            try {
283                blockEncoder.finish();
284                index.add(blockEncoder.getUnpaddedSize(),
285                        blockEncoder.getUncompressedSize());
286                blockEncoder = null;
287            } catch (IOException e) {
288                exception = e;
289                throw e;
290            }
291        }
292    }
293
294    /**
295     * Flushes the encoder and calls <code>out.flush()</code>.
296     * All buffered pending data will then be decompressible from
297     * the output stream.
298     * <p>
299     * Calling this function very often may increase the compressed
300     * file size a lot. The filter chain options may affect the size
301     * increase too. For example, with LZMA2 the HC4 match finder has
302     * smaller penalty with flushing than BT4.
303     * <p>
304     * Some filters don't support flushing. If the filter chain has
305     * such a filter, <code>flush()</code> will call <code>endBlock()</code>
306     * before flushing.
307     *
308     * @throws XZIOException       XZ Stream has grown too big
309     * @throws XZIOException       stream finished or closed
310     * @throws java.io.IOException may be thrown by the underlying output stream
311     */
312    public void flush() throws IOException {
313        if (exception != null) {
314            throw exception;
315        }
316
317        if (finished) {
318            throw new XZIOException("Stream finished or closed");
319        }
320
321        try {
322            if (blockEncoder != null) {
323                if (filtersSupportFlushing) {
324                    // This will eventually call out.flush() so
325                    // no need to do it here again.
326                    blockEncoder.flush();
327                } else {
328                    endBlock();
329                    out.flush();
330                }
331            } else {
332                out.flush();
333            }
334        } catch (IOException e) {
335            exception = e;
336            throw e;
337        }
338    }
339
340    /**
341     * Finishes compression without closing the underlying stream.
342     * No more data can be written to this stream after finishing
343     * (calling <code>write</code> with an empty buffer is OK).
344     * <p>
345     * Repeated calls to <code>finish()</code> do nothing unless
346     * an exception was thrown by this stream earlier. In that case
347     * the same exception is thrown again.
348     * <p>
349     * After finishing, the stream may be closed normally with
350     * <code>close()</code>. If the stream will be closed anyway, there
351     * usually is no need to call <code>finish()</code> separately.
352     *
353     * @throws XZIOException       XZ Stream has grown too big
354     * @throws java.io.IOException may be thrown by the underlying output stream
355     */
356    public void finish() throws IOException {
357        if (!finished) {
358            // This checks for pending exceptions so we don't need to
359            // worry about it here.
360            endBlock();
361
362            try {
363                index.encode(out);
364                encodeStreamFooter();
365            } catch (IOException e) {
366                exception = e;
367                throw e;
368            }
369
370            // Set it to true only if everything goes fine. Setting it earlier
371            // would cause repeated calls to finish() do nothing instead of
372            // throwing an exception to indicate an earlier error.
373            finished = true;
374        }
375    }
376
377    /**
378     * Finishes compression and closes the underlying stream.
379     * The underlying stream <code>out</code> is closed even if finishing
380     * fails. If both finishing and closing fail, the exception thrown
381     * by <code>finish()</code> is thrown and the exception from the failed
382     * <code>out.close()</code> is lost.
383     *
384     * @throws XZIOException       XZ Stream has grown too big
385     * @throws java.io.IOException may be thrown by the underlying output stream
386     */
387    public void close() throws IOException {
388        if (out != null) {
389            // If finish() throws an exception, it stores the exception to
390            // the variable "exception". So we can ignore the possible
391            // exception here.
392            try {
393                finish();
394            } catch (IOException e) {
395            }
396
397            try {
398                out.close();
399            } catch (IOException e) {
400                // Remember the exception but only if there is no previous
401                // pending exception.
402                if (exception == null) {
403                    exception = e;
404                }
405            }
406
407            out = null;
408        }
409
410        if (exception != null) {
411            throw exception;
412        }
413    }
414
415    private void encodeStreamFlags(byte[] buf, int off) {
416        buf[off] = 0x00;
417        buf[off + 1] = (byte) streamFlags.checkType;
418    }
419
420    private void encodeStreamHeader() throws IOException {
421        out.write(XZ.HEADER_MAGIC);
422
423        byte[] buf = new byte[2];
424        encodeStreamFlags(buf, 0);
425        out.write(buf);
426
427        EncoderUtil.writeCRC32(out, buf);
428    }
429
430    private void encodeStreamFooter() throws IOException {
431        byte[] buf = new byte[6];
432        long backwardSize = index.getIndexSize() / 4 - 1;
433        for (int i = 0; i < 4; ++i) {
434            buf[i] = (byte) (backwardSize >>> (i * 8));
435        }
436
437        encodeStreamFlags(buf, 4);
438
439        EncoderUtil.writeCRC32(out, buf);
440        out.write(buf);
441        out.write(XZ.FOOTER_MAGIC);
442    }
