010package org.xbib.io.compress.xz;
011
012/**
013 * Thrown when the compressed input data is corrupt.
014 * However, it is possible that some or all of the data
015 * already read from the input stream was corrupt too.
016 */
017public class CorruptedInputException extends XZIOException {
018    private static final long serialVersionUID = 3L;
019
020    /**
021     * Creates a new CorruptedInputException with
022     * the default error detail message.
023     */
024    public CorruptedInputException() {
025        super("Compressed data is corrupt");
026    }
027
028    /**
029     * Creates a new CorruptedInputException with
030     * the specified error detail message.
031     *
032     * @param       s           error detail message
033     */
034    public CorruptedInputException(String s) {
035        super(s);
036    }
037}
