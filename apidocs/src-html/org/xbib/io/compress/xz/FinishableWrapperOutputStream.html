002    package org.xbib.io.compress.xz;
003    
004    import java.io.IOException;
005    import java.io.OutputStream;
006    
007    /**
008     * Wraps an output stream to a finishable output stream for use with
009     * raw encoders. This is not needed for XZ compression and thus most
010     * people will never need this.
011     */
012    public class FinishableWrapperOutputStream extends FinishableOutputStream {
013        /**
014         * The {@link java.io.OutputStream OutputStream} that has been
015         * wrapped into a FinishableWrapperOutputStream.
016         */
017        protected OutputStream out;
018    
019        /**
020         * Creates a new output stream which support finishing.
021         * The <code>finish()</code> method will do nothing.
022         */
023        public FinishableWrapperOutputStream(OutputStream out) {
024            this.out = out;
025        }
026    
027        /**
028         * Calls {@link java.io.OutputStream#write(int) out.write(b)}.
029         */
030        public void write(int b) throws IOException {
031            out.write(b);
032        }
033    
034        /**
035         * Calls {@link java.io.OutputStream#write(byte[]) out.write(buf)}.
036         */
037        public void write(byte[] buf) throws IOException {
038            out.write(buf);
039        }
040    
041        /**
042         * Calls {@link java.io.OutputStream#write(byte[], int, int)
043         * out.write(buf, off, len)}.
044         */
045        public void write(byte[] buf, int off, int len) throws IOException {
046            out.write(buf, off, len);
047        }
048    
049        /**
050         * Calls {@link java.io.OutputStream#flush() out.flush()}.
051         */
052        public void flush() throws IOException {
053            out.flush();
054        }
055    
056        /**
057         * Calls {@link java.io.OutputStream#close() out.close()}.
058         */
059        public void close() throws IOException {
060            out.close();
061        }
062    }
