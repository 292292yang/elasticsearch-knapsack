001package org.xbib.io.compress.xz;
002
003import java.io.IOException;
004import java.io.OutputStream;
005
006/**
007 * Wraps an output stream to a finishable output stream for use with
008 * raw encoders. This is not needed for XZ compression and thus most
009 * people will never need this.
010 */
011public class FinishableWrapperOutputStream extends FinishableOutputStream {
012    /**
013     * The {@link java.io.OutputStream OutputStream} that has been
014     * wrapped into a FinishableWrapperOutputStream.
015     */
016    protected OutputStream out;
017
018    /**
019     * Creates a new output stream which support finishing.
020     * The <code>finish()</code> method will do nothing.
021     */
022    public FinishableWrapperOutputStream(OutputStream out) {
023        this.out = out;
024    }
025
026    /**
027     * Calls {@link java.io.OutputStream#write(int) out.write(b)}.
028     */
029    public void write(int b) throws IOException {
030        out.write(b);
031    }
032
033    /**
034     * Calls {@link java.io.OutputStream#write(byte[]) out.write(buf)}.
035     */
036    public void write(byte[] buf) throws IOException {
037        out.write(buf);
038    }
039
040    /**
041     * Calls {@link java.io.OutputStream#write(byte[], int, int)
042     * out.write(buf, off, len)}.
043     */
044    public void write(byte[] buf, int off, int len) throws IOException {
045        out.write(buf, off, len);
046    }
047
048    /**
049     * Calls {@link java.io.OutputStream#flush() out.flush()}.
050     */
051    public void flush() throws IOException {
052        out.flush();
053    }
054
055    /**
056     * Calls {@link java.io.OutputStream#close() out.close()}.
057     */
058    public void close() throws IOException {
059        out.close();
060    }
061}
