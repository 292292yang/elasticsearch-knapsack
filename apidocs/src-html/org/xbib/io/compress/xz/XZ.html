002    package org.xbib.io.compress.xz;
003    
004    /**
005     * XZ constants.
006     */
007    public class XZ {
008        /**
009         * XZ Header Magic Bytes begin a XZ file.
010         * This can be useful to detect XZ compressed data.
011         */
012        public static final byte[] HEADER_MAGIC = {
013                (byte) 0xFD, '7', 'z', 'X', 'Z', '\0'};
014    
015        /**
016         * XZ Footer Magic Bytes are the last bytes of a XZ Stream.
017         */
018        public static final byte[] FOOTER_MAGIC = {'Y', 'Z'};
019    
020        /**
021         * Integrity check ID indicating that no integrity check is calculated.
022         * <p/>
023         * Omitting the integrity check is strongly discouraged except when
024         * the integrity of the data will be verified by other means anyway,
025         * and calculating the check twice would be useless.
026         */
027        public static final int CHECK_NONE = 0;
028    
029        /**
030         * Integrity check ID for CRC32.
031         */
032        public static final int CHECK_CRC32 = 1;
033    
034        /**
035         * Integrity check ID for CRC64.
036         */
037        public static final int CHECK_CRC64 = 4;
038    
039        /**
040         * Integrity check ID for SHA-256.
041         */
042        public static final int CHECK_SHA256 = 10;
043    
044        private XZ() {
045        }
046    }
