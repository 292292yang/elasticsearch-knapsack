002    package org.xbib.io.compress.xz;
003    
004    import org.xbib.io.compress.xz.simple.PowerPC;
005    
006    import java.io.InputStream;
007    
008    /**
009     * BCJ filter for big endian PowerPC instructions.
010     */
011    public class PowerPCOptions extends BCJOptions {
012        private static final int ALIGNMENT = 4;
013    
014        public PowerPCOptions() {
015            super(ALIGNMENT);
016        }
017    
018        public FinishableOutputStream getOutputStream(FinishableOutputStream out) {
019            return new SimpleOutputStream(out, new PowerPC(true, startOffset));
020        }
021    
022        public InputStream getInputStream(InputStream in) {
023            return new SimpleInputStream(in, new PowerPC(false, startOffset));
024        }
025    
026        FilterEncoder getFilterEncoder() {
027            return new BCJEncoder(this, BCJCoder.POWERPC_FILTER_ID);
028        }
029    }
