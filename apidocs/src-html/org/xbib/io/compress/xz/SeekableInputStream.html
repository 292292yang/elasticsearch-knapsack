002    package org.xbib.io.compress.xz;
003    
004    import java.io.IOException;
005    import java.io.InputStream;
006    
007    /**
008     * Input stream with random access support.
009     */
010    public abstract class SeekableInputStream extends InputStream {
011        /**
012         * Seeks <code>n</code> bytes forward in this stream.
013         * <p/>
014         * This will not seek past the end of the file. If the current position
015         * is already at or past the end of the file, this doesn't seek at all
016         * and returns <code>0</code>. Otherwise, if skipping <code>n</code> bytes
017         * would cause the position to exceed the stream size, this will do
018         * equivalent of <code>seek(length())</code> and the return value will
019         * be adjusted accordingly.
020         * <p/>
021         * If <code>n</code> is negative, the position isn't changed and
022         * the return value is <code>0</code>. It doesn't seek backward
023         * because it would conflict with the specification of
024         * {@link java.io.InputStream#skip(long) InputStream.skip}.
025         *
026         * @return <code>0</code> if <code>n</code> is negative,
027         * less than <code>n</code> if skipping <code>n</code>
028         * bytes would seek past the end of the file,
029         * <code>n</code> otherwise
030         * @throws java.io.IOException might be thrown by {@link #seek(long)}
031         */
032        public long skip(long n) throws IOException {
033            if (n <= 0) {
034                return 0;
035            }
036    
037            long size = length();
038            long pos = position();
039            if (pos >= size) {
040                return 0;
041            }
042    
043            if (size - pos < n) {
044                n = size - pos;
045            }
046    
047            seek(pos + n);
048            return n;
049        }
050    
051        /**
052         * Gets the size of the stream.
053         */
054        public abstract long length() throws IOException;
055    
056        /**
057         * Gets the current position in the stream.
058         */
059        public abstract long position() throws IOException;
060    
061        /**
062         * Seeks to the specified absolute position in the stream.
063         * <p/>
064         * Seeking past the end of the file should be supported by the subclasses
065         * unless there is a good reason to do otherwise. If one has seeked
066         * past the end of the stream, <code>read</code> will return
067         * <code>-1</code> to indicate end of stream.
068         *
069         * @param pos new read position in the stream
070         * @throws java.io.IOException if <code>pos</code> is negative or if
071         *                             a stream-specific I/O error occurs
072         */
073        public abstract void seek(long pos) throws IOException;
074    }
