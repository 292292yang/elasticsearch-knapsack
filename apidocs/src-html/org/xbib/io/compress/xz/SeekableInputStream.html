010package org.xbib.io.compress.xz;
011
012import java.io.InputStream;
013import java.io.IOException;
014
015/**
016 * Input stream with random access support.
017 */
018public abstract class SeekableInputStream extends InputStream {
019    /**
020     * Seeks <code>n</code> bytes forward in this stream.
021     * <p>
022     * This will not seek past the end of the file. If the current position
023     * is already at or past the end of the file, this doesn't seek at all
024     * and returns <code>0</code>. Otherwise, if skipping <code>n</code> bytes
025     * would cause the position to exceed the stream size, this will do
026     * equivalent of <code>seek(length())</code> and the return value will
027     * be adjusted accordingly.
028     * <p>
029     * If <code>n</code> is negative, the position isn't changed and
030     * the return value is <code>0</code>. It doesn't seek backward
031     * because it would conflict with the specification of
032     * {@link java.io.InputStream#skip(long) InputStream.skip}.
033     *
034     * @return      <code>0</code> if <code>n</code> is negative,
035     *              less than <code>n</code> if skipping <code>n</code>
036     *              bytes would seek past the end of the file,
037     *              <code>n</code> otherwise
038     *
039     * @throws      IOException might be thrown by {@link #seek(long)}
040     */
041    public long skip(long n) throws IOException {
042        if (n <= 0)
043            return 0;
044
045        long size = length();
046        long pos = position();
047        if (pos >= size)
048            return 0;
049
050        if (size - pos < n)
051            n = size - pos;
052
053        seek(pos + n);
054        return n;
055    }
056
057    /**
058     * Gets the size of the stream.
059     */
060    public abstract long length() throws IOException;
061
062    /**
063     * Gets the current position in the stream.
064     */
065    public abstract long position() throws IOException;
066
067    /**
068     * Seeks to the specified absolute position in the stream.
069     * <p>
070     * Seeking past the end of the file should be supported by the subclasses
071     * unless there is a good reason to do otherwise. If one has seeked
072     * past the end of the stream, <code>read</code> will return
073     * <code>-1</code> to indicate end of stream.
074     *
075     * @param       pos         new read position in the stream
076     *
077     * @throws      IOException if <code>pos</code> is negative or if
078     *                          a stream-specific I/O error occurs
079     */
080    public abstract void seek(long pos) throws IOException;
081}
