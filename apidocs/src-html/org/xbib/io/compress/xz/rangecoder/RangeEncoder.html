002package org.xbib.io.compress.xz.rangecoder;
003
004import java.io.IOException;
005import java.io.OutputStream;
006
007public final class RangeEncoder extends RangeCoder {
008    private static final int MOVE_REDUCING_BITS = 4;
009    private static final int BIT_PRICE_SHIFT_BITS = 4;
010
011    private static final int[] prices
012            = new int[BIT_MODEL_TOTAL >>> MOVE_REDUCING_BITS];
013
014    private long low;
015    private int range;
016
017    // NOTE: int is OK for LZMA2 because a compressed chunk
018    // is not more than 64 KiB, but with LZMA1 there is no chunking
019    // so in theory cacheSize can grow very big. To be very safe,
020    // use long instead of int if you adapt this code for LZMA1.
021    private int cacheSize;
022    private byte cache;
023
024    private final byte[] buf;
025    private int bufPos;
026
027    static {
028        for (int i = (1 << MOVE_REDUCING_BITS) / 2; i < BIT_MODEL_TOTAL;
029             i += (1 << MOVE_REDUCING_BITS)) {
030            int w = i;
031            int bitCount = 0;
032
033            for (int j = 0; j < BIT_PRICE_SHIFT_BITS; ++j) {
034                w *= w;
035                bitCount <<= 1;
036
037                while ((w & 0xFFFF0000) != 0) {
038                    w >>>= 1;
039                    ++bitCount;
040                }
041            }
042
043            prices[i >> MOVE_REDUCING_BITS]
044                    = (BIT_MODEL_TOTAL_BITS << BIT_PRICE_SHIFT_BITS)
045                    - 15 - bitCount;
046        }
047    }
048
049    public RangeEncoder(int bufSize) {
050        buf = new byte[bufSize];
051        reset();
052    }
053
054    public void reset() {
055        low = 0;
056        range = 0xFFFFFFFF;
057        cache = 0x00;
058        cacheSize = 1;
059        bufPos = 0;
060    }
061
062    public int getPendingSize() {
063        return bufPos + cacheSize + 5 - 1;
064    }
065
066    public int finish() {
067        for (int i = 0; i < 5; ++i) {
068            shiftLow();
069        }
070
071        return bufPos;
072    }
073
074    public void write(OutputStream out) throws IOException {
075        out.write(buf, 0, bufPos);
076    }
077
078    private void shiftLow() {
079        int lowHi = (int) (low >>> 32);
080
081        if (lowHi != 0 || low < 0xFF000000L) {
082            int temp = cache;
083
084            do {
085                buf[bufPos++] = (byte) (temp + lowHi);
086                temp = 0xFF;
087            } while (--cacheSize != 0);
088
089            cache = (byte) (low >>> 24);
090        }
091
092        ++cacheSize;
093        low = (low & 0x00FFFFFF) << 8;
094    }
095
096    public void encodeBit(short[] probs, int index, int bit) {
097        int prob = probs[index];
098        int bound = (range >>> BIT_MODEL_TOTAL_BITS) * prob;
099
100        // NOTE: Any non-zero value for bit is taken as 1.
101        if (bit == 0) {
102            range = bound;
103            probs[index] = (short) (
104                    prob + ((BIT_MODEL_TOTAL - prob) >>> MOVE_BITS));
105        } else {
106            low += bound & 0xFFFFFFFFL;
107            range -= bound;
108            probs[index] = (short) (prob - (prob >>> MOVE_BITS));
109        }
110
111        if ((range & TOP_MASK) == 0) {
112            range <<= SHIFT_BITS;
113            shiftLow();
114        }
115    }
116
117    public static int getBitPrice(int prob, int bit) {
118        // NOTE: Unlike in encodeBit(), here bit must be 0 or 1.
119        assert bit == 0 || bit == 1;
120        return prices[(prob ^ ((-bit) & (BIT_MODEL_TOTAL - 1)))
121                >>> MOVE_REDUCING_BITS];
122    }
123
124    public void encodeBitTree(short[] probs, int symbol) {
125        int index = 1;
126        int mask = probs.length;
127
128        do {
129            mask >>>= 1;
130            int bit = symbol & mask;
131            encodeBit(probs, index, bit);
132
133            index <<= 1;
134            if (bit != 0) {
135                index |= 1;
136            }
137
138        } while (mask != 1);
139    }
140
141    public static int getBitTreePrice(short[] probs, int symbol) {
142        int price = 0;
143        symbol |= probs.length;
144
145        do {
146            int bit = symbol & 1;
147            symbol >>>= 1;
148            price += getBitPrice(probs[symbol], bit);
149        } while (symbol != 1);
150
151        return price;
152    }
153
154    public void encodeReverseBitTree(short[] probs, int symbol) {
155        int index = 1;
156        symbol |= probs.length;
157
158        do {
159            int bit = symbol & 1;
160            symbol >>>= 1;
161            encodeBit(probs, index, bit);
162            index = (index << 1) | bit;
163        } while (symbol != 1);
164    }
165
166    public static int getReverseBitTreePrice(short[] probs, int symbol) {
167        int price = 0;
168        int index = 1;
169        symbol |= probs.length;
170
171        do {
172            int bit = symbol & 1;
173            symbol >>>= 1;
174            price += getBitPrice(probs[index], bit);
175            index = (index << 1) | bit;
176        } while (symbol != 1);
177
178        return price;
179    }
180
181    public void encodeDirectBits(int value, int count) {
182        do {
183            range >>>= 1;
184            low += range & (0 - ((value >>> --count) & 1));
185
186            if ((range & TOP_MASK) == 0) {
187                range <<= SHIFT_BITS;
188                shiftLow();
189            }
190        } while (count != 0);
191    }
192
193    public static int getDirectBitsPrice(int count) {
194        return count << BIT_PRICE_SHIFT_BITS;
195    }
196}
