001package org.xbib.io.compress.xz.rangecoder;
002
003import org.xbib.io.compress.xz.CorruptedInputException;
004
005import java.io.DataInputStream;
006import java.io.IOException;
007
008public final class RangeDecoder extends RangeCoder {
009    private static final int INIT_SIZE = 5;
010
011    private final byte[] buf;
012    private int pos = 0;
013    private int end = 0;
014
015    private int range = 0;
016    private int code = 0;
017
018    public RangeDecoder(int inputSizeMax) {
019        buf = new byte[inputSizeMax - INIT_SIZE];
020    }
021
022    public void prepareInputBuffer(DataInputStream in, int len)
023            throws IOException {
024        if (len < INIT_SIZE) {
025            throw new CorruptedInputException();
026        }
027
028        if (in.readUnsignedByte() != 0x00) {
029            throw new CorruptedInputException();
030        }
031
032        code = in.readInt();
033        range = 0xFFFFFFFF;
034
035        pos = 0;
036        end = len - INIT_SIZE;
037        in.readFully(buf, 0, end);
038    }
039
040    public boolean isInBufferOK() {
041        return pos <= end;
042    }
043
044    public boolean isFinished() {
045        return pos == end && code == 0;
046    }
047
048    public void normalize() throws IOException {
049        if ((range & TOP_MASK) == 0) {
050            try {
051                // If the input is corrupt, this might throw
052                // ArrayIndexOutOfBoundsException.
053                code = (code << SHIFT_BITS) | (buf[pos++] & 0xFF);
054                range <<= SHIFT_BITS;
055            } catch (ArrayIndexOutOfBoundsException e) {
056                throw new CorruptedInputException();
057            }
058        }
059    }
060
061    public int decodeBit(short[] probs, int index) throws IOException {
062        normalize();
063
064        int prob = probs[index];
065        int bound = (range >>> BIT_MODEL_TOTAL_BITS) * prob;
066        int bit;
067
068        // Compare code and bound as if they were unsigned 32-bit integers.
069        if ((code ^ 0x80000000) < (bound ^ 0x80000000)) {
070            range = bound;
071            probs[index] = (short) (
072                    prob + ((BIT_MODEL_TOTAL - prob) >>> MOVE_BITS));
073            bit = 0;
074        } else {
075            range -= bound;
076            code -= bound;
077            probs[index] = (short) (prob - (prob >>> MOVE_BITS));
078            bit = 1;
079        }
080
081        return bit;
082    }
083
084    public int decodeBitTree(short[] probs) throws IOException {
085        int symbol = 1;
086
087        do {
088            symbol = (symbol << 1) | decodeBit(probs, symbol);
089        } while (symbol < probs.length);
090
091        return symbol - probs.length;
092    }
093
094    public int decodeReverseBitTree(short[] probs) throws IOException {
095        int symbol = 1;
096        int i = 0;
097        int result = 0;
098
099        do {
100            int bit = decodeBit(probs, symbol);
101            symbol = (symbol << 1) | bit;
102            result |= bit << i++;
103        } while (symbol < probs.length);
104
105        return result;
106    }
107
108    public int decodeDirectBits(int count) throws IOException {
109        int result = 0;
110
111        do {
112            normalize();
113
114            range >>>= 1;
115            int t = (code - range) >>> 31;
116            code -= range & (t - 1);
117            result = (result << 1) | (1 - t);
118        } while (--count != 0);
119
120        return result;
121    }
122}
