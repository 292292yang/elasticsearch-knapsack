011package org.xbib.io.compress.xz.rangecoder;
012
013import java.util.Arrays;
014
015public abstract class RangeCoder {
016    static final int SHIFT_BITS = 8;
017    static final int TOP_MASK = 0xFF000000;
018    static final int BIT_MODEL_TOTAL_BITS = 11;
019    static final int BIT_MODEL_TOTAL = 1 << BIT_MODEL_TOTAL_BITS;
020    static final short PROB_INIT = (short)(BIT_MODEL_TOTAL / 2);
021    static final int MOVE_BITS = 5;
022
023    public static final void initProbs(short[] probs) {
024        Arrays.fill(probs, PROB_INIT);
025    }
026}
