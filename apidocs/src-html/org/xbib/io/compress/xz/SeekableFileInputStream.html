002package org.xbib.io.compress.xz;
003
004import java.io.File;
005import java.io.FileNotFoundException;
006import java.io.IOException;
007import java.io.RandomAccessFile;
008
009/**
010 * Wraps a {@link java.io.RandomAccessFile RandomAccessFile}
011 * in a SeekableInputStream.
012 */
013public class SeekableFileInputStream extends SeekableInputStream {
014    /**
015     * The RandomAccessFile that has been wrapped
016     * into a SeekableFileInputStream.
017     */
018    protected RandomAccessFile randomAccessFile;
019
020    /**
021     * Creates a new seekable input stream that reads from the specified file.
022     */
023    public SeekableFileInputStream(File file) throws FileNotFoundException {
024        randomAccessFile = new RandomAccessFile(file, "r");
025    }
026
027    /**
028     * Creates a new seekable input stream that reads from a file with
029     * the specified name.
030     */
031    public SeekableFileInputStream(String name) throws FileNotFoundException {
032        randomAccessFile = new RandomAccessFile(name, "r");
033    }
034
035    /**
036     * Creates a new seekable input stream from an existing
037     * <code>RandomAccessFile</code> object.
038     */
039    public SeekableFileInputStream(RandomAccessFile randomAccessFile) {
040        this.randomAccessFile = randomAccessFile;
041    }
042
043    /**
044     * Calls {@link java.io.RandomAccessFile#read() randomAccessFile.read()}.
045     */
046    public int read() throws IOException {
047        return randomAccessFile.read();
048    }
049
050    /**
051     * Calls {@link java.io.RandomAccessFile#read(byte[]) randomAccessFile.read(buf)}.
052     */
053    public int read(byte[] buf) throws IOException {
054        return randomAccessFile.read(buf);
055    }
056
057    /**
058     * Calls
059     * {@link java.io.RandomAccessFile#read(byte[], int, int)
060     * randomAccessFile.read(buf, off, len)}.
061     */
062    public int read(byte[] buf, int off, int len) throws IOException {
063        return randomAccessFile.read(buf, off, len);
064    }
065
066    /**
067     * Calls {@link java.io.RandomAccessFile#close() randomAccessFile.close()}.
068     */
069    public void close() throws IOException {
070        randomAccessFile.close();
071    }
072
073    /**
074     * Calls {@link java.io.RandomAccessFile#length() randomAccessFile.length()}.
075     */
076    public long length() throws IOException {
077        return randomAccessFile.length();
078    }
079
080    /**
081     * Calls {@link java.io.RandomAccessFile#getFilePointer()
082     * randomAccessFile.getFilePointer()}.
083     */
084    public long position() throws IOException {
085        return randomAccessFile.getFilePointer();
086    }
087
088    /**
089     * Calls {@link java.io.RandomAccessFile#seek(long) randomAccessFile.seek(long)}.
090     */
091    public void seek(long pos) throws IOException {
092        randomAccessFile.seek(pos);
093    }
094}
