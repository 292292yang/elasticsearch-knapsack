002    package org.xbib.io.compress;
003    
004    import org.xbib.io.StreamCodec;
005    import org.xbib.io.compress.bzip2.Bzip2InputStream;
006    import org.xbib.io.compress.bzip2.Bzip2OutputStream;
007    
008    import java.io.IOException;
009    import java.io.InputStream;
010    import java.io.OutputStream;
011    
012    public class Bzip2StreamCodec implements StreamCodec<Bzip2InputStream, Bzip2OutputStream> {
013    
014        @Override
015        public String getName() {
016            return "bz2";
017        }
018    
019        @Override
020        public Bzip2InputStream decode(InputStream in) throws IOException {
021            return new Bzip2InputStream(in);
022        }
023    
024        @Override
025        public Bzip2InputStream decode(InputStream in, int bufsize) throws IOException {
026            return new Bzip2InputStream(in, bufsize);
027        }
028    
029        @Override
030        public Bzip2OutputStream encode(OutputStream out) throws IOException {
031            return new Bzip2OutputStream(out);
032        }
033    
034        @Override
035        public Bzip2OutputStream encode(OutputStream out, int bufsize) throws IOException {
036            return new Bzip2OutputStream(out, bufsize);
037        }
038    }
