032        /**
033         * This method adds a single key/value parameter to the query
034         * string of a given URI. Existing keys will be overwritten.
035         *
036         * @param uri      the URI
037         * @param key      the key
038         * @param value    the value
039         * @param encoding the encoding
040         * @return the URI
041         * @throws java.net.URISyntaxException
042         */
043        public static URI addParameter(URI uri, String key, String value, Charset encoding)
044                throws URISyntaxException {
045            Map<String, String> m = parseQueryString(uri, encoding);
046            m.put(key, value);
047            String query = renderQueryString(m);
048            return new URI(uri.getScheme(), uri.getAuthority(), uri.getPath(), query, uri.getFragment());
049        }

052         * This method adds a map of key/value parameters to the query
053         * string of a given URI. Existing keys will be overwritten.
054         *
055         * @param uri      the URI
056         * @param map      the map
057         * @param encoding the encoding
058         * @return the URI
059         * @throws java.net.URISyntaxException
060         */
061        public static URI addParameter(URI uri, Map<String, String> map, Charset encoding)
062                throws URISyntaxException {
063            Map<String, String> oldMap = parseQueryString(uri, encoding);
064            oldMap.putAll(map);
065            String query = renderQueryString(oldMap);
066            return new URI(uri.getScheme(), uri.getAuthority(), uri.getPath(), query, uri.getFragment());
067        }

069        private static String concat(Collection<String> c) {
070            StringBuilder sb = new StringBuilder();
071            for (String s : c) {
072                if (sb.length() > 0) {
073                    sb.append(",");
074                }
075                sb.append(s);
076            }
077            return sb.toString();
078        }

080        /**
081         * Decodes an octet according to RFC 2396. According to this spec,
082         * any characters outside the range 0x20 - 0x7E must be escaped because
083         * they are not printable characters, except for any characters in the
084         * fragment identifier. This method will translate any escaped characters
085         * back to the original.
086         *
087         * @param s      the URI to decode
088         * @param encoding the encoding to decode into
089         * @return The decoded URI
090         */
091        public static String decode(String s, Charset encoding) {
092            if (s == null || s.isEmpty()) {
093                return null;
094            }
095            StringBuilder sb = new StringBuilder();
096            boolean fragment = false;
097            for (int i = 0; i < s.length(); i++) {
098                char ch = s.charAt(i);
099                switch (ch) {
100                    case '+':
101                        sb.append(' ');
102                        break;
103                    case '#':
104                        sb.append(ch);
105                        fragment = true;
106                        break;
107                    case '%':
108                        if (!fragment) {
109                            // fast hex decode
110                            sb.append((char) ((Character.digit(s.charAt(++i), 16) << 4)
111                                    | Character.digit(s.charAt(++i), 16)));
112                        } else {
113                            sb.append(ch);
114                        }
115                        break;
116                    default:
117                        sb.append(ch);
118                        break;
119                }
120            }
121            return new String(sb.toString().getBytes(LATIN1), encoding);
122        }

124        /**
125         * <p>Escape a string into URI syntax</p>
126         * <p>This function applies the URI escaping rules defined in
127         * section 2 of [RFC 2396], as amended by [RFC 2732], to the string
128         * supplied as the first argument, which typically represents all or part
129         * of a URI, URI reference or IRI. The effect of the function is to
130         * replace any special character in the string by an escape sequence of
131         * the form %xx%yy..., where xxyy... is the hexadecimal representation of
132         * the octets used to represent the character in US-ASCII for characters
133         * in the ASCII repertoire, and a different character encoding for
134         * non-ASCII characters.</p>
135         * <p>If the second argument is true, all characters are escaped
136         * other than lower case letters a-z, upper case letters A-Z, digits 0-9,
137         * and the characters referred to in [RFC 2396] as "marks": specifically,
138         * "-" | "_" | "." | "!" | "~" | "" | "'" | "(" | ")". The "%" character
139         * itself is escaped only if it is not followed by two hexadecimal digits
140         * (that is, 0-9, a-f, and A-F).</p>
141         * <p>[RFC 2396] does not define whether escaped URIs should use
142         * lower case or upper case for hexadecimal digits. To ensure that escaped
143         * URIs can be compared using string comparison functions, this function
144         * must always use the upper-case letters A-F.</p>
145         * <p>The character encoding used as the basis for determining the
146         * octets depends on the setting of the second argument.</p>
147         *
148         * @param s        the String to convert
149         * @param encoding The encoding to use for unsafe characters
150         * @return The converted String
151         */
152        public static String encode(String s, Charset encoding) {
153            if (s == null || s.isEmpty()) {
154                return null;
155            }
156            int length = s.length();
157            int start = 0;
158            int i = 0;
159            StringBuilder result = new StringBuilder(length);
160            while (true) {
161                while ((i < length) && isSafe(s.charAt(i))) {
162                    i++;
163                }
164                // Safe character can just be added
165                result.append(s.substring(start, i));
166                // Are we done?
167                if (i >= length) {
168                    return result.toString();
169                } else if (s.charAt(i) == ' ') {
170                    result.append('+'); // Replace space char with plus symbol.
171                    i++;
172                } else {
173                    // Get all unsafe characters
174                    start = i;
175                    char c;
176                    while ((i < length) && ((c = s.charAt(i)) != ' ') && !isSafe(c)) {
177                        i++;
178                    }
179                    // Convert them to %XY encoded strings
180                    String unsafe = s.substring(start, i);
181                    byte[] bytes = unsafe.getBytes(encoding);
182                    for (byte aByte : bytes) {
183                        result.append('%');
184                        result.append(hex.charAt(((int) aByte & 0xf0) >> 4));
185                        result.append(hex.charAt((int) aByte & 0x0f));
186                    }
187                }
188                start = i;
189            }
190        }

192        /**
193         * This method takes a String of an URI with an unescaped query
194         * string and converts it into a URI with encoded query string format.
195         * Useful for processing command line input.
196         *
197         * @param s the URI string
198         * @return a string with the URL encoded data.
199         * @throws java.net.URISyntaxException
200         */
201        public static URI encodeQueryString(String s) throws URISyntaxException {
202            if (s == null) {
203                return null;
204            }
205            StringBuilder out = new StringBuilder();
206            int questionmark = s.indexOf('?');
207            if (questionmark > 0) {
208                StringTokenizer st = new StringTokenizer(s.substring(questionmark + 1), "&");
209                while (st.hasMoreTokens()) {
210                    String pair = st.nextToken();
211                    int pos = pair.indexOf('=');
212                    if (pos == -1) {
213                        throw new URISyntaxException(s, "missing '='");
214                    }
215                    if (out.length() > 0) {
216                        out.append("&");
217                    }
218                    out.append(pair.substring(0, pos + 1)).append(encode(pair.substring(pos + 1), UTF8));
219                }
220                return new URI(s.substring(0, questionmark + 1) + out.toString());
221            } else {
222                return new URI(s);
223            }
224        }

226        /**
227         * Get properties from URI
228         *
229         * @param uri the URI
230         * @return the properties
231         */
232        public static Properties getPropertiesFromURI(URI uri) {
233            if (uri == null) {
234                throw new IllegalArgumentException("uri must not be null");
235            }
236            Properties properties = new Properties();
237            properties.setProperty("uri", uri.toString());
238            String scheme = uri.getScheme();
239            // ensure scheme is not null
240            scheme = scheme != null ? scheme : "";
241            properties.setProperty("scheme", scheme);
242            String type;
243            if (scheme.startsWith("jdbc:")) {
244                int pos = scheme.substring(5).indexOf(':');
245                type = (pos > 0) ? scheme.substring(5).substring(0, pos) : "default";
246                properties.setProperty("type", type);
247            }
248            String host = uri.getHost();
249            host = (host != null) ? host : "";
250            int port = uri.getPort();
251            String path = uri.getPath();
252            String[] s = (path != null) ? path.split("/") : new String[]{};
253            String cluster = (s.length > 1) ? s[1] : "";
254            String collection = (s.length > 2) ? s[2] : "";
255            String userInfo = uri.getUserInfo();
256            String[] ui = (userInfo != null) ? userInfo.split(":") : new String[]{};
257            properties.setProperty("username", (ui.length > 0) ? ui[0] : "");
258            properties.setProperty("password", (ui.length > 1) ? ui[1] : "");
259            Map<String, String> m = parseQueryString(uri);
260            if ((m != null) && (m.size() > 0)) {
261                properties.setProperty("requestkeys", concat(m.keySet()));
262                properties.setProperty("requestquery", renderQueryString(m));
263                properties.putAll(m);
264            }
265            properties.setProperty("host", host);
266            if (port != 0) {
267                properties.setProperty("port", Integer.toString(port));
268            }
269            if (path != null) {
270                properties.setProperty("path", path);
271            }
272            if (uri.getFragment() != null) {
273                properties.setProperty("fragment", uri.getFragment());
274            }
275            if (cluster != null) {
276                properties.setProperty("cluster", cluster);
277                properties.setProperty("index", cluster);
278            }
279            if (collection != null) {
280                properties.setProperty("collection", collection);
281                properties.setProperty("type", collection);
282            }
283            return properties;
284        }

286        /**
287         * Returns true if the given char is
288         * either a uppercase or lowercase letter from 'a' till 'z', or a digit
289         * froim '0' till '9', or one of the characters '-', '_', '.' or ''. Such
290         * 'safe' character don't have to be url encoded.
291         *
292         * @param c the character
293         * @return true or false
294         */
295        private static boolean isSafe(char c) {
296            return (((c >= 'a') && (c <= 'z')) || ((c >= 'A') && (c <= 'Z'))
297                    || ((c >= '0') && (c <= '9')) || (c == '-') || (c == '_') || (c == '.') || (c == '*'));
298        }

325        /**
326         * This method parses a query string and returns a map of decoded
327         * request parameters. We do not rely on java.net.URI because it does not
328         * decode plus characters. A listener can process the parameters in order.
329         *
330         * @param uri      the URI to examine for request parameters
331         * @param encoding the encoding
332         * @param listener a listner for processing the URI parameters in order, or null
333         * @return a Map of parameters
334         */
335        public static Map<String, String> parseQueryString(URI uri, Charset encoding, ParameterListener listener) {
336            Map<String, String> m = new HashMap<String, String>();
337            if (uri == null) {
338                throw new IllegalArgumentException();
339            }
340            if (uri.getRawQuery() == null) {
341                return m;
342            }
343            // we use getRawQuery because we do our decoding by ourselves
344            StringTokenizer st = new StringTokenizer(uri.getRawQuery(), "&");
345            while (st.hasMoreTokens()) {
346                String pair = st.nextToken();
347                String k;
348                String v;
349                int pos = pair.indexOf('=');
350                if (pos < 0) {
351                    k = pair;
352                    v = null;
353                } else {
354                    k = pair.substring(0, pos);
355                    v = decode(pair.substring(pos + 1, pair.length()), encoding);
356                }
357                m.put(k, v);
358                if (listener != null) {
359                    listener.received(k, v);
360                }
361            }
362            return m;
363        }

365        /**
366         * This method takes a Map of key/value elements and converts it
367         * into a URL encoded querystring format.
368         *
369         * @param m a map of key/value arrays
370         * @return a string with the URL encoded data
371         */
372        public static String renderQueryString(Map<String, String> m) {
373            String key;
374            String value;
375            StringBuilder out = new StringBuilder();
376            Charset encoding = m.containsKey("encoding") ? Charset.forName(m.get("encoding")) : UTF8;
377            for (Map.Entry<String, String> me : m.entrySet()) {
378                key = me.getKey();
379                String o = me.getValue();
380                value = o != null ? encode(o, encoding) : null;
381                if (key != null) {
382                    if (out.length() > 0) {
383                        out.append("&");
384                    }
385                    out.append(key);
386                    if ((value != null) && (value.length() > 0)) {
387                        out.append("=").append(value);
388                    }
389                }
390            }
391            return out.toString();
392        }

394        /**
395         * This method takes a Map of key/value elements and generates a
396         * string for queries.
397         *
398         * @param m a map of key/value arrays.
399         * @return a string
400         */
401        public static String renderRawQueryString(Map<String, String> m) {
402            String key;
403            String value;
404            StringBuilder out = new StringBuilder();
405            for (Map.Entry<String, String> me : m.entrySet()) {
406                key = me.getKey();
407                value = me.getValue();
408                if ((key != null) && (value != null) && (value.length() > 0)) {
409                    if (out.length() > 0) {
410                        out.append("&");
411                    }
412                    out.append(key).append("=").append(value);
413                }
414            }
415            return out.toString();
416        }
