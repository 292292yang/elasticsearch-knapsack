018    static final ZipEncoding DEFAULT_ENCODING =
019            ZipEncodingHelper.getZipEncoding(null);

021    static final ZipEncoding FALLBACK_ENCODING = new ZipEncoding() {
022        public boolean canEncode(String name) {
023            return true;
024        }
025
026        public ByteBuffer encode(String name) {
027            final int length = name.length();
028            byte[] buf = new byte[length];
029
030            // copy until end of input or output is reached.
031            for (int i = 0; i < length; ++i) {
032                buf[i] = (byte) name.charAt(i);
033            }
034            return ByteBuffer.wrap(buf);
035        }
036
037        public String decode(byte[] buffer) {
038            final int length = buffer.length;
039            StringBuilder result = new StringBuilder(length);
040
041            for (byte b : buffer) {
042                if (b == 0) { // Trailing null
043                    break;
044                }
045                result.append((char) (b & 0xFF)); // Allow for sign-extension
046            }
047
048            return result.toString();
049        }
050    };

058    /**
059     * Parse an octal string from a buffer.
060     * <p/>
061     * <p>Leading spaces are ignored.
062     * The buffer must contain a trailing space or NUL,
063     * and may contain an additional trailing space or NUL.</p>
064     * <p/>
065     * <p>The input buffer is allowed to contain all NULs,
066     * in which case the method returns 0L
067     * (this allows for missing fields).</p>
068     * <p/>
069     * <p>To work-around some tar implementations that insert a
070     * leading NUL this method returns 0 if it detects a leading NUL.</p>
071     *
072     * @param buffer The buffer from which to parse.
073     * @param offset The offset into the buffer from which to parse.
074     * @param length The maximum number of bytes to parse - must be at least 2 bytes.
075     * @return The long value of the octal string.
076     * @throws IllegalArgumentException if the trailing space/NUL is missing or if a invalid byte is detected.
077     */
078    public static long parseOctal(final byte[] buffer, final int offset, final int length) {
079        long result = 0;
080        int end = offset + length;
081        int start = offset;
082
083        if (length < 2) {
084            throw new IllegalArgumentException("Length " + length + " must be at least 2");
085        }
086
087        if (buffer[start] == 0) {
088            return 0L;
089        }
090
091        // Skip leading spaces
092        while (start < end) {
093            if (buffer[start] == ' ') {
094                start++;
095            } else {
096                break;
097            }
098        }
099
100        // Must have trailing NUL or space
101        byte trailer;
102        trailer = buffer[end - 1];
103        if (trailer == 0 || trailer == ' ') {
104            end--;
105        } else {
106            throw new IllegalArgumentException(
107                    exceptionMessage(buffer, offset, length, end - 1, trailer));
108        }
109        // May have additional NUL or space
110        trailer = buffer[end - 1];
111        if (trailer == 0 || trailer == ' ') {
112            end--;
113        }
114
115        for (; start < end; start++) {
116            final byte currentByte = buffer[start];
117            // CheckStyle:MagicNumber OFF
118            if (currentByte < '0' || currentByte > '7') {
119                throw new IllegalArgumentException(
120                        exceptionMessage(buffer, offset, length, start, currentByte));
121            }
122            result = (result << 3) + (currentByte - '0'); // convert from ASCII
123            // CheckStyle:MagicNumber ON
124        }
125
126        return result;
127    }

129    /**
130     * Compute the value contained in a byte buffer.  If the most
131     * significant bit of the first byte in the buffer is set, this
132     * bit is ignored and the rest of the buffer is interpreted as a
133     * binary number.  Otherwise, the buffer is interpreted as an
134     * octal number as per the parseOctal function above.
135     *
136     * @param buffer The buffer from which to parse.
137     * @param offset The offset into the buffer from which to parse.
138     * @param length The maximum number of bytes to parse.
139     * @return The long value of the octal or binary string.
140     * @throws IllegalArgumentException if the trailing space/NUL is
141     *                                  missing or an invalid byte is detected in an octal number, or
142     *                                  if a binary number would exceed the size of a signed long
143     *                                  64-bit integer.
144     */
145    public static long parseOctalOrBinary(final byte[] buffer, final int offset,
146                                          final int length) {
147
148        if ((buffer[offset] & 0x80) == 0) {
149            return parseOctal(buffer, offset, length);
150        }
151        final boolean negative = buffer[offset] == (byte) 0xff;
152        if (length < 9) {
153            return parseBinaryLong(buffer, offset, length, negative);
154        }
155        return parseBinaryBigInteger(buffer, offset, length, negative);
156    }
157
158    private static long parseBinaryLong(final byte[] buffer, final int offset,
159                                        final int length,
160                                        final boolean negative) {
161        if (length >= 9) {
162            throw new IllegalArgumentException("At offset " + offset + ", "
163                    + length + " byte binary number"
164                    + " exceeds maximum signed long"
165                    + " value");
166        }
167        long val = 0;
168        for (int i = 1; i < length; i++) {
169            val = (val << 8) + (buffer[offset + i] & 0xff);
170        }
171        if (negative) {
172            // 2's complement
173            val--;
174            val ^= ((long) Math.pow(2, (length - 1) * 8) - 1);
175        }
176        return negative ? -val : val;
177    }
178
179    private static long parseBinaryBigInteger(final byte[] buffer,
180                                              final int offset,
181                                              final int length,
182                                              final boolean negative) {
183        byte[] remainder = new byte[length - 1];
184        System.arraycopy(buffer, offset + 1, remainder, 0, length - 1);
185        BigInteger val = new BigInteger(remainder);
186        if (negative) {
187            // 2's complement
188            val = val.add(BigInteger.valueOf(-1)).not();
189        }
190        if (val.bitLength() > 63) {
191            throw new IllegalArgumentException("At offset " + offset + ", "
192                    + length + " byte binary number"
193                    + " exceeds maximum signed long"
194                    + " value");
195        }
196        return negative ? -val.longValue() : val.longValue();
197    }

199    /**
200     * Parse a boolean byte from a buffer.
201     * Leading spaces and NUL are ignored.
202     * The buffer may contain trailing spaces or NULs.
203     *
204     * @param buffer The buffer from which to parse.
205     * @param offset The offset into the buffer from which to parse.
206     * @return The boolean value of the bytes.
207     * @throws IllegalArgumentException if an invalid byte is detected.
208     */
209    public static boolean parseBoolean(final byte[] buffer, final int offset) {
210        return buffer[offset] == 1;
211    }

213    // Helper method to generate the exception message
214    private static String exceptionMessage(byte[] buffer, final int offset,
215                                           final int length, int current, final byte currentByte) {
216        String string = new String(buffer, offset, length); // TODO default charset?
217        string = string.replaceAll("\0", "{NUL}"); // Replace NULs to allow string to be printed
218        final String s = "Invalid byte " + currentByte + " at offset " + (current - offset) + " in '" + string + "' len=" + length;
219        return s;
220    }

222    /**
223     * Parse an entry name from a buffer.
224     * Parsing stops when a NUL is found
225     * or the buffer length is reached.
226     *
227     * @param buffer The buffer from which to parse.
228     * @param offset The offset into the buffer from which to parse.
229     * @param length The maximum number of bytes to parse.
230     * @return The entry name.
231     */
232    public static String parseName(byte[] buffer, final int offset, final int length) {
233        try {
234            return parseName(buffer, offset, length, DEFAULT_ENCODING);
235        } catch (IOException ex) {
236            try {
237                return parseName(buffer, offset, length, FALLBACK_ENCODING);
238            } catch (IOException ex2) {
239                // impossible
240                throw new RuntimeException(ex2);
241            }
242        }
243    }

245    /**
246     * Parse an entry name from a buffer.
247     * Parsing stops when a NUL is found
248     * or the buffer length is reached.
249     *
250     * @param buffer   The buffer from which to parse.
251     * @param offset   The offset into the buffer from which to parse.
252     * @param length   The maximum number of bytes to parse.
253     * @param encoding name of the encoding to use for file names
254     * @return The entry name.
255     */
256    public static String parseName(byte[] buffer, final int offset,
257                                   final int length,
258                                   final ZipEncoding encoding)
259            throws IOException {
260
261        int len = length;
262        for (; len > 0; len--) {
263            if (buffer[offset + len - 1] != 0) {
264                break;
265            }
266        }
267        if (len > 0) {
268            byte[] b = new byte[len];
269            System.arraycopy(buffer, offset, b, 0, len);
270            return encoding.decode(b);
271        }
272        return "";
273    }

304    /**
305     * Copy a name into a buffer.
306     * Copies characters from the name into the buffer
307     * starting at the specified offset.
308     * If the buffer is longer than the name, the buffer
309     * is filled with trailing NULs.
310     * If the name is longer than the buffer,
311     * the output is truncated.
312     *
313     * @param name     The header name from which to copy the characters.
314     * @param buf      The buffer where the name is to be stored.
315     * @param offset   The starting offset into the buffer
316     * @param length   The maximum number of header bytes to copy.
317     * @param encoding name of the encoding to use for file names
318     * @return The updated offset, i.e. offset + length
319     */
320    public static int formatNameBytes(String name, byte[] buf, final int offset,
321                                      final int length,
322                                      final ZipEncoding encoding)
323            throws IOException {
324        int len = name.length();
325        ByteBuffer b = encoding.encode(name);
326        while (b.limit() > length && len > 0) {
327            b = encoding.encode(name.substring(0, --len));
328        }
329        final int limit = b.limit();
330        System.arraycopy(b.array(), b.arrayOffset(), buf, offset, limit);
331
332        // Pad any remaining output bytes with NUL
333        for (int i = limit; i < length; ++i) {
334            buf[offset + i] = 0;
335        }
336
337        return offset + length;
338    }

340    /**
341     * Fill buffer with unsigned octal number, padded with leading zeroes.
342     *
343     * @param value  number to convert to octal - treated as unsigned
344     * @param buffer destination buffer
345     * @param offset starting offset in buffer
346     * @param length length of buffer to fill
347     * @throws IllegalArgumentException if the value will not fit in the buffer
348     */
349    public static void formatUnsignedOctalString(final long value, byte[] buffer,
350                                                 final int offset, final int length) {
351        int remaining = length;
352        remaining--;
353        if (value == 0) {
354            buffer[offset + remaining--] = (byte) '0';
355        } else {
356            long val = value;
357            for (; remaining >= 0 && val != 0; --remaining) {
358                // CheckStyle:MagicNumber OFF
359                buffer[offset + remaining] = (byte) ((byte) '0' + (byte) (val & 7));
360                val = val >>> 3;
361                // CheckStyle:MagicNumber ON
362            }
363            if (val != 0) {
364                throw new IllegalArgumentException
365                        (value + "=" + Long.toOctalString(value) + " will not fit in octal number buffer of length " + length);
366            }
367        }
368
369        for (; remaining >= 0; --remaining) { // leading zeros
370            buffer[offset + remaining] = (byte) '0';
371        }
372    }

374    /**
375     * Write an octal integer into a buffer.
376     * <p/>
377     * Uses {@link #formatUnsignedOctalString} to format
378     * the value as an octal string with leading zeros.
379     * The converted number is followed by space and NUL
380     *
381     * @param value  The value to write
382     * @param buf    The buffer to receive the output
383     * @param offset The starting offset into the buffer
384     * @param length The size of the output buffer
385     * @return The updated offset, i.e offset+length
386     * @throws IllegalArgumentException if the value (and trailer) will not fit in the buffer
387     */
388    public static int formatOctalBytes(final long value, byte[] buf, final int offset, final int length) {
389
390        int idx = length - 2; // For space and trailing null
391        formatUnsignedOctalString(value, buf, offset, idx);
392
393        buf[offset + idx++] = (byte) ' '; // Trailing space
394        buf[offset + idx] = 0; // Trailing null
395
396        return offset + length;
397    }

423    /**
424     * Write an long integer into a buffer as an octal string if this
425     * will fit, or as a binary number otherwise.
426     * <p/>
427     * Uses {@link #formatUnsignedOctalString} to format
428     * the value as an octal string with leading zeros.
429     * The converted number is followed by a space.
430     *
431     * @param value  The value to write into the buffer.
432     * @param buf    The destination buffer.
433     * @param offset The starting offset into the buffer.
434     * @param length The length of the buffer.
435     * @return The updated offset.
436     * @throws IllegalArgumentException if the value (and trailer)
437     *                                  will not fit in the buffer.
438     */
439    public static int formatLongOctalOrBinaryBytes(
440            final long value, byte[] buf, final int offset, final int length) {
441
442        // Check whether we are dealing with UID/GID or SIZE field
443        final long maxAsOctalChar = length == TarConstants.UIDLEN ? TarConstants.MAXID : TarConstants.MAXSIZE;
444
445        final boolean negative = value < 0;
446        if (!negative && value <= maxAsOctalChar) { // OK to store as octal chars
447            return formatLongOctalBytes(value, buf, offset, length);
448        }
449
450        if (length < 9) {
451            formatLongBinary(value, buf, offset, length, negative);
452        }
453        formatBigIntegerBinary(value, buf, offset, length, negative);
454
455        buf[offset] = (byte) (negative ? 0xff : 0x80);
456        return offset + length;
457    }
458
459    private static void formatLongBinary(final long value, byte[] buf,
460                                         final int offset, final int length,
461                                         final boolean negative) {
462        final int bits = (length - 1) * 8;
463        final long max = 1l << bits;
464        long val = Math.abs(value);
465        if (val >= max) {
466            throw new IllegalArgumentException("Value " + value +
467                    " is too large for " + length + " byte field.");
468        }
469        if (negative) {
470            val ^= max - 1;
471            val |= 0xff << bits;
472            val++;
473        }
474        for (int i = offset + length - 1; i >= offset; i--) {
475            buf[i] = (byte) val;
476            val >>= 8;
477        }
478    }
479
480    private static void formatBigIntegerBinary(final long value, byte[] buf,
481                                               final int offset,
482                                               final int length,
483                                               final boolean negative) {
484        BigInteger val = BigInteger.valueOf(value);
485        final byte[] b = val.toByteArray();
486        final int len = b.length;
487        final int off = offset + length - len;
488        System.arraycopy(b, 0, buf, off, len);
489        final byte fill = (byte) (negative ? 0xff : 0);
490        for (int i = offset + 1; i < off; i++) {
491            buf[i] = fill;
492        }
493    }

495    /**
496     * Writes an octal value into a buffer.
497     * <p/>
498     * Uses {@link #formatUnsignedOctalString} to format
499     * the value as an octal string with leading zeros.
500     * The converted number is followed by NUL and then space.
501     *
502     * @param value  The value to convert
503     * @param buf    The destination buffer
504     * @param offset The starting offset into the buffer.
505     * @param length The size of the buffer.
506     * @return The updated value of offset, i.e. offset+length
507     * @throws IllegalArgumentException if the value (and trailer) will not fit in the buffer
508     */
509    public static int formatCheckSumOctalBytes(final long value, byte[] buf, final int offset, final int length) {
510
511        int idx = length - 2; // for NUL and space
512        formatUnsignedOctalString(value, buf, offset, idx);
513
514        buf[offset + idx++] = 0; // Trailing null
515        buf[offset + idx] = (byte) ' '; // Trailing space
516
517        return offset + length;
518    }

520    /**
521     * Compute the checksum of a tar entry header.
522     *
523     * @param buf The tar entry's header buffer.
524     * @return The computed checksum.
525     */
526    public static long computeCheckSum(final byte[] buf) {
527        long sum = 0;
528
529        for (int i = 0; i < buf.length; ++i) {
530            sum += BYTE_MASK & buf[i];
531        }
532
533        return sum;
534    }
