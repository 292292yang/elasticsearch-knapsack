002package org.xbib.io.archivers.tar;
003
004import java.io.IOException;
005
006/**
007 * This class represents a sparse entry in a Tar archive.
008 * <p/>
009 * <p/>
010 * The C structure for a sparse entry is:
011 * <pre>
012 * struct posix_header {
013 * struct sparse sp[21]; // TarConstants.SPARSELEN_GNU_SPARSE     - offset 0
014 * char isextended;      // TarConstants.ISEXTENDEDLEN_GNU_SPARSE - offset 504
015 * };
016 * </pre>
017 * Whereas, "struct sparse" is:
018 * <pre>
019 * struct sparse {
020 * char offset[12];   // offset 0
021 * char numbytes[12]; // offset 12
022 * };
023 * </pre>
024 */
025
026public class TarArchiveSparseEntry implements TarConstants {
027    /**
028     * If an extension sparse header follows.
029     */
030    private boolean isExtended;
031
032    /**
033     * Construct an entry from an archive's header bytes. File is set
034     * to null.
035     *
036     * @param headerBuf The header bytes from a tar archive entry.
037     * @throws java.io.IOException on unknown format
038     */
039    public TarArchiveSparseEntry(byte[] headerBuf) throws IOException {
040        int offset = 0;
041        offset += SPARSELEN_GNU_SPARSE;
042        isExtended = TarUtils.parseBoolean(headerBuf, offset);
043    }
044
045    public boolean isExtended() {
046        return isExtended;
047    }
048}
