017/**
018 * The TarOutputStream writes a UNIX tar archive as an OutputStream.
019 * Methods are provided to put entries, and then write their contents
020 * by writing to this stream using write().
021 */
022public class TarArchiveOutputStream extends ArchiveOutputStream {

148    public TarArchiveOutputStream(OutputStream os, int blockSize,
149                                  int recordSize, String encoding) {
150        out = new CountingOutputStream(os);
151        this.encoding = ZipEncodingHelper.getZipEncoding(encoding);
152
153        this.buffer = new TarBuffer(out, blockSize, recordSize);
154        this.assemLen = 0;
155        this.assemBuf = new byte[recordSize];
156        this.recordBuf = new byte[recordSize];
157    }

195    /**
196     * Ends the TAR archive without closing the underlying OutputStream.
197     * <p/>
198     * An archive consists of a series of file entries terminated by an
199     * end-of-archive entry, which consists of two 512 blocks of zero bytes.
200     * POSIX.1 requires two EOF records, like some other implementations.
201     *
202     * @throws java.io.IOException on error
203     */
204    @Override
205    public void finish() throws IOException {
206        if (finished) {
207            throw new IOException("This archive has already been finished");
208        }
209
210        if (haveUnclosedEntry) {
211            throw new IOException("This archives contains unclosed entries.");
212        }
213        writeEOFRecord();
214        writeEOFRecord();
215        buffer.flushBlock();
216        finished = true;
217    }
218
219    /**
220     * Closes the underlying OutputStream.
221     *
222     * @throws java.io.IOException on error
223     */
224    @Override
225    public void close() throws IOException {
226        if (!finished) {
227            finish();
228        }
229
230        if (!closed) {
231            buffer.close();
232            out.close();
233            closed = true;
234        }
235    }

246    /**
247     * Put an entry on the output stream. This writes the entry's
248     * header record and positions the output stream for writing
249     * the contents of the entry. Once this method is called, the
250     * stream is ready for calls to write() to write the entry's
251     * contents. Once the contents are written, closeArchiveEntry()
252     * <B>MUST</B> be called to ensure that all buffered data
253     * is completely written to the output stream.
254     *
255     * @param archiveEntry The TarEntry to be written to the archive.
256     * @throws java.io.IOException on error
257     * @throws ClassCastException  if archiveEntry is not an instance of TarArchiveEntry
258     */
259    @Override
260    public void putArchiveEntry(ArchiveEntry archiveEntry) throws IOException {
261        if (finished) {
262            throw new IOException("Stream has already been finished");
263        }
264        TarArchiveEntry entry = (TarArchiveEntry) archiveEntry;
265        Map<String, String> paxHeaders = new HashMap<String, String>();
266        final String entryName = entry.getName();
267        final byte[] nameBytes = encoding.encode(entryName).array();
268        boolean paxHeaderContainsPath = false;
269        if (nameBytes.length >= TarConstants.NAMELEN) {
270
271            if (longFileMode == LONGFILE_POSIX) {
272                paxHeaders.put("path", entryName);
273                paxHeaderContainsPath = true;
274            } else if (longFileMode == LONGFILE_GNU) {
275                // create a TarEntry for the LongLink, the contents
276                // of which are the entry's name
277                TarArchiveEntry longLinkEntry = new TarArchiveEntry(TarConstants.GNU_LONGLINK,
278                        TarConstants.LF_GNUTYPE_LONGNAME);
279
280                longLinkEntry.setEntrySize(nameBytes.length + 1); // +1 for NUL
281                putArchiveEntry(longLinkEntry);
282                write(nameBytes);
283                write(0); // NUL terminator
284                closeArchiveEntry();
285            } else if (longFileMode != LONGFILE_TRUNCATE) {
286                throw new RuntimeException("file name '" + entryName
287                        + "' is too long ( > "
288                        + TarConstants.NAMELEN + " bytes)");
289            }
290        }
291
292        if (bigNumberMode == BIGNUMBER_POSIX) {
293            addPaxHeadersForBigNumbers(paxHeaders, entry);
294        } else if (bigNumberMode != BIGNUMBER_STAR) {
295            failForBigNumbers(entry);
296        }
297
298        if (addPaxHeadersForNonAsciiNames && !paxHeaderContainsPath
299                && !ASCII.canEncode(entryName)) {
300            paxHeaders.put("path", entryName);
301        }
302
303        if (addPaxHeadersForNonAsciiNames
304                && (entry.isLink() || entry.isSymbolicLink())
305                && !ASCII.canEncode(entry.getLinkName())) {
306            paxHeaders.put("linkpath", entry.getLinkName());
307        }
308
309        if (paxHeaders.size() > 0) {
310            writePaxHeaders(entryName, paxHeaders);
311        }
312
313        entry.writeEntryHeader(recordBuf, encoding,
314                bigNumberMode == BIGNUMBER_STAR);
315        buffer.writeRecord(recordBuf);
316
317        currBytes = 0;
318
319        if (entry.isDirectory()) {
320            currSize = 0;
321        } else {
322            currSize = entry.getEntrySize();
323        }
324        currName = entryName;
325        haveUnclosedEntry = true;
326    }

328    /**
329     * Close an entry. This method MUST be called for all file
330     * entries that contain data. The reason is that we must
331     * buffer data written to the stream in order to satisfy
332     * the buffer's record based writes. Thus, there may be
333     * data fragments still being assembled that must be written
334     * to the output stream before this entry is closed and the
335     * next entry written.
336     *
337     * @throws java.io.IOException on error
338     */
339    @Override
340    public void closeArchiveEntry() throws IOException {
341        if (finished) {
342            throw new IOException("Stream has already been finished");
343        }
344        if (!haveUnclosedEntry) {
345            throw new IOException("No current entry to close");
346        }
347        if (assemLen > 0) {
348            for (int i = assemLen; i < assemBuf.length; ++i) {
349                assemBuf[i] = 0;
350            }
351
352            buffer.writeRecord(assemBuf);
353
354            currBytes += assemLen;
355            assemLen = 0;
356        }
357
358        if (currBytes < currSize) {
359            throw new IOException("entry '" + currName + "' closed at '"
360                    + currBytes
361                    + "' before the '" + currSize
362                    + "' bytes specified in the header were written");
363        }
364        haveUnclosedEntry = false;
365    }

367    /**
368     * Writes bytes to the current tar archive entry. This method
369     * is aware of the current entry and will throw an exception if
370     * you attempt to write bytes past the length specified for the
371     * current entry. The method is also (painfully) aware of the
372     * record buffering required by TarBuffer, and manages buffers
373     * that are not a multiple of recordsize in length, including
374     * assembling records from small buffers.
375     *
376     * @param wBuf       The buffer to write to the archive.
377     * @param wOffset    The offset in the buffer from which to get bytes.
378     * @param numToWrite The number of bytes to write.
379     * @throws java.io.IOException on error
380     */
381    @Override
382    public void write(byte[] wBuf, int wOffset, int numToWrite) throws IOException {
383        if ((currBytes + numToWrite) > currSize) {
384            throw new IOException("request to write '" + numToWrite
385                    + "' bytes exceeds size in header of '"
386                    + currSize + "' bytes for entry '"
387                    + currName + "'");
388
389            //
390            // We have to deal with assembly!!!
391            // The programmer can be writing little 32 byte chunks for all
392            // we know, and we must assemble complete records for writing.
393            // REVIEW Maybe this should be in TarBuffer? Could that help to
394            // eliminate some of the buffer copying.
395            //
396        }
397
398        if (assemLen > 0) {
399            if ((assemLen + numToWrite) >= recordBuf.length) {
400                int aLen = recordBuf.length - assemLen;
401
402                System.arraycopy(assemBuf, 0, recordBuf, 0,
403                        assemLen);
404                System.arraycopy(wBuf, wOffset, recordBuf,
405                        assemLen, aLen);
406                buffer.writeRecord(recordBuf);
407
408                currBytes += recordBuf.length;
409                wOffset += aLen;
410                numToWrite -= aLen;
411                assemLen = 0;
412            } else {
413                System.arraycopy(wBuf, wOffset, assemBuf, assemLen,
414                        numToWrite);
415
416                wOffset += numToWrite;
417                assemLen += numToWrite;
418                numToWrite = 0;
419            }
420        }
421
422        //
423        // When we get here we have EITHER:
424        // o An empty "assemble" buffer.
425        // o No bytes to write (numToWrite == 0)
426        //
427        while (numToWrite > 0) {
428            if (numToWrite < recordBuf.length) {
429                System.arraycopy(wBuf, wOffset, assemBuf, assemLen,
430                        numToWrite);
431
432                assemLen += numToWrite;
433
434                break;
435            }
436
437            buffer.writeRecord(wBuf, wOffset);
438
439            int num = recordBuf.length;
440
441            currBytes += num;
442            numToWrite -= num;
443            wOffset += num;
444        }
445    }

447    /**
448     * Writes a PAX extended header with the given map as contents.
449     */
450    void writePaxHeaders(String entryName,
451                         Map<String, String> headers) throws IOException {
452        String name = "./PaxHeaders.X/" + stripTo7Bits(entryName);
453        if (name.length() >= TarConstants.NAMELEN) {
454            name = name.substring(0, TarConstants.NAMELEN - 1);
455        }
456        TarArchiveEntry pex = new TarArchiveEntry(name,
457                TarConstants.LF_PAX_EXTENDED_HEADER_LC);
458
459        StringWriter w = new StringWriter();
460        for (Map.Entry<String, String> h : headers.entrySet()) {
461            String key = h.getKey();
462            String value = h.getValue();
463            int len = key.length() + value.length()
464                    + 3 /* blank, equals and newline */
465                    + 2 /* guess 9 < actual length < 100 */;
466            String line = len + " " + key + "=" + value + "\n";
467            int actualLength = line.getBytes(Charset.forName("UTF-8")).length;
468            while (len != actualLength) {
469                // Adjust for cases where length < 10 or > 100
470                // or where UTF-8 encoding isn't a single octet
471                // per character.
472                // Must be in loop as size may go from 99 to 100 in
473                // first pass so we'd need a second.
474                len = actualLength;
475                line = len + " " + key + "=" + value + "\n";
476                actualLength = line.getBytes(Charset.forName("UTF-8")).length;
477            }
478            w.write(line);
479        }
480        byte[] data = w.toString().getBytes(Charset.forName("UTF_8"));
481        pex.setEntrySize(data.length);
482        putArchiveEntry(pex);
483        write(data);
484        closeArchiveEntry();
485    }

487    private String stripTo7Bits(String name) {
488        final int length = name.length();
489        StringBuilder result = new StringBuilder(length);
490        for (int i = 0; i < length; i++) {
491            char stripped = (char) (name.charAt(i) & 0x7F);
492            if (stripped != 0) { // would be read as Trailing null
493                result.append(stripped);
494            }
495        }
496        return result.toString();
497    }

499    /**
500     * Write an EOF (end of archive) record to the tar archive.
501     * An EOF record consists of a record of all zeros.
502     */
503    private void writeEOFRecord() throws IOException {
504        for (int i = 0; i < recordBuf.length; ++i) {
505            recordBuf[i] = 0;
506        }
507
508        buffer.writeRecord(recordBuf);
509    }

516    private void addPaxHeadersForBigNumbers(Map<String, String> paxHeaders,
517                                            TarArchiveEntry entry) {
518        addPaxHeaderForBigNumber(paxHeaders, "size", entry.getEntrySize(),
519                TarConstants.MAXSIZE);
520        addPaxHeaderForBigNumber(paxHeaders, "gid", entry.getGroupId(),
521                TarConstants.MAXID);
522        addPaxHeaderForBigNumber(paxHeaders, "mtime",
523                entry.getLastModified().getTime() / 1000,
524                TarConstants.MAXSIZE);
525        addPaxHeaderForBigNumber(paxHeaders, "uid", entry.getUserId(),
526                TarConstants.MAXID);
527        // star extensions by J\u00f6rg Schilling
528        addPaxHeaderForBigNumber(paxHeaders, "SCHILY.devmajor",
529                entry.getDevMajor(), TarConstants.MAXID);
530        addPaxHeaderForBigNumber(paxHeaders, "SCHILY.devminor",
531                entry.getDevMinor(), TarConstants.MAXID);
532        // there is no PAX header for file mode
533        failForBigNumber("mode", entry.getMode(), TarConstants.MAXID);
534    }
535
536    private void addPaxHeaderForBigNumber(Map<String, String> paxHeaders,
537                                          String header, long value,
538                                          long maxValue) {
539        if (value < 0 || value > maxValue) {
540            paxHeaders.put(header, String.valueOf(value));
541        }
542    }
543
544    private void failForBigNumbers(TarArchiveEntry entry) {
545        failForBigNumber("entry size", entry.getEntrySize(), TarConstants.MAXSIZE);
546        failForBigNumber("group id", entry.getGroupId(), TarConstants.MAXID);
547        failForBigNumber("last modification time",
548                entry.getLastModified().getTime() / 1000,
549                TarConstants.MAXSIZE);
550        failForBigNumber("user id", entry.getUserId(), TarConstants.MAXID);
551        failForBigNumber("mode", entry.getMode(), TarConstants.MAXID);
552        failForBigNumber("major device number", entry.getDevMajor(),
553                TarConstants.MAXID);
554        failForBigNumber("minor device number", entry.getDevMinor(),
555                TarConstants.MAXID);
556    }
557
558    private void failForBigNumber(String field, long value, long maxValue) {
559        if (value < 0 || value > maxValue) {
560            throw new RuntimeException(field + " '" + value
561                    + "' is too big ( > "
562                    + maxValue + " )");
563        }
564    }
