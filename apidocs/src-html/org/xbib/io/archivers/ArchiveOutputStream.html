002package org.xbib.io.archivers;
003
004import java.io.IOException;
005import java.io.OutputStream;
006
007/**
008 * Archive output stream implementations are expected to override the
009 * {@link #write(byte[], int, int)} method to improve performance.
010 * They should also override {@link #close()} to ensure that any necessary
011 * trailers are added.
012 * <p/>
013 * <p>
014 * The normal sequence of calls for working with ArchiveOutputStreams is:
015 * + create ArchiveOutputStream object
016 * + write SFX header (optional, Zip only)
017 * + repeat as needed:
018 * - putArchiveEntry() (writes entry header)
019 * - write() (writes entry data)
020 * - closeArchiveEntry() (closes entry)
021 * + finish() (ends the addition of entries)
022 * + write additional data if format supports it (optional)
023 * + close()
024 * </p>
025 */
026public abstract class ArchiveOutputStream extends OutputStream {
027
028    /**
029     * Temporary buffer used for the {@link #write(int)} method
030     */
031    private final byte[] oneByte = new byte[1];
032
033
034    private static final int BYTE_MASK = 0xFF;
035
036    /**
037     * holds the number of bytes written to this stream
038     */
039    private long bytesWritten = 0;
040    // Methods specific to ArchiveOutputStream
041
042    /**
043     * Writes the headers for an archive entry to the output stream.
044     * The caller must then write the content to the stream and call
045     * {@link #closeArchiveEntry()} to complete the process.
046     *
047     * @param entry describes the entry
048     * @throws java.io.IOException
049     */
050    public abstract void putArchiveEntry(ArchiveEntry entry) throws IOException;
051
052    /**
053     * Closes the archive entry, writing any trailer information that may
054     * be required.
055     *
056     * @throws java.io.IOException
057     */
058    public abstract void closeArchiveEntry() throws IOException;
059
060    /**
061     * Finishes the addition of entries to this stream, without closing it.
062     * Additional data can be written, if the format supports it.
063     * <p/>
064     * The finish() method throws an Exception if the user forgets to close the entry
065     * .
066     *
067     * @throws java.io.IOException
068     */
069    public abstract void finish() throws IOException;
070
071    /**
072     * Writes a byte to the current archive entry.
073     * <p/>
074     * This method simply calls write( byte[], 0, 1 ).
075     * <p/>
076     * MUST be overridden if the {@link #write(byte[], int, int)} method
077     * is not overridden; may be overridden otherwise.
078     *
079     * @param b The byte to be written.
080     * @throws java.io.IOException on error
081     */
082    @Override
083    public void write(int b) throws IOException {
084        oneByte[0] = (byte) (b & BYTE_MASK);
085        write(oneByte, 0, 1);
086    }
087
088    /**
089     * Increments the counter of already written bytes.
090     * Doesn't increment if the EOF has been hit (read == -1)
091     *
092     * @param written the number of bytes written
093     */
094    protected void count(int written) {
095        count((long) written);
096    }
097
098    /**
099     * Increments the counter of already written bytes.
100     * Doesn't increment if the EOF has been hit (read == -1)
101     *
102     * @param written the number of bytes written
103     */
104    protected void count(long written) {
105        if (written != -1) {
106            bytesWritten = bytesWritten + written;
107        }
108    }
109
110    /**
111     * Returns the current number of bytes written to this stream.
112     *
113     * @return the number of written bytes
114     */
115    public long getBytesWritten() {
116        return bytesWritten;
117    }
118
119}
