002package org.xbib.io.archivers;
003
004import org.xbib.io.archivers.cpio.CpioArchiveEntry;
005import org.xbib.io.archivers.cpio.CpioArchiveInputStream;
006import org.xbib.io.archivers.cpio.CpioArchiveOutputStream;
007import org.xbib.io.archivers.cpio.CpioConstants;
008import org.xbib.io.archivers.tar.TarArchiveEntry;
009import org.xbib.io.archivers.tar.TarArchiveInputStream;
010import org.xbib.io.archivers.tar.TarArchiveOutputStream;
011import org.xbib.io.archivers.zip.ZipArchiveEntry;
012import org.xbib.io.archivers.zip.ZipArchiveInputStream;
013import org.xbib.io.archivers.zip.ZipArchiveOutputStream;
014
015import java.io.ByteArrayInputStream;
016import java.io.IOException;
017import java.io.InputStream;
018import java.io.OutputStream;
019
020/**
021 * <p>Factory to create Archive[In|Out]putStreams from names or the first bytes of
022 * the InputStream. In order add other implementations you should extend
023 * ArchiveStreamFactory and override the appropriate methods (and call their
024 * implementation from super of course).</p>
025 * <p/>
026 * Compressing a ZIP file:
027 * <p/>
028 * <pre>
029 * final OutputStream out = new FileOutputStream(output);
030 * ArchiveOutputStream os = ArchiveStreamFactory.createArchiveOutputStream(ArchiveStreamFactory.ZIP, out);
031 *
032 * os.putArchiveEntry(new ZipArchiveEntry("testdata/test1.xml"));
033 * IOUtils.copy(new FileInputStream(file1), os);
034 * os.closeArchiveEntry();
035 *
036 * os.putArchiveEntry(new ZipArchiveEntry("testdata/test2.xml"));
037 * IOUtils.copy(new FileInputStream(file2), os);
038 * os.closeArchiveEntry();
039 * os.close();
040 * </pre>
041 * <p/>
042 * Decompressing a ZIP file:
043 * <p/>
044 * <pre>
045 * final InputStream is = new FileInputStream(input);
046 * ArchiveInputStream in = ArchiveStreamFactory.createArchiveInputStream(ArchiveStreamFactory.ZIP, is);
047 * ZipArchiveEntry entry = (ZipArchiveEntry)in.getNextEntry();
048 * OutputStream out = new FileOutputStream(new File(dir, entry.getName()));
049 * IOUtils.copy(in, out);
050 * out.close();
051 * in.close();
052 * </pre>
053 */
054public final class ArchiveFactory {
055
056    /**
057     * Constant used to identify the CPIO archive format.
058     */
059    public static final String CPIO = "cpio";
060    /**
061     * Constant used to identify the TAR archive format.
062     */
063    public static final String TAR = "tar";
064    /**
065     * Constant used to identify the ZIP archive format.
066     */
067    public static final String ZIP = "zip";
068
069    private ArchiveFactory() {
070    }
071
072    public static ArchiveEntry createArchiveEntry(final String archiverName)
073            throws IOException {
074        if (archiverName == null) {
075            throw new IllegalArgumentException("archiver name must not be null");
076        }
077        if (ZIP.equalsIgnoreCase(archiverName)) {
078            return new ZipArchiveEntry();
079        }
080        if (TAR.equalsIgnoreCase(archiverName)) {
081            return new TarArchiveEntry();
082        }
083        if (CPIO.equalsIgnoreCase(archiverName)) {
084            return new CpioArchiveEntry(CpioConstants.FORMAT_NEW);
085        }
086        throw new IOException("archiver " + archiverName + " not found");
087    }
088
089    /**
090     * Create an archive input stream from an archiver name and an input stream.
091     *
092     * @param archiverName the archive name, i.e. "ar", "zip", "tar", "jar", "dump" or "cpio"
093     * @param in           the input stream
094     * @return the archive input stream
095     * @throws java.io.IOException      if the archiver name is not known
096     * @throws IllegalArgumentException if the archiver name or stream is null
097     */
098    public static ArchiveInputStream createArchiveInputStream(final String archiverName, final InputStream in)
099            throws IOException {
100        if (archiverName == null) {
101            throw new IllegalArgumentException("archiver name must not be null");
102        }
103        if (in == null) {
104            throw new IllegalArgumentException("input stream must not be null");
105        }
106        if (ZIP.equalsIgnoreCase(archiverName)) {
107            return new ZipArchiveInputStream(in);
108        }
109        if (TAR.equalsIgnoreCase(archiverName)) {
110            return new TarArchiveInputStream(in);
111        }
112        if (CPIO.equalsIgnoreCase(archiverName)) {
113            return new CpioArchiveInputStream(in);
114        }
115        throw new IOException("archiver " + archiverName + " not found");
116    }
117
118    /**
119     * Create an archive output stream from an archiver name and an input stream.
120     *
121     * @param archiverName the archive name, i.e. "ar", "zip", "tar", "jar" or "cpio"
122     * @param out          the output stream
123     * @return the archive output stream
124     * @throws java.io.IOException      if the archiver name is not known
125     * @throws IllegalArgumentException if the archiver name or stream is null
126     */
127    public static ArchiveOutputStream createArchiveOutputStream(final String archiverName, final OutputStream out)
128            throws IOException {
129        if (archiverName == null) {
130            throw new IllegalArgumentException("archiver name must not be null");
131        }
132        if (out == null) {
133            throw new IllegalArgumentException("output stream must not be null");
134        }
135        if (ZIP.equalsIgnoreCase(archiverName)) {
136            return new ZipArchiveOutputStream(out);
137        }
138        if (TAR.equalsIgnoreCase(archiverName)) {
139            return new TarArchiveOutputStream(out);
140        }
141        if (CPIO.equalsIgnoreCase(archiverName)) {
142            return new CpioArchiveOutputStream(out);
143        }
144        throw new IOException("archiver " + archiverName + " not found");
145    }
146
147    /**
148     * Create an archive input stream from an input stream and autodetect
149     * the archive type from the first few bytes of the stream. The InputStream
150     * must support the mark()/reset() operation like BufferedInputStream.
151     *
152     * @param in the input stream
153     * @return the archive input stream
154     * @throws java.io.IOException      if the archiver name is not known
155     * @throws IllegalArgumentException if the stream is null or does not support mark
156     */
157    public static ArchiveInputStream createArchiveInputStream(final InputStream in)
158            throws IOException {
159        if (in == null) {
160            throw new IllegalArgumentException("input stream must not be null");
161        }
162        if (!in.markSupported()) {
163            throw new IllegalArgumentException("mark is not supported");
164        }
165        final byte[] signature = new byte[12];
166        in.mark(signature.length);
167        int signatureLength = in.read(signature);
168        in.reset();
169        if (ZipArchiveInputStream.matches(signature, signatureLength)) {
170            return new ZipArchiveInputStream(in);
171        } else if (CpioArchiveInputStream.matches(signature, signatureLength)) {
172            return new CpioArchiveInputStream(in);
173        }
174        // Dump needs a bigger buffer to check the signature
175        final byte[] dumpsig = new byte[32];
176        in.mark(dumpsig.length);
177        signatureLength = in.read(dumpsig);
178        in.reset();
179        // Tar needs an even bigger buffer to check the signature; read the first block
180        final byte[] tarheader = new byte[512];
181        in.mark(tarheader.length);
182        signatureLength = in.read(tarheader);
183        in.reset();
184        if (TarArchiveInputStream.matches(tarheader, signatureLength)) {
185            return new TarArchiveInputStream(in);
186        }
187        if (signatureLength >= 512) {
188            try {
189                TarArchiveInputStream tais = new TarArchiveInputStream(new ByteArrayInputStream(tarheader));
190                tais.getNextEntry();
191                return new TarArchiveInputStream(in);
192            } catch (Exception e) {
193                // ignored
194            }
195        }
196        throw new IOException("no archiver found for the stream signature");
197    }
198}
