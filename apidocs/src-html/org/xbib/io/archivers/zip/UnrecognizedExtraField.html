002package org.xbib.io.archivers.zip;
003
004/**
005 * Simple placeholder for all those extra fields we don't want to deal
006 * with.
007 * <p/>
008 * <p>Assumes local file data and central directory entries are
009 * identical - unless told the opposite.</p>
010 */
011public class UnrecognizedExtraField implements ZipExtraField {
012
013    /**
014     * The Header-ID.
015     */
016    private ZipShort headerId;
017
018    /**
019     * Set the header id.
020     *
021     * @param headerId the header id to use
022     */
023    public void setHeaderId(ZipShort headerId) {
024        this.headerId = headerId;
025    }
026
027    /**
028     * Get the header id.
029     *
030     * @return the header id
031     */
032    public ZipShort getHeaderId() {
033        return headerId;
034    }
035
036    /**
037     * Extra field data in local file data - without
038     * Header-ID or length specifier.
039     */
040    private byte[] localData;
041
042    /**
043     * Set the extra field data in the local file data -
044     * without Header-ID or length specifier.
045     *
046     * @param data the field data to use
047     */
048    public void setLocalFileDataData(byte[] data) {
049        localData = ZipUtil.copy(data);
050    }
051
052    /**
053     * Get the length of the local data.
054     *
055     * @return the length of the local data
056     */
057    public ZipShort getLocalFileDataLength() {
058        return new ZipShort(localData.length);
059    }
060
061    /**
062     * Get the local data.
063     *
064     * @return the local data
065     */
066    public byte[] getLocalFileDataData() {
067        return ZipUtil.copy(localData);
068    }
069
070    /**
071     * Extra field data in central directory - without
072     * Header-ID or length specifier.
073     */
074    private byte[] centralData;
075
076    /**
077     * Set the extra field data in central directory.
078     *
079     * @param data the data to use
080     */
081    public void setCentralDirectoryData(byte[] data) {
082        centralData = ZipUtil.copy(data);
083    }
084
085    /**
086     * Get the central data length.
087     * If there is no central data, get the local file data length.
088     *
089     * @return the central data length
090     */
091    public ZipShort getCentralDirectoryLength() {
092        if (centralData != null) {
093            return new ZipShort(centralData.length);
094        }
095        return getLocalFileDataLength();
096    }
097
098    /**
099     * Get the central data.
100     *
101     * @return the central data if present, else return the local file data
102     */
103    public byte[] getCentralDirectoryData() {
104        if (centralData != null) {
105            return ZipUtil.copy(centralData);
106        }
107        return getLocalFileDataData();
108    }
109
110    /**
111     * @param data   the array of bytes.
112     * @param offset the source location in the data array.
113     * @param length the number of bytes to use in the data array.
114     * @see ZipExtraField#parseFromLocalFileData(byte[], int, int)
115     */
116    public void parseFromLocalFileData(byte[] data, int offset, int length) {
117        byte[] tmp = new byte[length];
118        System.arraycopy(data, offset, tmp, 0, length);
119        setLocalFileDataData(tmp);
120    }
121
122    /**
123     * @param data   the array of bytes.
124     * @param offset the source location in the data array.
125     * @param length the number of bytes to use in the data array.
126     * @see ZipExtraField#parseFromCentralDirectoryData(byte[], int, int)
127     */
128    public void parseFromCentralDirectoryData(byte[] data, int offset,
129                                              int length) {
130        byte[] tmp = new byte[length];
131        System.arraycopy(data, offset, tmp, 0, length);
132        setCentralDirectoryData(tmp);
133        if (localData == null) {
134            setLocalFileDataData(tmp);
135        }
136    }
137
138}
