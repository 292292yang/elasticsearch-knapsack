002    package org.xbib.io.archivers.zip;
003    
004    /**
005     * Info-ZIP Unicode Path Extra Field (0x7075):
006     * <p/>
007     * <p>Stores the UTF-8 version of the file name field as stored in the
008     * local header and central directory header.</p>
009     * <p/>
010     * <pre>
011     *         Value         Size        Description
012     *         -----         ----        -----------
013     * (UPath) 0x7075        Short       tag for this extra block type ("up")
014     *         TSize         Short       total data size for this block
015     *         Version       1 byte      version of this extra field, currently 1
016     *         NameCRC32     4 bytes     File Name Field CRC32 Checksum
017     *         UnicodeName   Variable    UTF-8 version of the entry File Name
018     * </pre>
019     */
020    public class UnicodePathExtraField extends AbstractUnicodeExtraField {
021    
022        public static final ZipShort UPATH_ID = new ZipShort(0x7075);
023    
024        public UnicodePathExtraField() {
025        }
026    
027        /**
028         * Assemble as unicode path extension from the name given as
029         * text as well as the encoded bytes actually written to the archive.
030         *
031         * @param text  The file name
032         * @param bytes the bytes actually written to the archive
033         * @param off   The offset of the encoded filename in <code>bytes</code>.
034         * @param len   The length of the encoded filename or comment in
035         *              <code>bytes</code>.
036         */
037        public UnicodePathExtraField(String text, byte[] bytes, int off, int len) {
038            super(text, bytes, off, len);
039        }
040    
041        /**
042         * Assemble as unicode path extension from the name given as
043         * text as well as the encoded bytes actually written to the archive.
044         *
045         * @param name  The file name
046         * @param bytes the bytes actually written to the archive
047         */
048        public UnicodePathExtraField(String name, byte[] bytes) {
049            super(name, bytes);
050        }
051    
052        public ZipShort getHeaderId() {
053            return UPATH_ID;
054        }
055    }
