002    package org.xbib.io.archivers.zip;
003    
004    import java.io.IOException;
005    import java.nio.ByteBuffer;
006    
007    /**
008     * An interface for encoders that do a pretty encoding of ZIP
009     * filenames.
010     * <p/>
011     * <p>There are mostly two implementations, one that uses java.nio
012     * {@link java.nio.charset.Charset Charset} and one implementation,
013     * which copes with simple 8 bit charsets, because java-1.4 did not
014     * support Cp437 in java.nio.</p>
015     * <p/>
016     * <p>The main reason for defining an own encoding layer comes from
017     * the problems with {@link String#getBytes(String)
018     * String.getBytes}, which encodes unknown characters as ASCII
019     * quotation marks ('?'). Quotation marks are per definition an
020     * invalid filename on some operating systems  like Windows, which
021     * leads to ignored ZIP entries.</p>
022     * <p/>
023     * <p>All implementations should implement this interface in a
024     * reentrant way.</p>
025     */
026    public interface ZipEncoding {
027        /**
028         * Check, whether the given string may be losslessly encoded using this
029         * encoding.
030         *
031         * @param name A filename or ZIP comment.
032         * @return Whether the given name may be encoded with out any losses.
033         */
034        boolean canEncode(String name);
035    
036        /**
037         * Encode a filename or a comment to a byte array suitable for
038         * storing it to a zip entry.
039         * <p/>
040         * <p>Examples for CP 437 (in pseudo-notation, right hand side is
041         * C-style notation):</p>
042         * <pre>
043         *  encode("\u20AC_for_Dollar.txt") = "%U20AC_for_Dollar.txt"
044         *  encode("\u00D6lf\u00E4sser.txt") = "\231lf\204sser.txt"
045         * </pre>
046         *
047         * @param name A filename or ZIP comment.
048         * @return A byte buffer with a backing array containing the
049         * encoded name.  Unmappable characters or malformed
050         * character sequences are mapped to a sequence of utf-16
051         * words encoded in the format <code>%Uxxxx</code>.  It is
052         * assumed, that the byte buffer is positioned at the
053         * beginning of the encoded result, the byte buffer has a
054         * backing array and the limit of the byte buffer points
055         * to the end of the encoded result.
056         * @throws java.io.IOException
057         */
058        ByteBuffer encode(String name) throws IOException;
059    
060        /**
061         * @param data The byte values to decode.
062         * @return The decoded string.
063         * @throws java.io.IOException
064         */
065        String decode(byte[] data) throws IOException;
066    }
