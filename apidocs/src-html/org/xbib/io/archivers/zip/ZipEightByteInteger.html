002package org.xbib.io.archivers.zip;
003
004import java.math.BigInteger;
005
006import static org.xbib.io.archivers.zip.ZipConstants.BYTE_MASK;
007
008/**
009 * Utility class that represents an eight byte integer with conversion
010 * rules for the big endian byte order of ZIP files.
011 */
012public final class ZipEightByteInteger {
013
014    private static final int BYTE_1 = 1;
015    private static final int BYTE_1_MASK = 0xFF00;
016    private static final int BYTE_1_SHIFT = 8;
017
018    private static final int BYTE_2 = 2;
019    private static final int BYTE_2_MASK = 0xFF0000;
020    private static final int BYTE_2_SHIFT = 16;
021
022    private static final int BYTE_3 = 3;
023    private static final long BYTE_3_MASK = 0xFF000000L;
024    private static final int BYTE_3_SHIFT = 24;
025
026    private static final int BYTE_4 = 4;
027    private static final long BYTE_4_MASK = 0xFF00000000L;
028    private static final int BYTE_4_SHIFT = 32;
029
030    private static final int BYTE_5 = 5;
031    private static final long BYTE_5_MASK = 0xFF0000000000L;
032    private static final int BYTE_5_SHIFT = 40;
033
034    private static final int BYTE_6 = 6;
035    private static final long BYTE_6_MASK = 0xFF000000000000L;
036    private static final int BYTE_6_SHIFT = 48;
037
038    private static final int BYTE_7 = 7;
039    private static final long BYTE_7_MASK = 0x7F00000000000000L;
040    private static final int BYTE_7_SHIFT = 56;
041
042    private static final int LEFTMOST_BIT_SHIFT = 63;
043    private static final byte LEFTMOST_BIT = (byte) 0x80;
044
045    private final BigInteger value;
046
047    public static final ZipEightByteInteger ZERO = new ZipEightByteInteger(0);
048
049    /**
050     * Create instance from a number.
051     *
052     * @param value the long to store as a ZipEightByteInteger
053     */
054    public ZipEightByteInteger(long value) {
055        this(BigInteger.valueOf(value));
056    }
057
058    /**
059     * Create instance from a number.
060     *
061     * @param value the BigInteger to store as a ZipEightByteInteger
062     */
063    public ZipEightByteInteger(BigInteger value) {
064        this.value = value;
065    }
066
067    /**
068     * Create instance from bytes.
069     *
070     * @param bytes the bytes to store as a ZipEightByteInteger
071     */
072    public ZipEightByteInteger(byte[] bytes) {
073        this(bytes, 0);
074    }
075
076    /**
077     * Create instance from the eight bytes starting at offset.
078     *
079     * @param bytes  the bytes to store as a ZipEightByteInteger
080     * @param offset the offset to start
081     */
082    public ZipEightByteInteger(byte[] bytes, int offset) {
083        value = ZipEightByteInteger.getValue(bytes, offset);
084    }
085
086    /**
087     * Get value as eight bytes in big endian byte order.
088     *
089     * @return value as eight bytes in big endian order
090     */
091    public byte[] getBytes() {
092        return ZipEightByteInteger.getBytes(value);
093    }
094
095    /**
096     * Get value as Java long.
097     *
098     * @return value as a long
099     */
100    public long getLongValue() {
101        return value.longValue();
102    }
103
104    /**
105     * Get value as Java long.
106     *
107     * @return value as a long
108     */
109    public BigInteger getValue() {
110        return value;
111    }
112
113    /**
114     * Get value as eight bytes in big endian byte order.
115     *
116     * @param value the value to convert
117     * @return value as eight bytes in big endian byte order
118     */
119    public static byte[] getBytes(long value) {
120        return getBytes(BigInteger.valueOf(value));
121    }
122
123    /**
124     * Get value as eight bytes in big endian byte order.
125     *
126     * @param value the value to convert
127     * @return value as eight bytes in big endian byte order
128     */
129    public static byte[] getBytes(BigInteger value) {
130        byte[] result = new byte[8];
131        long val = value.longValue();
132        result[0] = (byte) ((val & BYTE_MASK));
133        result[BYTE_1] = (byte) ((val & BYTE_1_MASK) >> BYTE_1_SHIFT);
134        result[BYTE_2] = (byte) ((val & BYTE_2_MASK) >> BYTE_2_SHIFT);
135        result[BYTE_3] = (byte) ((val & BYTE_3_MASK) >> BYTE_3_SHIFT);
136        result[BYTE_4] = (byte) ((val & BYTE_4_MASK) >> BYTE_4_SHIFT);
137        result[BYTE_5] = (byte) ((val & BYTE_5_MASK) >> BYTE_5_SHIFT);
138        result[BYTE_6] = (byte) ((val & BYTE_6_MASK) >> BYTE_6_SHIFT);
139        result[BYTE_7] = (byte) ((val & BYTE_7_MASK) >> BYTE_7_SHIFT);
140        if (value.testBit(LEFTMOST_BIT_SHIFT)) {
141            result[BYTE_7] |= LEFTMOST_BIT;
142        }
143        return result;
144    }
145
146    /**
147     * Helper method to get the value as a Java long from eight bytes
148     * starting at given array offset
149     *
150     * @param bytes  the array of bytes
151     * @param offset the offset to start
152     * @return the corresponding Java long value
153     */
154    public static long getLongValue(byte[] bytes, int offset) {
155        return getValue(bytes, offset).longValue();
156    }
157
158    /**
159     * Helper method to get the value as a Java BigInteger from eight
160     * bytes starting at given array offset
161     *
162     * @param bytes  the array of bytes
163     * @param offset the offset to start
164     * @return the corresponding Java BigInteger value
165     */
166    public static BigInteger getValue(byte[] bytes, int offset) {
167        long value = ((long) bytes[offset + BYTE_7] << BYTE_7_SHIFT) & BYTE_7_MASK;
168        value += ((long) bytes[offset + BYTE_6] << BYTE_6_SHIFT) & BYTE_6_MASK;
169        value += ((long) bytes[offset + BYTE_5] << BYTE_5_SHIFT) & BYTE_5_MASK;
170        value += ((long) bytes[offset + BYTE_4] << BYTE_4_SHIFT) & BYTE_4_MASK;
171        value += ((long) bytes[offset + BYTE_3] << BYTE_3_SHIFT) & BYTE_3_MASK;
172        value += ((long) bytes[offset + BYTE_2] << BYTE_2_SHIFT) & BYTE_2_MASK;
173        value += ((long) bytes[offset + BYTE_1] << BYTE_1_SHIFT) & BYTE_1_MASK;
174        value += ((long) bytes[offset] & BYTE_MASK);
175        BigInteger val = BigInteger.valueOf(value);
176        return (bytes[offset + BYTE_7] & LEFTMOST_BIT) == LEFTMOST_BIT
177                ? val.setBit(LEFTMOST_BIT_SHIFT) : val;
178    }
179
180    /**
181     * Helper method to get the value as a Java long from an eight-byte array
182     *
183     * @param bytes the array of bytes
184     * @return the corresponding Java long value
185     */
186    public static long getLongValue(byte[] bytes) {
187        return getLongValue(bytes, 0);
188    }
189
190    /**
191     * Helper method to get the value as a Java long from an eight-byte array
192     *
193     * @param bytes the array of bytes
194     * @return the corresponding Java BigInteger value
195     */
196    public static BigInteger getValue(byte[] bytes) {
197        return getValue(bytes, 0);
198    }
199
200    /**
201     * Override to make two instances with same value equal.
202     *
203     * @param o an object to compare
204     * @return true if the objects are equal
205     */
206    @Override
207    public boolean equals(Object o) {
208        return !(o == null || !(o instanceof ZipEightByteInteger))
209                && value.equals(((ZipEightByteInteger) o).getValue());
210    }
211
212    /**
213     * Override to make two instances with same value equal.
214     *
215     * @return the hashCode of the value stored in the ZipEightByteInteger
216     */
217    @Override
218    public int hashCode() {
219        return value.hashCode();
220    }
221
222    @Override
223    public String toString() {
224        return "ZipEightByteInteger value: " + value;
225    }
226}
