002    package org.xbib.io.archivers.zip;
003    
004    import java.util.zip.ZipException;
005    
006    /**
007     * General format of extra field data. <p>
008     * <p/>
009     * Extra fields usually appear twice per file, once in the local file data and
010     * once in the central directory. Usually they are the same, but they don't have
011     * to be. {@link java.util.zip.ZipOutputStream java.util.zip.ZipOutputStream}
012     * will only use the local file data in both places.</p>
013     */
014    public interface ZipExtraField {
015        /**
016         * The Header-ID.
017         *
018         * @return The HeaderId value
019         */
020        ZipShort getHeaderId();
021    
022        /**
023         * Length of the extra field in the local file data - without Header-ID or
024         * length specifier.
025         *
026         * @return The LocalFileDataLength value
027         */
028        ZipShort getLocalFileDataLength();
029    
030        /**
031         * Length of the extra field in the central directory - without Header-ID or
032         * length specifier.
033         *
034         * @return The CentralDirectoryLength value
035         */
036        ZipShort getCentralDirectoryLength();
037    
038        /**
039         * The actual data to put into local file data - without Header-ID or length
040         * specifier.
041         *
042         * @return The LocalFileDataData value
043         */
044        byte[] getLocalFileDataData();
045    
046        /**
047         * The actual data to put into central directory - without Header-ID or
048         * length specifier.
049         *
050         * @return The CentralDirectoryData value
051         */
052        byte[] getCentralDirectoryData();
053    
054        /**
055         * Populate data from this array as if it was in local file data.
056         *
057         * @param buffer the buffer to read data from
058         * @param offset offset into buffer to read data
059         * @param length the length of data
060         * @throws java.util.zip.ZipException on error
061         */
062        void parseFromLocalFileData(byte[] buffer, int offset, int length)
063                throws ZipException;
064    
065        /**
066         * Populate data from this array as if it was in central directory data.
067         *
068         * @param buffer the buffer to read data from
069         * @param offset offset into buffer to read data
070         * @param length the length of data
071         * @throws java.util.zip.ZipException on error
072         */
073        void parseFromCentralDirectoryData(byte[] buffer, int offset, int length)
074                throws ZipException;
075    }
