002    package org.xbib.io.archivers.zip;
003    
004    import java.nio.charset.Charset;
005    import java.util.zip.CRC32;
006    import java.util.zip.ZipException;
007    
008    /**
009     * A common base class for Unicode extra information extra fields.
010     */
011    public abstract class AbstractUnicodeExtraField implements ZipExtraField {
012    
013        private long nameCRC32;
014    
015        private byte[] unicodeName;
016    
017        private byte[] data;
018    
019        protected AbstractUnicodeExtraField() {
020        }
021    
022        /**
023         * Assemble as unicode extension from the name/comment and
024         * encoding of the orginal zip entry.
025         *
026         * @param text  The file name or comment.
027         * @param bytes The encoded of the filename or comment in the zip
028         *              file.
029         * @param off   The offset of the encoded filename or comment in
030         *              <code>bytes</code>.
031         * @param len   The length of the encoded filename or commentin
032         *              <code>bytes</code>.
033         */
034        protected AbstractUnicodeExtraField(String text, byte[] bytes, int off, int len) {
035            CRC32 crc32 = new CRC32();
036            crc32.update(bytes, off, len);
037            nameCRC32 = crc32.getValue();
038    
039            unicodeName = text.getBytes(Charset.forName("UTF-8"));
040        }
041    
042        /**
043         * Assemble as unicode extension from the name/comment and
044         * encoding of the orginal zip entry.
045         *
046         * @param text  The file name or comment.
047         * @param bytes The encoded of the filename or comment in the zip
048         *              file.
049         */
050        protected AbstractUnicodeExtraField(String text, byte[] bytes) {
051            this(text, bytes, 0, bytes.length);
052        }
053    
054        private void assembleData() {
055            if (unicodeName == null) {
056                return;
057            }
058    
059            data = new byte[5 + unicodeName.length];
060            // version 1
061            data[0] = 0x01;
062            System.arraycopy(ZipLong.getBytes(nameCRC32), 0, data, 1, 4);
063            System.arraycopy(unicodeName, 0, data, 5, unicodeName.length);
064        }
065    
066        /**
067         * @return The CRC32 checksum of the filename or comment as
068         * encoded in the central directory of the zip file.
069         */
070        public long getNameCRC32() {
071            return nameCRC32;
072        }
073    
074        /**
075         * @param nameCRC32 The CRC32 checksum of the filename as encoded
076         *                  in the central directory of the zip file to set.
077         */
078        public void setNameCRC32(long nameCRC32) {
079            this.nameCRC32 = nameCRC32;
080            data = null;
081        }
082    
083        /**
084         * @return The utf-8 encoded name.
085         */
086        public byte[] getUnicodeName() {
087            byte[] b = null;
088            if (unicodeName != null) {
089                b = new byte[unicodeName.length];
090                System.arraycopy(unicodeName, 0, b, 0, b.length);
091            }
092            return b;
093        }
094    
095        /**
096         * @param unicodeName The utf-8 encoded name to set.
097         */
098        public void setUnicodeName(byte[] unicodeName) {
099            if (unicodeName != null) {
100                this.unicodeName = new byte[unicodeName.length];
101                System.arraycopy(unicodeName, 0, this.unicodeName, 0,
102                        unicodeName.length);
103            } else {
104                this.unicodeName = null;
105            }
106            data = null;
107        }
108    
109        public byte[] getCentralDirectoryData() {
110            if (data == null) {
111                this.assembleData();
112            }
113            byte[] b = null;
114            if (data != null) {
115                b = new byte[data.length];
116                System.arraycopy(data, 0, b, 0, b.length);
117            }
118            return b;
119        }
120    
121        public ZipShort getCentralDirectoryLength() {
122            if (data == null) {
123                assembleData();
124            }
125            return new ZipShort(data.length);
126        }
127    
128        public byte[] getLocalFileDataData() {
129            return getCentralDirectoryData();
130        }
131    
132        public ZipShort getLocalFileDataLength() {
133            return getCentralDirectoryLength();
134        }
135    
136        public void parseFromLocalFileData(byte[] buffer, int offset, int length)
137                throws ZipException {
138    
139            if (length < 5) {
140                throw new ZipException("UniCode path extra data must have at least 5 bytes.");
141            }
142    
143            int version = buffer[offset];
144    
145            if (version != 0x01) {
146                throw new ZipException("Unsupported version [" + version
147                        + "] for UniCode path extra data.");
148            }
149    
150            nameCRC32 = ZipLong.getValue(buffer, offset + 1);
151            unicodeName = new byte[length - 5];
152            System.arraycopy(buffer, offset + 5, unicodeName, 0, length - 5);
153            data = null;
154        }
155    
156        /**
157         * Doesn't do anything special since this class always uses the
158         * same data in central directory and local file data.
159         */
160        public void parseFromCentralDirectoryData(byte[] buffer, int offset,
161                                                  int length)
162                throws ZipException {
163            parseFromLocalFileData(buffer, offset, length);
164        }
165    }
