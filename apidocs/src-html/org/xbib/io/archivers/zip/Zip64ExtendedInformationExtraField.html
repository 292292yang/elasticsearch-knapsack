001    
002    package org.xbib.io.archivers.zip;
003    
004    import java.util.zip.ZipException;
005    
006    import static org.xbib.io.archivers.zip.ZipConstants.DWORD;
007    import static org.xbib.io.archivers.zip.ZipConstants.WORD;
008    
009    /**
010     * Holds size and other extended information for entries that use Zip64
011     * features.
012     * <p/>
013     * <p>From {@link "http://www.pkware.com/documents/casestudies/APPNOTE.TXT PKWARE's APPNOTE.TXT"}
014     * <pre>
015     * Zip64 Extended Information Extra Field (0x0001):
016     *
017     *          The following is the layout of the zip64 extended
018     *          information "extra" block. If one of the size or
019     *          offset fields in the Local or Central directory
020     *          record is too small to hold the required data,
021     *          a Zip64 extended information record is created.
022     *          The order of the fields in the zip64 extended
023     *          information record is fixed, but the fields will
024     *          only appear if the corresponding Local or Central
025     *          directory record field is set to 0xFFFF or 0xFFFFFFFF.
026     *
027     *          Note: all fields stored in Intel low-byte/high-byte order.
028     *
029     *          Value      Size       Description
030     *          -----      ----       -----------
031     *  (ZIP64) 0x0001     2 bytes    Tag for this "extra" block type
032     *          Size       2 bytes    Size of this "extra" block
033     *          Original
034     *          Size       8 bytes    Original uncompressed file size
035     *          Compressed
036     *          Size       8 bytes    Size of compressed data
037     *          Relative Header
038     *          Offset     8 bytes    Offset of local header record
039     *          Disk Start
040     *          Number     4 bytes    Number of the disk on which
041     *                                this file starts
042     *
043     *          This entry in the Local header must include BOTH original
044     *          and compressed file size fields. If encrypting the
045     *          central directory and bit 13 of the general purpose bit
046     *          flag is set indicating masking, the value stored in the
047     *          Local Header for the original file size will be zero.
048     * </pre></p>
049     *
050     * <p>Currently this code doesn't support encrypting the
051     * central directory so the not about masking doesn't apply.</p>
052     *
053     * <p>The implementation relies on data being read from the local file
054     * header and assumes that both size values are always present.</p>
055     */
056    public class Zip64ExtendedInformationExtraField implements ZipExtraField {
057    
058        static final ZipShort HEADER_ID = new ZipShort(0x0001);
059    
060        private static final String LFH_MUST_HAVE_BOTH_SIZES_MSG =
061                "Zip64 extended information must contain"
062                        + " both size values in the local file header.";
063    
064        private ZipEightByteInteger size, compressedSize, relativeHeaderOffset;
065        private ZipLong diskStart;
066    
067        /**
068         * Stored in {@link #parseFromCentralDirectoryData
069         * parseFromCentralDirectoryData} so it can be reused when ZipFile
070         * calls {@link #reparseCentralDirectoryData
071         * reparseCentralDirectoryData}.
072         * <p/>
073         * <p>Not used for anything else</p>
074         */
075        private byte[] rawCentralDirectoryData;
076    
077        public Zip64ExtendedInformationExtraField() {
078        }
079    
080        /**
081         * Creates an extra field based on the original and compressed size.
082         *
083         * @param size           the entry's original size
084         * @param compressedSize the entry's compressed size
085         * @throws IllegalArgumentException if size or compressedSize is null
086         */
087        public Zip64ExtendedInformationExtraField(ZipEightByteInteger size,
088                                                  ZipEightByteInteger compressedSize) {
089            this(size, compressedSize, null, null);
090        }
091    
092        /**
093         * Creates an extra field based on all four possible values.
094         *
095         * @param size           the entry's original size
096         * @param compressedSize the entry's compressed size
097         * @throws IllegalArgumentException if size or compressedSize is null
098         */
099        public Zip64ExtendedInformationExtraField(ZipEightByteInteger size,
100                                                  ZipEightByteInteger compressedSize,
101                                                  ZipEightByteInteger relativeHeaderOffset,
102                                                  ZipLong diskStart) {
103            this.size = size;
104            this.compressedSize = compressedSize;
105            this.relativeHeaderOffset = relativeHeaderOffset;
106            this.diskStart = diskStart;
107        }
108    
109        public ZipShort getHeaderId() {
110            return HEADER_ID;
111        }
112    
113        public ZipShort getLocalFileDataLength() {
114            return new ZipShort(size != null ? 2 * DWORD : 0);
115        }
116    
117        public ZipShort getCentralDirectoryLength() {
118            return new ZipShort((size != null ? DWORD : 0)
119                    + (compressedSize != null ? DWORD : 0)
120                    + (relativeHeaderOffset != null ? DWORD : 0)
121                    + (diskStart != null ? WORD : 0));
122        }
123    
124        public byte[] getLocalFileDataData() {
125            if (size != null || compressedSize != null) {
126                if (size == null || compressedSize == null) {
127                    throw new IllegalArgumentException(LFH_MUST_HAVE_BOTH_SIZES_MSG);
128                }
129                byte[] data = new byte[2 * DWORD];
130                addSizes(data);
131                return data;
132            }
133            return new byte[0];
134        }
135    
136        public byte[] getCentralDirectoryData() {
137            byte[] data = new byte[getCentralDirectoryLength().getValue()];
138            int off = addSizes(data);
139            if (relativeHeaderOffset != null) {
140                System.arraycopy(relativeHeaderOffset.getBytes(), 0, data, off, DWORD);
141                off += DWORD;
142            }
143            if (diskStart != null) {
144                System.arraycopy(diskStart.getBytes(), 0, data, off, WORD);
145                off += WORD;
146            }
147            return data;
148        }
149    
150        public void parseFromLocalFileData(byte[] buffer, int offset, int length)
151                throws ZipException {
152            if (length == 0) {
153                // no local file data at all, may happen if an archive
154                // only holds a ZIP64 extended information extra field
155                // inside the central directory but not inside the local
156                // file header
157                return;
158            }
159            if (length < 2 * DWORD) {
160                throw new ZipException(LFH_MUST_HAVE_BOTH_SIZES_MSG);
161            }
162            size = new ZipEightByteInteger(buffer, offset);
163            offset += DWORD;
164            compressedSize = new ZipEightByteInteger(buffer, offset);
165            offset += DWORD;
166            int remaining = length - 2 * DWORD;
167            if (remaining >= DWORD) {
168                relativeHeaderOffset = new ZipEightByteInteger(buffer, offset);
169                offset += DWORD;
170                remaining -= DWORD;
171            }
172            if (remaining >= WORD) {
173                diskStart = new ZipLong(buffer, offset);
174                offset += WORD;
175                remaining -= WORD;
176            }
177        }
178    
179        public void parseFromCentralDirectoryData(byte[] buffer, int offset,
180                                                  int length)
181                throws ZipException {
182            // store for processing in reparseCentralDirectoryData
183            rawCentralDirectoryData = new byte[length];
184            System.arraycopy(buffer, offset, rawCentralDirectoryData, 0, length);
185    
186            // if there is no size information in here, we are screwed and
187            // can only hope things will get resolved by LFH data later
188            // But there are some cases that can be detected
189            // * all data is there
190            // * length == 24 -> both sizes and offset
191            // * length % 8 == 4 -> at least we can identify the diskStart field
192            if (length >= 3 * DWORD + WORD) {
193                parseFromLocalFileData(buffer, offset, length);
194            } else if (length == 3 * DWORD) {
195                size = new ZipEightByteInteger(buffer, offset);
196                offset += DWORD;
197                compressedSize = new ZipEightByteInteger(buffer, offset);
198                offset += DWORD;
199                relativeHeaderOffset = new ZipEightByteInteger(buffer, offset);
200            } else if (length % DWORD == WORD) {
201                diskStart = new ZipLong(buffer, offset + length - WORD);
202            }
203        }
204    
205        /**
206         * Parses the raw bytes read from the central directory extra
207         * field with knowledge which fields are expected to be there.
208         * <p/>
209         * <p>All four fields inside the zip64 extended information extra
210         * field are optional and only present if their corresponding
211         * entry inside the central directory contains the correct magic
212         * value.</p>
213         */
214        public void reparseCentralDirectoryData(boolean hasUncompressedSize,
215                                                boolean hasCompressedSize,
216                                                boolean hasRelativeHeaderOffset,
217                                                boolean hasDiskStart)
218                throws ZipException {
219            if (rawCentralDirectoryData != null) {
220                int expectedLength = (hasUncompressedSize ? DWORD : 0)
221                        + (hasCompressedSize ? DWORD : 0)
222                        + (hasRelativeHeaderOffset ? DWORD : 0)
223                        + (hasDiskStart ? WORD : 0);
224                if (rawCentralDirectoryData.length != expectedLength) {
225                    throw new ZipException("central directory zip64 extended"
226                            + " information extra field's length"
227                            + " doesn't match central directory"
228                            + " data.  Expected length "
229                            + expectedLength + " but is "
230                            + rawCentralDirectoryData.length);
231                }
232                int offset = 0;
233                if (hasUncompressedSize) {
234                    size = new ZipEightByteInteger(rawCentralDirectoryData, offset);
235                    offset += DWORD;
236                }
237                if (hasCompressedSize) {
238                    compressedSize = new ZipEightByteInteger(rawCentralDirectoryData,
239                            offset);
240                    offset += DWORD;
241                }
242                if (hasRelativeHeaderOffset) {
243                    relativeHeaderOffset =
244                            new ZipEightByteInteger(rawCentralDirectoryData, offset);
245                    offset += DWORD;
246                }
247                if (hasDiskStart) {
248                    diskStart = new ZipLong(rawCentralDirectoryData, offset);
249                    offset += WORD;
250                }
251            }
252        }
253    
254        /**
255         * The uncompressed size stored in this extra field.
256         */
257        public ZipEightByteInteger getSize() {
258            return size;
259        }
260    
261        /**
262         * The uncompressed size stored in this extra field.
263         */
264        public void setSize(ZipEightByteInteger size) {
265            this.size = size;
266        }
267    
268        /**
269         * The compressed size stored in this extra field.
270         */
271        public ZipEightByteInteger getCompressedSize() {
272            return compressedSize;
273        }
274    
275        /**
276         * The uncompressed size stored in this extra field.
277         */
278        public void setCompressedSize(ZipEightByteInteger compressedSize) {
279            this.compressedSize = compressedSize;
280        }
281    
282        /**
283         * The relative header offset stored in this extra field.
284         */
285        public ZipEightByteInteger getRelativeHeaderOffset() {
286            return relativeHeaderOffset;
287        }
288    
289        /**
290         * The relative header offset stored in this extra field.
291         */
292        public void setRelativeHeaderOffset(ZipEightByteInteger rho) {
293            relativeHeaderOffset = rho;
294        }
295    
296        /**
297         * The disk start number stored in this extra field.
298         */
299        public ZipLong getDiskStartNumber() {
300            return diskStart;
301        }
302    
303        /**
304         * The disk start number stored in this extra field.
305         */
306        public void setDiskStartNumber(ZipLong ds) {
307            diskStart = ds;
308        }
309    
310        private int addSizes(byte[] data) {
311            int off = 0;
312            if (size != null) {
313                System.arraycopy(size.getBytes(), 0, data, 0, DWORD);
314                off += DWORD;
315            }
316            if (compressedSize != null) {
317                System.arraycopy(compressedSize.getBytes(), 0, data, off, DWORD);
318                off += DWORD;
319            }
320            return off;
321        }
322    }
