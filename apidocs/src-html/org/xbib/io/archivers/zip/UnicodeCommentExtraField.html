002    package org.xbib.io.archivers.zip;
003    
004    /**
005     * Info-ZIP Unicode Comment Extra Field (0x6375):
006     * <p/>
007     * <p>Stores the UTF-8 version of the file comment as stored in the
008     * central directory header.</p>
009     * <p/>
010     * <pre>
011     *         Value         Size        Description
012     *         -----         ----        -----------
013     *  (UCom) 0x6375        Short       tag for this extra block type ("uc")
014     *         TSize         Short       total data size for this block
015     *         Version       1 byte      version of this extra field, currently 1
016     *         ComCRC32      4 bytes     Comment Field CRC32 Checksum
017     *         UnicodeCom    Variable    UTF-8 version of the entry comment
018     * </pre>
019     */
020    public class UnicodeCommentExtraField extends AbstractUnicodeExtraField {
021    
022        public static final ZipShort UCOM_ID = new ZipShort(0x6375);
023    
024        public UnicodeCommentExtraField() {
025        }
026    
027        /**
028         * Assemble as unicode comment extension from the name given as
029         * text as well as the encoded bytes actually written to the archive.
030         *
031         * @param text  The file name
032         * @param bytes the bytes actually written to the archive
033         * @param off   The offset of the encoded comment in <code>bytes</code>.
034         * @param len   The length of the encoded comment or comment in
035         *              <code>bytes</code>.
036         */
037        public UnicodeCommentExtraField(String text, byte[] bytes, int off, int len) {
038            super(text, bytes, off, len);
039        }
040    
041        public ZipShort getHeaderId() {
042            return UCOM_ID;
043        }
044    
045    }
