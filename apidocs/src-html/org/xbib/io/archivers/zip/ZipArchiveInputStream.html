024/**
025 * Implements an input stream that can read Zip archives.
026 * <p/>
027 * <p>Note that {@link ZipArchiveEntry#getSize()} may return -1 if the
028 * DEFLATE algorithm is used, as the size information is not available
029 * from the header.</p>
030 * <p/>
031 * <p>The {@link ZipFile} class is preferred when reading from files.</p>
032 * <p/>
033 * <p>This code transparently supports Zip64
034 * extensions and thus individual entries and archives larger than 4
035 * GB or with more than 65536 entries.</p>
036 *
037 * @see ZipFile
038 */
039public class ZipArchiveInputStream extends ArchiveInputStream {

098      local file header signature     4 bytes  (0x04034b50)

103      last mod file date              2 bytes
104      crc-32                          4 bytes
105      compressed size                 4 bytes
106      uncompressed size               4 bytes

148    public ZipArchiveEntry getNextZipEntry() throws IOException {
149        if (closed) {
150            throw new IOException("stream is closed");
151        }
152        if (hitCentralDirectory) {
153            return null;
154        }
155        if (current != null) {
156            closeEntry();
157        }
158        byte[] lfh = new byte[LFH_LEN];
159        try {
160            readFully(lfh);
161        } catch (EOFException e) {
162            return null;
163        }
164        ZipLong sig = new ZipLong(lfh);
165        if (sig.equals(ZipLong.CFH_SIG)) {
166            hitCentralDirectory = true;
167            return null;
168        }
169        if (!sig.equals(ZipLong.LFH_SIG)) {
170            return null;
171        }
172
173        int off = WORD;
174        current = new CurrentEntry();
175
176        int versionMadeBy = ZipShort.getValue(lfh, off);
177        off += SHORT;
178        current.entry.setPlatform((versionMadeBy >> ZipFile.BYTE_SHIFT)
179                & ZipFile.NIBLET_MASK);
180
181        final GeneralPurposeBit gpFlag = GeneralPurposeBit.parse(lfh, off);
182        final boolean hasUTF8Flag = gpFlag.usesUTF8ForNames();
183        final ZipEncoding entryEncoding =
184                hasUTF8Flag ? ZipEncodingHelper.UTF8_ZIP_ENCODING : zipEncoding;
185        current.hasDataDescriptor = gpFlag.usesDataDescriptor();
186        current.entry.setGeneralPurposeBit(gpFlag);
187
188        off += SHORT;
189
190        current.entry.setMethod(ZipShort.getValue(lfh, off));
191        off += SHORT;
192
193        long time = ZipUtil.dosToJavaTime(ZipLong.getValue(lfh, off));
194        current.entry.setTime(time);
195        off += WORD;
196
197        ZipLong size = null, cSize = null;
198        if (!current.hasDataDescriptor) {
199            current.entry.setCrc(ZipLong.getValue(lfh, off));
200            off += WORD;
201
202            cSize = new ZipLong(lfh, off);
203            off += WORD;
204
205            size = new ZipLong(lfh, off);
206            off += WORD;
207        } else {
208            off += 3 * WORD;
209        }
210
211        int fileNameLen = ZipShort.getValue(lfh, off);
212
213        off += SHORT;
214
215        int extraLen = ZipShort.getValue(lfh, off);
216        off += SHORT;
217
218        byte[] fileName = new byte[fileNameLen];
219        readFully(fileName);
220        current.entry.setName(entryEncoding.decode(fileName), fileName);
221
222        byte[] extraData = new byte[extraLen];
223        readFully(extraData);
224        current.entry.setExtra(extraData);
225
226        if (!hasUTF8Flag && useUnicodeExtraFields) {
227            ZipUtil.setNameAndCommentFromExtraFields(current.entry, fileName,
228                    null);
229        }
230
231        processZip64Extra(size, cSize);
232        return current.entry;
233    }

235    /**
236     * Records whether a Zip64 extra is present and sets the size
237     * information from it if sizes are 0xFFFFFFFF and the entry
238     * doesn't use a data descriptor.
239     */
240    private void processZip64Extra(ZipLong size, ZipLong cSize) {
241        Zip64ExtendedInformationExtraField z64 =
242                (Zip64ExtendedInformationExtraField)
243                        current.entry.getExtraField(Zip64ExtendedInformationExtraField
244                                .HEADER_ID);
245        current.usesZip64 = z64 != null;
246        if (!current.hasDataDescriptor) {
247            if (current.usesZip64 && (cSize.equals(ZipLong.ZIP64_MAGIC)
248                    || size.equals(ZipLong.ZIP64_MAGIC))
249                    ) {
250                current.entry.setCompressedSize(z64.getCompressedSize() // z64 cannot be null here
251                        .getLongValue());
252                current.entry.setSize(z64.getSize().getLongValue());
253            } else {
254                current.entry.setCompressedSize(cSize.getValue());
255                current.entry.setSize(size.getValue());
256            }
257        }
258    }

265    @Override
266    public int read(byte[] buffer, int start, int length) throws IOException {
267        if (closed) {
268            throw new IOException("stream is closed");
269        }
270        if (inf.finished() || current == null) {
271            return -1;
272        }
273
274        // avoid int overflow, check null buffer
275        if (start <= buffer.length && length >= 0 && start >= 0
276                && buffer.length - start >= length) {
277            ZipUtil.checkRequestedFeatures(current.entry);
278            if (!supportsDataDescriptorFor(current.entry)) {
279                throw new UnsupportedZipFeatureException(UnsupportedZipFeatureException
280                        .Feature
281                        .DATA_DESCRIPTOR,
282                        current.entry);
283            }
284
285            if (current.entry.getMethod() == ZipArchiveOutputStream.STORED) {
286                return readStored(buffer, start, length);
287            }
288            return readDeflated(buffer, start, length);
289        }
290        throw new ArrayIndexOutOfBoundsException();
291    }
292
293    /**
294     * Implementation of read for STORED entries.
295     */
296    private int readStored(byte[] buffer, int start, int length)
297            throws IOException {
298
299        if (current.hasDataDescriptor) {
300            if (lastStoredEntry == null) {
301                readStoredEntry();
302            }
303            return lastStoredEntry.read(buffer, start, length);
304        }
305
306        long csize = current.entry.getSize();
307        if (current.bytesRead >= csize) {
308            return -1;
309        }
310
311        if (buf.offsetInBuffer >= buf.lengthOfLastRead) {
312            buf.offsetInBuffer = 0;
313            if ((buf.lengthOfLastRead = in.read(buf.buf)) == -1) {
314                return -1;
315            }
316            count(buf.lengthOfLastRead);
317            current.bytesReadFromStream += buf.lengthOfLastRead;
318        }
319
320        int toRead = length > buf.lengthOfLastRead
321                ? buf.lengthOfLastRead - buf.offsetInBuffer
322                : length;
323        if ((csize - current.bytesRead) < toRead) {
324            // if it is smaller than toRead then it fits into an int
325            toRead = (int) (csize - current.bytesRead);
326        }
327        System.arraycopy(buf.buf, buf.offsetInBuffer, buffer, start, toRead);
328        buf.offsetInBuffer += toRead;
329        current.bytesRead += toRead;
330        crc.update(buffer, start, toRead);
331        return toRead;
332    }
333
334    /**
335     * Implementation of read for DEFLATED entries.
336     */
337    private int readDeflated(byte[] buffer, int start, int length)
338            throws IOException {
339        if (inf.needsInput()) {
340            fill();
341            if (buf.lengthOfLastRead > 0) {
342                current.bytesReadFromStream += buf.lengthOfLastRead;
343            }
344        }
345        int read;
346        try {
347            read = inf.inflate(buffer, start, length);
348        } catch (DataFormatException e) {
349            throw new ZipException(e.getMessage());
350        }
351        if (read == 0) {
352            if (inf.finished()) {
353                return -1;
354            } else if (buf.lengthOfLastRead == -1) {
355                throw new IOException("Truncated ZIP file");
356            }
357        }
358        crc.update(buffer, start, read);
359        return read;
360    }

371    /**
372     * Skips over and discards value bytes of data from this input
373     * stream.
374     * <p/>
375     * <p>This implementation may end up skipping over some smaller
376     * number of bytes, possibly 0, if and only if it reaches the end
377     * of the underlying stream.</p>
378     * <p/>
379     * <p>The actual number of bytes skipped is returned.</p>
380     *
381     * @param value the number of bytes to be skipped.
382     * @return the actual number of bytes skipped.
383     * @throws java.io.IOException      - if an I/O error occurs.
384     * @throws IllegalArgumentException - if value is negative.
385     */
386    @Override
387    public long skip(long value) throws IOException {
388        if (value >= 0) {
389            long skipped = 0;
390            byte[] b = new byte[1024];
391            while (skipped < value) {
392                long rem = value - skipped;
393                int x = read(b, 0, (int) (b.length > rem ? rem : b.length));
394                if (x == -1) {
395                    return skipped;
396                }
397                skipped += x;
398            }
399            return skipped;
400        }
401        throw new IllegalArgumentException();
402    }

404    /**
405     * Checks if the signature matches what is expected for a zip file.
406     * Does not currently handle self-extracting zips which may have arbitrary
407     * leading content.
408     *
409     * @param signature the bytes to check
410     * @param length    the number of bytes to check
411     * @return true, if this stream is a zip archive stream, false otherwise
412     */
413    public static boolean matches(byte[] signature, int length) {
414        if (length < ZipArchiveOutputStream.LFH_SIG.length) {
415            return false;
416        }
417
418        return checksig(signature, ZipArchiveOutputStream.LFH_SIG) // normal file
419                || checksig(signature, ZipArchiveOutputStream.EOCD_SIG); // empty zip
420    }
421
422    private static boolean checksig(byte[] signature, byte[] expected) {
423        for (int i = 0; i < expected.length; i++) {
424            if (signature[i] != expected[i]) {
425                return false;
426            }
427        }
428        return true;
429    }

431    /**
432     * Closes the current ZIP archive entry and positions the underlying
433     * stream to the beginning of the next entry. All per-entry variables
434     * and data structures are cleared.
435     * <p/>
436     * If the compressed size of this entry is included in the entry header,
437     * then any outstanding bytes are simply skipped from the underlying
438     * stream without uncompressing them. This allows an entry to be safely
439     * closed even if the compression method is unsupported.
440     * <p/>
441     * In case we don't know the compressed size of this entry or have
442     * already buffered too much data from the underlying stream to support
443     * uncompression, then the uncompression process is completed and the
444     * end position of the stream is adjusted based on the result of that
445     * process.
446     *
447     * @throws java.io.IOException if an error occurs
448     */
449    private void closeEntry() throws IOException {
450        if (closed) {
451            throw new IOException("The stream is closed");
452        }
453        if (current == null) {
454            return;
455        }
456
457        // Ensure all entry bytes are read
458        if (current.bytesReadFromStream <= current.entry.getCompressedSize()
459                && !current.hasDataDescriptor) {
460            drainCurrentEntryData();
461        } else {
462            skip(Long.MAX_VALUE);
463
464            long inB =
465                    current.entry.getMethod() == ZipArchiveOutputStream.DEFLATED
466                            ? getBytesInflated() : current.bytesRead;
467
468            // this is at most a single read() operation and can't
469            // exceed the range of int
470            int diff = (int) (current.bytesReadFromStream - inB);
471
472            // Pushback any required bytes
473            if (diff > 0) {
474                pushback(buf.buf, buf.lengthOfLastRead - diff, diff);
475            }
476        }
477
478        if (lastStoredEntry == null && current.hasDataDescriptor) {
479            readDataDescriptor();
480        }
481
482        inf.reset();
483        buf.reset();
484        crc.reset();
485        current = null;
486        lastStoredEntry = null;
487    }
488
489    /**
490     * Read all data of the current entry from the underlying stream
491     * that hasn't been read, yet.
492     */
493    private void drainCurrentEntryData() throws IOException {
494        long remaining = current.entry.getCompressedSize()
495                - current.bytesReadFromStream;
496        while (remaining > 0) {
497            long n = in.read(buf.buf, 0, (int) Math.min(buf.buf.length,
498                    remaining));
499            if (n < 0) {
500                throw new EOFException(
501                        "Truncated ZIP entry: " + current.entry.getName());
502            } else {
503                count(n);
504                remaining -= n;
505            }
506        }
507    }

509    /**
510     * Get the number of bytes Inflater has actually processed.
511     * <p/>
512     * <p>for Java &lt; Java7 the getBytes* methods in
513     * Inflater/Deflater seem to return unsigned ints rather than
514     * longs that start over with 0 at 2^32.</p>
515     * <p/>
516     * <p>The stream knows how many bytes it has read, but not how
517     * many the Inflater actually consumed - it should be between the
518     * total number of bytes read for the entry and the total number
519     * minus the last read operation.  Here we just try to make the
520     * value close enough to the bytes we've read by assuming the
521     * number of bytes consumed must be smaller than (or equal to) the
522     * number of bytes read but not smaller by more than 2^32.</p>
523     */
524    private long getBytesInflated() {
525        long inB = inf.getBytesRead();
526        if (current.bytesReadFromStream >= TWO_EXP_32) {
527            while (inB + TWO_EXP_32 <= current.bytesReadFromStream) {
528                inB += TWO_EXP_32;
529            }
530        }
531        return inB;
532    }

534    private void fill() throws IOException {
535        if (closed) {
536            throw new IOException("The stream is closed");
537        }
538        if ((buf.lengthOfLastRead = in.read(buf.buf)) > 0) {
539            count(buf.lengthOfLastRead);
540            inf.setInput(buf.buf, 0, buf.lengthOfLastRead);
541        }
542    }
543
544    private void readFully(byte[] b) throws IOException {
545        int count = 0, x = 0;
546        while (count != b.length) {
547            count += x = in.read(b, count, b.length - count);
548            if (x == -1) {
549                throw new EOFException();
550            }
551            count(x);
552        }
553    }

555    private void readDataDescriptor() throws IOException {
556        byte[] b = new byte[WORD];
557        readFully(b);
558        ZipLong val = new ZipLong(b);
559        if (ZipLong.DD_SIG.equals(val)) {
560            // data descriptor with signature, skip sig
561            readFully(b);
562            val = new ZipLong(b);
563        }
564        current.entry.setCrc(val.getValue());
565
566        // if there is a ZIP64 extra field, sizes are eight bytes
567        // each, otherwise four bytes each.  Unfortunately some
568        // implementations - namely Java7 - use eight bytes without
569        // using a ZIP64 extra field -
570        // http://bugs.sun.com/bugdatabase/view_bug.do?bug_id=7073588
571
572        // just read 16 bytes and check whether bytes nine to twelve
573        // look like one of the signatures of what could follow a data
574        // descriptor (ignoring archive decryption headers for now).
575        // If so, push back eight bytes and assume sizes are four
576        // bytes, otherwise sizes are eight bytes each.
577        b = new byte[2 * DWORD];
578        readFully(b);
579        ZipLong potentialSig = new ZipLong(b, DWORD);
580        if (potentialSig.equals(ZipLong.CFH_SIG)
581                || potentialSig.equals(ZipLong.LFH_SIG)) {
582            pushback(b, DWORD, DWORD);
583            current.entry.setCompressedSize(ZipLong.getValue(b));
584            current.entry.setSize(ZipLong.getValue(b, WORD));
585        } else {
586            current.entry
587                    .setCompressedSize(ZipEightByteInteger.getLongValue(b));
588            current.entry.setSize(ZipEightByteInteger.getLongValue(b, DWORD));
589        }
590    }

592    /**
593     * Whether this entry requires a data descriptor this library can work with.
594     *
595     * @return true if allowStoredEntriesWithDataDescriptor is true,
596     * the entry doesn't require any data descriptor or the method is
597     * DEFLATED.
598     */
599    private boolean supportsDataDescriptorFor(ZipArchiveEntry entry) {
600        return allowStoredEntriesWithDataDescriptor ||
601                !entry.getGeneralPurposeBit().usesDataDescriptor()
602                || entry.getMethod() == ZipEntry.DEFLATED;
603    }

605    /**
606     * Caches a stored entry that uses the data descriptor.
607     * <p/>
608     * <ul>
609     * <li>Reads a stored entry until the signature of a local file
610     * header, central directory header or data descriptor has been
611     * found.</li>
612     * <li>Stores all entry data in lastStoredEntry.</p>
613     * <li>Rewinds the stream to position at the data
614     * descriptor.</li>
615     * <li>reads the data descriptor</li>
616     * </ul>
617     * <p/>
618     * <p>After calling this method the entry should know its size,
619     * the entry's data is cached and the stream is positioned at the
620     * next local file or central directory header.</p>
621     */
622    private void readStoredEntry() throws IOException {
623        ByteArrayOutputStream bos = new ByteArrayOutputStream();
624        int off = 0;
625        boolean done = false;
626
627        // length of DD without signature
628        int ddLen = current.usesZip64 ? WORD + 2 * DWORD : 3 * WORD;
629
630        while (!done) {
631            int r = in.read(buf.buf, off,
632                    ZipArchiveOutputStream.BUFFER_SIZE - off);
633            if (r <= 0) {
634                // read the whole archive without ever finding a
635                // central directory
636                throw new IOException("Truncated ZIP file");
637            }
638            if (r + off < 4) {
639                // buf is too small to check for a signature, loop
640                off += r;
641                continue;
642            }
643
644            done = bufferContainsSignature(bos, off, r, ddLen);
645            if (!done) {
646                off = cacheBytesRead(bos, off, r, ddLen);
647            }
648        }
649
650        byte[] b = bos.toByteArray();
651        lastStoredEntry = new ByteArrayInputStream(b);
652    }

654    private static final byte[] LFH = ZipLong.LFH_SIG.getBytes();
655    private static final byte[] CFH = ZipLong.CFH_SIG.getBytes();
656    private static final byte[] DD = ZipLong.DD_SIG.getBytes();
657
658    /**
659     * Checks whether the current buffer contains the signature of a
660     * &quot;data decsriptor&quot;, &quot;local file header&quot; or
661     * &quot;central directory entry&quot;.
662     * <p/>
663     * <p>If it contains such a signature, reads the data descriptor
664     * and positions the stream right after the data descriptor.</p>
665     */
666    private boolean bufferContainsSignature(ByteArrayOutputStream bos,
667                                            int offset, int lastRead,
668                                            int expectedDDLen)
669            throws IOException {
670        boolean done = false;
671        int readTooMuch = 0;
672        for (int i = 0; !done && i < lastRead - 4; i++) {
673            if (buf.buf[i] == LFH[0] && buf.buf[i + 1] == LFH[1]) {
674                if ((buf.buf[i + 2] == LFH[2] && buf.buf[i + 3] == LFH[3])
675                        || (buf.buf[i] == CFH[2] && buf.buf[i + 3] == CFH[3])) {
676                    // found a LFH or CFH:
677                    readTooMuch = offset + lastRead - i - expectedDDLen;
678                    done = true;
679                } else if (buf.buf[i + 2] == DD[2] && buf.buf[i + 3] == DD[3]) {
680                    // found DD:
681                    readTooMuch = offset + lastRead - i;
682                    done = true;
683                }
684                if (done) {
685                    // * push back bytes read in excess as well as the data
686                    //   descriptor
687                    // * copy the remaining bytes to cache
688                    // * read data descriptor
689                    pushback(buf.buf, offset + lastRead - readTooMuch,
690                            readTooMuch);
691                    bos.write(buf.buf, 0, i);
692                    readDataDescriptor();
693                }
694            }
695        }
696        return done;
697    }

699    /**
700     * If the last read bytes could hold a data descriptor and an
701     * incomplete signature then save the last bytes to the front of
702     * the buffer and cache everything in front of the potential data
703     * descriptor into the given ByteArrayOutputStream.
704     * <p/>
705     * <p>Data descriptor plus incomplete signature (3 bytes in the
706     * worst case) can be 20 bytes max.</p>
707     */
708    private int cacheBytesRead(ByteArrayOutputStream bos, int offset,
709                               int lastRead, int expecteDDLen) {
710        final int cacheable = offset + lastRead - expecteDDLen - 3;
711        if (cacheable > 0) {
712            bos.write(buf.buf, 0, cacheable);
713            System.arraycopy(buf.buf, cacheable, buf.buf, 0,
714                    expecteDDLen + 3);
715            offset = expecteDDLen + 3;
716        } else {
717            offset += lastRead;
718        }
719        return offset;
720    }

722    private void pushback(byte[] buf, int offset, int length)
723            throws IOException {
724        ((PushbackInputStream) in).unread(buf, offset, length);
725        pushedBackBytes(length);
726    }

763     * housekeeping.

765    private static final class Buffer {

767         * Buffer used as temporary buffer when reading from the stream.

770        /**
771         * {@link #buf buf} may contain data the client hasnt read, yet,

774        private int offsetInBuffer = 0;
775        /**
776         * Number of bytes read from the wrapped stream into {@link #buf
