002    package org.xbib.io.archivers.zip;
003    
004    /**
005     * Constants from stat.h on Unix systems.
006     */
007    public interface UnixStat {
008        /**
009         * Bits used for permissions (and sticky bit)
010         */
011        int PERM_MASK = 07777;
012        /**
013         * Indicates symbolic links.
014         */
015        int LINK_FLAG = 0120000;
016        /**
017         * Indicates plain files.
018         */
019        int FILE_FLAG = 0100000;
020        /**
021         * Indicates directories.
022         */
023        int DIR_FLAG = 040000;
024    
025    }
