002package org.xbib.io.archivers.zip;
003
004import java.util.zip.ZipException;
005
006/**
007 * Exception thrown when attempting to write data that requires Zip64
008 * support to an archive and {@link ZipArchiveOutputStream#setUseZip64
009 * UseZip64} has been set to {@link Zip64Mode#Never Never}.
010 */
011public class Zip64RequiredException extends ZipException {
012
013    /**
014     * Helper to format "entry too big" messages.
015     */
016    static String getEntryTooBigMessage(ZipArchiveEntry ze) {
017        return ze.getName() + "'s size exceeds the limit of 4GByte.";
018    }
019
020    static final String ARCHIVE_TOO_BIG_MESSAGE =
021            "archive's size exceeds the limit of 4GByte.";
022
023    static final String TOO_MANY_ENTRIES_MESSAGE =
024            "archive contains more than 65535 entries.";
025
026    public Zip64RequiredException(String reason) {
027        super(reason);
028    }
029}
