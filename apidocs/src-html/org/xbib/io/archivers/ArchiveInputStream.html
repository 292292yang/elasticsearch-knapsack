002    package org.xbib.io.archivers;
003    
004    import java.io.IOException;
005    import java.io.InputStream;
006    
007    /**
008     * Archive input streams <b>MUST</b> override the
009     * {@link #read(byte[], int, int)} - or {@link #read()} -
010     * method so that reading from the stream generates EOF for the end of
011     * data in each entry as well as at the end of the file proper.
012     * <p/>
013     * The {@link #getNextEntry()} method is used to reset the input stream
014     * ready for reading the data from the next entry.
015     * <p/>
016     * The input stream classes must also implement a method with the signature:
017     * <pre>
018     * public static boolean matches(byte[] signature, int length)
019     * </pre>
020     * which is used by the {@link ArchiveFactory} to autodetect
021     * the archive type from the first few bytes of a stream.
022     */
023    public abstract class ArchiveInputStream extends InputStream {
024    
025        private byte[] SINGLE = new byte[1];
026        private static final int BYTE_MASK = 0xFF;
027    
028        /**
029         * holds the number of bytes read in this stream
030         */
031        private long bytesRead = 0;
032    
033        /**
034         * Returns the next Archive Entry in this Stream.
035         *
036         * @return the next entry,
037         * or {@code null} if there are no more entries
038         * @throws java.io.IOException if the next entry could not be read
039         */
040        public abstract ArchiveEntry getNextEntry() throws IOException;
041    
042        /**
043         * Reads a byte of data. This method will block until enough input is
044         * available.
045         * <p/>
046         * Simply calls the {@link #read(byte[], int, int)} method.
047         * <p/>
048         * MUST be overridden if the {@link #read(byte[], int, int)} method
049         * is not overridden; may be overridden otherwise.
050         *
051         * @return the byte read, or -1 if end of input is reached
052         * @throws java.io.IOException if an I/O error has occurred
053         */
054        @Override
055        public int read() throws IOException {
056            int num = read(SINGLE, 0, 1);
057            return num == -1 ? -1 : SINGLE[0] & BYTE_MASK;
058        }
059    
060        /**
061         * Increments the counter of already read bytes.
062         * Doesn't increment if the EOF has been hit (read == -1)
063         *
064         * @param read the number of bytes read
065         */
066        protected void count(int read) {
067            count((long) read);
068        }
069    
070        /**
071         * Increments the counter of already read bytes.
072         * Doesn't increment if the EOF has been hit (read == -1)
073         *
074         * @param read the number of bytes read
075         */
076        protected void count(long read) {
077            if (read != -1) {
078                bytesRead = bytesRead + read;
079            }
080        }
081    
082        /**
083         * Decrements the counter of already read bytes.
084         *
085         * @param pushedBack the number of bytes pushed back.
086         */
087        protected void pushedBackBytes(long pushedBack) {
088            bytesRead -= pushedBack;
089        }
090    
091        /**
092         * Returns the current number of bytes read from this stream.
093         *
094         * @return the number of read bytes
095         */
096        public long getBytesRead() {
097            return bytesRead;
098        }
099    
100    }
