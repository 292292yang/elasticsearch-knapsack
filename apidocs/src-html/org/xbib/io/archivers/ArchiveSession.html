002package org.xbib.io.archivers;
003
004import org.xbib.io.ObjectPacket;
005import org.xbib.io.Packet;
006import org.xbib.io.Session;
007import org.xbib.io.StreamCodecService;
008import org.xbib.io.StreamUtil;
009
010import java.io.ByteArrayOutputStream;
011import java.io.File;
012import java.io.FileInputStream;
013import java.io.FileNotFoundException;
014import java.io.FileOutputStream;
015import java.io.IOException;
016import java.net.URI;
017import java.util.Date;
018import java.util.Set;
019
020import static org.xbib.io.archivers.ArchiveFactory.createArchiveEntry;
021import static org.xbib.io.archivers.ArchiveFactory.createArchiveInputStream;
022import static org.xbib.io.archivers.ArchiveFactory.createArchiveOutputStream;
023
024/**
025 * A basic archive session
026 */
027public abstract class ArchiveSession implements Session {
028
029    private final static StreamCodecService codecFactory = StreamCodecService.getInstance();
030
031    private boolean isOpen;
032
033    private ArchiveInputStream in;
034
035    private ArchiveOutputStream out;
036
037    private URI uri;
038
039    protected ArchiveSession() {
040    }
041
042    public ArchiveSession setURI(URI uri) {
043        this.uri = uri;
044        return this;
045    }
046
047    protected abstract String getSuffix();
048
049
050    @Override
051    public synchronized void open(Mode mode) throws IOException {
052        if (isOpen) {
053            return;
054        }
055        final String suffix = getSuffix();
056        final String scheme = uri.getScheme();
057        final String part = uri.getSchemeSpecificPart();
058        switch (mode) {
059            case READ: {
060                FileInputStream fin;
061                this.in = null;
062                if (scheme.equals(suffix) || part.endsWith("." + suffix)) {
063                    fin = createFileInputStream(uri, "." + suffix);
064                    this.in = createArchiveInputStream(suffix, fin);
065                } else {
066                    Set<String> codecs = StreamCodecService.getCodecs();
067                    for (String codec : codecs) {
068                        if (scheme.equals(suffix + codec) || part.endsWith("." + suffix + "." + codec)) {
069                            fin = createFileInputStream(uri, "." + suffix + "." + codec);
070                            this.in = createArchiveInputStream(suffix, codecFactory.getCodec(codec).decode(fin));
071                        }
072                    }
073                }
074                this.isOpen = in != null;
075                if (!isOpen) {
076                    throw new FileNotFoundException("can't open for input, check existence or access rights: " + uri);
077                }
078                break;
079            }
080            case WRITE: {
081                FileOutputStream fout;
082                this.out = null;
083                if (scheme.equals(suffix) || part.endsWith("." + suffix)) {
084                    fout = createFileOutputStream(uri, "." + suffix);
085                    this.out = createArchiveOutputStream(suffix, fout);
086                } else {
087                    Set<String> codecs = StreamCodecService.getCodecs();
088                    for (String codec : codecs) {
089                        if (scheme.equals(suffix + codec) || part.endsWith("." + suffix + "." + codec)) {
090                            fout = createFileOutputStream(uri, "." + suffix + "." + codec);
091                            this.out = createArchiveOutputStream(suffix, codecFactory.getCodec(codec).encode(fout));
092                        }
093                    }
094                }
095                this.isOpen = out != null;
096                if (!isOpen) {
097                    throw new FileNotFoundException("can't open for output, check existence or access rights: " + uri);
098                }
099                break;
100            }
101        }
102    }
103
104    @Override
105    public Packet newPacket() {
106        return new ObjectPacket();
107    }
108
109    @Override
110    public synchronized Packet read() throws IOException {
111        if (!isOpen()) {
112            throw new IOException("not open");
113        }
114        if (in == null) {
115            throw new IOException("no input stream found");
116        }
117        ArchiveEntry entry = in.getNextEntry();
118        if (entry == null) {
119            return null;
120        }
121        ObjectPacket packet = new ObjectPacket();
122        String name = entry.getName();
123        packet.name(name);
124        ByteArrayOutputStream bout = new ByteArrayOutputStream();
125        StreamUtil.copy(in, bout);
126        packet.packet(new String(bout.toByteArray()));
127        return packet;
128    }
129
130    @Override
131    public synchronized void write(Packet packet) throws IOException {
132        if (!isOpen()) {
133            throw new IOException("not open");
134        }
135        if (out == null) {
136            throw new IOException("no output stream found");
137        }
138        if (packet == null || packet.toString() == null) {
139            throw new IOException("no payload to write for entry");
140        }
141        byte[] buf = packet.toString().getBytes();
142        if (buf.length > 0) {
143            String name = packet.name();
144            ArchiveEntry entry = createArchiveEntry(getSuffix());
145            entry.setName(name);
146            entry.setLastModified(new Date());
147            entry.setEntrySize(buf.length);
148            out.putArchiveEntry(entry);
149            out.write(buf);
150            out.closeArchiveEntry();
151        }
152    }
153
154    /**
155     * Close session
156     */
157    @Override
158    public synchronized void close() throws IOException {
159        if (!isOpen) {
160            return;
161        }
162        if (out != null) {
163            out.close();
164            out = null;
165        }
166        if (in != null) {
167            in.close();
168            in = null;
169        }
170        this.isOpen = false;
171    }
172
173    @Override
174    public boolean isOpen() {
175        return isOpen;
176    }
177
178    public static boolean canOpen(URI uri, String suffix, boolean withCodecs) {
179        final String scheme = uri.getScheme();
180        final String part = uri.getSchemeSpecificPart();
181        if (scheme.equals(suffix) ||
182                (scheme.equals("file") && part.endsWith("." + suffix.toLowerCase())) ||
183                (scheme.equals("file") && part.endsWith("." + suffix.toUpperCase()))) {
184            return true;
185        }
186        if (withCodecs) {
187            Set<String> codecs = StreamCodecService.getCodecs();
188            for (String codec : codecs) {
189                String s = "." + suffix + "." + codec;
190                if (part.endsWith(s) || part.endsWith(s.toLowerCase()) || part.endsWith(s.toUpperCase())) {
191                    return true;
192                }
193            }
194        }
195        return false;
196    }
197
198
199    /**
200     * Helper method for creating the FileInputStream
201     *
202     * @param uri    the URI
203     * @param suffix the suffix
204     * @return a FileInputStream
205     * @throws java.io.IOException if existence or access rights do not suffice
206     */
207    private FileInputStream createFileInputStream(URI uri, String suffix) throws IOException {
208        String part = uri.getSchemeSpecificPart();
209        String s = part.endsWith(suffix) ? part : part + suffix;
210        File f = new File(s);
211        if (f.isFile() && f.canRead()) {
212            return new FileInputStream(f);
213        } else {
214            throw new FileNotFoundException("can't open for input, check existence or access rights: " + s);
215        }
216    }
217
218    /**
219     * Helper method for creating the FileOutputStream. Creates the directory if
220     * it does not exist.
221     *
222     * @param uri    the URI
223     * @param suffix the suffix
224     * @throws java.io.IOException
225     */
226    private FileOutputStream createFileOutputStream(URI uri, String suffix) throws IOException {
227        String part = uri.getSchemeSpecificPart();
228        String s = part.endsWith(suffix) ? part : part + suffix;
229        File f = new File(s);
230        if (!f.getAbsoluteFile().getParentFile().exists()
231                && !f.getAbsoluteFile().getParentFile().mkdirs()) {
232            throw new IOException("could not create directories to write: " + f);
233        }
234        if (!f.exists()) {
235            return new FileOutputStream(f);
236        } else {
237            throw new IOException("file " + f.getAbsolutePath() + " already exists");
238        }
239    }
