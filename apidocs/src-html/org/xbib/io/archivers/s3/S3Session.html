002package org.xbib.io.archivers.s3;
003
004import com.amazonaws.auth.BasicAWSCredentials;
005import com.amazonaws.auth.DefaultAWSCredentialsProviderChain;
006import com.amazonaws.services.s3.AmazonS3Client;
007
008import org.xbib.io.ObjectPacket;
009import org.xbib.io.Packet;
010import org.xbib.io.Session;
011import org.xbib.io.StreamUtil;
012import org.xbib.io.URIUtil;
013import org.xbib.io.archivers.ArchiveEntry;
014
015import java.io.File;
016import java.io.FileNotFoundException;
017import java.io.FileOutputStream;
018import java.io.IOException;
019import java.net.URI;
020import java.util.Date;
021import java.util.Map;
022
023public class S3Session implements Session {
024
025    private boolean isOpen;
026
027    private URI uri;
028
029    private AmazonS3Client s3client;
030
031    private S3ArchiveInputStream inputStream;
032
033    private S3ArchiveOutputStream outputStream;
034
035    private String accessKey;
036
037    private String secretKey;
038
039    private String bucketName;
040
041    private String key;
042
043    public S3Session() {
044    }
045
046    public S3Session setURI(URI uri) {
047        this.uri = uri;
048        // authorization
049        String auth = uri.getAuthority();
050        if (auth != null) {
051            String[] s = auth.split(":");
052            this.accessKey = s[0];
053            this.secretKey = s.length > 1 ? s[1] : null;
054        }
055        this.s3client = accessKey != null && secretKey != null ?
056                new AmazonS3Client(new BasicAWSCredentials(accessKey, secretKey)) :
057                new AmazonS3Client(new DefaultAWSCredentialsProviderChain());
058        // s3://host --> host
059        s3client.setEndpoint(uri.getHost());
060        return this;
061    }
062
063    @Override
064    public synchronized void open(Mode mode) throws IOException {
065        if (isOpen) {
066            return;
067        }
068        // map URI params to input/output stream services
069        switch (mode) {
070            case READ: {
071                Map<String,String> params = URIUtil.parseQueryString(uri);
072                bucketName = params.get("bucketName");
073                key = params.get("key");
074                inputStream = new S3ArchiveInputStream(s3client, bucketName, key);
075                File f = new File(bucketName + File.separator + key);
076                if (!f.exists()) {
077                    f.mkdirs();
078                }
079                if (!f.canWrite()) {
080                    throw new IOException("can't write to " + f.getAbsolutePath());
081                }
082                isOpen = f.canWrite();
083                break;
084            }
085            case WRITE: {
086                Map<String,String> params = URIUtil.parseQueryString(uri);
087                bucketName = params.get("bucketName");
088                key = params.get("key");
089                outputStream = new S3ArchiveOutputStream(s3client, bucketName, key);
090                isOpen = true;
091                break;
092            }
093        }
094    }
095
096    @Override
097    public Packet<Object> newPacket() {
098        return new ObjectPacket();
099    }
100
101    @Override
102    public synchronized Packet read() throws IOException {
103        if (!isOpen()) {
104            throw new IOException("not open");
105        }
106        if (inputStream == null) {
107            throw new IOException("no input stream found");
108        }
109        ArchiveEntry entry = inputStream.getNextEntry();
110        if (entry == null) {
111            return null;
112        }
113        Packet<Object> packet = newPacket();
114        String name = entry.getName();
115        packet.name(name);
116        packet.packet(bucketName + File.separator + key);
117        File f = new File(bucketName + File.separator + key);
118        if (!f.exists()) {
119            f.mkdirs();
120        }
121        FileOutputStream out = new FileOutputStream(f);
122        StreamUtil.copy(inputStream.getInputStream(), out);
123        out.close();
124        inputStream.close();
125        // packet name = S3 key
126        // packet payload = path where the S3 file is now copied to
127        return packet;
128    }
129
130    @Override
131    public synchronized void write(Packet packet) throws IOException {
132        if (!isOpen()) {
133            throw new IOException("not open");
134        }
135        if (outputStream == null) {
136            throw new IOException("no output stream found");
137        }
138        if (packet == null) {
139            throw new IOException("no packet to write");
140        }
141        // we only need a packet name
142        ArchiveEntry entry = new S3ArchiveEntry();
143        File f = new File(packet.name());
144        if (!f.exists()) {
145            throw new FileNotFoundException("file not found: " + packet.name());
146        }
147        entry.setName(packet.name());
148        entry.setLastModified(new Date(f.lastModified()));
149        entry.setEntrySize(f.length());
150        outputStream.putArchiveEntry(entry); // move file up to S3
151        outputStream.closeArchiveEntry(); // do nothing
152    }
153
154    @Override
155    public void close() throws IOException {
156        s3client.shutdown();
157    }
158
159    @Override
160    public boolean isOpen() {
161        return isOpen;
162    }
163}
