026public class TarSession implements Session {

028    private static final String CHARSET_NAME = "UTF-8";
029    private final StreamCodecService codecFactory = StreamCodecService.getInstance();
030    private boolean isOpen;
031    private FileInputStream fin;
032    private FileOutputStream fout;

034    private TarArchiveOutputStream out;

050    @Override
051    public synchronized void open(Mode mode) throws IOException {
052        if (isOpen) {
053            return;
054        }
055        switch (mode) {
056            case READ:
057                if (scheme.startsWith("targz")) {
058                    String s = part + ".tar.gz";
059                    File f = new File(s);
060                    if (f.isFile() && f.canRead()) {
061                        this.fin = new FileInputStream(f);
062                        this.in = new TarArchiveInputStream(codecFactory.getCodec("gz").decode(fin), "UTF-8");
063                        this.isOpen = true;
064                    } else {
065                        throw new FileNotFoundException("check existence or access rights: " + s);
066                    }
067                } else if (scheme.startsWith("tarbz2")) {
068                    String s = part + ".tar.bz2";
069                    File f = new File(s);
070                    if (f.isFile() && f.canRead()) {
071                        this.fin = new FileInputStream(f);
072                        this.in = new TarArchiveInputStream(codecFactory.getCodec("bz2").decode(fin), "UTF-8");
073                        this.isOpen = true;
074                    } else {
075                        throw new FileNotFoundException("check existence or access rights: " + s);
076                    }
077                } else if (scheme.startsWith("tarxz")) {
078                    String s = part + ".tar.xz";
079                    File f = new File(s);
080                    if (f.isFile() && f.canRead()) {
081                        this.fin = new FileInputStream(f);
082                        this.in = new TarArchiveInputStream(codecFactory.getCodec("xz").decode(fin), "UTF-8");
083                        this.isOpen = true;
084                    } else {
085                        throw new FileNotFoundException("check existence or access rights: " + s);
086                    }
087                } else {
088                    String s = part + ".tar";
089                    File f = new File(s);
090                    if (f.isFile() && f.canRead()) {
091                        this.fin = new FileInputStream(f);
092                        this.in = new TarArchiveInputStream(fin, "UTF-8");
093                        this.isOpen = true;
094                    } else {
095                        throw new FileNotFoundException("check existence or access rights: " + s);
096                    }
097                }
098                break;
099            case WRITE:
100                if (scheme.equals("targz")) {
101                    createFileOutputStream(".tar.gz");
102                    this.out = new TarArchiveOutputStream(codecFactory.getCodec("gz").encode(fout), "UTF-8");
103                    out.setLongFileMode(TarArchiveOutputStream.LONGFILE_GNU);
104                    this.isOpen = true;
105                } else if (scheme.equals("tarbz2")) {
106                    createFileOutputStream(".tar.bz2");
107                    this.out = new TarArchiveOutputStream(codecFactory.getCodec("bz2").encode(fout), "UTF-8");
108                    out.setLongFileMode(TarArchiveOutputStream.LONGFILE_GNU);
109                    this.isOpen = true;
110                } else if (scheme.equals("tarxz")) {
111                    createFileOutputStream(".tar.xz");
112                    this.out = new TarArchiveOutputStream(codecFactory.getCodec("xz").encode(fout), "UTF-8");
113                    out.setLongFileMode(TarArchiveOutputStream.LONGFILE_GNU);
114                    this.isOpen = true;
115                } else {
116                    createFileOutputStream(".tar");
117                    this.out = new TarArchiveOutputStream(fout, "UTF-8");
118                    out.setLongFileMode(TarArchiveOutputStream.LONGFILE_GNU);
119                    this.isOpen = true;
120                }
121                break;
122        }
123    }
124
125    /**
126     * Close Tar session
127     */
128    @Override
129    public synchronized void close() throws IOException {
130        if (!isOpen) {
131            return;
132        }
133        if (out != null) {
134            out.flush();
135            out.close();
136            out = null;
137        }
138        if (in != null) {
139            in.close();
140        }
141        this.isOpen = false;
142    }
143
144    @Override
145    public boolean isOpen() {
146        return isOpen;
147    }
148
149    TarArchiveInputStream getInputStream() {
150        return in;
151    }
152
153    TarArchiveOutputStream getOutputStream() {
154        return out;
155    }
156
157    /**
158     * Helper method for creating the FileOutputStream. Creates the directory if
159     * it doesnot exist.
160     *
161     * @param suffix
162     * @throws java.io.IOException
163     */
164    private synchronized void createFileOutputStream(String suffix) throws IOException {
165        File f = new File(part + (part.endsWith(suffix) ? "" : suffix));
166        if (!f.getAbsoluteFile().getParentFile().exists()
167                && !f.getAbsoluteFile().getParentFile().mkdirs()) {
168            throw new RuntimeException(
169                    "Could not create directories to store file: " + f);
170        }
171        if (f.exists()) {
172            throw new IOException("file " + f.getAbsolutePath() + " already exists");
173        }
174        this.fout = new FileOutputStream(f);
175    }
176
177    @Override
178    public Packet read() throws IOException {
179        if (!isOpen()) {
180            throw new IOException("not open");
181        }
182        if (getInputStream() == null) {
183            throw new IOException("no tar input stream");
184        }
185        TarArchiveEntry entry = getInputStream().getNextTarEntry();
186        if (entry == null) {
187            return null;
188        }
189        ObjectPacket packet = new ObjectPacket();
190        String name = entry.getName();
191        packet.setName(name);
192        ByteArrayOutputStream bout = new ByteArrayOutputStream();
193        IOUtils.copy(getInputStream(), bout);
194        packet.setPacket(new String(bout.toByteArray() , CHARSET_NAME));
195        return packet;
196    }
197
198    @Override
199    public void write(Packet packet) throws IOException {
200        if (packet == null || packet.toString() == null) {
201            throw new IOException("no packet to write");
202        }
203        byte[] buf = packet.toString().getBytes(CHARSET_NAME);
204        if (buf.length > 0) {
205            String name = createEntryName(packet.getName());
206            TarArchiveEntry entry = new TarArchiveEntry(name);
207            entry.setModTime(new Date());
208            entry.setSize(buf.length);
209            getOutputStream().putArchiveEntry(entry);
210            getOutputStream().write(buf);
211            getOutputStream().closeArchiveEntry();
212        }
213    }

216    private static final NumberFormat nf = DecimalFormat.getIntegerInstance();
217
218    static {
219        nf.setGroupingUsed(false);
220        nf.setMinimumIntegerDigits(12);

223
224    private String createEntryName(String name) {
225        StringBuilder sb = new StringBuilder();
226        if (name != null && name.length() > 0) {
227            // normalize name if it contains relative "." or ".."
228            try {
229                URI n = new URI("file:" + name).normalize();
230                sb.append(n.getSchemeSpecificPart());
231            } catch (URISyntaxException e) {
232                sb.append(name);
233            }
234        }
235        return sb.toString();
236    }
237
238    @Override
239    public Packet newPacket() {
240        return new ObjectPacket();
241    }
