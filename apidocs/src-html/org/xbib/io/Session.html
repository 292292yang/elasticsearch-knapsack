002package org.xbib.io;
003
004import java.io.IOException;
005import java.nio.file.Path;
006import java.util.EnumSet;
007
008/**
009 * The Session interface is used for being opened, receive
010 * operations, and being closed. Sessions must be opened before the first
011 * operation and closed after the last operation.
012 */
013public interface Session<P extends Packet> {
014
015    enum Mode {
016
017        READ, WRITE, READ_WRITE, APPEND, DEFERRED_WRITE, OVERWRITE, URI_ENCODED, NONE;
018    }
019
020    /**
021     * Open valve with a given input/output mode
022     *
023     * @throws java.io.IOException if valve can not be opened
024     */
025    void open(EnumSet<Mode> mode, Path path) throws IOException;
026
027    P newPacket();
028
029    P read() throws IOException;
030
031    void write(P packet) throws IOException;
032
033    /**
034     * Close valve
035     *
036     * @throws java.io.IOException if valve can not be closed
037     */
038    void close() throws IOException;
039
040    /**
041     * Checks if this session has been successfully opened.
042     *
043     * @return true if the session is open
044     */
045    boolean isOpen();
046}
