001    
002    package org.xbib.io;
003    
004    import java.io.IOException;
005    import java.net.URI;
006    import java.util.Map;
007    import java.util.ServiceConfigurationError;
008    import java.util.ServiceLoader;
009    import java.util.WeakHashMap;
010    
011    /**
012     * The Connection service
013     */
014    public final class ConnectionService<S extends Session> {
015    
016        private final Map<String, ConnectionFactory<S>> factories = new WeakHashMap<String, ConnectionFactory<S>>();
017    
018        private final static ConnectionService instance = new ConnectionService();
019    
020        private ConnectionService() {
021            ServiceLoader<ConnectionFactory> loader = ServiceLoader.load(ConnectionFactory.class);
022            for (ConnectionFactory factory : loader) {
023                if (!factories.containsKey(factory.getName())) {
024                    factories.put(factory.getName(), factory);
025                }
026            }
027        }
028    
029        public static <S extends Session> ConnectionService<S> getInstance() {
030            return instance;
031        }
032    
033        public ConnectionFactory<S> getConnectionFactory(String name)
034                throws IOException {
035            if (factories.containsKey(name)) {
036                return factories.get(name);
037            }
038            throw new ServiceConfigurationError("no connection factory found for scheme " + name);
039        }
040    
041        public ConnectionFactory<S> getConnectionFactory(URI uri) {
042            for (ConnectionFactory<S> factory : factories.values()) {
043                if (factory.canOpen(uri)) {
044                    return factory;
045                }
046            }
047            throw new ServiceConfigurationError("no connection factory found for " + uri);
048        }
049    }
