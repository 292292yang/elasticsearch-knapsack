029    @Test
030    public void testBulk() throws Exception {
031        File exportFile = File.createTempFile("knapsack-bulk-", ".bulk");
032        Path exportPath = Paths.get(URI.create("file:" + exportFile.getAbsolutePath()));
033        client("1").index(new IndexRequest().index("index1").type("test1").id("doc1").source("content","Hello World").refresh(true)).actionGet();
034        KnapsackExportRequestBuilder requestBuilder = new KnapsackExportRequestBuilder(client("1"))
035                .setPath(exportPath)
036                .setOverwriteAllowed(true);
037        KnapsackExportResponse knapsackExportResponse = requestBuilder.execute().actionGet();
038        if (!knapsackExportResponse.isRunning()) {
039            logger.error(knapsackExportResponse.getReason());
040        }
041        assertTrue(knapsackExportResponse.isRunning());
042        KnapsackStateRequestBuilder knapsackStateRequestBuilder =
043               new KnapsackStateRequestBuilder(client("1"));
044        KnapsackStateResponse knapsackStateResponse = knapsackStateRequestBuilder.execute().actionGet();
045        knapsackStateResponse.isExportActive(exportPath);
046        Thread.sleep(1000L);
047        // delete index
048        client("1").admin().indices().delete(new DeleteIndexRequest("index1")).actionGet();
049        KnapsackImportRequestBuilder knapsackImportRequestBuilder =
050                new KnapsackImportRequestBuilder(client("1"))
051                .setPath(exportPath);
052        KnapsackImportResponse knapsackImportResponse = knapsackImportRequestBuilder.execute().actionGet();
053        if (!knapsackImportResponse.isRunning()) {
054            logger.error(knapsackImportResponse.getReason());
055        }
056        assertTrue(knapsackImportResponse.isRunning());
057        Thread.sleep(1000L);
058        // count
059        long count = client("1").prepareCount("index1").setQuery(QueryBuilders.matchAllQuery()).execute().actionGet().getCount();
060        assertEquals(1L, count);
061    }
