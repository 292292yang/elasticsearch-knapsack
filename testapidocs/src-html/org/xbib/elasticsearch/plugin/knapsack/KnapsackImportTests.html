033    @Test
034    public void testMinimalImport() throws Exception {
035        File exportFile = File.createTempFile("minimal-import-", ".bulk");
036        Path exportPath = Paths.get(URI.create("file:" + exportFile.getAbsolutePath()));
037        client("1").index(new IndexRequest().index("index1").type("test1").id("doc1")
038                .source("content", "Hello World").refresh(true)).actionGet();
039        KnapsackExportRequestBuilder requestBuilder = new KnapsackExportRequestBuilder(client("1"))
040                .setPath(exportPath)
041                .setOverwriteAllowed(true);
042        KnapsackExportResponse knapsackExportResponse = requestBuilder.execute().actionGet();
043        if (!knapsackExportResponse.isRunning()) {
044            logger.error(knapsackExportResponse.getReason());
045        }
046        assertTrue(knapsackExportResponse.isRunning());
047        KnapsackStateRequestBuilder knapsackStateRequestBuilder =
048               new KnapsackStateRequestBuilder(client("2"));
049        KnapsackStateResponse knapsackStateResponse = knapsackStateRequestBuilder.execute().actionGet();
050        knapsackStateResponse.isExportActive(exportPath);
051        Thread.sleep(2000L);
052        BufferedReader reader = new BufferedReader(new FileReader(exportFile));
053        assertEquals("{\"index\":{\"_index\":\"index1\",\"_type\":\"test1\",\"_id\":\"doc1\"}", reader.readLine());
054        assertEquals("{\"content\":\"Hello World\"}", reader.readLine());
055        reader.close();
056        // delete index
057        client("1").admin().indices().prepareDelete("index1").execute().actionGet();
058        KnapsackImportRequestBuilder knapsackImportRequestBuilder = new KnapsackImportRequestBuilder(client("1"))
059                .setPath(exportPath);
060        KnapsackImportResponse knapsackImportResponse = knapsackImportRequestBuilder.execute().actionGet();
061        if (!knapsackImportResponse.isRunning()) {
062            logger.error(knapsackImportResponse.getReason());
063        }
064        Thread.sleep(2000L);
065        // count
066        long count = client("1").prepareCount("index1").setQuery(QueryBuilders.matchAllQuery()).execute().actionGet().getCount();
067        logger.info("testMinimalImport count={}", count);
068        assertEquals(1L, count);
069    }
