030    @Test
031    public void testTar() throws Exception {
032        File exportFile = File.createTempFile("knapsack-tar-", ".tar");
033        Path exportPath = Paths.get(URI.create("file:" + exportFile.getAbsolutePath()));
034        client("1").index(new IndexRequest().index("index1").type("test1").id("doc1").source("content","Hello World").refresh(true)).actionGet();
035        KnapsackExportRequestBuilder requestBuilder = new KnapsackExportRequestBuilder(client("1"))
036                .setPath(exportPath)
037                .setOverwriteAllowed(true);
038        KnapsackExportResponse knapsackExportResponse = requestBuilder.execute().actionGet();
039        if (!knapsackExportResponse.isRunning()) {
040            logger.error(knapsackExportResponse.getReason());
041        }
042        assertTrue(knapsackExportResponse.isRunning());
043        KnapsackStateRequestBuilder knapsackStateRequestBuilder =
044               new KnapsackStateRequestBuilder(client("2"));
045        KnapsackStateResponse knapsackStateResponse = knapsackStateRequestBuilder.execute().actionGet();
046        knapsackStateResponse.isExportActive(exportPath);
047        Thread.sleep(1000L);
048        // delete index
049        client("1").admin().indices().delete(new DeleteIndexRequest("index1")).actionGet();
050        KnapsackImportRequestBuilder knapsackImportRequestBuilder = new KnapsackImportRequestBuilder(client("1"))
051                .setPath(exportPath);
052        KnapsackImportResponse knapsackImportResponse = knapsackImportRequestBuilder.execute().actionGet();
053        if (!knapsackImportResponse.isRunning()) {
054            logger.error(knapsackImportResponse.getReason());
055        }
056        assertTrue(knapsackImportResponse.isRunning());
057        Thread.sleep(1000L);
058        // count
059        long count = client("1").prepareCount("index1").setQuery(QueryBuilders.matchAllQuery()).execute().actionGet().getCount();
060        assertEquals(1L, count);
061    }

178                .setPath(path);
