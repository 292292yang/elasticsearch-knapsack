014import java.io.BufferedReader;
015import java.io.File;
016import java.io.FileReader;
017import java.io.IOException;
018import java.net.URI;
019import java.nio.file.Path;
020import java.nio.file.Paths;
021
022import static org.junit.Assert.assertEquals;
023import static org.junit.Assert.assertFalse;
024import static org.junit.Assert.assertTrue;
025
026public class KnapsackExportTests extends AbstractNodeTestHelper {
027
028    private final static ESLogger logger = ESLoggerFactory.getLogger(KnapsackExportTests.class.getName());
029
030    @Test
031    public void testMinimalExport() throws Exception {
032        File exportFile = File.createTempFile("minimal-export-", ".bulk");
033        Path exportPath = Paths.get(URI.create("file:" + exportFile.getAbsolutePath()));
034        client("1").index(new IndexRequest().index("index1").type("test1").id("doc1").source("content", "Hello World").refresh(true)).actionGet();
035        logger.info("waiting for recovery");
036        ClientHelper.waitForRecovery(client("1"), "index1");
037        logger.info("recovered");
038        KnapsackExportRequestBuilder requestBuilder = new KnapsackExportRequestBuilder(client("1"))
039                .setPath(exportPath)
040                .setOverwriteAllowed(true);
041        KnapsackExportResponse knapsackExportResponse = requestBuilder.execute().actionGet();
042        if (!knapsackExportResponse.isRunning()) {
043            logger.error(knapsackExportResponse.getReason());
044            throw new IOException("knapsack export not running");
045        }
046        assertTrue(knapsackExportResponse.isRunning());
047        Thread.sleep(2000L);
048        KnapsackStateRequestBuilder knapsackStateRequestBuilder =
049               new KnapsackStateRequestBuilder(client("1"));
050        KnapsackStateResponse knapsackStateResponse = knapsackStateRequestBuilder.execute().actionGet();
051        logger.info("response={}", knapsackStateResponse);
052        assertFalse(knapsackStateResponse.isExportActive(exportPath));
053//        Thread.sleep(2000L);
054        BufferedReader reader = new BufferedReader(new FileReader(exportFile));
055        assertEquals("{\"index\":{\"_index\":\"index1\",\"_type\":\"test1\",\"_id\":\"doc1\"}", reader.readLine());
056        assertEquals("{\"content\":\"Hello World\"}", reader.readLine());
057        reader.close();
058    }
059
060}
