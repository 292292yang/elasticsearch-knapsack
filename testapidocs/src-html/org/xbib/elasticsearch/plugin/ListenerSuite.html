001package org.xbib.elasticsearch.plugin;
002
003import org.junit.runner.Runner;
004import org.junit.runner.notification.RunNotifier;
005import org.junit.runners.Suite;
006import org.junit.runners.model.InitializationError;
007import org.junit.runners.model.RunnerBuilder;
008
009public class ListenerSuite extends Suite {
010
011    private final TestListener listener = new TestListener();
012
013    public ListenerSuite(Class<?> klass, RunnerBuilder builder) throws InitializationError {
014        super(klass, builder);
015    }
016
017    @Override
018    protected void runChild(Runner runner, RunNotifier notifier) {
019        notifier.addListener(listener);
020        runner.run(notifier);
021        notifier.removeListener(listener);
022    }
023}
